026package java.nio.file.attribute;
027
028/**
029 * Defines the bits for use with the {@link PosixFileAttributes#permissions()
030 * permissions} attribute.
031 *
032 * <p> The {@link PosixFilePermissions} class defines methods for manipulating
033 * set of permissions.
034 *
035 * @since 1.7
036 */
037
038public enum PosixFilePermission {
039
040    /**
041     * Read permission, owner.
042     */
043    OWNER_READ,
044
045    /**
046     * Write permission, owner.
047     */
048    OWNER_WRITE,
049
050    /**
051     * Execute/search permission, owner.
052     */
053    OWNER_EXECUTE,
054
055    /**
056     * Read permission, group.
057     */
058    GROUP_READ,
059
060    /**
061     * Write permission, group.
062     */
063    GROUP_WRITE,
064
065    /**
066     * Execute/search permission, group.
067     */
068    GROUP_EXECUTE,
069
070    /**
071     * Read permission, others.
072     */
073    OTHERS_READ,
074
075    /**
076     * Write permission, others.
077     */
078    OTHERS_WRITE,
079
080    /**
081     * Execute/search permission, others.
082     */
083    OTHERS_EXECUTE;
084}
