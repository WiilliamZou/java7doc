026package java.nio.file.attribute;
027
028/**
029 * Basic attributes associated with a file in a file system.
030 *
031 * <p> Basic file attributes are attributes that are common to many file systems
032 * and consist of mandatory and optional file attributes as defined by this
033 * interface.
034 *
035 * <p> <b>Usage Example:</b>
036 * <pre>
037 *    Path file = ...
038 *    BasicFileAttributes attrs = Files.readAttributes(file, BasicFileAttributes.class);
039 * </pre>
040 *
041 * @since 1.7
042 *
043 * @see BasicFileAttributeView
044 */
045
046public interface BasicFileAttributes {
047
048    /**
049     * Returns the time of last modification.
050     *
051     * <p> If the file system implementation does not support a time stamp
052     * to indicate the time of last modification then this method returns an
053     * implementation specific default value, typically a {@code FileTime}
054     * representing the epoch (1970-01-01T00:00:00Z).
055     *
056     * @return  a {@code FileTime} representing the time the file was last
057     *          modified
058     */
059    FileTime lastModifiedTime();
060
061    /**
062     * Returns the time of last access.
063     *
064     * <p> If the file system implementation does not support a time stamp
065     * to indicate the time of last access then this method returns
066     * an implementation specific default value, typically the {@link
067     * #lastModifiedTime() last-modified-time} or a {@code FileTime}
068     * representing the epoch (1970-01-01T00:00:00Z).
069     *
070     * @return  a {@code FileTime} representing the time of last access
071     */
072    FileTime lastAccessTime();
073
074    /**
075     * Returns the creation time. The creation time is the time that the file
076     * was created.
077     *
078     * <p> If the file system implementation does not support a time stamp
079     * to indicate the time when the file was created then this method returns
080     * an implementation specific default value, typically the {@link
081     * #lastModifiedTime() last-modified-time} or a {@code FileTime}
082     * representing the epoch (1970-01-01T00:00:00Z).
083     *
084     * @return   a {@code FileTime} representing the time the file was created
085     */
086    FileTime creationTime();
087
088    /**
089     * Tells whether the file is a regular file with opaque content.
090     */
091    boolean isRegularFile();
092
093    /**
094     * Tells whether the file is a directory.
095     */
096    boolean isDirectory();
097
098    /**
099     * Tells whether the file is a symbolic link.
100     */
101    boolean isSymbolicLink();
102
103    /**
104     * Tells whether the file is something other than a regular file, directory,
105     * or symbolic link.
106     */
107    boolean isOther();
108
109    /**
110     * Returns the size of the file (in bytes). The size may differ from the
111     * actual size on the file system due to compression, support for sparse
112     * files, or other reasons. The size of files that are not {@link
113     * #isRegularFile regular} files is implementation specific and
114     * therefore unspecified.
115     *
116     * @return  the file size, in bytes
117     */
118    long size();
119
120    /**
121     * Returns an object that uniquely identifies the given file, or {@code
122     * null} if a file key is not available. On some platforms or file systems
123     * it is possible to use an identifier, or a combination of identifiers to
124     * uniquely identify a file. Such identifiers are important for operations
125     * such as file tree traversal in file systems that support <a
126     * href="../package-summary.html#links">symbolic links</a> or file systems
127     * that allow a file to be an entry in more than one directory. On UNIX file
128     * systems, for example, the <em>device ID</em> and <em>inode</em> are
129     * commonly used for such purposes.
130     *
131     * <p> The file key returned by this method can only be guaranteed to be
132     * unique if the file system and files remain static. Whether a file system
133     * re-uses identifiers after a file is deleted is implementation dependent and
134     * therefore unspecified.
135     *
136     * <p> File keys returned by this method can be compared for equality and are
137     * suitable for use in collections. If the file system and files remain static,
138     * and two files are the {@link java.nio.file.Files#isSameFile same} with
139     * non-{@code null} file keys, then their file keys are equal.
140     *
141     * @see java.nio.file.Files#walkFileTree
142     */
143    Object fileKey();
144}
