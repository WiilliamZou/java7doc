026package java.nio.file.attribute;
027
028import java.security.Principal;
029
030/**
031 * A {@code Principal} representing an identity used to determine access rights
032 * to objects in a file system.
033 *
034 * <p> On many platforms and file systems an entity requires appropriate access
035 * rights or permissions in order to access objects in a file system. The
036 * access rights are generally performed by checking the identity of the entity.
037 * For example, on implementations that use Access Control Lists (ACLs) to
038 * enforce privilege separation then a file in the file system may have an
039 * associated ACL that determines the access rights of identities specified in
040 * the ACL.
041 *
042 * <p> A {@code UserPrincipal} object is an abstract representation of an
043 * identity. It has a {@link #getName() name} that is typically the username or
044 * account name that it represents. User principal objects may be obtained using
045 * a {@link UserPrincipalLookupService}, or returned by {@link
046 * FileAttributeView} implementations that provide access to identity related
047 * attributes. For example, the {@link AclFileAttributeView} and {@link
048 * PosixFileAttributeView} provide access to a file's {@link
049 * PosixFileAttributes#owner owner}.
050 *
051 * @since 1.7
052 */
053
054public interface UserPrincipal extends Principal { }
