026package java.nio.file;
027
028/**
029 * Defines the standard open options.
030 *
031 * @since 1.7
032 */
033
034public enum StandardOpenOption implements OpenOption {
035    /**
036     * Open for read access.
037     */
038    READ,
039
040    /**
041     * Open for write access.
042     */
043    WRITE,
044
045    /**
046     * If the file is opened for {@link #WRITE} access then bytes will be written
047     * to the end of the file rather than the beginning.
048     *
049     * <p> If the file is opened for write access by other programs, then it
050     * is file system specific if writing to the end of the file is atomic.
051     */
052    APPEND,
053
054    /**
055     * If the file already exists and it is opened for {@link #WRITE}
056     * access, then its length is truncated to 0. This option is ignored
057     * if the file is opened only for {@link #READ} access.
058     */
059    TRUNCATE_EXISTING,
060
061    /**
062     * Create a new file if it does not exist.
063     * This option is ignored if the {@link #CREATE_NEW} option is also set.
064     * The check for the existence of the file and the creation of the file
065     * if it does not exist is atomic with respect to other file system
066     * operations.
067     */
068    CREATE,
069
070    /**
071     * Create a new file, failing if the file already exists.
072     * The check for the existence of the file and the creation of the file
073     * if it does not exist is atomic with respect to other file system
074     * operations.
075     */
076    CREATE_NEW,
077
078    /**
079     * Delete on close. When this option is present then the implementation
080     * makes a <em>best effort</em> attempt to delete the file when closed
081     * by the appropriate {@code close} method. If the {@code close} method is
082     * not invoked then a <em>best effort</em> attempt is made to delete the
083     * file when the Java virtual machine terminates (either normally, as
084     * defined by the Java Language Specification, or where possible, abnormally).
085     * This option is primarily intended for use with <em>work files</em> that
086     * are used solely by a single instance of the Java virtual machine. This
087     * option is not recommended for use when opening files that are open
088     * concurrently by other entities. Many of the details as to when and how
089     * the file is deleted are implementation specific and therefore not
090     * specified. In particular, an implementation may be unable to guarantee
091     * that it deletes the expected file when replaced by an attacker while the
092     * file is open. Consequently, security sensitive applications should take
093     * care when using this option.
094     *
095     * <p> For security reasons, this option may imply the {@link
096     * LinkOption#NOFOLLOW_LINKS} option. In other words, if the option is present
097     * when opening an existing file that is a symbolic link then it may fail
098     * (by throwing {@link java.io.IOException}).
099     */
100    DELETE_ON_CLOSE,
101
102    /**
103     * Sparse file. When used with the {@link #CREATE_NEW} option then this
104     * option provides a <em>hint</em> that the new file will be sparse. The
105     * option is ignored when the file system does not support the creation of
106     * sparse files.
107     */
108    SPARSE,
109
110    /**
111     * Requires that every update to the file's content or metadata be written
112     * synchronously to the underlying storage device.
113     *
114     * @see <a href="package-summary.html#integrity">Synchronized I/O file integrity</a>
115     */
116    SYNC,
117
118    /**
119     * Requires that every update to the file's content be written
120     * synchronously to the underlying storage device.
121     *
122     * @see <a href="package-summary.html#integrity">Synchronized I/O file integrity</a>
123     */
124    DSYNC;
125}
