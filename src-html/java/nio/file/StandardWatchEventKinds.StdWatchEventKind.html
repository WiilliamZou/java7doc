026package java.nio.file;
027
028/**
029 * Defines the <em>standard</em> event kinds.
030 *
031 * @since 1.7
032 */
033
034public final class StandardWatchEventKinds {
035    private StandardWatchEventKinds() { }
036
037    /**
038     * A special event to indicate that events may have been lost or
039     * discarded.
040     *
041     * <p> The {@link WatchEvent#context context} for this event is
042     * implementation specific and may be {@code null}. The event {@link
043     * WatchEvent#count count} may be greater than {@code 1}.
044     *
045     * @see WatchService
046     */
047    public static final WatchEvent.Kind<Object> OVERFLOW =
048        new StdWatchEventKind<Object>("OVERFLOW", Object.class);
049
050    /**
051     * Directory entry created.
052     *
053     * <p> When a directory is registered for this event then the {@link WatchKey}
054     * is queued when it is observed that an entry is created in the directory
055     * or renamed into the directory. The event {@link WatchEvent#count count}
056     * for this event is always {@code 1}.
057     */
058    public static final WatchEvent.Kind<Path> ENTRY_CREATE =
059        new StdWatchEventKind<Path>("ENTRY_CREATE", Path.class);
060
061    /**
062     * Directory entry deleted.
063     *
064     * <p> When a directory is registered for this event then the {@link WatchKey}
065     * is queued when it is observed that an entry is deleted or renamed out of
066     * the directory. The event {@link WatchEvent#count count} for this event
067     * is always {@code 1}.
068     */
069    public static final WatchEvent.Kind<Path> ENTRY_DELETE =
070        new StdWatchEventKind<Path>("ENTRY_DELETE", Path.class);
071
072    /**
073     * Directory entry modified.
074     *
075     * <p> When a directory is registered for this event then the {@link WatchKey}
076     * is queued when it is observed that an entry in the directory has been
077     * modified. The event {@link WatchEvent#count count} for this event is
078     * {@code 1} or greater.
079     */
080    public static final WatchEvent.Kind<Path> ENTRY_MODIFY =
081        new StdWatchEventKind<Path>("ENTRY_MODIFY", Path.class);
082
083    private static class StdWatchEventKind<T> implements WatchEvent.Kind<T> {
084        private final String name;
085        private final Class<T> type;
086        StdWatchEventKind(String name, Class<T> type) {
087            this.name = name;
088            this.type = type;
089        }
090        @Override public String name() { return name; }
091        @Override public Class<T> type() { return type; }
092        @Override public String toString() { return name; }
093    }
094}
