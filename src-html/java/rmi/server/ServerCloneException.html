026package java.rmi.server;
027
028/**
029 * A <code>ServerCloneException</code> is thrown if a remote exception occurs
030 * during the cloning of a <code>UnicastRemoteObject</code>.
031 *
032 * <p>As of release 1.4, this exception has been retrofitted to conform to
033 * the general purpose exception-chaining mechanism.  The "nested exception"
034 * that may be provided at construction time and accessed via the public
035 * {@link #detail} field is now known as the <i>cause</i>, and may be
036 * accessed via the {@link Throwable#getCause()} method, as well as
037 * the aforementioned "legacy field."
038 *
039 * <p>Invoking the method {@link Throwable#initCause(Throwable)} on an
040 * instance of <code>ServerCloneException</code> always throws {@link
041 * IllegalStateException}.
042 *
043 * @author  Ann Wollrath
044 * @since   JDK1.1
045 * @see     java.rmi.server.UnicastRemoteObject#clone()
046 */
047public class ServerCloneException extends CloneNotSupportedException {
048
049    /**
050     * The cause of the exception.
051     *
052     * <p>This field predates the general-purpose exception chaining facility.
053     * The {@link Throwable#getCause()} method is now the preferred means of
054     * obtaining this information.
055     *
056     * @serial
057     */
058    public Exception detail;
059
060    /* indicate compatibility with JDK 1.1.x version of class */
061    private static final long serialVersionUID = 6617456357664815945L;
062
063    /**
064     * Constructs a <code>ServerCloneException</code> with the specified
065     * detail message.
066     *
067     * @param s the detail message.
068     */
069    public ServerCloneException(String s) {
070        super(s);
071        initCause(null);  // Disallow subsequent initCause
072    }
073
074    /**
075     * Constructs a <code>ServerCloneException</code> with the specified
076     * detail message and cause.
077     *
078     * @param s the detail message.
079     * @param cause the cause
080     */
081    public ServerCloneException(String s, Exception cause) {
082        super(s);
083        initCause(null);  // Disallow subsequent initCause
084        detail = cause;
085    }
086
087    /**
088     * Returns the detail message, including the message from the cause, if
089     * any, of this exception.
090     *
091     * @return the detail message
092     */
093    public String getMessage() {
094        if (detail == null)
095            return super.getMessage();
096        else
097            return super.getMessage() +
098                "; nested exception is: \n\t" +
099                detail.toString();
100    }
101
102    /**
103     * Returns the cause of this exception.  This method returns the value
104     * of the {@link #detail} field.
105     *
106     * @return  the cause, which may be <tt>null</tt>.
107     * @since   1.4
108     */
109    public Throwable getCause() {
110        return detail;
111    }
112}
