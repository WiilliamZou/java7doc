026package java.rmi.server;
027
028import java.rmi.RemoteException;
029
030/**
031 * A <code>SkeletonNotFoundException</code> is thrown if the
032 * <code>Skeleton</code> corresponding to the remote object being
033 * exported is not found.  Skeletons are no longer required, so this
034 * exception is never thrown.
035 *
036 * @since   JDK1.1
037 * @deprecated no replacement.  Skeletons are no longer required for remote
038 * method calls in the Java 2 platform v1.2 and greater.
039 */
040@Deprecated
041public class SkeletonNotFoundException extends RemoteException {
042
043    /* indicate compatibility with JDK 1.1.x version of class */
044    private static final long serialVersionUID = -7860299673822761231L;
045
046    /**
047     * Constructs a <code>SkeletonNotFoundException</code> with the specified
048     * detail message.
049     *
050     * @param s the detail message.
051     * @since JDK1.1
052     */
053    public SkeletonNotFoundException(String s) {
054        super(s);
055    }
056
057    /**
058     * Constructs a <code>SkeletonNotFoundException</code> with the specified
059     * detail message and nested exception.
060     *
061     * @param s the detail message.
062     * @param ex the nested exception
063     * @since JDK1.1
064     */
065    public SkeletonNotFoundException(String s, Exception ex) {
066        super(s, ex);
067    }
068}
