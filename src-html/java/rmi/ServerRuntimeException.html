026package java.rmi;
027
028/**
029 * From a server executing on JDK&nbsp;1.1, a
030 * <code>ServerRuntimeException</code> is thrown as a result of a
031 * remote method invocation when a <code>RuntimeException</code> is
032 * thrown while processing the invocation on the server, either while
033 * unmarshalling the arguments, executing the remote method itself, or
034 * marshalling the return value.
035 *
036 * A <code>ServerRuntimeException</code> instance contains the original
037 * <code>RuntimeException</code> that occurred as its cause.
038 *
039 * <p>A <code>ServerRuntimeException</code> is not thrown from servers
040 * executing on the Java 2 platform v1.2 or later versions.
041 *
042 * @author  Ann Wollrath
043 * @since   JDK1.1
044 * @deprecated no replacement
045 */
046@Deprecated
047public class ServerRuntimeException extends RemoteException {
048
049    /* indicate compatibility with JDK 1.1.x version of class */
050    private static final long serialVersionUID = 7054464920481467219L;
051
052    /**
053     * Constructs a <code>ServerRuntimeException</code> with the specified
054     * detail message and nested exception.
055     *
056     * @param s the detail message
057     * @param ex the nested exception
058     * @deprecated no replacement
059     * @since JDK1.1
060     */
061    @Deprecated
062    public ServerRuntimeException(String s, Exception ex) {
063        super(s, ex);
064    }
065}
