026package java.rmi.registry;
027
028import java.rmi.RemoteException;
029import java.rmi.UnknownHostException;
030
031/**
032 * <code>RegistryHandler</code> is an interface used internally by the RMI
033 * runtime in previous implementation versions.  It should never be accessed
034 * by application code.
035 *
036 * @author  Ann Wollrath
037 * @since   JDK1.1
038 * @deprecated no replacement
039 */
040@Deprecated
041public interface RegistryHandler {
042
043    /**
044     * Returns a "stub" for contacting a remote registry
045     * on the specified host and port.
046     *
047     * @deprecated no replacement.  As of the Java 2 platform v1.2, RMI no
048     * longer uses the <code>RegistryHandler</code> to obtain the registry's
049     * stub.
050     * @param host name of remote registry host
051     * @param port remote registry port
052     * @return remote registry stub
053     * @throws RemoteException if a remote error occurs
054     * @throws UnknownHostException if unable to resolve given hostname
055     */
056    @Deprecated
057    Registry registryStub(String host, int port)
058        throws RemoteException, UnknownHostException;
059
060    /**
061     * Constructs and exports a Registry on the specified port.
062     * The port must be non-zero.
063     *
064     * @deprecated no replacement.  As of the Java 2 platform v1.2, RMI no
065     * longer uses the <code>RegistryHandler</code> to obtain the registry's
066     * implementation.
067     * @param port port to export registry on
068     * @return registry stub
069     * @throws RemoteException if a remote error occurs
070     */
071    @Deprecated
072    Registry registryImpl(int port) throws RemoteException;
073}
