026package java.rmi.activation;
027
028import java.rmi.MarshalledObject;
029import java.rmi.Remote;
030import java.rmi.RemoteException;
031
032/**
033 * An <code>ActivationInstantiator</code> is responsible for creating
034 * instances of "activatable" objects. A concrete subclass of
035 * <code>ActivationGroup</code> implements the <code>newInstance</code>
036 * method to handle creating objects within the group.
037 *
038 * @author      Ann Wollrath
039 * @see         ActivationGroup
040 * @since       1.2
041 */
042public interface ActivationInstantiator extends Remote {
043
044   /**
045    * The activator calls an instantiator's <code>newInstance</code>
046    * method in order to recreate in that group an object with the
047    * activation identifier, <code>id</code>, and descriptor,
048    * <code>desc</code>. The instantiator is responsible for: <ul>
049    *
050    * <li> determining the class for the object using the descriptor's
051    * <code>getClassName</code> method,
052    *
053    * <li> loading the class from the code location obtained from the
054    * descriptor (using the <code>getLocation</code> method),
055    *
056    * <li> creating an instance of the class by invoking the special
057    * "activation" constructor of the object's class that takes two
058    * arguments: the object's <code>ActivationID</code>, and the
059    * <code>MarshalledObject</code> containing object specific
060    * initialization data, and
061    *
062    * <li> returning a MarshalledObject containing the stub for the
063    * remote object it created </ul>
064    *
065    * @param id the object's activation identifier
066    * @param desc the object's descriptor
067    * @return a marshalled object containing the serialized
068    * representation of remote object's stub
069    * @exception ActivationException if object activation fails
070    * @exception RemoteException if remote call fails
071    * @since 1.2
072    */
073    public MarshalledObject<? extends Remote> newInstance(ActivationID id,
074                                                          ActivationDesc desc)
075        throws ActivationException, RemoteException;
076}
