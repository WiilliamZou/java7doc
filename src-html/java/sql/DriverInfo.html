036/**
037 * <P>The basic service for managing a set of JDBC drivers.<br>
038 * <B>NOTE:</B> The {@link <code>DataSource</code>} interface, new in the
039 * JDBC 2.0 API, provides another way to connect to a data source.
040 * The use of a <code>DataSource</code> object is the preferred means of
041 * connecting to a data source.
042 *
043 * <P>As part of its initialization, the <code>DriverManager</code> class will
044 * attempt to load the driver classes referenced in the "jdbc.drivers"
045 * system property. This allows a user to customize the JDBC Drivers
046 * used by their applications. For example in your
047 * ~/.hotjava/properties file you might specify:
048 * <pre>
049 * <CODE>jdbc.drivers=foo.bah.Driver:wombat.sql.Driver:bad.taste.ourDriver</CODE>
050 * </pre>
051 *<P> The <code>DriverManager</code> methods <code>getConnection</code> and
052 * <code>getDrivers</code> have been enhanced to support the Java Standard Edition
053 * <a href="../../../technotes/guides/jar/jar.html#Service%20Provider">Service Provider</a> mechanism. JDBC 4.0 Drivers must
054 * include the file <code>META-INF/services/java.sql.Driver</code>. This file contains the name of the JDBC drivers
055 * implementation of <code>java.sql.Driver</code>.  For example, to load the <code>my.sql.Driver</code> class,
056 * the <code>META-INF/services/java.sql.Driver</code> file would contain the entry:
057 * <pre>
058 * <code>my.sql.Driver</code>
059 * </pre>
060 *
061 * <P>Applications no longer need to explictly load JDBC drivers using <code>Class.forName()</code>. Existing programs
062 * which currently load JDBC drivers using <code>Class.forName()</code> will continue to work without
063 * modification.
064 *
065 * <P>When the method <code>getConnection</code> is called,
066 * the <code>DriverManager</code> will attempt to
067 * locate a suitable driver from amongst those loaded at
068 * initialization and those loaded explicitly using the same classloader
069 * as the current applet or application.
070 *
071 * <P>
072 * Starting with the Java 2 SDK, Standard Edition, version 1.3, a
073 * logging stream can be set only if the proper
074 * permission has been granted.  Normally this will be done with
075 * the tool PolicyTool, which can be used to grant <code>permission
076 * java.sql.SQLPermission "setLog"</code>.
077 * @see Driver
078 * @see Connection
079 */
080public class DriverManager {

128    /**
129     * Sets the logging/tracing <code>PrintWriter</code> object
130     * that is used by the <code>DriverManager</code> and all drivers.
131     * <P>
132     * There is a minor versioning problem created by the introduction
133     * of the method <code>setLogWriter</code>.  The
134     * method <code>setLogWriter</code> cannot create a <code>PrintStream</code> object
135     * that will be returned by <code>getLogStream</code>---the Java platform does
136     * not provide a backward conversion.  As a result, a new application
137     * that uses <code>setLogWriter</code> and also uses a JDBC 1.0 driver that uses
138     * <code>getLogStream</code> will likely not see debugging information written
139     * by that driver.
140     *<P>
141     * Starting with the Java 2 SDK, Standard Edition, version 1.3 release, this method checks
142     * to see that there is an <code>SQLPermission</code> object before setting
143     * the logging stream.  If a <code>SecurityManager</code> exists and its
144     * <code>checkPermission</code> method denies setting the log writer, this
145     * method throws a <code>java.lang.SecurityException</code>.
146     *
147     * @param out the new logging/tracing <code>PrintStream</code> object;
148     *      <code>null</code> to disable logging and tracing
149     * @throws SecurityException
150     *    if a security manager exists and its
151     *    <code>checkPermission</code> method denies
152     *    setting the log writer
153     *
154     * @see SecurityManager#checkPermission
155     * @see #getLogWriter
156     * @since 1.2
157     */
158    public static void setLogWriter(java.io.PrintWriter out) {
159
160        SecurityManager sec = System.getSecurityManager();
161        if (sec != null) {
162            sec.checkPermission(SET_LOG_PERMISSION);
163        }
164            logStream = null;
165            logWriter = out;
166    }

236    /**
237     * Attempts to locate a driver that understands the given URL.
238     * The <code>DriverManager</code> attempts to select an appropriate driver from
239     * the set of registered JDBC drivers.
240     *
241     * @param url a database URL of the form
242     *     <code>jdbc:<em>subprotocol</em>:<em>subname</em></code>
243     * @return a <code>Driver</code> object representing a driver
244     * that can connect to the given URL
245     * @exception SQLException if a database access error occurs
246     */
247    @CallerSensitive
248    public static Driver getDriver(String url)
249        throws SQLException {
250
251        println("DriverManager.getDriver(\"" + url + "\")");
252
253        Class<?> callerClass = Reflection.getCallerClass();
254
255        // Walk through the loaded registeredDrivers attempting to locate someone
256        // who understands the given URL.
257        for (DriverInfo aDriver : registeredDrivers) {
258            // If the caller does not have permission to load the driver then
259            // skip it.
260            if(isDriverAllowed(aDriver.driver, callerClass)) {
261                try {
262                    if(aDriver.driver.acceptsURL(url)) {
263                        // Success!
264                        println("getDriver returning " + aDriver.driver.getClass().getName());
265                    return (aDriver.driver);
266                    }
267
268                } catch(SQLException sqe) {
269                    // Drop through and try the next driver.
270                }
271            } else {
272                println("    skipping: " + aDriver.driver.getClass().getName());
273            }
274
275        }
276
277        println("getDriver: no suitable driver");
278        throw new SQLException("No suitable driver", "08001");
279    }

282    /**
283     * Registers the given driver with the <code>DriverManager</code>.
284     * A newly-loaded driver class should call
285     * the method <code>registerDriver</code> to make itself
286     * known to the <code>DriverManager</code>.
287     *
288     * @param driver the new JDBC Driver that is to be registered with the
289     *               <code>DriverManager</code>
290     * @exception SQLException if a database access error occurs
291     */
292    public static synchronized void registerDriver(java.sql.Driver driver)
293        throws SQLException {
294
295        /* Register the driver if it has not already been added to our list */
296        if(driver != null) {
297            registeredDrivers.addIfAbsent(new DriverInfo(driver));
298        } else {
299            // This is for compatibility with the original DriverManager
300            throw new NullPointerException();
301        }
302
303        println("registerDriver: " + driver);
304
305    }

307    /**
308     * Drops a driver from the <code>DriverManager</code>'s list.
309     *  Applets can only deregister drivers from their own classloaders.
310     *
311     * @param driver the JDBC Driver to drop
312     * @exception SQLException if a database access error occurs
313     */
314    @CallerSensitive
315    public static synchronized void deregisterDriver(Driver driver)
316        throws SQLException {
317        if (driver == null) {
318            return;
319        }
320
321        println("DriverManager.deregisterDriver: " + driver);
322
323        DriverInfo aDriver = new DriverInfo(driver);
324        if(registeredDrivers.contains(aDriver)) {
325            if (isDriverAllowed(driver, Reflection.getCallerClass())) {
326                 registeredDrivers.remove(aDriver);
327            } else {
328                // If the caller does not have permission to load the driver then
329                // throw a SecurityException.
330                throw new SecurityException();
331            }
332        } else {
333            println("    couldn't find driver to unload");
334        }
335    }

337    /**
338     * Retrieves an Enumeration with all of the currently loaded JDBC drivers
339     * to which the current caller has access.
340     *
341     * <P><B>Note:</B> The classname of a driver can be found using
342     * <CODE>d.getClass().getName()</CODE>
343     *
344     * @return the list of JDBC Drivers loaded by the caller's class loader
345     */
346    @CallerSensitive
347    public static java.util.Enumeration<Driver> getDrivers() {
348        java.util.Vector<Driver> result = new java.util.Vector<Driver>();
349
350        Class<?> callerClass = Reflection.getCallerClass();
351
352        // Walk through the loaded registeredDrivers.
353        for(DriverInfo aDriver : registeredDrivers) {
354            // If the caller does not have permission to load the driver then
355            // skip it.
356            if(isDriverAllowed(aDriver.driver, callerClass)) {
357                result.addElement(aDriver.driver);
358            } else {
359                println("    skipping: " + aDriver.getClass().getName());
360            }
361        }
362        return (result.elements());
363    }

416            logWriter = new java.io.PrintWriter(out);
417        else
418            logWriter = null;

436     * @param message a log or tracing message

438    public static void println(String message) {
439        synchronized (logSync) {

444                logWriter.flush();
445            }

451    // Indicates whether the class object that would be created if the code calling
452    // DriverManager is accessible.
453    private static boolean isDriverAllowed(Driver driver, Class<?> caller) {
454        ClassLoader callerCL = caller != null ? caller.getClassLoader() : null;
455        return isDriverAllowed(driver, callerCL);
456    }
457
458    private static boolean isDriverAllowed(Driver driver, ClassLoader classLoader) {
459        boolean result = false;
460        if(driver != null) {
461            Class<?> aClass = null;
462            try {
463                aClass =  Class.forName(driver.getClass().getName(), true, classLoader);
464            } catch (Exception ex) {
465                result = false;
466            }
467
468             result = ( aClass == driver.getClass() ) ? true : false;
469        }
470
471        return result;
472    }

474    private static void loadInitialDrivers() {
475        String drivers;
476        try {
477            drivers = AccessController.doPrivileged(new PrivilegedAction<String>() {
478                public String run() {
479                    return System.getProperty("jdbc.drivers");
480                }
481            });
482        } catch (Exception ex) {
483            drivers = null;
484        }
485        // If the driver is packaged as a Service Provider, load it.
486        // Get all the drivers through the classloader
487        // exposed as a java.sql.Driver.class service.
488        // ServiceLoader.load() replaces the sun.misc.Providers()
489
490        AccessController.doPrivileged(new PrivilegedAction<Void>() {
491            public Void run() {
492
493                ServiceLoader<Driver> loadedDrivers = ServiceLoader.load(Driver.class);
494                Iterator driversIterator = loadedDrivers.iterator();
495
496                /* Load these drivers, so that they can be instantiated.
497                 * It may be the case that the driver class may not be there
498                 * i.e. there may be a packaged driver with the service class
499                 * as implementation of java.sql.Driver but the actual class
500                 * may be missing. In that case a java.util.ServiceConfigurationError
501                 * will be thrown at runtime by the VM trying to locate
502                 * and load the service.
503                 *
504                 * Adding a try catch block to catch those runtime errors
505                 * if driver not available in classpath but it's
506                 * packaged as service and that service is there in classpath.
507                 */
508                try{
509                    while(driversIterator.hasNext()) {
510                        driversIterator.next();
511                    }
512                } catch(Throwable t) {
513                // Do nothing
514                }
515                return null;
516            }
517        });
518
519        println("DriverManager.initialize: jdbc.drivers = " + drivers);
520
521        if (drivers == null || drivers.equals("")) {
522            return;
523        }
524        String[] driversList = drivers.split(":");
525        println("number of Drivers:" + driversList.length);
526        for (String aDriver : driversList) {
527            try {
528                println("DriverManager.Initialize: loading " + aDriver);
529                Class.forName(aDriver, true,
530                        ClassLoader.getSystemClassLoader());
531            } catch (Exception ex) {
532                println("DriverManager.Initialize: load failed: " + ex);
533            }
534        }
535    }

538    //  Worker method called by the public getConnection() methods.
539    private static Connection getConnection(
540        String url, java.util.Properties info, Class<?> caller) throws SQLException {
541        /*
542         * When callerCl is null, we should check the application's
543         * (which is invoking this class indirectly)
544         * classloader, so that the JDBC driver class outside rt.jar
545         * can be loaded from here.
546         */
547        ClassLoader callerCL = caller != null ? caller.getClassLoader() : null;
548        synchronized (DriverManager.class) {
549            // synchronize loading of the correct classloader.
550            if (callerCL == null) {
551                callerCL = Thread.currentThread().getContextClassLoader();
552            }
553        }
554
555        if(url == null) {
556            throw new SQLException("The url cannot be null", "08001");
557        }
558
559        println("DriverManager.getConnection(\"" + url + "\")");
560
561        // Walk through the loaded registeredDrivers attempting to make a connection.
562        // Remember the first exception that gets raised so we can reraise it.
563        SQLException reason = null;
564
565        for(DriverInfo aDriver : registeredDrivers) {
566            // If the caller does not have permission to load the driver then
567            // skip it.
568            if(isDriverAllowed(aDriver.driver, callerCL)) {
569                try {
570                    println("    trying " + aDriver.driver.getClass().getName());
571                    Connection con = aDriver.driver.connect(url, info);
572                    if (con != null) {
573                        // Success!
574                        println("getConnection returning " + aDriver.driver.getClass().getName());
575                        return (con);
576                    }
577                } catch (SQLException ex) {
578                    if (reason == null) {
579                        reason = ex;
580                    }
581                }
582
583            } else {
584                println("    skipping: " + aDriver.getClass().getName());
585            }
586
587        }
588
589        // if we got here nobody could connect.
590        if (reason != null)    {
591            println("getConnection failed: " + reason);
592            throw reason;
593        }
594
595        println("getConnection: no suitable driver found for "+ url);
596        throw new SQLException("No suitable driver found for "+ url, "08001");
597    }

600/*
601 * Wrapper class for registered Drivers in order to not expose Driver.equals()
602 * to avoid the capture of the Driver it being compared to as it might not
603 * normally have access.
604 */
605class DriverInfo {
606
607    final Driver driver;
608    DriverInfo(Driver driver) {
609        this.driver = driver;
610    }
611
612    public boolean equals(Object other) {
613        return (other instanceof DriverInfo)
614                && this.driver == ((DriverInfo) other).driver;
615    }
616
617    public int hashCode() {
618        return driver.hashCode();
619    }
620
621    public String toString() {
622        return ("driver[className="  + driver + "]");
623    }
