026package java.sql;
027
028/**
029 * The representation of a savepoint, which is a point within
030 * the current transaction that can be referenced from the
031 * <code>Connection.rollback</code> method. When a transaction
032 * is rolled back to a savepoint all changes made after that
033 * savepoint are undone.
034 * <p>
035 * Savepoints can be either named or unnamed. Unnamed savepoints
036 * are identified by an ID generated by the underlying data source.
037 *
038 * @since 1.4
039 */
040
041public interface Savepoint {
042
043    /**
044     * Retrieves the generated ID for the savepoint that this
045     * <code>Savepoint</code> object represents.
046     * @return the numeric ID of this savepoint
047     * @exception SQLException if this is a named savepoint
048     * @since 1.4
049     */
050    int getSavepointId() throws SQLException;
051
052    /**
053     * Retrieves the name of the savepoint that this <code>Savepoint</code>
054     * object represents.
055     * @return the name of this savepoint
056     * @exception SQLException if this is an un-named savepoint
057     * @since 1.4
058     */
059    String getSavepointName() throws SQLException;
060}
