026package java.sql;
027
028/**
029 * <p>Driver properties for making a connection. The
030 * <code>DriverPropertyInfo</code> class is of interest only to advanced programmers
031 * who need to interact with a Driver via the method
032 * <code>getDriverProperties</code> to discover
033 * and supply properties for connections.
034 */
035
036public class DriverPropertyInfo {
037
038    /**
039     * Constructs a <code>DriverPropertyInfo</code> object with a  given
040     * name and value.  The <code>description</code> and <code>choices</code>
041     * are intialized to <code>null</code> and <code>required</code> is initialized
042     * to <code>false</code>.
043     *
044     * @param name the name of the property
045     * @param value the current value, which may be null
046     */
047    public DriverPropertyInfo(String name, String value) {
048        this.name = name;
049        this.value = value;
050    }
051
052    /**
053     * The name of the property.
054     */
055    public String name;
056
057    /**
058     * A brief description of the property, which may be null.
059     */
060    public String description = null;
061
062    /**
063     * The <code>required</code> field is <code>true</code> if a value must be
064         * supplied for this property
065     * during <code>Driver.connect</code> and <code>false</code> otherwise.
066     */
067    public boolean required = false;
068
069    /**
070     * The <code>value</code> field specifies the current value of
071         * the property, based on a combination of the information
072         * supplied to the method <code>getPropertyInfo</code>, the
073     * Java environment, and the driver-supplied default values.  This field
074     * may be null if no value is known.
075     */
076    public String value = null;
077
078    /**
079     * An array of possible values if the value for the field
080         * <code>DriverPropertyInfo.value</code> may be selected
081         * from a particular set of values; otherwise null.
082     */
083    public String[] choices = null;
084}
