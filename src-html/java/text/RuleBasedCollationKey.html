039package java.text;
040
041/**
042 * A RuleBasedCollationKey is a concrete implementation of CollationKey class.
043 * The RuleBasedCollationKey class is used by the RuleBasedCollator class.
044 */
045
046final class RuleBasedCollationKey extends CollationKey {
047    /**
048     * Compare this RuleBasedCollationKey to target. The collation rules of the
049     * Collator object which created these keys are applied. <strong>Note:</strong>
050     * RuleBasedCollationKeys created by different Collators can not be compared.
051     * @param target target RuleBasedCollationKey
052     * @return Returns an integer value. Value is less than zero if this is less
053     * than target, value is zero if this and target are equal and value is greater than
054     * zero if this is greater than target.
055     * @see java.text.Collator#compare
056     */
057    public int compareTo(CollationKey target)
058    {
059        int result = key.compareTo(((RuleBasedCollationKey)(target)).key);
060        if (result <= Collator.LESS)
061            return Collator.LESS;
062        else if (result >= Collator.GREATER)
063            return Collator.GREATER;
064        return Collator.EQUAL;
065    }
066
067    /**
068     * Compare this RuleBasedCollationKey and the target for equality.
069     * The collation rules of the Collator object which created these keys are applied.
070     * <strong>Note:</strong> RuleBasedCollationKeys created by different Collators can not be
071     * compared.
072     * @param target the RuleBasedCollationKey to compare to.
073     * @return Returns true if two objects are equal, false otherwise.
074     */
075    public boolean equals(Object target) {
076        if (this == target) return true;
077        if (target == null || !getClass().equals(target.getClass())) {
078            return false;
079        }
080        RuleBasedCollationKey other = (RuleBasedCollationKey)target;
081        return key.equals(other.key);
082    }
083
084    /**
085     * Creates a hash code for this RuleBasedCollationKey. The hash value is calculated on the
086     * key itself, not the String from which the key was created.  Thus
087     * if x and y are RuleBasedCollationKeys, then x.hashCode(x) == y.hashCode() if
088     * x.equals(y) is true.  This allows language-sensitive comparison in a hash table.
089     * See the CollatinKey class description for an example.
090     * @return the hash value based on the string's collation order.
091     */
092    public int hashCode() {
093        return (key.hashCode());
094    }
095
096    /**
097     * Converts the RuleBasedCollationKey to a sequence of bits. If two RuleBasedCollationKeys
098     * could be legitimately compared, then one could compare the byte arrays
099     * for each of those keys to obtain the same result.  Byte arrays are
100     * organized most significant byte first.
101     */
102    public byte[] toByteArray() {
103
104        char[] src = key.toCharArray();
105        byte[] dest = new byte[ 2*src.length ];
106        int j = 0;
107        for( int i=0; i<src.length; i++ ) {
108            dest[j++] = (byte)(src[i] >>> 8);
109            dest[j++] = (byte)(src[i] & 0x00ff);
110        }
111        return dest;
112    }
113
114    /**
115     * A RuleBasedCollationKey can only be generated by Collator objects.
116     */
117    RuleBasedCollationKey(String source, String key) {
118        super(source);
119        this.key = key;
120    }
121    private String key = null;
122
123}
