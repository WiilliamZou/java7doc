041/**
042 * An HttpCookie object represents an http cookie, which carries state
043 * information between server and user agent. Cookie is widely adopted
044 * to create stateful sessions.
045 *
046 * <p>There are 3 http cookie specifications:
047 * <blockquote>
048 *   Netscape draft<br>
049 *   RFC 2109 - <a href="http://www.ietf.org/rfc/rfc2109.txt">
050 * <i>http://www.ietf.org/rfc/rfc2109.txt</i></a><br>
051 *   RFC 2965 - <a href="http://www.ietf.org/rfc/rfc2965.txt">
052 * <i>http://www.ietf.org/rfc/rfc2965.txt</i></a>
053 * </blockquote>
054 *
055 * <p>HttpCookie class can accept all these 3 forms of syntax.
056 *
057 * @author Edward Wang
058 * @since 1.6
059 */
060public final class HttpCookie implements Cloneable {

125    /**
126     * Constructs a cookie with a specified name and value.
127     *
128     * <p>The name must conform to RFC 2965. That means it can contain
129     * only ASCII alphanumeric characters and cannot contain commas,
130     * semicolons, or white space or begin with a $ character. The cookie's
131     * name cannot be changed after creation.
132     *
133     * <p>The value can be anything the server chooses to send. Its
134     * value is probably of interest only to the server. The cookie's
135     * value can be changed after creation with the
136     * <code>setValue</code> method.
137     *
138     * <p>By default, cookies are created according to the RFC 2965
139     * cookie specification. The version can be changed with the
140     * <code>setVersion</code> method.
141     *
142     *
143     * @param name                      a <code>String</code> specifying the name of the cookie
144     *
145     * @param value                     a <code>String</code> specifying the value of the cookie
146     *
147     * @throws IllegalArgumentException if the cookie name contains illegal characters
148     *                                  or it is one of the tokens reserved for use
149     *                                  by the cookie protocol
150     * @throws NullPointerException     if <tt>name</tt> is <tt>null</tt>
151     * @see #setValue
152     * @see #setVersion
153     *
154     */
155
156    public HttpCookie(String name, String value) {
157        this(name, value, null /*header*/);
158    }
159
160    private HttpCookie(String name, String value, String header) {
161        name = name.trim();
162        if (name.length() == 0 || !isToken(name) || name.charAt(0) == '$') {
163            throw new IllegalArgumentException("Illegal cookie name");
164        }
165
166        this.name = name;
167        this.value = value;
168        toDiscard = false;
169        secure = false;
170
171        whenCreated = System.currentTimeMillis();
172        portlist = null;
173        this.header = header;
174    }

177    /**
178     * Constructs cookies from set-cookie or set-cookie2 header string.
179     * RFC 2965 section 3.2.2 set-cookie2 syntax indicates that one header line
180     * may contain more than one cookie definitions, so this is a static
181     * utility method instead of another constructor.
182     *
183     * @param header    a <tt>String</tt> specifying the set-cookie header.
184     *                  The header should start with "set-cookie", or "set-cookie2"
185     *                  token; or it should have no leading token at all.
186     * @return          a List of cookie parsed from header line string
187     * @throws IllegalArgumentException if header string violates the cookie
188     *                                  specification's syntax, or the cookie
189     *                                  name contains llegal characters, or
190     *                                  the cookie name is one of the tokens
191     *                                  reserved for use by the cookie protocol
192     * @throws NullPointerException     if the header string is <tt>null</tt>
193     */
194    public static List<HttpCookie> parse(String header) {
195        return parse(header, false);
196    }
197
198    // Private version of parse() that will store the original header used to
199    // create the cookie, in the cookie itself. This can be useful for filtering
200    // Set-Cookie[2] headers, using the internal parsing logic defined in this
201    // class.
202    private static List<HttpCookie> parse(String header, boolean retainHeader) {
203
204        int version = guessCookieVersion(header);
205
206        // if header start with set-cookie or set-cookie2, strip it off
207        if (startsWithIgnoreCase(header, SET_COOKIE2)) {
208            header = header.substring(SET_COOKIE2.length());
209        } else if (startsWithIgnoreCase(header, SET_COOKIE)) {
210            header = header.substring(SET_COOKIE.length());
211        }
212
213
214        List<HttpCookie> cookies = new java.util.ArrayList<HttpCookie>();
215        // The Netscape cookie may have a comma in its expires attribute,
216        // while the comma is the delimiter in rfc 2965/2109 cookie header string.
217        // so the parse logic is slightly different
218        if (version == 0) {
219            // Netscape draft cookie
220            HttpCookie cookie = parseInternal(header, retainHeader);
221            cookie.setVersion(0);
222            cookies.add(cookie);
223        } else {
224            // rfc2965/2109 cookie
225            // if header string contains more than one cookie,
226            // it'll separate them with comma
227            List<String> cookieStrings = splitMultiCookies(header);
228            for (String cookieStr : cookieStrings) {
229                HttpCookie cookie = parseInternal(cookieStr, retainHeader);
230                cookie.setVersion(1);
231                cookies.add(cookie);
232            }
233        }
234
235        return cookies;
236    }

244    /**
245     * Reports whether this http cookie has expired or not.
246     *
247     * @return  <tt>true</tt> to indicate this http cookie has expired;
248     *          otherwise, <tt>false</tt>
249     */
250    public boolean hasExpired() {
251        if (maxAge == 0) return true;
252
253        // if not specify max-age, this cookie should be
254        // discarded when user agent is to be closed, but
255        // it is not expired.
256        if (maxAge == MAX_AGE_UNSPECIFIED) return false;
257
258        long deltaSecond = (System.currentTimeMillis() - whenCreated) / 1000;
259        if (deltaSecond > maxAge)
260            return true;
261        else
262            return false;
263    }

712    /**
713     * The utility method to check whether a host name is in a domain
714     * or not.
715     *
716     * <p>This concept is described in the cookie specification.
717     * To understand the concept, some terminologies need to be defined first:
718     * <blockquote>
719     * effective host name = hostname if host name contains dot<br>
720     * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;or = hostname.local if not
721     * </blockquote>
722     * <p>Host A's name domain-matches host B's if:
723     * <blockquote><ul>
724     *   <li>their host name strings string-compare equal; or</li>
725     *   <li>A is a HDN string and has the form NB, where N is a non-empty
726     *   name string, B has the form .B', and B' is a HDN string.  (So,
727     *   x.y.com domain-matches .Y.com but not Y.com.)</li>
728     * </ul></blockquote>
729     *
730     * <p>A host isn't in a domain (RFC 2965 sec. 3.3.2) if:
731     * <blockquote><ul>
732     *   <li>The value for the Domain attribute contains no embedded dots,
733     *   and the value is not .local.</li>
734     *   <li>The effective host name that derives from the request-host does
735     *   not domain-match the Domain attribute.</li>
736     *   <li>The request-host is a HDN (not IP address) and has the form HD,
737     *   where D is the value of the Domain attribute, and H is a string
738     *   that contains one or more dots.</li>
739     * </ul></blockquote>
740     *
741     * <p>Examples:
742     * <blockquote><ul>
743     *   <li>A Set-Cookie2 from request-host y.x.foo.com for Domain=.foo.com
744     *   would be rejected, because H is y.x and contains a dot.</li>
745     *   <li>A Set-Cookie2 from request-host x.foo.com for Domain=.foo.com
746     *   would be accepted.</li>
747     *   <li>A Set-Cookie2 with Domain=.com or Domain=.com., will always be
748     *   rejected, because there is no embedded dot.</li>
749     *   <li>A Set-Cookie2 with Domain=ajax.com will be accepted, and the
750     *   value for Domain will be taken to be .ajax.com, because a dot
751     *   gets prepended to the value.</li>
752     *   <li>A Set-Cookie2 from request-host example for Domain=.local will
753     *   be accepted, because the effective host name for the request-
754     *   host is example.local, and example.local domain-matches .local.</li>
755     * </ul></blockquote>
756     *
757     * @param domain    the domain name to check host name with
758     * @param host      the host name in question
759     * @return          <tt>true</tt> if they domain-matches; <tt>false</tt> if not
760     */
761    public static boolean domainMatches(String domain, String host) {
762        if (domain == null || host == null)
763            return false;
764
765        // if there's no embedded dot in domain and domain is not .local
766        boolean isLocalDomain = ".local".equalsIgnoreCase(domain);
767        int embeddedDotInDomain = domain.indexOf('.');
768        if (embeddedDotInDomain == 0)
769            embeddedDotInDomain = domain.indexOf('.', 1);
770        if (!isLocalDomain
771            && (embeddedDotInDomain == -1 || embeddedDotInDomain == domain.length() - 1))
772            return false;
773
774        // if the host name contains no dot and the domain name is .local
775        int firstDotInHost = host.indexOf('.');
776        if (firstDotInHost == -1 && isLocalDomain)
777            return true;
778
779        int domainLength = domain.length();
780        int lengthDiff = host.length() - domainLength;
781        if (lengthDiff == 0) {
782            // if the host name and the domain name are just string-compare euqal
783            return host.equalsIgnoreCase(domain);
784        }
785        else if (lengthDiff > 0) {
786            // need to check H & D component
787            String H = host.substring(0, lengthDiff);
788            String D = host.substring(lengthDiff);
789
790            return (H.indexOf('.') == -1 && D.equalsIgnoreCase(domain));
791        }
792        else if (lengthDiff == -1) {
793            // if domain is actually .host
794            return (domain.charAt(0) == '.' &&
795                        host.equalsIgnoreCase(domain.substring(1)));
796        }
797
798        return false;
799    }

821     *
822     * <p> The result is <tt>true</tt> only if two cookies
823     * come from same domain (case-insensitive),
824     * have same name (case-insensitive),

826     *
827     * @return          <tt>true</tt> if 2 http cookies equal to each other;
828     *                  otherwise, <tt>false</tt>

830    @Override
831    public boolean equals(Object obj) {
832        if (obj == this)
833            return true;
834        if (!(obj instanceof HttpCookie))
835            return false;
836        HttpCookie other = (HttpCookie)obj;
837
838        // One http cookie equals to another cookie (RFC 2965 sec. 3.3.3) if:
839        //   1. they come from same domain (case-insensitive),
840        //   2. have same name (case-insensitive),
841        //   3. and have same path (case-sensitive).
842        return equalsIgnoreCase(getName(), other.getName()) &&
843               equalsIgnoreCase(getDomain(), other.getDomain()) &&
844               Objects.equals(getPath(), other.getPath());

849     * Return hash code of this http cookie. The result is the sum of
850     * hash code value of three significant components of this cookie:
851     * name, domain, and path.
852     * That is, the hash code is the value of the expression:
853     * <blockquote>
854     * getName().toLowerCase().hashCode()<br>
855     * + getDomain().toLowerCase().hashCode()<br>
856     * + getPath().hashCode()
857     * </blockquote>

864        int h2 = (domain!=null) ? domain.toLowerCase().hashCode() : 0;
865        int h3 = (path!=null) ? path.hashCode() : 0;
866
867        return h1 + h2 + h3;

893    /*
894     * Tests a string and returns true if the string counts as a
895     * token.
896     *
897     * @param value             the <code>String</code> to be tested
898     *
899     * @return                  <code>true</code> if the <code>String</code> is
900     *                          a token; <code>false</code> if it is not
901     */
902
903    private static boolean isToken(String value) {
904        int len = value.length();
905
906        for (int i = 0; i < len; i++) {
907            char c = value.charAt(i);
908
909            if (c < 0x20 || c >= 0x7f || tspecials.indexOf(c) != -1)
910                return false;
911        }
912        return true;
913    }

916    /*
917     * Parse header string to cookie object.
918     *
919     * @param header    header string; should contain only one NAME=VALUE pair
920     *
921     * @return          an HttpCookie being extracted
922     *
923     * @throws IllegalArgumentException if header string violates the cookie
924     *                                  specification
925     */
926    private static HttpCookie parseInternal(String header,
927                                            boolean retainHeader)
928    {
929        HttpCookie cookie = null;
930        String namevaluePair = null;
931
932        StringTokenizer tokenizer = new StringTokenizer(header, ";");
933
934        // there should always have at least on name-value pair;
935        // it's cookie's name
936        try {
937            namevaluePair = tokenizer.nextToken();
938            int index = namevaluePair.indexOf('=');
939            if (index != -1) {
940                String name = namevaluePair.substring(0, index).trim();
941                String value = namevaluePair.substring(index + 1).trim();
942                if (retainHeader)
943                    cookie = new HttpCookie(name,
944                                            stripOffSurroundingQuote(value),
945                                            header);
946                else
947                    cookie = new HttpCookie(name,
948                                            stripOffSurroundingQuote(value));
949            } else {
950                // no "=" in name-value pair; it's an error
951                throw new IllegalArgumentException("Invalid cookie name-value pair");
952            }
953        } catch (NoSuchElementException ignored) {
954            throw new IllegalArgumentException("Empty cookie header string");
955        }
956
957        // remaining name-value pairs are cookie's attributes
958        while (tokenizer.hasMoreTokens()) {
959            namevaluePair = tokenizer.nextToken();
960            int index = namevaluePair.indexOf('=');
961            String name, value;
962            if (index != -1) {
963                name = namevaluePair.substring(0, index).trim();
964                value = namevaluePair.substring(index + 1).trim();
965            } else {
966                name = namevaluePair.trim();
967                value = null;
968            }
969
970            // assign attribute to cookie
971            assignAttribute(cookie, name, value);
972        }
973
974        return cookie;
975    }

1125    /*
1126     * @param dateString        a date string in one of the formats
1127     *                          defined in Netscape cookie spec
1128     *
1129     * @return                  delta seconds between this cookie's creation
1130     *                          time and the time specified by dateString
1131     */
1132    private long expiryDate2DeltaSeconds(String dateString) {
1133        Calendar cal = new GregorianCalendar(GMT);
1134        for (int i = 0; i < COOKIE_DATE_FORMATS.length; i++) {
1135            SimpleDateFormat df = new SimpleDateFormat(COOKIE_DATE_FORMATS[i],
1136                                                       Locale.US);
1137            cal.set(1970, 0, 1, 0, 0, 0);
1138            df.setTimeZone(GMT);
1139            df.setLenient(false);
1140            df.set2DigitYearStart(cal.getTime());
1141            try {
1142                cal.setTime(df.parse(dateString));
1143                if (!COOKIE_DATE_FORMATS[i].contains("yyyy")) {
1144                    // 2-digit years following the standard set
1145                    // out it rfc 6265
1146                    int year = cal.get(Calendar.YEAR);
1147                    year %= 100;
1148                    if (year < 70) {
1149                        year += 2000;
1150                    } else {
1151                        year += 1900;
1152                    }
1153                    cal.set(Calendar.YEAR, year);
1154                }
1155                return (cal.getTimeInMillis() - whenCreated) / 1000;
1156            } catch (Exception e) {
1157                // Ignore, try the next date format
1158            }
1159        }
1160        return 0;
1161    }

1164    /*
1165     * try to guess the cookie version through set-cookie header string
1166     */
1167    private static int guessCookieVersion(String header) {
1168        int version = 0;
1169
1170        header = header.toLowerCase();
1171        if (header.indexOf("expires=") != -1) {
1172            // only netscape cookie using 'expires'
1173            version = 0;
1174        } else if (header.indexOf("version=") != -1) {
1175            // version is mandatory for rfc 2965/2109 cookie
1176            version = 1;
1177        } else if (header.indexOf("max-age") != -1) {
1178            // rfc 2965/2109 use 'max-age'
1179            version = 1;
1180        } else if (startsWithIgnoreCase(header, SET_COOKIE2)) {
1181            // only rfc 2965 cookie starts with 'set-cookie2'
1182            version = 1;
1183        }
1184
1185        return version;
1186    }

1188    private static String stripOffSurroundingQuote(String str) {
1189        if (str != null && str.length() > 2 &&
1190            str.charAt(0) == '"' && str.charAt(str.length() - 1) == '"') {
1191            return str.substring(1, str.length() - 1);
1192        }
1193        if (str != null && str.length() > 2 &&
1194            str.charAt(0) == '\'' && str.charAt(str.length() - 1) == '\'') {
1195            return str.substring(1, str.length() - 1);
1196        }
1197        return str;

1219    /*
1220     * Split cookie header string according to rfc 2965:
1221     *   1) split where it is a comma;
1222     *   2) but not the comma surrounding by double-quotes, which is the comma
1223     *      inside port list or embeded URIs.
1224     *
1225     * @param header            the cookie header string to split
1226     *
1227     * @return                  list of strings; never null
1228     *
1229     */
1230    private static List<String> splitMultiCookies(String header) {
1231        List<String> cookies = new java.util.ArrayList<String>();
1232        int quoteCount = 0;
1233        int p, q;
1234
1235        for (p = 0, q = 0; p < header.length(); p++) {
1236            char c = header.charAt(p);
1237            if (c == '"') quoteCount++;
1238            if (c == ',' && (quoteCount % 2 == 0)) {      // it is comma and not surrounding by double-quotes
1239                cookies.add(header.substring(q, p));
1240                q = p + 1;
1241            }
1242        }
1243
1244        cookies.add(header.substring(q));
1245
1246        return cookies;
1247    }
