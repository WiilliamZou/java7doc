026package java.net;
027
028/**
029 * This interface defines a factory for socket implementations. It
030 * is used by the classes <code>Socket</code> and
031 * <code>ServerSocket</code> to create actual socket
032 * implementations.
033 *
034 * @author  Arthur van Hoff
035 * @see     java.net.Socket
036 * @see     java.net.ServerSocket
037 * @since   JDK1.0
038 */
039public
040interface SocketImplFactory {
041    /**
042     * Creates a new <code>SocketImpl</code> instance.
043     *
044     * @return  a new instance of <code>SocketImpl</code>.
045     * @see     java.net.SocketImpl
046     */
047    SocketImpl createSocketImpl();
048}
