037/**
038 * Class <code>URL</code> represents a Uniform Resource
039 * Locator, a pointer to a "resource" on the World
040 * Wide Web. A resource can be something as simple as a file or a
041 * directory, or it can be a reference to a more complicated object,
042 * such as a query to a database or to a search engine. More
043 * information on the types of URLs and their formats can be found at:
044 * <a href=
045 * "http://web.archive.org/web/20051219043731/http://archive.ncsa.uiuc.edu/SDG/Software/Mosaic/Demo/url-primer.html">
046 * <i>Types of URL</i></a>
047 * <p>
048 * In general, a URL can be broken into several parts. Consider the
049 * following example:
050 * <blockquote><pre>
051 *     http://www.example.com/docs/resource1.html
052 * </pre></blockquote>
053 * <p>
054 * The URL above indicates that the protocol to use is
055 * <code>http</code> (HyperText Transfer Protocol) and that the
056 * information resides on a host machine named
057 * <code>www.example.com</code>. The information on that host
058 * machine is named <code>/docs/resource1.html</code>. The exact
059 * meaning of this name on the host machine is both protocol
060 * dependent and host dependent. The information normally resides in
061 * a file, but it could be generated on the fly. This component of
062 * the URL is called the <i>path</i> component.
063 * <p>
064 * A URL can optionally specify a "port", which is the
065 * port number to which the TCP connection is made on the remote host
066 * machine. If the port is not specified, the default port for
067 * the protocol is used instead. For example, the default port for
068 * <code>http</code> is <code>80</code>. An alternative port could be
069 * specified as:
070 * <blockquote><pre>
071 *     http://www.example.com:1080/docs/resource1.html
072 * </pre></blockquote>
073 * <p>
074 * The syntax of <code>URL</code> is defined by  <a
075 * href="http://www.ietf.org/rfc/rfc2396.txt"><i>RFC&nbsp;2396: Uniform
076 * Resource Identifiers (URI): Generic Syntax</i></a>, amended by <a
077 * href="http://www.ietf.org/rfc/rfc2732.txt"><i>RFC&nbsp;2732: Format for
078 * Literal IPv6 Addresses in URLs</i></a>. The Literal IPv6 address format
079 * also supports scope_ids. The syntax and usage of scope_ids is described
080 * <a href="Inet6Address.html#scoped">here</a>.
081 * <p>
082 * A URL may have appended to it a "fragment", also known
083 * as a "ref" or a "reference". The fragment is indicated by the sharp
084 * sign character "#" followed by more characters. For example,
085 * <blockquote><pre>
086 *     http://java.sun.com/index.html#chapter1
087 * </pre></blockquote>
088 * <p>
089 * This fragment is not technically part of the URL. Rather, it
090 * indicates that after the specified resource is retrieved, the
091 * application is specifically interested in that part of the
092 * document that has the tag <code>chapter1</code> attached to it. The
093 * meaning of a tag is resource specific.
094 * <p>
095 * An application can also specify a "relative URL",
096 * which contains only enough information to reach the resource
097 * relative to another URL. Relative URLs are frequently used within
098 * HTML pages. For example, if the contents of the URL:
099 * <blockquote><pre>
100 *     http://java.sun.com/index.html
101 * </pre></blockquote>
102 * contained within it the relative URL:
103 * <blockquote><pre>
104 *     FAQ.html
105 * </pre></blockquote>
106 * it would be a shorthand for:
107 * <blockquote><pre>
108 *     http://java.sun.com/FAQ.html
109 * </pre></blockquote>
110 * <p>
111 * The relative URL need not specify all the components of a URL. If
112 * the protocol, host name, or port number is missing, the value is
113 * inherited from the fully specified URL. The file component must be
114 * specified. The optional fragment is not inherited.
115 * <p>
116 * The URL class does not itself encode or decode any URL components
117 * according to the escaping mechanism defined in RFC2396. It is the
118 * responsibility of the caller to encode any fields, which need to be
119 * escaped prior to calling URL, and also to decode any escaped fields,
120 * that are returned from URL. Furthermore, because URL has no knowledge
121 * of URL escaping, it does not recognise equivalence between the encoded
122 * or decoded form of the same URL. For example, the two URLs:<br>
123 * <pre>    http://foo.com/hello world/ and http://foo.com/hello%20world</pre>
124 * would be considered not equal to each other.
125 * <p>
126 * Note, the {@link java.net.URI} class does perform escaping of its
127 * component fields in certain circumstances. The recommended way
128 * to manage the encoding and decoding of URLs is to use {@link java.net.URI},
129 * and to convert between these two classes using {@link #toURI()} and
130 * {@link URI#toURL()}.
131 * <p>
132 * The {@link URLEncoder} and {@link URLDecoder} classes can also be
133 * used, but only for HTML form encoding, which is not the same
134 * as the encoding scheme defined in RFC2396.
135 *
136 * @author  James Gosling
137 * @since JDK1.0
138 */

225    /**
226     * Creates a <code>URL</code> object from the specified
227     * <code>protocol</code>, <code>host</code>, <code>port</code>
228     * number, and <code>file</code>.<p>
229     *
230     * <code>host</code> can be expressed as a host name or a literal
231     * IP address. If IPv6 literal address is used, it should be
232     * enclosed in square brackets (<tt>'['</tt> and <tt>']'</tt>), as
233     * specified by <a
234     * href="http://www.ietf.org/rfc/rfc2732.txt">RFC&nbsp;2732</a>;
235     * However, the literal IPv6 address format defined in <a
236     * href="http://www.ietf.org/rfc/rfc2373.txt"><i>RFC&nbsp;2373: IP
237     * Version 6 Addressing Architecture</i></a> is also accepted.<p>
238     *
239     * Specifying a <code>port</code> number of <code>-1</code>
240     * indicates that the URL should use the default port for the
241     * protocol.<p>
242     *
243     * If this is the first URL object being created with the specified
244     * protocol, a <i>stream protocol handler</i> object, an instance of
245     * class <code>URLStreamHandler</code>, is created for that protocol:
246     * <ol>
247     * <li>If the application has previously set up an instance of
248     *     <code>URLStreamHandlerFactory</code> as the stream handler factory,
249     *     then the <code>createURLStreamHandler</code> method of that instance
250     *     is called with the protocol string as an argument to create the
251     *     stream protocol handler.
252     * <li>If no <code>URLStreamHandlerFactory</code> has yet been set up,
253     *     or if the factory's <code>createURLStreamHandler</code> method
254     *     returns <code>null</code>, then the constructor finds the
255     *     value of the system property:
256     *     <blockquote><pre>
257     *         java.protocol.handler.pkgs
258     *     </pre></blockquote>
259     *     If the value of that system property is not <code>null</code>,
260     *     it is interpreted as a list of packages separated by a vertical
261     *     slash character '<code>|</code>'. The constructor tries to load
262     *     the class named:
263     *     <blockquote><pre>
264     *         &lt;<i>package</i>&gt;.&lt;<i>protocol</i>&gt;.Handler
265     *     </pre></blockquote>
266     *     where &lt;<i>package</i>&gt; is replaced by the name of the package
267     *     and &lt;<i>protocol</i>&gt; is replaced by the name of the protocol.
268     *     If this class does not exist, or if the class exists but it is not
269     *     a subclass of <code>URLStreamHandler</code>, then the next package
270     *     in the list is tried.
271     * <li>If the previous step fails to find a protocol handler, then the
272     *     constructor tries to load from a system default package.
273     *     <blockquote><pre>
274     *         &lt;<i>system default package</i>&gt;.&lt;<i>protocol</i>&gt;.Handler
275     *     </pre></blockquote>
276     *     If this class does not exist, or if the class exists but it is not a
277     *     subclass of <code>URLStreamHandler</code>, then a
278     *     <code>MalformedURLException</code> is thrown.
279     * </ol>
280     *
281     * <p>Protocol handlers for the following protocols are guaranteed
282     * to exist on the search path :-
283     * <blockquote><pre>
284     *     http, https, ftp, file, and jar
285     * </pre></blockquote>
286     * Protocol handlers for additional protocols may also be
287     * available.
288     *
289     * <p>No validation of the inputs is performed by this constructor.
290     *
291     * @param      protocol   the name of the protocol to use.
292     * @param      host       the name of the host.
293     * @param      port       the port number on the host.
294     * @param      file       the file on the host
295     * @exception  MalformedURLException  if an unknown protocol is specified.
296     * @see        java.lang.System#getProperty(java.lang.String)
297     * @see        java.net.URL#setURLStreamHandlerFactory(
298     *                  java.net.URLStreamHandlerFactory)
299     * @see        java.net.URLStreamHandler
300     * @see        java.net.URLStreamHandlerFactory#createURLStreamHandler(
301     *                  java.lang.String)
302     */
303    public URL(String protocol, String host, int port, String file)
304        throws MalformedURLException
305    {
306        this(protocol, host, port, file, null);
307    }

332    /**
333     * Creates a <code>URL</code> object from the specified
334     * <code>protocol</code>, <code>host</code>, <code>port</code>
335     * number, <code>file</code>, and <code>handler</code>. Specifying
336     * a <code>port</code> number of <code>-1</code> indicates that
337     * the URL should use the default port for the protocol. Specifying
338     * a <code>handler</code> of <code>null</code> indicates that the URL
339     * should use a default stream handler for the protocol, as outlined
340     * for:
341     *     java.net.URL#URL(java.lang.String, java.lang.String, int,
342     *                      java.lang.String)
343     *
344     * <p>If the handler is not null and there is a security manager,
345     * the security manager's <code>checkPermission</code>
346     * method is called with a
347     * <code>NetPermission("specifyStreamHandler")</code> permission.
348     * This may result in a SecurityException.
349     *
350     * No validation of the inputs is performed by this constructor.
351     *
352     * @param      protocol   the name of the protocol to use.
353     * @param      host       the name of the host.
354     * @param      port       the port number on the host.
355     * @param      file       the file on the host
356     * @param      handler    the stream handler for the URL.
357     * @exception  MalformedURLException  if an unknown protocol is specified.
358     * @exception  SecurityException
359     *        if a security manager exists and its
360     *        <code>checkPermission</code> method doesn't allow
361     *        specifying a stream handler explicitly.
362     * @see        java.lang.System#getProperty(java.lang.String)
363     * @see        java.net.URL#setURLStreamHandlerFactory(
364     *                  java.net.URLStreamHandlerFactory)
365     * @see        java.net.URLStreamHandler
366     * @see        java.net.URLStreamHandlerFactory#createURLStreamHandler(
367     *                  java.lang.String)
368     * @see        SecurityManager#checkPermission
369     * @see        java.net.NetPermission
370     */
371    public URL(String protocol, String host, int port, String file,
372               URLStreamHandler handler) throws MalformedURLException {
373        if (handler != null) {
374            SecurityManager sm = System.getSecurityManager();
375            if (sm != null) {
376                // check for permission to specify a handler
377                checkSpecifyHandler(sm);
378            }
379        }
380
381        protocol = protocol.toLowerCase();
382        this.protocol = protocol;
383        if (host != null) {
384
385            /**
386             * if host is a literal IPv6 address,
387             * we will make it conform to RFC 2732
388             */
389            if (host.indexOf(':') >= 0 && !host.startsWith("[")) {
390                host = "["+host+"]";
391            }
392            this.host = host;
393
394            if (port < -1) {
395                throw new MalformedURLException("Invalid port number :" +
396                                                    port);
397            }
398            this.port = port;
399            authority = (port == -1) ? host : host + ":" + port;
400        }
401
402        Parts parts = new Parts(file);
403        path = parts.getPath();
404        query = parts.getQuery();
405
406        if (query != null) {
407            this.file = path + "?" + query;
408        } else {
409            this.file = path;
410        }
411        ref = parts.getRef();
412
413        // Note: we don't do validation of the URL here. Too risky to change
414        // right now, but worth considering for future reference. -br
415        if (handler == null &&
416            (handler = getURLStreamHandler(protocol)) == null) {
417            throw new MalformedURLException("unknown protocol: " + protocol);
418        }
419        this.handler = handler;
420    }

438    /**
439     * Creates a URL by parsing the given spec within a specified context.
440     *
441     * The new URL is created from the given context URL and the spec
442     * argument as described in
443     * RFC2396 &quot;Uniform Resource Identifiers : Generic * Syntax&quot; :
444     * <blockquote><pre>
445     *          &lt;scheme&gt;://&lt;authority&gt;&lt;path&gt;?&lt;query&gt;#&lt;fragment&gt;
446     * </pre></blockquote>
447     * The reference is parsed into the scheme, authority, path, query and
448     * fragment parts. If the path component is empty and the scheme,
449     * authority, and query components are undefined, then the new URL is a
450     * reference to the current document. Otherwise, the fragment and query
451     * parts present in the spec are used in the new URL.
452     * <p>
453     * If the scheme component is defined in the given spec and does not match
454     * the scheme of the context, then the new URL is created as an absolute
455     * URL based on the spec alone. Otherwise the scheme component is inherited
456     * from the context URL.
457     * <p>
458     * If the authority component is present in the spec then the spec is
459     * treated as absolute and the spec authority and path will replace the
460     * context authority and path. If the authority component is absent in the
461     * spec then the authority of the new URL will be inherited from the
462     * context.
463     * <p>
464     * If the spec's path component begins with a slash character
465     * &quot;/&quot; then the
466     * path is treated as absolute and the spec path replaces the context path.
467     * <p>
468     * Otherwise, the path is treated as a relative path and is appended to the
469     * context path, as described in RFC2396. Also, in this case,
470     * the path is canonicalized through the removal of directory
471     * changes made by occurences of &quot;..&quot; and &quot;.&quot;.
472     * <p>
473     * For a more detailed description of URL parsing, refer to RFC2396.
474     *
475     * @param      context   the context in which to parse the specification.
476     * @param      spec      the <code>String</code> to parse as a URL.
477     * @exception  MalformedURLException  if no protocol is specified, or an
478     *               unknown protocol is found, or <tt>spec</tt> is <tt>null</tt>.
479     * @see        java.net.URL#URL(java.lang.String, java.lang.String,
480     *                  int, java.lang.String)
481     * @see        java.net.URLStreamHandler
482     * @see        java.net.URLStreamHandler#parseURL(java.net.URL,
483     *                  java.lang.String, int, int)
484     */
485    public URL(URL context, String spec) throws MalformedURLException {
486        this(context, spec, null);
487    }

489    /**
490     * Creates a URL by parsing the given spec with the specified handler
491     * within a specified context. If the handler is null, the parsing
492     * occurs as with the two argument constructor.
493     *
494     * @param      context   the context in which to parse the specification.
495     * @param      spec      the <code>String</code> to parse as a URL.
496     * @param      handler   the stream handler for the URL.
497     * @exception  MalformedURLException  if no protocol is specified, or an
498     *               unknown protocol is found, or <tt>spec</tt> is <tt>null</tt>.
499     * @exception  SecurityException
500     *        if a security manager exists and its
501     *        <code>checkPermission</code> method doesn't allow
502     *        specifying a stream handler.
503     * @see        java.net.URL#URL(java.lang.String, java.lang.String,
504     *                  int, java.lang.String)
505     * @see        java.net.URLStreamHandler
506     * @see        java.net.URLStreamHandler#parseURL(java.net.URL,
507     *                  java.lang.String, int, int)
508     */
509    public URL(URL context, String spec, URLStreamHandler handler)
510        throws MalformedURLException
511    {
512        String original = spec;
513        int i, limit, c;
514        int start = 0;
515        String newProtocol = null;
516        boolean aRef=false;
517        boolean isRelative = false;
518
519        // Check for permission to specify a handler
520        if (handler != null) {
521            SecurityManager sm = System.getSecurityManager();
522            if (sm != null) {
523                checkSpecifyHandler(sm);
524            }
525        }
526
527        try {
528            limit = spec.length();
529            while ((limit > 0) && (spec.charAt(limit - 1) <= ' ')) {
530                limit--;        //eliminate trailing whitespace
531            }
532            while ((start < limit) && (spec.charAt(start) <= ' ')) {
533                start++;        // eliminate leading whitespace
534            }
535
536            if (spec.regionMatches(true, start, "url:", 0, 4)) {
537                start += 4;
538            }
539            if (start < spec.length() && spec.charAt(start) == '#') {
540                /* we're assuming this is a ref relative to the context URL.
541                 * This means protocols cannot start w/ '#', but we must parse
542                 * ref URL's like: "hello:there" w/ a ':' in them.
543                 */
544                aRef=true;
545            }
546            for (i = start ; !aRef && (i < limit) &&
547                     ((c = spec.charAt(i)) != '/') ; i++) {
548                if (c == ':') {
549
550                    String s = spec.substring(start, i).toLowerCase();
551                    if (isValidProtocol(s)) {
552                        newProtocol = s;
553                        start = i + 1;
554                    }
555                    break;
556                }
557            }
558
559            // Only use our context if the protocols match.
560            protocol = newProtocol;
561            if ((context != null) && ((newProtocol == null) ||
562                            newProtocol.equalsIgnoreCase(context.protocol))) {
563                // inherit the protocol handler from the context
564                // if not specified to the constructor
565                if (handler == null) {
566                    handler = context.handler;
567                }
568
569                // If the context is a hierarchical URL scheme and the spec
570                // contains a matching scheme then maintain backwards
571                // compatibility and treat it as if the spec didn't contain
572                // the scheme; see 5.2.3 of RFC2396
573                if (context.path != null && context.path.startsWith("/"))
574                    newProtocol = null;
575
576                if (newProtocol == null) {
577                    protocol = context.protocol;
578                    authority = context.authority;
579                    userInfo = context.userInfo;
580                    host = context.host;
581                    port = context.port;
582                    file = context.file;
583                    path = context.path;
584                    isRelative = true;
585                }
586            }
587
588            if (protocol == null) {
589                throw new MalformedURLException("no protocol: "+original);
590            }
591
592            // Get the protocol handler if not specified or the protocol
593            // of the context could not be used
594            if (handler == null &&
595                (handler = getURLStreamHandler(protocol)) == null) {
596                throw new MalformedURLException("unknown protocol: "+protocol);
597            }
598
599            this.handler = handler;
600
601            i = spec.indexOf('#', start);
602            if (i >= 0) {
603                ref = spec.substring(i + 1, limit);
604                limit = i;
605            }
606
607            /*
608             * Handle special case inheritance of query and fragment
609             * implied by RFC2396 section 5.2.2.
610             */
611            if (isRelative && start == limit) {
612                query = context.query;
613                if (ref == null) {
614                    ref = context.ref;
615                }
616            }
617
618            handler.parseURL(this, spec, start, limit);
619
620        } catch(MalformedURLException e) {
621            throw e;
622        } catch(Exception e) {
623            MalformedURLException exception = new MalformedURLException(e.getMessage());
624            exception.initCause(e);
625            throw exception;
626        }
627    }

629    /*
630     * Returns true if specified string is a valid protocol name.
631     */
632    private boolean isValidProtocol(String protocol) {
633        int len = protocol.length();
634        if (len < 1)
635            return false;
636        char c = protocol.charAt(0);
637        if (!Character.isLetter(c))
638            return false;
639        for (int i = 1; i < len; i++) {
640            c = protocol.charAt(i);
641            if (!Character.isLetterOrDigit(c) && c != '.' && c != '+' &&
642                c != '-') {

1138    // package name of the JDK implementation protocol handlers
1139    private static final String JDK_PACKAGE_PREFIX =  "sun.net.www.protocol";

1141    /**
1142     * Returns the Stream Handler.
1143     * @param protocol the protocol to use
1144     */
1145    static URLStreamHandler getURLStreamHandler(String protocol) {
1146
1147        URLStreamHandler handler = (URLStreamHandler)handlers.get(protocol);
1148        if (handler == null) {
1149
1150            boolean checkedWithFactory = false;
1151
1152            // Use the factory (if any)
1153            if (factory != null) {
1154                handler = factory.createURLStreamHandler(protocol);
1155                checkedWithFactory = true;
1156            }
1157
1158            // Try java protocol handler
1159            if (handler == null) {
1160                String packagePrefixList = null;
1161
1162                packagePrefixList
1163                    = java.security.AccessController.doPrivileged(
1164                    new sun.security.action.GetPropertyAction(
1165                        protocolPathProp,""));
1166                if (packagePrefixList != "") {
1167                    packagePrefixList += "|";
1168                }
1169
1170                // REMIND: decide whether to allow the "null" class prefix
1171                // or not.
1172                packagePrefixList += JDK_PACKAGE_PREFIX;
1173
1174                StringTokenizer packagePrefixIter =
1175                    new StringTokenizer(packagePrefixList, "|");
1176
1177                while (handler == null &&
1178                       packagePrefixIter.hasMoreTokens()) {
1179
1180                    String packagePrefix =
1181                      packagePrefixIter.nextToken().trim();
1182
1183                    // do not try to instantiate the JDK gopher handler
1184                    // unless the system property had been explicitly set
1185                    if (protocol.equalsIgnoreCase(GOPHER) &&
1186                        packagePrefix.equals(JDK_PACKAGE_PREFIX) &&
1187                        !enableGopher) {
1188                            continue;
1189                    }
1190                    try {
1191                        String clsName = packagePrefix + "." + protocol +
1192                          ".Handler";
1193                        Class cls = null;
1194                        try {
1195                            cls = Class.forName(clsName);
1196                        } catch (ClassNotFoundException e) {
1197                            ClassLoader cl = ClassLoader.getSystemClassLoader();
1198                            if (cl != null) {
1199                                cls = cl.loadClass(clsName);
1200                            }
1201                        }
1202                        if (cls != null) {
1203                            handler  =
1204                              (URLStreamHandler)cls.newInstance();
1205                        }
1206                    } catch (Exception e) {
1207                        // any number of exceptions can get thrown here
1208                    }
1209                }
1210            }
1211
1212            synchronized (streamHandlerLock) {
1213
1214                URLStreamHandler handler2 = null;
1215
1216                // Check again with hashtable just in case another
1217                // thread created a handler since we last checked
1218                handler2 = (URLStreamHandler)handlers.get(protocol);
1219
1220                if (handler2 != null) {
1221                    return handler2;
1222                }
1223
1224                // Check with factory if another thread set a
1225                // factory since our last check
1226                if (!checkedWithFactory && factory != null) {
1227                    handler2 = factory.createURLStreamHandler(protocol);
1228                }
1229
1230                if (handler2 != null) {
1231                    // The handler from the factory must be given more
1232                    // importance. Discard the default handler that
1233                    // this thread created.
1234                    handler = handler2;
1235                }
1236
1237                // Insert this handler into the hashtable
1238                if (handler != null) {
1239                    handlers.put(protocol, handler);
1240                }
1241
1242            }
1243        }
1244
1245        return handler;
1246
1247    }

1265    /**
1266     * readObject is called to restore the state of the URL from the
1267     * stream.  It reads the components of the URL and finds the local
1268     * stream handler.
1269     */
1270    private synchronized void readObject(java.io.ObjectInputStream s)
1271         throws IOException, ClassNotFoundException
1272    {
1273        s.defaultReadObject();  // read the fields
1274        if ((handler = getURLStreamHandler(protocol)) == null) {
1275            throw new IOException("unknown protocol: " + protocol);
1276        }
1277
1278        // Construct authority part
1279        if (authority == null &&
1280            ((host != null && host.length() > 0) || port != -1)) {
1281            if (host == null)
1282                host = "";
1283            authority = (port == -1) ? host : host + ":" + port;
1284
1285            // Handle hosts with userInfo in them
1286            int at = host.lastIndexOf('@');
1287            if (at != -1) {
1288                userInfo = host.substring(0, at);
1289                host = host.substring(at+1);
1290            }
1291        } else if (authority != null) {
1292            // Construct user info part
1293            int ind = authority.indexOf('@');
1294            if (ind != -1)
1295                userInfo = authority.substring(0, ind);
1296        }
1297
1298        // Construct path and query part
1299        path = null;
1300        query = null;
1301        if (file != null) {
1302            // Fix: only do this if hierarchical?
1303            int q = file.lastIndexOf('?');
1304            if (q != -1) {
1305                query = file.substring(q+1);
1306                path = file.substring(0, q);
1307            } else
1308                path = file;
1309        }
1310    }
