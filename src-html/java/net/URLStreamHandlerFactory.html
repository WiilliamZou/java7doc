026package java.net;
027
028/**
029 * This interface defines a factory for <code>URL</code> stream
030 * protocol handlers.
031 * <p>
032 * It is used by the <code>URL</code> class to create a
033 * <code>URLStreamHandler</code> for a specific protocol.
034 *
035 * @author  Arthur van Hoff
036 * @see     java.net.URL
037 * @see     java.net.URLStreamHandler
038 * @since   JDK1.0
039 */
040public interface URLStreamHandlerFactory {
041    /**
042     * Creates a new <code>URLStreamHandler</code> instance with the specified
043     * protocol.
044     *
045     * @param   protocol   the protocol ("<code>ftp</code>",
046     *                     "<code>http</code>", "<code>nntp</code>", etc.).
047     * @return  a <code>URLStreamHandler</code> for the specific protocol.
048     * @see     java.net.URLStreamHandler
049     */
050    URLStreamHandler createURLStreamHandler(String protocol);
051}
