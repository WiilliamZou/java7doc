052/**
053 * This class represents access to a network via sockets.
054 * A SocketPermission consists of a
055 * host specification and a set of "actions" specifying ways to
056 * connect to that host. The host is specified as
057 * <pre>
058 *    host = (hostname | IPv4address | iPv6reference) [:portrange]
059 *    portrange = portnumber | -portnumber | portnumber-[portnumber]
060 * </pre>
061 * The host is expressed as a DNS name, as a numerical IP address,
062 * or as "localhost" (for the local machine).
063 * The wildcard "*" may be included once in a DNS name host
064 * specification. If it is included, it must be in the leftmost
065 * position, as in "*.sun.com".
066 * <p>
067 * The format of the IPv6reference should follow that specified in <a
068 * href="http://www.ietf.org/rfc/rfc2732.txt"><i>RFC&nbsp;2732: Format
069 * for Literal IPv6 Addresses in URLs</i></a>:
070 * <pre>
071 *    ipv6reference = "[" IPv6address "]"
072 *</pre>
073 * For example, you can construct a SocketPermission instance
074 * as the following:
075 * <pre>
076 *    String hostAddress = inetaddress.getHostAddress();
077 *    if (inetaddress instanceof Inet6Address) {
078 *        sp = new SocketPermission("[" + hostAddress + "]:" + port, action);
079 *    } else {
080 *        sp = new SocketPermission(hostAddress + ":" + port, action);
081 *    }
082 * </pre>
083 * or
084 * <pre>
085 *    String host = url.getHost();
086 *    sp = new SocketPermission(host + ":" + port, action);
087 * </pre>
088 * <p>
089 * The <A HREF="Inet6Address.html#lform">full uncompressed form</A> of
090 * an IPv6 literal address is also valid.
091 * <p>
092 * The port or portrange is optional. A port specification of the
093 * form "N-", where <i>N</i> is a port number, signifies all ports
094 * numbered <i>N</i> and above, while a specification of the
095 * form "-N" indicates all ports numbered <i>N</i> and below.
096 * <p>
097 * The possible ways to connect to the host are
098 * <pre>
099 * accept
100 * connect
101 * listen
102 * resolve
103 * </pre>
104 * The "listen" action is only meaningful when used with "localhost".
105 * The "resolve" action is implied when any of the other actions are present.
106 * The action "resolve" refers to host/ip name service lookups.
107 * <P>
108 * The actions string is converted to lowercase before processing.
109 * <p>As an example of the creation and meaning of SocketPermissions,
110 * note that if the following permission:
111 *
112 * <pre>
113 *   p1 = new SocketPermission("puffin.eng.sun.com:7777", "connect,accept");
114 * </pre>
115 *
116 * is granted to some code, it allows that code to connect to port 7777 on
117 * <code>puffin.eng.sun.com</code>, and to accept connections on that port.
118 *
119 * <p>Similarly, if the following permission:
120 *
121 * <pre>
122 *   p2 = new SocketPermission("localhost:1024-", "accept,connect,listen");
123 * </pre>
124 *
125 * is granted to some code, it allows that code to
126 * accept connections on, connect to, or listen on any port between
127 * 1024 and 65535 on the local host.
128 *
129 * <p>Note: Granting code permission to accept or make connections to remote
130 * hosts may be dangerous because malevolent code can then more easily
131 * transfer and share confidential data among parties who may not
132 * otherwise have access to the data.
133 *
134 * @see java.security.Permissions
135 * @see SocketPermission
136 *
137 *
138 * @author Marianne Mueller
139 * @author Roland Schemers
140 *
141 * @serial exclude
142 */
143

257    /**
258     * Creates a new SocketPermission object with the specified actions.
259     * The host is expressed as a DNS name, or as a numerical IP address.
260     * Optionally, a port or a portrange may be supplied (separated
261     * from the DNS name or IP address by a colon).
262     * <p>
263     * To specify the local machine, use "localhost" as the <i>host</i>.
264     * Also note: An empty <i>host</i> String ("") is equivalent to "localhost".
265     * <p>
266     * The <i>actions</i> parameter contains a comma-separated list of the
267     * actions granted for the specified host (and port(s)). Possible actions are
268     * "connect", "listen", "accept", "resolve", or
269     * any combination of those. "resolve" is automatically added
270     * when any of the other three are specified.
271     * <p>
272     * Examples of SocketPermission instantiation are the following:
273     * <pre>
274     *    nr = new SocketPermission("www.catalog.com", "connect");
275     *    nr = new SocketPermission("www.sun.com:80", "connect");
276     *    nr = new SocketPermission("*.sun.com", "connect");
277     *    nr = new SocketPermission("*.edu", "resolve");
278     *    nr = new SocketPermission("204.160.241.0", "connect");
279     *    nr = new SocketPermission("localhost:1024-65535", "listen");
280     *    nr = new SocketPermission("204.160.241.0:1024-65535", "connect");
281     * </pre>
282     *
283     * @param host the hostname or IPaddress of the computer, optionally
284     * including a colon followed by a port or port range.
285     * @param action the action string.
286     */
287    public SocketPermission(String host, String action) {
288        super(getHost(host));
289        // name initialized to getHost(host); NPE detected in getHost()
290        init(getName(), getMask(action));

304    private static String getHost(String host)
305    {
306        if (host.equals("")) {
307            return "localhost";
308        } else {
309            /* IPv6 literal address used in this context should follow
310             * the format specified in RFC 2732;
311             * if not, we try to solve the unambiguous case
312             */
313            int ind;
314            if (host.charAt(0) != '[') {
315                if ((ind = host.indexOf(':')) != host.lastIndexOf(':')) {
316                    /* More than one ":", meaning IPv6 address is not
317                     * in RFC 2732 format;
318                     * We will rectify user errors for all unambiguious cases
319                     */
320                    StringTokenizer st = new StringTokenizer(host, ":");
321                    int tokens = st.countTokens();
322                    if (tokens == 9) {
323                        // IPv6 address followed by port
324                        ind = host.lastIndexOf(':');
325                        host = "[" + host.substring(0, ind) + "]" +
326                            host.substring(ind);
327                    } else if (tokens == 8 && host.indexOf("::") == -1) {
328                        // IPv6 address only, not followed by port
329                        host = "[" + host + "]";
330                    } else {
331                        // could be ambiguous
332                        throw new IllegalArgumentException("Ambiguous"+
333                                                           " hostport part");
334                    }
335                }
336            }
337            return host;
338        }
339    }

341    private int[] parsePort(String port)
342        throws Exception
343    {
344
345        if (port == null || port.equals("") || port.equals("*")) {
346            return new int[] {PORT_MIN, PORT_MAX};
347        }
348
349        int dash = port.indexOf('-');
350
351        if (dash == -1) {
352            int p = Integer.parseInt(port);
353            return new int[] {p, p};
354        } else {
355            String low = port.substring(0, dash);
356            String high = port.substring(dash+1);
357            int l,h;
358
359            if (low.equals("")) {
360                l = PORT_MIN;
361            } else {
362                l = Integer.parseInt(low);
363            }
364
365            if (high.equals("")) {
366                h = PORT_MAX;
367            } else {
368                h = Integer.parseInt(high);
369            }
370            if (l < 0 || h < 0 || h<l)
371                throw new IllegalArgumentException("invalid port range");
372
373            return new int[] {l, h};
374        }
375    }

385    /**
386     * Initialize the SocketPermission object. We don't do any DNS lookups
387     * as this point, instead we hold off until the implies method is
388     * called.
389     */
390    private void init(String host, int mask) {
391        // Set the integer mask that represents the actions
392
393        if ((mask & ALL) != mask)
394            throw new IllegalArgumentException("invalid actions mask");
395
396        // always OR in RESOLVE if we allow any of the others
397        this.mask = mask | RESOLVE;
398
399        // Parse the host name.  A name has up to three components, the
400        // hostname, a port number, or two numbers representing a port
401        // range.   "www.sun.com:8080-9090" is a valid host name.
402
403        // With IPv6 an address can be 2010:836B:4179::836B:4179
404        // An IPv6 address needs to be enclose in []
405        // For ex: [2010:836B:4179::836B:4179]:8080-9090
406        // Refer to RFC 2732 for more information.
407
408        int rb = 0 ;
409        int start = 0, end = 0;
410        int sep = -1;
411        String hostport = host;
412        if (host.charAt(0) == '[') {
413            start = 1;
414            rb = host.indexOf(']');
415            if (rb != -1) {
416                host = host.substring(start, rb);
417            } else {
418                throw new
419                    IllegalArgumentException("invalid host/port: "+host);
420            }
421            sep = hostport.indexOf(':', rb+1);
422        } else {
423            start = 0;
424            sep = host.indexOf(':', rb);
425            end = sep;
426            if (sep != -1) {
427                host = host.substring(start, end);
428            }
429        }
430
431        if (sep != -1) {
432            String port = hostport.substring(sep+1);
433            try {
434                portrange = parsePort(port);
435            } catch (Exception e) {
436                throw new
437                    IllegalArgumentException("invalid port range: "+port);
438            }
439        } else {
440            portrange = new int[] { PORT_MIN, PORT_MAX };
441        }
442
443        hostname = host;
444
445        // is this a domain wildcard specification
446        if (host.lastIndexOf('*') > 0) {
447            throw new
448               IllegalArgumentException("invalid host wildcard specification");
449        } else if (host.startsWith("*")) {
450            wildcard = true;
451            if (host.equals("*")) {
452                cname = "";
453            } else if (host.startsWith("*.")) {
454                cname = host.substring(1).toLowerCase();
455            } else {
456              throw new
457               IllegalArgumentException("invalid host wildcard specification");
458            }
459            return;
460        } else {
461            if (host.length() > 0) {
462                // see if we are being initialized with an IP address.
463                char ch = host.charAt(0);
464                if (ch == ':' || Character.digit(ch, 16) != -1) {
465                    byte ip[] = IPAddressUtil.textToNumericFormatV4(host);
466                    if (ip == null) {
467                        ip = IPAddressUtil.textToNumericFormatV6(host);
468                    }
469                    if (ip != null) {
470                        try {
471                            addresses =
472                                new InetAddress[]
473                                {InetAddress.getByAddress(ip) };
474                            init_with_ip = true;
475                        } catch (UnknownHostException uhe) {
476                            // this shouldn't happen
477                            invalid = true;
478                        }
479                    }
480                }
481            }
482        }
483    }

485    /**
486     * Convert an action string to an integer actions mask.
487     *
488     * @param action the action string
489     * @return the action mask
490     */
491    private static int getMask(String action) {
492
493        if (action == null) {
494            throw new NullPointerException("action can't be null");
495        }
496
497        if (action.equals("")) {
498            throw new IllegalArgumentException("action can't be empty");
499        }
500
501        int mask = NONE;
502
503        // Check against use of constants (used heavily within the JDK)
504        if (action == SecurityConstants.SOCKET_RESOLVE_ACTION) {
505            return RESOLVE;
506        } else if (action == SecurityConstants.SOCKET_CONNECT_ACTION) {
507            return CONNECT;
508        } else if (action == SecurityConstants.SOCKET_LISTEN_ACTION) {
509            return LISTEN;
510        } else if (action == SecurityConstants.SOCKET_ACCEPT_ACTION) {
511            return ACCEPT;
512        } else if (action == SecurityConstants.SOCKET_CONNECT_ACCEPT_ACTION) {
513            return CONNECT|ACCEPT;
514        }
515
516        char[] a = action.toCharArray();
517
518        int i = a.length - 1;
519        if (i < 0)
520            return mask;
521
522        while (i != -1) {
523            char c;
524
525            // skip whitespace
526            while ((i!=-1) && ((c = a[i]) == ' ' ||
527                               c == '\r' ||
528                               c == '\n' ||
529                               c == '\f' ||
530                               c == '\t'))
531                i--;
532
533            // check for the known strings
534            int matchlen;
535
536            if (i >= 6 && (a[i-6] == 'c' || a[i-6] == 'C') &&
537                          (a[i-5] == 'o' || a[i-5] == 'O') &&
538                          (a[i-4] == 'n' || a[i-4] == 'N') &&
539                          (a[i-3] == 'n' || a[i-3] == 'N') &&
540                          (a[i-2] == 'e' || a[i-2] == 'E') &&
541                          (a[i-1] == 'c' || a[i-1] == 'C') &&
542                          (a[i] == 't' || a[i] == 'T'))
543            {
544                matchlen = 7;
545                mask |= CONNECT;
546
547            } else if (i >= 6 && (a[i-6] == 'r' || a[i-6] == 'R') &&
548                                 (a[i-5] == 'e' || a[i-5] == 'E') &&
549                                 (a[i-4] == 's' || a[i-4] == 'S') &&
550                                 (a[i-3] == 'o' || a[i-3] == 'O') &&
551                                 (a[i-2] == 'l' || a[i-2] == 'L') &&
552                                 (a[i-1] == 'v' || a[i-1] == 'V') &&
553                                 (a[i] == 'e' || a[i] == 'E'))
554            {
555                matchlen = 7;
556                mask |= RESOLVE;
557
558            } else if (i >= 5 && (a[i-5] == 'l' || a[i-5] == 'L') &&
559                                 (a[i-4] == 'i' || a[i-4] == 'I') &&
560                                 (a[i-3] == 's' || a[i-3] == 'S') &&
561                                 (a[i-2] == 't' || a[i-2] == 'T') &&
562                                 (a[i-1] == 'e' || a[i-1] == 'E') &&
563                                 (a[i] == 'n' || a[i] == 'N'))
564            {
565                matchlen = 6;
566                mask |= LISTEN;
567
568            } else if (i >= 5 && (a[i-5] == 'a' || a[i-5] == 'A') &&
569                                 (a[i-4] == 'c' || a[i-4] == 'C') &&
570                                 (a[i-3] == 'c' || a[i-3] == 'C') &&
571                                 (a[i-2] == 'e' || a[i-2] == 'E') &&
572                                 (a[i-1] == 'p' || a[i-1] == 'P') &&
573                                 (a[i] == 't' || a[i] == 'T'))
574            {
575                matchlen = 6;
576                mask |= ACCEPT;
577
578            } else {
579                // parse error
580                throw new IllegalArgumentException(
581                        "invalid permission: " + action);
582            }
583
584            // make sure we didn't just match the tail of a word
585            // like "ackbarfaccept".  Also, skip to the comma.
586            boolean seencomma = false;
587            while (i >= matchlen && !seencomma) {
588                switch(a[i-matchlen]) {
589                case ',':
590                    seencomma = true;
591                    /*FALLTHROUGH*/
592                case ' ': case '\r': case '\n':
593                case '\f': case '\t':
594                    break;
595                default:
596                    throw new IllegalArgumentException(
597                            "invalid permission: " + action);
598                }
599                i--;
600            }
601
602            // point i at the location of the comma minus one (or -1).
603            i -= matchlen;
604        }
605
606        return mask;
607    }

609    private boolean isUntrusted()
610        throws UnknownHostException
611    {
612        if (trusted) return false;
613        if (invalid || untrusted) return true;
614        try {
615            if (!trustNameService && (defaultDeny ||
616                sun.net.www.URLConnection.isProxiedHost(hostname))) {
617                if (this.cname == null) {
618                    this.getCanonName();
619                }
620                if (!match(cname, hostname)) {
621                    // Last chance
622                    if (!authorized(hostname, addresses[0].getAddress())) {
623                        untrusted = true;
624                        Debug debug = getDebug();
625                        if (debug != null && Debug.isOn("failure")) {
626                            debug.println("socket access restriction: proxied host " + "(" + addresses[0] + ")" + " does not match " + cname + " from reverse lookup");
627                        }
628                        return true;
629                    }
630                }
631                trusted = true;
632            }
633        } catch (UnknownHostException uhe) {
634            invalid = true;
635            throw uhe;
636        }
637        return false;
638    }

640    /**
641     * attempt to get the fully qualified domain name
642     *
643     */
644    void getCanonName()
645        throws UnknownHostException
646    {
647        if (cname != null || invalid || untrusted) return;
648
649        // attempt to get the canonical name
650
651        try {
652            // first get the IP addresses if we don't have them yet
653            // this is because we need the IP address to then get
654            // FQDN.
655            if (addresses == null) {
656                getIP();
657            }
658
659            // we have to do this check, otherwise we might not
660            // get the fully qualified domain name
661            if (init_with_ip) {
662                cname = addresses[0].getHostName(false).toLowerCase();
663            } else {
664             cname = InetAddress.getByName(addresses[0].getHostAddress()).
665                                              getHostName(false).toLowerCase();
666            }
667        } catch (UnknownHostException uhe) {
668            invalid = true;
669            throw uhe;
670        }
671    }

675    private boolean match(String cname, String hname) {
676        String a = cname.toLowerCase();
677        String b = hname.toLowerCase();
678        if (a.startsWith(b)  &&
679            ((a.length() == b.length()) || (a.charAt(b.length()) == '.')))
680            return true;
681        if (cdomain == null) {
682            cdomain = RegisteredDomain.getRegisteredDomain(a);
683        }
684        if (hdomain == null) {
685            hdomain = RegisteredDomain.getRegisteredDomain(b);
686        }
687
688        return cdomain.length() != 0 && hdomain.length() != 0
689                        && cdomain.equals(hdomain);
690    }

692    private boolean authorized(String cname, byte[] addr) {
693        if (addr.length == 4)
694            return authorizedIPv4(cname, addr);
695        else if (addr.length == 16)
696            return authorizedIPv6(cname, addr);
697        else
698            return false;
699    }

701    private boolean authorizedIPv4(String cname, byte[] addr) {
702            String authHost = "";
703            InetAddress auth;
704
705        try {
706            authHost = "auth." +
707                        (addr[3] & 0xff) + "." + (addr[2] & 0xff) + "." +
708                        (addr[1] & 0xff) + "." + (addr[0] & 0xff) +
709                        ".in-addr.arpa";
710            // Following check seems unnecessary
711            // auth = InetAddress.getAllByName0(authHost, false)[0];
712            authHost = hostname + '.' + authHost;
713            auth = InetAddress.getAllByName0(authHost, false)[0];
714            if (auth.equals(InetAddress.getByAddress(addr))) {
715                return true;
716            }
717            Debug debug = getDebug();
718            if (debug != null && Debug.isOn("failure")) {
719                debug.println("socket access restriction: IP address of " + auth + " != " + InetAddress.getByAddress(addr));
720            }
721        } catch (UnknownHostException uhe) {
722            Debug debug = getDebug();
723            if (debug != null && Debug.isOn("failure")) {
724                debug.println("socket access restriction: forward lookup failed for " + authHost);
725            }
726        }
727        return false;

799    /**
800     * Checks if this socket permission object "implies" the
801     * specified permission.
802     * <P>
803     * More specifically, this method first ensures that all of the following
804     * are true (and returns false if any of them are not):<p>
805     * <ul>
806     * <li> <i>p</i> is an instanceof SocketPermission,<p>
807     * <li> <i>p</i>'s actions are a proper subset of this
808     * object's actions, and<p>
809     * <li> <i>p</i>'s port range is included in this port range. Note:
810     * port range is ignored when p only contains the action, 'resolve'.<p>
811     * </ul>
812     *
813     * Then <code>implies</code> checks each of the following, in order,
814     * and for each returns true if the stated condition is true:<p>
815     * <ul>
816     * <li> If this object was initialized with a single IP address and one of <i>p</i>'s
817     * IP addresses is equal to this object's IP address.<p>
818     * <li>If this object is a wildcard domain (such as *.sun.com), and
819     * <i>p</i>'s canonical name (the name without any preceding *)
820     * ends with this object's canonical host name. For example, *.sun.com
821     * implies *.eng.sun.com..<p>
822     * <li>If this object was not initialized with a single IP address, and one of this
823     * object's IP addresses equals one of <i>p</i>'s IP addresses.<p>
824     * <li>If this canonical name equals <i>p</i>'s canonical name.<p>
825     * </ul>
826     *
827     * If none of the above are true, <code>implies</code> returns false.
828     * @param p the permission to check against.
829     *
830     * @return true if the specified permission is implied by this object,
831     * false if not.
832     */
833
834    public boolean implies(Permission p) {
835        int i,j;
836
837        if (!(p instanceof SocketPermission))
838            return false;
839
840        if (p == this)
841            return true;
842
843        SocketPermission that = (SocketPermission) p;
844
845        return ((this.mask & that.mask) == that.mask) &&
846                                        impliesIgnoreMask(that);
847    }

849    /**
850     * Checks if the incoming Permission's action are a proper subset of
851     * the this object's actions.
852     * <P>
853     * Check, in the following order:
854     * <ul>
855     * <li> Checks that "p" is an instanceof a SocketPermission
856     * <li> Checks that "p"'s actions are a proper subset of the
857     * current object's actions.
858     * <li> Checks that "p"'s port range is included in this port range
859     * <li> If this object was initialized with an IP address, checks that
860     *      one of "p"'s IP addresses is equal to this object's IP address.
861     * <li> If either object is a wildcard domain (i.e., "*.sun.com"),
862     *      attempt to match based on the wildcard.
863     * <li> If this object was not initialized with an IP address, attempt
864     *      to find a match based on the IP addresses in both objects.
865     * <li> Attempt to match on the canonical hostnames of both objects.
866     * </ul>
867     * @param p the incoming permission request
868     *
869     * @return true if "permission" is a proper subset of the current object,
870     * false if not.
871     */
872
873    boolean impliesIgnoreMask(SocketPermission that) {
874        int i,j;
875
876        if ((that.mask & RESOLVE) != that.mask) {
877
878            // check simple port range
879            if ((that.portrange[0] < this.portrange[0]) ||
880                    (that.portrange[1] > this.portrange[1])) {
881
882                // if either includes the ephemeral range, do full check
883                if (this.includesEphemerals() || that.includesEphemerals()) {
884                    if (!inRange(this.portrange[0], this.portrange[1],
885                                     that.portrange[0], that.portrange[1]))
886                    {
887                                return false;
888                    }
889                } else {
890                    return false;
891                }
892            }
893        }
894
895        // allow a "*" wildcard to always match anything
896        if (this.wildcard && "".equals(this.cname))
897            return true;
898
899        // return if either one of these NetPerm objects are invalid...
900        if (this.invalid || that.invalid) {
901            return compareHostnames(that);
902        }
903
904        try {
905            if (this.init_with_ip) { // we only check IP addresses
906                if (that.wildcard)
907                    return false;
908
909                if (that.init_with_ip) {
910                    return (this.addresses[0].equals(that.addresses[0]));
911                } else {
912                    if (that.addresses == null) {
913                        that.getIP();
914                    }
915                    for (i=0; i < that.addresses.length; i++) {
916                        if (this.addresses[0].equals(that.addresses[i]))
917                            return true;
918                    }
919                }
920                // since "this" was initialized with an IP address, we
921                // don't check any other cases
922                return false;
923            }
924
925            // check and see if we have any wildcards...
926            if (this.wildcard || that.wildcard) {
927                // if they are both wildcards, return true iff
928                // that's cname ends with this cname (i.e., *.sun.com
929                // implies *.eng.sun.com)
930                if (this.wildcard && that.wildcard)
931                    return (that.cname.endsWith(this.cname));
932
933                // a non-wildcard can't imply a wildcard
934                if (that.wildcard)
935                    return false;
936
937                // this is a wildcard, lets see if that's cname ends with
938                // it...
939                if (that.cname == null) {
940                    that.getCanonName();
941                }
942                return (that.cname.endsWith(this.cname));
943            }
944
945            // comapare IP addresses
946            if (this.addresses == null) {
947                this.getIP();
948            }
949
950            if (that.addresses == null) {
951                that.getIP();
952            }
953
954            if (!(that.init_with_ip && this.isUntrusted())) {
955                for (j = 0; j < this.addresses.length; j++) {
956                    for (i=0; i < that.addresses.length; i++) {
957                        if (this.addresses[j].equals(that.addresses[i]))
958                            return true;
959                    }
960                }
961
962                // XXX: if all else fails, compare hostnames?
963                // Do we really want this?
964                if (this.cname == null) {
965                    this.getCanonName();
966                }
967
968                if (that.cname == null) {
969                    that.getCanonName();
970                }
971
972                return (this.cname.equalsIgnoreCase(that.cname));
973            }
974
975        } catch (UnknownHostException uhe) {
976            return compareHostnames(that);
977        }
978
979        // make sure the first thing that is done here is to return
980        // false. If not, uncomment the return false in the above catch.
981
982        return false;
983    }

997    /**
998     * Checks two SocketPermission objects for equality.
999     * <P>
1000     * @param obj the object to test for equality with this object.
1001     *
1002     * @return true if <i>obj</i> is a SocketPermission, and has the
1003     *  same hostname, port range, and actions as this
1004     *  SocketPermission object. However, port range will be ignored
1005     *  in the comparison if <i>obj</i> only contains the action, 'resolve'.
1006     */
1007    public boolean equals(Object obj) {
1008        if (obj == this)
1009            return true;
1010
1011        if (! (obj instanceof SocketPermission))
1012            return false;
1013
1014        SocketPermission that = (SocketPermission) obj;
1015
1016        //this is (overly?) complex!!!
1017
1018        // check the mask first
1019        if (this.mask != that.mask) return false;
1020
1021        if ((that.mask & RESOLVE) != that.mask) {
1022            // now check the port range...
1023            if ((this.portrange[0] != that.portrange[0]) ||
1024                (this.portrange[1] != that.portrange[1])) {
1025                return false;
1026            }
1027        }
1028
1029        // short cut. This catches:
1030        //  "crypto" equal to "crypto", or
1031        // "1.2.3.4" equal to "1.2.3.4.", or
1032        //  "*.edu" equal to "*.edu", but it
1033        //  does not catch "crypto" equal to
1034        // "crypto.eng.sun.com".
1035
1036        if (this.getName().equalsIgnoreCase(that.getName())) {
1037            return true;
1038        }
1039
1040        // we now attempt to get the Canonical (FQDN) name and
1041        // compare that. If this fails, about all we can do is return
1042        // false.
1043
1044        try {
1045            this.getCanonName();
1046            that.getCanonName();
1047        } catch (UnknownHostException uhe) {
1048            return false;
1049        }
1050
1051        if (this.invalid || that.invalid)
1052            return false;
1053
1054        if (this.cname != null) {
1055            return this.cname.equalsIgnoreCase(that.cname);
1056        }
1057
1058        return false;
1059    }

1225    /**
1226     * Check if the target range is within the policy range
1227     * together with the ephemeral range for this platform
1228     * (if policy includes ephemeral range)
1229     */
1230    private static boolean inRange(
1231        int policyLow, int policyHigh, int targetLow, int targetHigh
1232    )
1233    {
1234        if (targetLow == 0) {
1235            // check policy includes ephemeral range
1236            if (!inRange(policyLow, policyHigh, ephemeralLow, ephemeralHigh)) {
1237                return false;
1238            }
1239            if (targetHigh == 0) {
1240                // nothing left to do
1241                return true;
1242            }
1243            // continue check with first real port number
1244            targetLow = 1;
1245        }
1246
1247        if (policyLow == 0 && policyHigh == 0) {
1248            // ephemeral range only
1249            return targetLow >= ephemeralLow && targetHigh <= ephemeralHigh;
1250        }
1251
1252        if (policyLow != 0) {
1253            // simple check of policy only
1254            return targetLow >= policyLow && targetHigh <= policyHigh;
1255        }
1256
1257        // policyLow == 0 which means possibly two ranges to check
1258
1259        // first check if policy and ephem range overlap/contiguous
1260
1261        if (policyHigh >= ephemeralLow - 1) {
1262            return targetHigh <= ephemeralHigh;
1263        }
1264
1265        // policy and ephem range do not overlap
1266
1267        // target range must lie entirely inside policy range or eph range
1268
1269        return  (targetLow <= policyHigh && targetHigh <= policyHigh) ||
1270                (targetLow >= ephemeralLow && targetHigh <= ephemeralHigh);
1271    }

1378     * @return true if "permission" is a proper subset of a permission in

1381

1383    {
1384        if (! (permission instanceof SocketPermission))
1385                return false;
1386
1387        SocketPermission np = (SocketPermission) permission;
1388
1389        int desired = np.getMask();
1390        int effective = 0;
1391        int needed = desired;
1392
1393        synchronized (this) {
1394            int len = perms.size();
1395            //System.out.println("implies "+np);
1396            for (int i = 0; i < len; i++) {

1400                    effective |=  x.getMask();
1401                    if ((effective & desired) == desired)
1402                        return true;
1403                    needed = (desired ^ effective);
1404                }

1406        }
1407        return false;
