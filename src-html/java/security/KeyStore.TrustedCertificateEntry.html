039/**
040 * This class represents a storage facility for cryptographic
041 * keys and certificates.
042 *
043 * <p> A <code>KeyStore</code> manages different types of entries.
044 * Each type of entry implements the <code>KeyStore.Entry</code> interface.
045 * Three basic <code>KeyStore.Entry</code> implementations are provided:
046 *
047 * <ul>
048 * <li><b>KeyStore.PrivateKeyEntry</b>
049 * <p> This type of entry holds a cryptographic <code>PrivateKey</code>,
050 * which is optionally stored in a protected format to prevent
051 * unauthorized access.  It is also accompanied by a certificate chain
052 * for the corresponding public key.
053 *
054 * <p> Private keys and certificate chains are used by a given entity for
055 * self-authentication. Applications for this authentication include software
056 * distribution organizations which sign JAR files as part of releasing
057 * and/or licensing software.
058 *
059 * <li><b>KeyStore.SecretKeyEntry</b>
060 * <p> This type of entry holds a cryptographic <code>SecretKey</code>,
061 * which is optionally stored in a protected format to prevent
062 * unauthorized access.
063 *
064 * <li><b>KeyStore.TrustedCertificateEntry</b>
065 * <p> This type of entry contains a single public key <code>Certificate</code>
066 * belonging to another party. It is called a <i>trusted certificate</i>
067 * because the keystore owner trusts that the public key in the certificate
068 * indeed belongs to the identity identified by the <i>subject</i> (owner)
069 * of the certificate.
070 *
071 * <p>This type of entry can be used to authenticate other parties.
072 * </ul>
073 *
074 * <p> Each entry in a keystore is identified by an "alias" string. In the
075 * case of private keys and their associated certificate chains, these strings
076 * distinguish among the different ways in which the entity may authenticate
077 * itself. For example, the entity may authenticate itself using different
078 * certificate authorities, or using different public key algorithms.
079 *
080 * <p> Whether aliases are case sensitive is implementation dependent. In order
081 * to avoid problems, it is recommended not to use aliases in a KeyStore that
082 * only differ in case.
083 *
084 * <p> Whether keystores are persistent, and the mechanisms used by the
085 * keystore if it is persistent, are not specified here. This allows
086 * use of a variety of techniques for protecting sensitive (e.g., private or
087 * secret) keys. Smart cards or other integrated cryptographic engines
088 * (SafeKeyper) are one option, and simpler mechanisms such as files may also
089 * be used (in a variety of formats).
090 *
091 * <p> Typical ways to request a KeyStore object include
092 * relying on the default type and providing a specific keystore type.
093 *
094 * <ul>
095 * <li>To rely on the default type:
096 * <pre>
097 *    KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
098 * </pre>
099 * The system will return a keystore implementation for the default type.
100 * <p>
101 *
102 * <li>To provide a specific keystore type:
103 * <pre>
104 *      KeyStore ks = KeyStore.getInstance("JKS");
105 * </pre>
106 * The system will return the most preferred implementation of the
107 * specified keystore type available in the environment. <p>
108 * </ul>
109 *
110 * <p> Before a keystore can be accessed, it must be
111 * {@link #load(java.io.InputStream, char[]) loaded}.
112 * <pre>
113 *    KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
114 *
115 *    // get user password and file input stream
116 *    char[] password = getPassword();
117 *
118 *    java.io.FileInputStream fis = null;
119 *    try {
120 *        fis = new java.io.FileInputStream("keyStoreName");
121 *        ks.load(fis, password);
122 *    } finally {
123 *        if (fis != null) {
124 *            fis.close();
125 *        }
126 *    }
127 * </pre>
128 *
129 * To create an empty keystore using the above <code>load</code> method,
130 * pass <code>null</code> as the <code>InputStream</code> argument.
131 *
132 * <p> Once the keystore has been loaded, it is possible
133 * to read existing entries from the keystore, or to write new entries
134 * into the keystore:
135 * <pre>
136 *    KeyStore.ProtectionParameter protParam =
137 *        new KeyStore.PasswordProtection(password);
138 *
139 *    // get my private key
140 *    KeyStore.PrivateKeyEntry pkEntry = (KeyStore.PrivateKeyEntry)
141 *        ks.getEntry("privateKeyAlias", protParam);
142 *    PrivateKey myPrivateKey = pkEntry.getPrivateKey();
143 *
144 *    // save my secret key
145 *    javax.crypto.SecretKey mySecretKey;
146 *    KeyStore.SecretKeyEntry skEntry =
147 *        new KeyStore.SecretKeyEntry(mySecretKey);
148 *    ks.setEntry("secretKeyAlias", skEntry, protParam);
149 *
150 *    // store away the keystore
151 *    java.io.FileOutputStream fos = null;
152 *    try {
153 *        fos = new java.io.FileOutputStream("newKeyStoreName");
154 *        ks.store(fos, password);
155 *    } finally {
156 *        if (fos != null) {
157 *            fos.close();
158 *        }
159 *    }
160 * </pre>
161 *
162 * Note that although the same password may be used to
163 * load the keystore, to protect the private key entry,
164 * to protect the secret key entry, and to store the keystore
165 * (as is shown in the sample code above),
166 * different passwords or other protection parameters
167 * may also be used.
168 *
169 * <p> Every implementation of the Java platform is required to support
170 * the following standard <code>KeyStore</code> type:
171 * <ul>
172 * <li><tt>PKCS12</tt></li>
173 * </ul>
174 * This type is described in the <a href=
175 * "{@docRoot}/../technotes/guides/security/StandardNames.html#KeyStore">
176 * KeyStore section</a> of the
177 * Java Cryptography Architecture Standard Algorithm Name Documentation.
178 * Consult the release documentation for your implementation to see if any
179 * other types are supported.
180 *
181 * @author Jan Luehe
182 *
183 * @see java.security.PrivateKey
184 * @see javax.crypto.SecretKey
185 * @see java.security.cert.Certificate
186 *
187 * @since 1.2
188 */

251    /**
252     * A password-based implementation of <code>ProtectionParameter</code>.
253     *
254     * @since 1.5
255     */
256    public static class PasswordProtection implements
257                ProtectionParameter, javax.security.auth.Destroyable {
258
259        private final char[] password;
260        private volatile boolean destroyed = false;
261
262        /**
263         * Creates a password parameter.
264         *
265         * <p> The specified <code>password</code> is cloned before it is stored
266         * in the new <code>PasswordProtection</code> object.
267         *
268         * @param password the password, which may be <code>null</code>
269         */
270        public PasswordProtection(char[] password) {
271            this.password = (password == null) ? null : password.clone();
272        }
273
274        /**
275         * Gets the password.
276         *
277         * <p>Note that this method returns a reference to the password.
278         * If a clone of the array is created it is the caller's
279         * responsibility to zero out the password information
280         * after it is no longer needed.
281         *
282         * @see #destroy()
283         * @return the password, which may be <code>null</code>
284         * @exception IllegalStateException if the password has
285         *              been cleared (destroyed)
286         */
287        public synchronized char[] getPassword() {
288            if (destroyed) {
289                throw new IllegalStateException("password has been cleared");
290            }
291            return password;
292        }
293
294        /**
295         * Clears the password.
296         *
297         * @exception DestroyFailedException if this method was unable
298         *      to clear the password
299         */
300        public synchronized void destroy()
301                throws javax.security.auth.DestroyFailedException {
302            destroyed = true;
303            if (password != null) {
304                Arrays.fill(password, ' ');
305            }
306        }
307
308        /**
309         * Determines if password has been cleared.
310         *
311         * @return true if the password has been cleared, false otherwise
312         */
313        public synchronized boolean isDestroyed() {
314            return destroyed;
315        }
316    }

360    /**
361     * A <code>KeyStore</code> entry that holds a <code>PrivateKey</code>
362     * and corresponding certificate chain.
363     *
364     * @since 1.5
365     */
366    public static final class PrivateKeyEntry implements Entry {
367
368        private final PrivateKey privKey;
369        private final Certificate[] chain;
370
371        /**
372         * Constructs a <code>PrivateKeyEntry</code> with a
373         * <code>PrivateKey</code> and corresponding certificate chain.
374         *
375         * <p> The specified <code>chain</code> is cloned before it is stored
376         * in the new <code>PrivateKeyEntry</code> object.
377         *
378         * @param privateKey the <code>PrivateKey</code>
379         * @param chain an array of <code>Certificate</code>s
380         *      representing the certificate chain.
381         *      The chain must be ordered and contain a
382         *      <code>Certificate</code> at index 0
383         *      corresponding to the private key.
384         *
385         * @exception NullPointerException if
386         *      <code>privateKey</code> or <code>chain</code>
387         *      is <code>null</code>
388         * @exception IllegalArgumentException if the specified chain has a
389         *      length of 0, if the specified chain does not contain
390         *      <code>Certificate</code>s of the same type,
391         *      or if the <code>PrivateKey</code> algorithm
392         *      does not match the algorithm of the <code>PublicKey</code>
393         *      in the end entity <code>Certificate</code> (at index 0)
394         */
395        public PrivateKeyEntry(PrivateKey privateKey, Certificate[] chain) {
396            if (privateKey == null || chain == null) {
397                throw new NullPointerException("invalid null input");
398            }
399            if (chain.length == 0) {
400                throw new IllegalArgumentException
401                                ("invalid zero-length input chain");
402            }
403
404            Certificate[] clonedChain = chain.clone();
405            String certType = clonedChain[0].getType();
406            for (int i = 1; i < clonedChain.length; i++) {
407                if (!certType.equals(clonedChain[i].getType())) {
408                    throw new IllegalArgumentException
409                                ("chain does not contain certificates " +
410                                "of the same type");
411                }
412            }
413            if (!privateKey.getAlgorithm().equals
414                        (clonedChain[0].getPublicKey().getAlgorithm())) {
415                throw new IllegalArgumentException
416                                ("private key algorithm does not match " +
417                                "algorithm of public key in end entity " +
418                                "certificate (at index 0)");
419            }
420            this.privKey = privateKey;
421
422            if (clonedChain[0] instanceof X509Certificate &&
423                !(clonedChain instanceof X509Certificate[])) {
424
425                this.chain = new X509Certificate[clonedChain.length];
426                System.arraycopy(clonedChain, 0,
427                                this.chain, 0, clonedChain.length);
428            } else {
429                this.chain = clonedChain;
430            }
431        }
432
433        /**
434         * Gets the <code>PrivateKey</code> from this entry.
435         *
436         * @return the <code>PrivateKey</code> from this entry
437         */
438        public PrivateKey getPrivateKey() {
439            return privKey;
440        }
441
442        /**
443         * Gets the <code>Certificate</code> chain from this entry.
444         *
445         * <p> The stored chain is cloned before being returned.
446         *
447         * @return an array of <code>Certificate</code>s corresponding
448         *      to the certificate chain for the public key.
449         *      If the certificates are of type X.509,
450         *      the runtime type of the returned array is
451         *      <code>X509Certificate[]</code>.
452         */
453        public Certificate[] getCertificateChain() {
454            return chain.clone();
455        }
456
457        /**
458         * Gets the end entity <code>Certificate</code>
459         * from the certificate chain in this entry.
460         *
461         * @return the end entity <code>Certificate</code> (at index 0)
462         *      from the certificate chain in this entry.
463         *      If the certificate is of type X.509,
464         *      the runtime type of the returned certificate is
465         *      <code>X509Certificate</code>.
466         */
467        public Certificate getCertificate() {
468            return chain[0];
469        }
470
471        /**
472         * Returns a string representation of this PrivateKeyEntry.
473         * @return a string representation of this PrivateKeyEntry.
474         */
475        public String toString() {
476            StringBuilder sb = new StringBuilder();
477            sb.append("Private key entry and certificate chain with "
478                + chain.length + " elements:\r\n");
479            for (Certificate cert : chain) {
480                sb.append(cert);
481                sb.append("\r\n");
482            }
483            return sb.toString();
484        }
485
486    }

595    /**
596     * Returns a keystore object of the specified type.
597     *
598     * <p> This method traverses the list of registered security Providers,
599     * starting with the most preferred Provider.
600     * A new KeyStore object encapsulating the
601     * KeyStoreSpi implementation from the first
602     * Provider that supports the specified type is returned.
603     *
604     * <p> Note that the list of registered providers may be retrieved via
605     * the {@link Security#getProviders() Security.getProviders()} method.
606     *
607     * @param type the type of keystore.
608     * See the KeyStore section in the <a href=
609     * "{@docRoot}/../technotes/guides/security/StandardNames.html#KeyStore">
610     * Java Cryptography Architecture Standard Algorithm Name Documentation</a>
611     * for information about standard keystore types.
612     *
613     * @return a keystore object of the specified type.
614     *
615     * @exception KeyStoreException if no Provider supports a
616     *          KeyStoreSpi implementation for the
617     *          specified type.
618     *
619     * @see Provider
620     */
621    public static KeyStore getInstance(String type)
622        throws KeyStoreException
623    {
624        try {
625            Object[] objs = Security.getImpl(type, "KeyStore", (String)null);
626            return new KeyStore((KeyStoreSpi)objs[0], (Provider)objs[1], type);
627        } catch (NoSuchAlgorithmException nsae) {
628            throw new KeyStoreException(type + " not found", nsae);
629        } catch (NoSuchProviderException nspe) {
630            throw new KeyStoreException(type + " not found", nspe);
631        }
632    }

721    /**
722     * Returns the default keystore type as specified in the Java security
723     * properties file, or the string
724     * &quot;jks&quot; (acronym for &quot;Java keystore&quot;)
725     * if no such property exists.
726     * The Java security properties file is located in the file named
727     * &lt;JAVA_HOME&gt;/lib/security/java.security.
728     * &lt;JAVA_HOME&gt; refers to the value of the java.home system property,
729     * and specifies the directory where the JRE is installed.
730     *
731     * <p>The default keystore type can be used by applications that do not
732     * want to use a hard-coded keystore type when calling one of the
733     * <code>getInstance</code> methods, and want to provide a default keystore
734     * type in case a user does not specify its own.
735     *
736     * <p>The default keystore type can be changed by setting the value of the
737     * "keystore.type" security property (in the Java security properties
738     * file) to the desired keystore type.
739     *
740     * @return the default keystore type as specified in the
741     * Java security properties file, or the string &quot;jks&quot;
742     * if no such property exists.
743     */
744    public final static String getDefaultType() {
745        String kstype;
746        kstype = AccessController.doPrivileged(new PrivilegedAction<String>() {
747            public String run() {
748                return Security.getProperty(KEYSTORE_TYPE);
749            }
750        });
751        if (kstype == null) {

1422         * @throws KeyStoreException if an error occured during the
1423         *   operation
1424         * @throws IllegalStateException if the getKeyStore method has
1425         *   not been invoked prior to calling this method
1426         */
1427        public abstract ProtectionParameter getProtectionParameter(String alias)
1428            throws KeyStoreException;

1433         * will return <code>keyStore</code>, the {@linkplain
1434         * #getProtectionParameter getProtectionParameter()} method will

1441         * @param keyStore the KeyStore to be encapsulated
1442         * @param protectionParameter the ProtectionParameter used to

1479        /**
1480         * Returns a new Builder object.
1481         *
1482         * <p>The first call to the {@link #getKeyStore} method on the returned
1483         * builder will create a KeyStore of type <code>type</code> and call
1484         * its {@link KeyStore#load load()} method.
1485         * The <code>inputStream</code> argument is constructed from
1486         * <code>file</code>.
1487         * If <code>protection</code> is a
1488         * <code>PasswordProtection</code>, the password is obtained by
1489         * calling the <code>getPassword</code> method.
1490         * Otherwise, if <code>protection</code> is a
1491         * <code>CallbackHandlerProtection</code>, the password is obtained
1492         * by invoking the CallbackHandler.
1493         *
1494         * <p>Subsequent calls to {@link #getKeyStore} return the same object
1495         * as the initial call. If the initial call to failed with a
1496         * KeyStoreException, subsequent calls also throw a
1497         * KeyStoreException.
1498         *
1499         * <p>The KeyStore is instantiated from <code>provider</code> if
1500         * non-null. Otherwise, all installed providers are searched.
1501         *
1502         * <p>Calls to {@link #getProtectionParameter getProtectionParameter()}
1503         * will return a {@link KeyStore.PasswordProtection PasswordProtection}
1504         * object encapsulating the password that was used to invoke the
1505         * <code>load</code> method.
1506         *
1507         * <p><em>Note</em> that the {@link #getKeyStore} method is executed
1508         * within the {@link AccessControlContext} of the code invoking this
1509         * method.
1510         *
1511         * @return a new Builder object
1512         * @param type the type of KeyStore to be constructed
1513         * @param provider the provider from which the KeyStore is to
1514         *   be instantiated (or null)
1515         * @param file the File that contains the KeyStore data
1516         * @param protection the ProtectionParameter securing the KeyStore data
1517         * @throws NullPointerException if type, file or protection is null
1518         * @throws IllegalArgumentException if protection is not an instance
1519         *   of either PasswordProtection or CallbackHandlerProtection; or
1520         *   if file does not exist or does not refer to a normal file
1521         */
1522        public static Builder newInstance(String type, Provider provider,
1523                File file, ProtectionParameter protection) {
1524            if ((type == null) || (file == null) || (protection == null)) {
1525                throw new NullPointerException();
1526            }
1527            if ((protection instanceof PasswordProtection == false) &&
1528                (protection instanceof CallbackHandlerProtection == false)) {
1529                throw new IllegalArgumentException
1530                ("Protection must be PasswordProtection or " +
1531                 "CallbackHandlerProtection");
1532            }
1533            if (file.isFile() == false) {
1534                throw new IllegalArgumentException
1535                    ("File does not exist or it does not refer " +
1536                     "to a normal file: " + file);
1537            }
1538            return new FileBuilder(type, provider, file, protection,
1539                AccessController.getContext());
1540        }

1542        private static final class FileBuilder extends Builder {
1543
1544            private final String type;
1545            private final Provider provider;
1546            private final File file;
1547            private ProtectionParameter protection;
1548            private ProtectionParameter keyProtection;
1549            private final AccessControlContext context;
1550
1551            private KeyStore keyStore;
1552
1553            private Throwable oldException;
1554
1555            FileBuilder(String type, Provider provider, File file,
1556                    ProtectionParameter protection,
1557                    AccessControlContext context) {
1558                this.type = type;
1559                this.provider = provider;
1560                this.file = file;
1561                this.protection = protection;
1562                this.context = context;
1563            }
1564
1565            public synchronized KeyStore getKeyStore() throws KeyStoreException
1566            {
1567                if (keyStore != null) {
1568                    return keyStore;
1569                }
1570                if (oldException != null) {
1571                    throw new KeyStoreException
1572                        ("Previous KeyStore instantiation failed",
1573                         oldException);
1574                }
1575                PrivilegedExceptionAction<KeyStore> action =
1576                        new PrivilegedExceptionAction<KeyStore>() {
1577                    public KeyStore run() throws Exception {
1578                        if (protection instanceof CallbackHandlerProtection == false) {
1579                            return run0();
1580                        }
1581                        // when using a CallbackHandler,
1582                        // reprompt if the password is wrong
1583                        int tries = 0;
1584                        while (true) {
1585                            tries++;
1586                            try {
1587                                return run0();
1588                            } catch (IOException e) {
1589                                if ((tries < MAX_CALLBACK_TRIES)
1590                                        && (e.getCause() instanceof UnrecoverableKeyException)) {
1591                                    continue;
1592                                }
1593                                throw e;
1594                            }
1595                        }
1596                    }
1597                    public KeyStore run0() throws Exception {
1598                        KeyStore ks;
1599                        if (provider == null) {
1600                            ks = KeyStore.getInstance(type);
1601                        } else {
1602                            ks = KeyStore.getInstance(type, provider);
1603                        }
1604                        InputStream in = null;
1605                        char[] password = null;
1606                        try {
1607                            in = new FileInputStream(file);
1608                            if (protection instanceof PasswordProtection) {
1609                                password =
1610                                ((PasswordProtection)protection).getPassword();
1611                                keyProtection = protection;
1612                            } else {
1613                                CallbackHandler handler =
1614                                    ((CallbackHandlerProtection)protection)
1615                                    .getCallbackHandler();
1616                                PasswordCallback callback = new PasswordCallback
1617                                    ("Password for keystore " + file.getName(),
1618                                    false);
1619                                handler.handle(new Callback[] {callback});
1620                                password = callback.getPassword();
1621                                if (password == null) {
1622                                    throw new KeyStoreException("No password" +
1623                                                                " provided");
1624                                }
1625                                callback.clearPassword();
1626                                keyProtection = new PasswordProtection(password);
1627                            }
1628                            ks.load(in, password);
1629                            return ks;
1630                        } finally {
1631                            if (in != null) {
1632                                in.close();
1633                            }
1634                        }
1635                    }
1636                };
1637                try {
1638                    keyStore = AccessController.doPrivileged(action, context);
1639                    return keyStore;
1640                } catch (PrivilegedActionException e) {
1641                    oldException = e.getCause();
1642                    throw new KeyStoreException
1643                        ("KeyStore instantiation failed", oldException);
1644                }
1645            }
1646
1647            public synchronized ProtectionParameter
1648                        getProtectionParameter(String alias) {
1649                if (alias == null) {
1650                    throw new NullPointerException();
1651                }
1652                if (keyStore == null) {
1653                    throw new IllegalStateException
1654                        ("getKeyStore() must be called first");
1655                }
1656                return keyProtection;
1657            }
1658        }

1660        /**
1661         * Returns a new Builder object.
1662         *
1663         * <p>Each call to the {@link #getKeyStore} method on the returned
1664         * builder will return a new KeyStore object of type <code>type</code>.
1665         * Its {@link KeyStore#load(KeyStore.LoadStoreParameter) load()}
1666         * method is invoked using a
1667         * <code>LoadStoreParameter</code> that encapsulates
1668         * <code>protection</code>.
1669         *
1670         * <p>The KeyStore is instantiated from <code>provider</code> if
1671         * non-null. Otherwise, all installed providers are searched.
1672         *
1673         * <p>Calls to {@link #getProtectionParameter getProtectionParameter()}
1674         * will return <code>protection</code>.
1675         *
1676         * <p><em>Note</em> that the {@link #getKeyStore} method is executed
1677         * within the {@link AccessControlContext} of the code invoking this
1678         * method.
1679         *
1680         * @return a new Builder object
1681         * @param type the type of KeyStore to be constructed
1682         * @param provider the provider from which the KeyStore is to
1683         *   be instantiated (or null)
1684         * @param protection the ProtectionParameter securing the Keystore
1685         * @throws NullPointerException if type or protection is null
1686         */
1687        public static Builder newInstance(final String type,
1688                final Provider provider, final ProtectionParameter protection) {
1689            if ((type == null) || (protection == null)) {
1690                throw new NullPointerException();
1691            }
1692            final AccessControlContext context = AccessController.getContext();
1693            return new Builder() {
1694                private volatile boolean getCalled;
1695                private IOException oldException;
1696
1697                private final PrivilegedExceptionAction<KeyStore> action
1698                        = new PrivilegedExceptionAction<KeyStore>() {
1699
1700                    public KeyStore run() throws Exception {
1701                        KeyStore ks;
1702                        if (provider == null) {
1703                            ks = KeyStore.getInstance(type);
1704                        } else {
1705                            ks = KeyStore.getInstance(type, provider);
1706                        }
1707                        LoadStoreParameter param = new SimpleLoadStoreParameter(protection);
1708                        if (protection instanceof CallbackHandlerProtection == false) {
1709                            ks.load(param);
1710                        } else {
1711                            // when using a CallbackHandler,
1712                            // reprompt if the password is wrong
1713                            int tries = 0;
1714                            while (true) {
1715                                tries++;
1716                                try {
1717                                    ks.load(param);
1718                                    break;
1719                                } catch (IOException e) {
1720                                    if (e.getCause() instanceof UnrecoverableKeyException) {
1721                                        if (tries < MAX_CALLBACK_TRIES) {
1722                                            continue;
1723                                        } else {
1724                                            oldException = e;
1725                                        }
1726                                    }
1727                                    throw e;
1728                                }
1729                            }
1730                        }
1731                        getCalled = true;
1732                        return ks;
1733                    }
1734                };
1735
1736                public synchronized KeyStore getKeyStore()
1737                        throws KeyStoreException {
1738                    if (oldException != null) {
1739                        throw new KeyStoreException
1740                            ("Previous KeyStore instantiation failed",
1741                             oldException);
1742                    }
1743                    try {
1744                        return AccessController.doPrivileged(action, context);
1745                    } catch (PrivilegedActionException e) {
1746                        Throwable cause = e.getCause();
1747                        throw new KeyStoreException
1748                            ("KeyStore instantiation failed", cause);
1749                    }
1750                }
1751
1752                public ProtectionParameter getProtectionParameter(String alias)
1753                {
1754                    if (alias == null) {
1755                        throw new NullPointerException();
1756                    }
1757                    if (getCalled == false) {
1758                        throw new IllegalStateException
1759                            ("getKeyStore() must be called first");
1760                    }
1761                    return protection;
1762                }
1763            };
1764        }
