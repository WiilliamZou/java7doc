026package java.security.spec;
027
028import java.security.spec.AlgorithmParameterSpec;
029
030/**
031 * This class specifies the set of parameters used with mask generation
032 * function MGF1 in OAEP Padding and RSA-PSS signature scheme, as
033 * defined in the
034 * <a href="http://www.ietf.org/rfc/rfc3447.txt">PKCS #1 v2.1</a>
035 * standard.
036 *
037 * <p>Its ASN.1 definition in PKCS#1 standard is described below:
038 * <pre>
039 * MGF1Parameters ::= OAEP-PSSDigestAlgorthms
040 * </pre>
041 * where
042 * <pre>
043 * OAEP-PSSDigestAlgorithms    ALGORITHM-IDENTIFIER ::= {
044 *   { OID id-sha1 PARAMETERS NULL   }|
045 *   { OID id-sha256 PARAMETERS NULL }|
046 *   { OID id-sha384 PARAMETERS NULL }|
047 *   { OID id-sha512 PARAMETERS NULL },
048 *   ...  -- Allows for future expansion --
049 * }
050 * </pre>
051 * @see PSSParameterSpec
052 * @see javax.crypto.spec.OAEPParameterSpec
053 *
054 * @author Valerie Peng
055 *
056 * @since 1.5
057 */
058public class MGF1ParameterSpec implements AlgorithmParameterSpec {
059
060    /**
061     * The MGF1ParameterSpec which uses "SHA-1" message digest.
062     */
063    public static final MGF1ParameterSpec SHA1 =
064        new MGF1ParameterSpec("SHA-1");
065    /**
066     * The MGF1ParameterSpec which uses "SHA-256" message digest.
067     */
068    public static final MGF1ParameterSpec SHA256 =
069        new MGF1ParameterSpec("SHA-256");
070    /**
071     * The MGF1ParameterSpec which uses "SHA-384" message digest.
072     */
073    public static final MGF1ParameterSpec SHA384 =
074        new MGF1ParameterSpec("SHA-384");
075    /**
076     * The MGF1ParameterSpec which uses SHA-512 message digest.
077     */
078    public static final MGF1ParameterSpec SHA512 =
079        new MGF1ParameterSpec("SHA-512");
080
081    private String mdName;
082
083    /**
084     * Constructs a parameter set for mask generation function MGF1
085     * as defined in the PKCS #1 standard.
086     *
087     * @param mdName the algorithm name for the message digest
088     * used in this mask generation function MGF1.
089     * @exception NullPointerException if <code>mdName</code> is null.
090     */
091    public MGF1ParameterSpec(String mdName) {
092        if (mdName == null) {
093            throw new NullPointerException("digest algorithm is null");
094        }
095        this.mdName = mdName;
096    }
097
098    /**
099     * Returns the algorithm name of the message digest used by the mask
100     * generation function.
101     *
102     * @return the algorithm name of the message digest.
103     */
104    public String getDigestAlgorithm() {
105        return mdName;
106    }
107}
