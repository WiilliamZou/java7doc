025package java.security.interfaces;
026
027import java.math.BigInteger;
028import java.security.PrivateKey;
029
030/**
031 * The interface to an elliptic curve (EC) private key.
032 *
033 * @author Valerie Peng
034 *
035 *
036 * @see PrivateKey
037 * @see ECKey
038 *
039 * @since 1.5
040 */
041public interface ECPrivateKey extends PrivateKey, ECKey {
042   /**
043    * The class fingerprint that is set to indicate
044    * serialization compatibility.
045    */
046    static final long serialVersionUID = -7896394956925609184L;
047
048    /**
049     * Returns the private value S.
050     * @return the private value S.
051     */
052    BigInteger getS();
053}
