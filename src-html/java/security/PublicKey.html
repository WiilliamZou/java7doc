026package java.security;
027
028/**
029 * <p>A public key. This interface contains no methods or constants.
030 * It merely serves to group (and provide type safety for) all public key
031 * interfaces.
032 *
033 * Note: The specialized public key interfaces extend this interface.
034 * See, for example, the DSAPublicKey interface in
035 * <code>java.security.interfaces</code>.
036 *
037 * @see Key
038 * @see PrivateKey
039 * @see Certificate
040 * @see Signature#initVerify
041 * @see java.security.interfaces.DSAPublicKey
042 * @see java.security.interfaces.RSAPublicKey
043 *
044 */
045
046public interface PublicKey extends Key {
047    // Declare serialVersionUID to be compatible with JDK1.1
048    /**
049     * The class fingerprint that is set to indicate serialization
050     * compatibility with a previous version of the class.
051     */
052    static final long serialVersionUID = 7187392471159151072L;
053}
