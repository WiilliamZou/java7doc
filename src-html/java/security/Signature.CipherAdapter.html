049/**
050 * The Signature class is used to provide applications the functionality
051 * of a digital signature algorithm. Digital signatures are used for
052 * authentication and integrity assurance of digital data.
053 *
054 * <p> The signature algorithm can be, among others, the NIST standard
055 * DSA, using DSA and SHA-1. The DSA algorithm using the
056 * SHA-1 message digest algorithm can be specified as <tt>SHA1withDSA</tt>.
057 * In the case of RSA, there are multiple choices for the message digest
058 * algorithm, so the signing algorithm could be specified as, for example,
059 * <tt>MD2withRSA</tt>, <tt>MD5withRSA</tt>, or <tt>SHA1withRSA</tt>.
060 * The algorithm name must be specified, as there is no default.
061 *
062 * <p> A Signature object can be used to generate and verify digital
063 * signatures.
064 *
065 * <p> There are three phases to the use of a Signature object for
066 * either signing data or verifying a signature:<ol>
067 *
068 * <li>Initialization, with either
069 *
070 *     <ul>
071 *
072 *     <li>a public key, which initializes the signature for
073 *     verification (see {@link #initVerify(PublicKey) initVerify}), or
074 *
075 *     <li>a private key (and optionally a Secure Random Number Generator),
076 *     which initializes the signature for signing
077 *     (see {@link #initSign(PrivateKey)}
078 *     and {@link #initSign(PrivateKey, SecureRandom)}).
079 *
080 *     </ul><p>
081 *
082 * <li>Updating<p>
083 *
084 * <p>Depending on the type of initialization, this will update the
085 * bytes to be signed or verified. See the
086 * {@link #update(byte) update} methods.<p>
087 *
088 * <li>Signing or Verifying a signature on all updated bytes. See the
089 * {@link #sign() sign} methods and the {@link #verify(byte[]) verify}
090 * method.
091 *
092 * </ol>
093 *
094 * <p>Note that this class is abstract and extends from
095 * <code>SignatureSpi</code> for historical reasons.
096 * Application developers should only take notice of the methods defined in
097 * this <code>Signature</code> class; all the methods in
098 * the superclass are intended for cryptographic service providers who wish to
099 * supply their own implementations of digital signature algorithms.
100 *
101 * <p> Every implementation of the Java platform is required to support the
102 * following standard <code>Signature</code> algorithms:
103 * <ul>
104 * <li><tt>SHA1withDSA</tt></li>
105 * <li><tt>SHA1withRSA</tt></li>
106 * <li><tt>SHA256withRSA</tt></li>
107 * </ul>
108 * These algorithms are described in the <a href=
109 * "{@docRoot}/../technotes/guides/security/StandardNames.html#Signature">
110 * Signature section</a> of the
111 * Java Cryptography Architecture Standard Algorithm Name Documentation.
112 * Consult the release documentation for your implementation to see if any
113 * other algorithms are supported.
114 *
115 * @author Benjamin Renaud
116 *
117 */
118

192    /**
193     * Returns a Signature object that implements the specified signature
194     * algorithm.
195     *
196     * <p> This method traverses the list of registered security Providers,
197     * starting with the most preferred Provider.
198     * A new Signature object encapsulating the
199     * SignatureSpi implementation from the first
200     * Provider that supports the specified algorithm is returned.
201     *
202     * <p> Note that the list of registered providers may be retrieved via
203     * the {@link Security#getProviders() Security.getProviders()} method.
204     *
205     * @param algorithm the standard name of the algorithm requested.
206     * See the Signature section in the <a href=
207     * "{@docRoot}/../technotes/guides/security/StandardNames.html#Signature">
208     * Java Cryptography Architecture Standard Algorithm Name Documentation</a>
209     * for information about standard algorithm names.
210     *
211     * @return the new Signature object.
212     *
213     * @exception NoSuchAlgorithmException if no Provider supports a
214     *          Signature implementation for the
215     *          specified algorithm.
216     *
217     * @see Provider
218     */
219    public static Signature getInstance(String algorithm)
220            throws NoSuchAlgorithmException {
221        List<Service> list;
222        if (algorithm.equalsIgnoreCase(RSA_SIGNATURE)) {
223            list = GetInstance.getServices(rsaIds);
224        } else {
225            list = GetInstance.getServices("Signature", algorithm);
226        }
227        Iterator<Service> t = list.iterator();
228        if (t.hasNext() == false) {
229            throw new NoSuchAlgorithmException
230                (algorithm + " Signature not available");
231        }
232        // try services until we find an Spi or a working Signature subclass
233        NoSuchAlgorithmException failure;
234        do {
235            Service s = t.next();
236            if (isSpi(s)) {
237                return new Delegate(s, t, algorithm);
238            } else {
239                // must be a subclass of Signature, disable dynamic selection
240                try {
241                    Instance instance =
242                        GetInstance.getInstance(s, SignatureSpi.class);
243                    return getInstance(instance, algorithm);
244                } catch (NoSuchAlgorithmException e) {
245                    failure = e;
246                }
247            }
248        } while (t.hasNext());
249        throw failure;
250    }

252    private static Signature getInstance(Instance instance, String algorithm) {
253        Signature sig;
254        if (instance.impl instanceof Signature) {
255            sig = (Signature)instance.impl;
256        } else {
257            SignatureSpi spi = (SignatureSpi)instance.impl;
258            sig = new Delegate(spi, algorithm);
259        }
260        sig.provider = instance.provider;
261        return sig;
262    }

282    private static boolean isSpi(Service s) {
283        if (s.getType().equals("Cipher")) {
284            // must be a CipherSpi, which we can wrap with the CipherAdapter
285            return true;
286        }
287        String className = s.getClassName();
288        Boolean result = signatureInfo.get(className);
289        if (result == null) {
290            try {
291                Object instance = s.newInstance(null);
292                // Signature extends SignatureSpi
293                // so it is a "real" Spi if it is an
294                // instance of SignatureSpi but not Signature
295                boolean r = (instance instanceof SignatureSpi)
296                                && (instance instanceof Signature == false);
297                if ((debug != null) && (r == false)) {
298                    debug.println("Not a SignatureSpi " + className);
299                    debug.println("Delayed provider selection may not be "
300                        + "available for algorithm " + s.getAlgorithm());
301                }
302                result = Boolean.valueOf(r);
303                signatureInfo.put(className, result);
304            } catch (Exception e) {
305                // something is wrong, assume not an SPI
306                return false;
307            }
308        }
309        return result.booleanValue();
310    }

312    /**
313     * Returns a Signature object that implements the specified signature
314     * algorithm.
315     *
316     * <p> A new Signature object encapsulating the
317     * SignatureSpi implementation from the specified provider
318     * is returned.  The specified provider must be registered
319     * in the security provider list.
320     *
321     * <p> Note that the list of registered providers may be retrieved via
322     * the {@link Security#getProviders() Security.getProviders()} method.
323     *
324     * @param algorithm the name of the algorithm requested.
325     * See the Signature section in the <a href=
326     * "{@docRoot}/../technotes/guides/security/StandardNames.html#Signature">
327     * Java Cryptography Architecture Standard Algorithm Name Documentation</a>
328     * for information about standard algorithm names.
329     *
330     * @param provider the name of the provider.
331     *
332     * @return the new Signature object.
333     *
334     * @exception NoSuchAlgorithmException if a SignatureSpi
335     *          implementation for the specified algorithm is not
336     *          available from the specified provider.
337     *
338     * @exception NoSuchProviderException if the specified provider is not
339     *          registered in the security provider list.
340     *
341     * @exception IllegalArgumentException if the provider name is null
342     *          or empty.
343     *
344     * @see Provider
345     */
346    public static Signature getInstance(String algorithm, String provider)
347            throws NoSuchAlgorithmException, NoSuchProviderException {
348        if (algorithm.equalsIgnoreCase(RSA_SIGNATURE)) {
349            // exception compatibility with existing code
350            if ((provider == null) || (provider.length() == 0)) {
351                throw new IllegalArgumentException("missing provider");
352            }
353            Provider p = Security.getProvider(provider);
354            if (p == null) {
355                throw new NoSuchProviderException
356                    ("no such provider: " + provider);
357            }
358            return getInstanceRSA(p);
359        }
360        Instance instance = GetInstance.getInstance
361                ("Signature", SignatureSpi.class, algorithm, provider);
362        return getInstance(instance, algorithm);
363    }

408    // return an implementation for NONEwithRSA, which is a special case
409    // because of the Cipher.RSA/ECB/PKCS1Padding compatibility wrapper
410    private static Signature getInstanceRSA(Provider p)
411            throws NoSuchAlgorithmException {
412        // try Signature first
413        Service s = p.getService("Signature", RSA_SIGNATURE);
414        if (s != null) {
415            Instance instance = GetInstance.getInstance(s, SignatureSpi.class);
416            return getInstance(instance, RSA_SIGNATURE);
417        }
418        // check Cipher
419        try {
420            Cipher c = Cipher.getInstance(RSA_CIPHER, p);
421            return new Delegate(new CipherAdapter(c), RSA_SIGNATURE);
422        } catch (GeneralSecurityException e) {
423            // throw Signature style exception message to avoid confusion,
424            // but append Cipher exception as cause
425            throw new NoSuchAlgorithmException("no such algorithm: "
426                + RSA_SIGNATURE + " for provider " + p.getName(), e);
427        }
428    }

465    /**
466     * Initializes this object for verification, using the public key from
467     * the given certificate.
468     * <p>If the certificate is of type X.509 and has a <i>key usage</i>
469     * extension field marked as critical, and the value of the <i>key usage</i>
470     * extension field implies that the public key in
471     * the certificate and its corresponding private key are not
472     * supposed to be used for digital signatures, an
473     * <code>InvalidKeyException</code> is thrown.
474     *
475     * @param certificate the certificate of the identity whose signature is
476     * going to be verified.
477     *
478     * @exception InvalidKeyException  if the public key in the certificate
479     * is not encoded properly or does not include required  parameter
480     * information or cannot be used for digital signature purposes.
481     * @since 1.3
482     */
483    public final void initVerify(Certificate certificate)
484            throws InvalidKeyException {
485        // If the certificate is of type X509Certificate,
486        // we should check whether it has a Key Usage
487        // extension marked as critical.
488        if (certificate instanceof java.security.cert.X509Certificate) {
489            // Check whether the cert has a key usage extension
490            // marked as a critical extension.
491            // The OID for KeyUsage extension is 2.5.29.15.
492            X509Certificate cert = (X509Certificate)certificate;
493            Set<String> critSet = cert.getCriticalExtensionOIDs();
494
495            if (critSet != null && !critSet.isEmpty()
496                && critSet.contains("2.5.29.15")) {
497                boolean[] keyUsageInfo = cert.getKeyUsage();
498                // keyUsageInfo[0] is for digitalSignature.
499                if ((keyUsageInfo != null) && (keyUsageInfo[0] == false))
500                    throw new InvalidKeyException("Wrong key usage");
501            }
502        }
503
504        PublicKey publicKey = certificate.getPublicKey();
505        engineInitVerify(publicKey);
506        state = VERIFY;
507
508        if (!skipDebug && pdebug != null) {
509            pdebug.println("Signature." + algorithm +
510                " verification algorithm from: " + this.provider.getName());
511        }
512    }

584    /**
585     * Finishes the signature operation and stores the resulting signature
586     * bytes in the provided buffer <code>outbuf</code>, starting at
587     * <code>offset</code>.
588     * The format of the signature depends on the underlying
589     * signature scheme.
590     *
591     * <p>This signature object is reset to its initial state (the state it
592     * was in after a call to one of the <code>initSign</code> methods) and
593     * can be reused to generate further signatures with the same private key.
594     *
595     * @param outbuf buffer for the signature result.
596     *
597     * @param offset offset into <code>outbuf</code> where the signature is
598     * stored.
599     *
600     * @param len number of bytes within <code>outbuf</code> allotted for the
601     * signature.
602     *
603     * @return the number of bytes placed into <code>outbuf</code>.
604     *
605     * @exception SignatureException if this signature object is not
606     * initialized properly, if this signature algorithm is unable to
607     * process the input data provided, or if <code>len</code> is less
608     * than the actual signature length.
609     *
610     * @since 1.2
611     */
612    public final int sign(byte[] outbuf, int offset, int len)
613        throws SignatureException {
614        if (outbuf == null) {
615            throw new IllegalArgumentException("No output buffer given");
616        }
617        if (offset < 0 || len < 0) {
618            throw new IllegalArgumentException("offset or len is less than 0");
619        }
620        if (outbuf.length - offset < len) {
621            throw new IllegalArgumentException
622                ("Output buffer too small for specified offset and length");
623        }
624        if (state != SIGN) {
625            throw new SignatureException("object not initialized for " +
626                                         "signing");
627        }
628        return engineSign(outbuf, offset, len);
629    }

632     * Verifies the passed-in signature.
633     *
634     * <p>A call to this method resets this signature object to the state
635     * it was in when previously initialized for verification via a
636     * call to <code>initVerify(PublicKey)</code>. That is, the object is

638     * whose public key was specified in the call to <code>initVerify</code>.
639     *
640     * @param signature the signature bytes to be verified.
641     *
642     * @return true if the signature was verified, false if not.
643     *
644     * @exception SignatureException if this signature object is not
645     * initialized properly, the passed-in signature is improperly
646     * encoded or of the wrong type, if this signature algorithm is unable to

649    public final boolean verify(byte[] signature) throws SignatureException {

651            return engineVerify(signature);

655    }

657    /**
658     * Verifies the passed-in signature in the specified array
659     * of bytes, starting at the specified offset.
660     *
661     * <p>A call to this method resets this signature object to the state
662     * it was in when previously initialized for verification via a
663     * call to <code>initVerify(PublicKey)</code>. That is, the object is
664     * reset and available to verify another signature from the identity
665     * whose public key was specified in the call to <code>initVerify</code>.
666     *
667     *
668     * @param signature the signature bytes to be verified.
669     * @param offset the offset to start from in the array of bytes.
670     * @param length the number of bytes to use, starting at offset.
671     *
672     * @return true if the signature was verified, false if not.
673     *
674     * @exception SignatureException if this signature object is not
675     * initialized properly, the passed-in signature is improperly
676     * encoded or of the wrong type, if this signature algorithm is unable to
677     * process the input data provided, etc.
678     * @exception IllegalArgumentException if the <code>signature</code>
679     * byte array is null, or the <code>offset</code> or <code>length</code>
680     * is less than 0, or the sum of the <code>offset</code> and
681     * <code>length</code> is greater than the length of the
682     * <code>signature</code> byte array.
683     * @since 1.4
684     */
685    public final boolean verify(byte[] signature, int offset, int length)
686        throws SignatureException {
687        if (state == VERIFY) {
688            if (signature == null) {
689                throw new IllegalArgumentException("signature is null");
690            }
691            if (offset < 0 || length < 0) {
692                throw new IllegalArgumentException
693                    ("offset or length is less than 0");
694            }
695            if (signature.length - offset < length) {
696                throw new IllegalArgumentException
697                    ("signature too small for specified offset and length");
698            }
699
700            return engineVerify(signature, offset, length);
701        }
702        throw new SignatureException("object not initialized for " +
703                                     "verification");
704    }

707     * Updates the data to be signed or verified by a byte.
708     *
709     * @param b the byte to use for the update.
710     *
711     * @exception SignatureException if this signature object is not

716            engineUpdate(b);
717        } else {

719                                         + "signature or verification");

736    /**
737     * Updates the data to be signed or verified, using the specified
738     * array of bytes, starting at the specified offset.
739     *
740     * @param data the array of bytes.
741     * @param off the offset to start from in the array of bytes.
742     * @param len the number of bytes to use, starting at offset.
743     *
744     * @exception SignatureException if this signature object is not
745     * initialized properly.
746     */
747    public final void update(byte[] data, int off, int len)
748            throws SignatureException {
749        if (state == SIGN || state == VERIFY) {
750            if (data == null) {
751                throw new IllegalArgumentException("data is null");
752            }
753            if (off < 0 || len < 0) {
754                throw new IllegalArgumentException("off or len is less than 0");
755            }
756            if (data.length - off < len) {
757                throw new IllegalArgumentException
758                    ("data too small for specified offset and length");
759            }
760            engineUpdate(data, off, len);
761        } else {
762            throw new SignatureException("object not initialized for "
763                                         + "signature or verification");
764        }
765    }

768     * Updates the data to be signed or verified using the specified
769     * ByteBuffer. Processes the <code>data.remaining()</code> bytes
770     * starting at at <code>data.position()</code>.
771     * Upon return, the buffer's position will be equal to its limit;
772     * its limit will not have changed.
773     *
774     * @param data the ByteBuffer
775     *
776     * @exception SignatureException if this signature object is not
777     * initialized properly.

781        if ((state != SIGN) && (state != VERIFY)) {
782            throw new SignatureException("object not initialized for "

785        if (data == null) {
786            throw new NullPointerException();

788        engineUpdate(data);

933    /*
934     * The following class allows providers to extend from SignatureSpi
935     * rather than from Signature. It represents a Signature with an
936     * encapsulated, provider-supplied SPI object (of type SignatureSpi).
937     * If the provider implementation is an instance of SignatureSpi, the
938     * getInstance() methods above return an instance of this class, with
939     * the SPI object encapsulated.
940     *
941     * Note: All SPI methods from the original Signature class have been
942     * moved up the hierarchy into a new class (SignatureSpi), which has
943     * been interposed in the hierarchy between the API (Signature)
944     * and its original parent (Object).
945     */
946
947    private static class Delegate extends Signature {
948
949        // The provider implementation (delegate)
950        // filled in once the provider is selected
951        private SignatureSpi sigSpi;
952
953        // lock for mutex during provider selection
954        private final Object lock;
955
956        // next service to try in provider selection
957        // null once provider is selected
958        private Service firstService;
959
960        // remaining services to try in provider selection
961        // null once provider is selected
962        private Iterator<Service> serviceIterator;
963
964        // constructor
965        Delegate(SignatureSpi sigSpi, String algorithm) {
966            super(algorithm);
967            this.sigSpi = sigSpi;
968            this.lock = null; // no lock needed
969        }
970
971        // used with delayed provider selection
972        Delegate(Service service,
973                        Iterator<Service> iterator, String algorithm) {
974            super(algorithm);
975            this.firstService = service;
976            this.serviceIterator = iterator;
977            this.lock = new Object();
978        }

980        /**
981         * Returns a clone if the delegate is cloneable.
982         *
983         * @return a clone if the delegate is cloneable.
984         *
985         * @exception CloneNotSupportedException if this is called on a
986         * delegate that does not support <code>Cloneable</code>.
987         */
988        public Object clone() throws CloneNotSupportedException {
989            chooseFirstProvider();
990            if (sigSpi instanceof Cloneable) {
991                SignatureSpi sigSpiClone = (SignatureSpi)sigSpi.clone();
992                // Because 'algorithm' and 'provider' are private
993                // members of our supertype, we must perform a cast to
994                // access them.
995                Signature that =
996                    new Delegate(sigSpiClone, ((Signature)this).algorithm);
997                that.provider = ((Signature)this).provider;
998                return that;
999            } else {
1000                throw new CloneNotSupportedException();
1001            }
1002        }

1004        private static SignatureSpi newInstance(Service s)
1005                throws NoSuchAlgorithmException {
1006            if (s.getType().equals("Cipher")) {
1007                // must be NONEwithRSA
1008                try {
1009                    Cipher c = Cipher.getInstance(RSA_CIPHER, s.getProvider());
1010                    return new CipherAdapter(c);
1011                } catch (NoSuchPaddingException e) {
1012                    throw new NoSuchAlgorithmException(e);
1013                }
1014            } else {
1015                Object o = s.newInstance(null);
1016                if (o instanceof SignatureSpi == false) {
1017                    throw new NoSuchAlgorithmException
1018                        ("Not a SignatureSpi: " + o.getClass().getName());
1019                }
1020                return (SignatureSpi)o;
1021            }
1022        }

1024        // max number of debug warnings to print from chooseFirstProvider()
1025        private static int warnCount = 10;
1026
1027        /**
1028         * Choose the Spi from the first provider available. Used if
1029         * delayed provider selection is not possible because initSign()/
1030         * initVerify() is not the first method called.
1031         */
1032        void chooseFirstProvider() {
1033            if (sigSpi != null) {
1034                return;
1035            }
1036            synchronized (lock) {
1037                if (sigSpi != null) {
1038                    return;
1039                }
1040                if (debug != null) {
1041                    int w = --warnCount;
1042                    if (w >= 0) {
1043                        debug.println("Signature.init() not first method "
1044                            + "called, disabling delayed provider selection");
1045                        if (w == 0) {
1046                            debug.println("Further warnings of this type will "
1047                                + "be suppressed");
1048                        }
1049                        new Exception("Call trace").printStackTrace();
1050                    }
1051                }
1052                Exception lastException = null;
1053                while ((firstService != null) || serviceIterator.hasNext()) {
1054                    Service s;
1055                    if (firstService != null) {
1056                        s = firstService;
1057                        firstService = null;
1058                    } else {
1059                        s = serviceIterator.next();
1060                    }
1061                    if (isSpi(s) == false) {
1062                        continue;
1063                    }
1064                    try {
1065                        sigSpi = newInstance(s);
1066                        provider = s.getProvider();
1067                        // not needed any more
1068                        firstService = null;
1069                        serviceIterator = null;
1070                        return;
1071                    } catch (NoSuchAlgorithmException e) {
1072                        lastException = e;
1073                    }
1074                }
1075                ProviderException e = new ProviderException
1076                        ("Could not construct SignatureSpi instance");
1077                if (lastException != null) {
1078                    e.initCause(lastException);
1079                }
1080                throw e;
1081            }
1082        }

1084        private void chooseProvider(int type, Key key, SecureRandom random)
1085                throws InvalidKeyException {
1086            synchronized (lock) {
1087                if (sigSpi != null) {
1088                    init(sigSpi, type, key, random);
1089                    return;
1090                }
1091                Exception lastException = null;
1092                while ((firstService != null) || serviceIterator.hasNext()) {
1093                    Service s;
1094                    if (firstService != null) {
1095                        s = firstService;
1096                        firstService = null;
1097                    } else {
1098                        s = serviceIterator.next();
1099                    }
1100                    // if provider says it does not support this key, ignore it
1101                    if (s.supportsParameter(key) == false) {
1102                        continue;
1103                    }
1104                    // if instance is not a SignatureSpi, ignore it
1105                    if (isSpi(s) == false) {
1106                        continue;
1107                    }
1108                    try {
1109                        SignatureSpi spi = newInstance(s);
1110                        init(spi, type, key, random);
1111                        provider = s.getProvider();
1112                        sigSpi = spi;
1113                        firstService = null;
1114                        serviceIterator = null;
1115                        return;
1116                    } catch (Exception e) {
1117                        // NoSuchAlgorithmException from newInstance()
1118                        // InvalidKeyException from init()
1119                        // RuntimeException (ProviderException) from init()
1120                        if (lastException == null) {
1121                            lastException = e;
1122                        }
1123                    }
1124                }
1125                // no working provider found, fail
1126                if (lastException instanceof InvalidKeyException) {
1127                    throw (InvalidKeyException)lastException;
1128                }
1129                if (lastException instanceof RuntimeException) {
1130                    throw (RuntimeException)lastException;
1131                }
1132                String k = (key != null) ? key.getClass().getName() : "(null)";
1133                throw new InvalidKeyException
1134                    ("No installed provider supports this key: "
1135                    + k, lastException);
1136            }
1137        }

1143        private void init(SignatureSpi spi, int type, Key  key,
1144                SecureRandom random) throws InvalidKeyException {
1145            switch (type) {
1146            case I_PUB:
1147                spi.engineInitVerify((PublicKey)key);
1148                break;
1149            case I_PRIV:
1150                spi.engineInitSign((PrivateKey)key);
1151                break;
1152            case I_PRIV_SR:
1153                spi.engineInitSign((PrivateKey)key, random);
1154                break;
1155            default:
1156                throw new AssertionError("Internal error: " + type);
1157            }
1158        }

1250    // adapter for RSA/ECB/PKCS1Padding ciphers
1251    private static class CipherAdapter extends SignatureSpi {
1252
1253        private final Cipher cipher;
1254
1255        private ByteArrayOutputStream data;
1256
1257        CipherAdapter(Cipher cipher) {
1258            this.cipher = cipher;
1259        }
1260
1261        protected void engineInitVerify(PublicKey publicKey)
1262                throws InvalidKeyException {
1263            cipher.init(Cipher.DECRYPT_MODE, publicKey);
1264            if (data == null) {
1265                data = new ByteArrayOutputStream(128);
1266            } else {
1267                data.reset();
1268            }
1269        }
1270
1271        protected void engineInitSign(PrivateKey privateKey)
1272                throws InvalidKeyException {
1273            cipher.init(Cipher.ENCRYPT_MODE, privateKey);
1274            data = null;
1275        }
1276
1277        protected void engineInitSign(PrivateKey privateKey,
1278                SecureRandom random) throws InvalidKeyException {
1279            cipher.init(Cipher.ENCRYPT_MODE, privateKey, random);
1280            data = null;
1281        }
1282
1283        protected void engineUpdate(byte b) throws SignatureException {
1284            engineUpdate(new byte[] {b}, 0, 1);
1285        }
1286
1287        protected void engineUpdate(byte[] b, int off, int len)
1288                throws SignatureException {
1289            if (data != null) {
1290                data.write(b, off, len);
1291                return;
1292            }
1293            byte[] out = cipher.update(b, off, len);
1294            if ((out != null) && (out.length != 0)) {
1295                throw new SignatureException
1296                    ("Cipher unexpectedly returned data");
1297            }
1298        }
1299
1300        protected byte[] engineSign() throws SignatureException {
1301            try {
1302                return cipher.doFinal();
1303            } catch (IllegalBlockSizeException e) {
1304                throw new SignatureException("doFinal() failed", e);
1305            } catch (BadPaddingException e) {
1306                throw new SignatureException("doFinal() failed", e);
1307            }
1308        }
1309
1310        protected boolean engineVerify(byte[] sigBytes)
1311                throws SignatureException {
1312            try {
1313                byte[] out = cipher.doFinal(sigBytes);
1314                byte[] dataBytes = data.toByteArray();
1315                data.reset();
1316                return Arrays.equals(out, dataBytes);
1317            } catch (BadPaddingException e) {
1318                // e.g. wrong public key used
1319                // return false rather than throwing exception
1320                return false;
1321            } catch (IllegalBlockSizeException e) {
1322                throw new SignatureException("doFinal() failed", e);
1323            }
1324        }
1325
1326        protected void engineSetParameter(String param, Object value)
1327                throws InvalidParameterException {
1328            throw new InvalidParameterException("Parameters not supported");
1329        }
1330
1331        protected Object engineGetParameter(String param)
1332                throws InvalidParameterException {
1333            throw new InvalidParameterException("Parameters not supported");
1334        }
