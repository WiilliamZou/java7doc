027package java.security;
028
029/**
030 * This is the exception for invalid Keys (invalid encoding, wrong
031 * length, uninitialized, etc).
032 *
033 * @author Benjamin Renaud
034 */
035
036public class InvalidKeyException extends KeyException {
037
038    private static final long serialVersionUID = 5698479920593359816L;
039
040    /**
041     * Constructs an InvalidKeyException with no detail message. A
042     * detail message is a String that describes this particular
043     * exception.
044     */
045    public InvalidKeyException() {
046        super();
047    }
048
049    /**
050     * Constructs an InvalidKeyException with the specified detail
051     * message. A detail message is a String that describes this
052     * particular exception.
053     *
054     * @param msg the detail message.
055     */
056    public InvalidKeyException(String msg) {
057        super(msg);
058    }
059
060    /**
061     * Creates a <code>InvalidKeyException</code> with the specified
062     * detail message and cause.
063     *
064     * @param message the detail message (which is saved for later retrieval
065     *        by the {@link #getMessage()} method).
066     * @param cause the cause (which is saved for later retrieval by the
067     *        {@link #getCause()} method).  (A <tt>null</tt> value is permitted,
068     *        and indicates that the cause is nonexistent or unknown.)
069     * @since 1.5
070     */
071    public InvalidKeyException(String message, Throwable cause) {
072        super(message, cause);
073    }
074
075    /**
076     * Creates a <code>InvalidKeyException</code> with the specified cause
077     * and a detail message of <tt>(cause==null ? null : cause.toString())</tt>
078     * (which typically contains the class and detail message of
079     * <tt>cause</tt>).
080     *
081     * @param cause the cause (which is saved for later retrieval by the
082     *        {@link #getCause()} method).  (A <tt>null</tt> value is permitted,
083     *        and indicates that the cause is nonexistent or unknown.)
084     * @since 1.5
085     */
086    public InvalidKeyException(Throwable cause) {
087        super(cause);
088    }
089}
