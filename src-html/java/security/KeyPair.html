026package java.security;
027
028import java.util.*;
029
030/**
031 * This class is a simple holder for a key pair (a public key and a
032 * private key). It does not enforce any security, and, when initialized,
033 * should be treated like a PrivateKey.
034 *
035 * @see PublicKey
036 * @see PrivateKey
037 *
038 * @author Benjamin Renaud
039 */
040
041public final class KeyPair implements java.io.Serializable {
042
043    private static final long serialVersionUID = -7565189502268009837L;
044
045    private PrivateKey privateKey;
046    private PublicKey publicKey;
047
048    /**
049     * Constructs a key pair from the given public key and private key.
050     *
051     * <p>Note that this constructor only stores references to the public
052     * and private key components in the generated key pair. This is safe,
053     * because <code>Key</code> objects are immutable.
054     *
055     * @param publicKey the public key.
056     *
057     * @param privateKey the private key.
058     */
059    public KeyPair(PublicKey publicKey, PrivateKey privateKey) {
060        this.publicKey = publicKey;
061        this.privateKey = privateKey;
062    }
063
064    /**
065     * Returns a reference to the public key component of this key pair.
066     *
067     * @return a reference to the public key.
068     */
069    public PublicKey getPublic() {
070        return publicKey;
071    }
072
073     /**
074     * Returns a reference to the private key component of this key pair.
075     *
076     * @return a reference to the private key.
077     */
078   public PrivateKey getPrivate() {
079        return privateKey;
080    }
081}
