026package java.security.cert;
027
028import java.security.GeneralSecurityException;
029
030/**
031 * This exception indicates one of a variety of certificate problems.
032 *
033 * @author Hemma Prafullchandra
034 * @see Certificate
035 */
036public class CertificateException extends GeneralSecurityException {
037
038    private static final long serialVersionUID = 3192535253797119798L;
039
040    /**
041     * Constructs a certificate exception with no detail message. A detail
042     * message is a String that describes this particular exception.
043     */
044    public CertificateException() {
045        super();
046    }
047
048    /**
049     * Constructs a certificate exception with the given detail
050     * message. A detail message is a String that describes this
051     * particular exception.
052     *
053     * @param msg the detail message.
054     */
055    public CertificateException(String msg) {
056        super(msg);
057    }
058
059    /**
060     * Creates a <code>CertificateException</code> with the specified
061     * detail message and cause.
062     *
063     * @param message the detail message (which is saved for later retrieval
064     *        by the {@link #getMessage()} method).
065     * @param cause the cause (which is saved for later retrieval by the
066     *        {@link #getCause()} method).  (A <tt>null</tt> value is permitted,
067     *        and indicates that the cause is nonexistent or unknown.)
068     * @since 1.5
069     */
070    public CertificateException(String message, Throwable cause) {
071        super(message, cause);
072    }
073
074    /**
075     * Creates a <code>CertificateException</code> with the specified cause
076     * and a detail message of <tt>(cause==null ? null : cause.toString())</tt>
077     * (which typically contains the class and detail message of
078     * <tt>cause</tt>).
079     *
080     * @param cause the cause (which is saved for later retrieval by the
081     *        {@link #getCause()} method).  (A <tt>null</tt> value is permitted,
082     *        and indicates that the cause is nonexistent or unknown.)
083     * @since 1.5
084     */
085    public CertificateException(Throwable cause) {
086        super(cause);
087    }
088}
