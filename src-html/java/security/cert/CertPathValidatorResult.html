026package java.security.cert;
027
028/**
029 * A specification of the result of a certification path validator algorithm.
030 * <p>
031 * The purpose of this interface is to group (and provide type safety
032 * for) all certification path validator results. All results returned
033 * by the {@link CertPathValidator#validate CertPathValidator.validate}
034 * method must implement this interface.
035 *
036 * @see CertPathValidator
037 *
038 * @since       1.4
039 * @author      Yassir Elley
040 */
041public interface CertPathValidatorResult extends Cloneable {
042
043    /**
044     * Makes a copy of this <code>CertPathValidatorResult</code>. Changes to the
045     * copy will not affect the original and vice versa.
046     *
047     * @return a copy of this <code>CertPathValidatorResult</code>
048     */
049    Object clone();
050}
