026package java.security.cert;
027
028/**
029 * A specification of <code>CertStore</code> parameters.
030 * <p>
031 * The purpose of this interface is to group (and provide type safety for)
032 * all <code>CertStore</code> parameter specifications. All
033 * <code>CertStore</code> parameter specifications must implement this
034 * interface.
035 * <p>
036 * Typically, a <code>CertStoreParameters</code> object is passed as a parameter
037 * to one of the {@link CertStore#getInstance CertStore.getInstance} methods.
038 * The <code>getInstance</code> method returns a <code>CertStore</code> that
039 * is used for retrieving <code>Certificate</code>s and <code>CRL</code>s. The
040 * <code>CertStore</code> that is returned is initialized with the specified
041 * parameters. The type of parameters needed may vary between different types
042 * of <code>CertStore</code>s.
043 *
044 * @see CertStore#getInstance
045 *
046 * @since       1.4
047 * @author      Steve Hanna
048 */
049public interface CertStoreParameters extends Cloneable {
050
051    /**
052     * Makes a copy of this <code>CertStoreParameters</code>.
053     * <p>
054     * The precise meaning of "copy" may depend on the class of
055     * the <code>CertStoreParameters</code> object. A typical implementation
056     * performs a "deep copy" of this object, but this is not an absolute
057     * requirement. Some implementations may perform a "shallow copy" of some
058     * or all of the fields of this object.
059     * <p>
060     * Note that the <code>CertStore.getInstance</code> methods make a copy
061     * of the specified <code>CertStoreParameters</code>. A deep copy
062     * implementation of <code>clone</code> is safer and more robust, as it
063     * prevents the caller from corrupting a shared <code>CertStore</code> by
064     * subsequently modifying the contents of its initialization parameters.
065     * However, a shallow copy implementation of <code>clone</code> is more
066     * appropriate for applications that need to hold a reference to a
067     * parameter contained in the <code>CertStoreParameters</code>. For example,
068     * a shallow copy clone allows an application to release the resources of
069     * a particular <code>CertStore</code> initialization parameter immediately,
070     * rather than waiting for the garbage collection mechanism. This should
071     * be done with the utmost care, since the <code>CertStore</code> may still
072     * be in use by other threads.
073     * <p>
074     * Each subclass should state the precise behavior of this method so
075     * that users and developers know what to expect.
076     *
077     * @return a copy of this <code>CertStoreParameters</code>
078     */
079    Object clone();
080}
