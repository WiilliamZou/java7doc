036/**
037 * An AccessControlContext is used to make system resource access decisions
038 * based on the context it encapsulates.
039 *
040 * <p>More specifically, it encapsulates a context and
041 * has a single method, <code>checkPermission</code>,
042 * that is equivalent to the <code>checkPermission</code> method
043 * in the AccessController class, with one difference: The AccessControlContext
044 * <code>checkPermission</code> method makes access decisions based on the
045 * context it encapsulates,
046 * rather than that of the current execution thread.
047 *
048 * <p>Thus, the purpose of AccessControlContext is for those situations where
049 * a security check that should be made within a given context
050 * actually needs to be done from within a
051 * <i>different</i> context (for example, from within a worker thread).
052 *
053 * <p> An AccessControlContext is created by calling the
054 * <code>AccessController.getContext</code> method.
055 * The <code>getContext</code> method takes a "snapshot"
056 * of the current calling context, and places
057 * it in an AccessControlContext object, which it returns. A sample call is
058 * the following:
059 *
060 * <pre>
061 *   AccessControlContext acc = AccessController.getContext()
062 * </pre>
063 *
064 * <p>
065 * Code within a different context can subsequently call the
066 * <code>checkPermission</code> method on the
067 * previously-saved AccessControlContext object. A sample call is the
068 * following:
069 *
070 * <pre>
071 *   acc.checkPermission(permission)
072 * </pre>
073 *
074 * @see AccessController
075 *
076 * @author Roland Schemers
077 */
078
079public final class AccessControlContext {

109    /**
110     * Create an AccessControlContext with the given array of ProtectionDomains.
111     * Context must not be null. Duplicate domains will be removed from the
112     * context.
113     *
114     * @param context the ProtectionDomains associated with this context.
115     * The non-duplicate domains are copied from the array. Subsequent
116     * changes to the array will not affect this AccessControlContext.
117     * @throws NullPointerException if <code>context</code> is <code>null</code>
118     */
119    public AccessControlContext(ProtectionDomain context[])
120    {
121        if (context.length == 0) {
122            this.context = null;
123        } else if (context.length == 1) {
124            if (context[0] != null) {
125                this.context = context.clone();
126            } else {
127                this.context = null;
128            }
129        } else {
130            List<ProtectionDomain> v = new ArrayList<>(context.length);
131            for (int i =0; i< context.length; i++) {
132                if ((context[i] != null) &&  (!v.contains(context[i])))
133                    v.add(context[i]);
134            }
135            if (!v.isEmpty()) {
136                this.context = new ProtectionDomain[v.size()];
137                this.context = v.toArray(this.context);
138            }
139        }
140    }

142    /**
143     * Create a new <code>AccessControlContext</code> with the given
144     * <code>AccessControlContext</code> and <code>DomainCombiner</code>.
145     * This constructor associates the provided
146     * <code>DomainCombiner</code> with the provided
147     * <code>AccessControlContext</code>.
148     *
149     * <p>
150     *
151     * @param acc the <code>AccessControlContext</code> associated
152     *          with the provided <code>DomainCombiner</code>.
153     *
154     * @param combiner the <code>DomainCombiner</code> to be associated
155     *          with the provided <code>AccessControlContext</code>.
156     *
157     * @exception NullPointerException if the provided
158     *          <code>context</code> is <code>null</code>.
159     *
160     * @exception SecurityException if a security manager is installed and the
161     *          caller does not have the "createAccessControlContext"
162     *          {@link SecurityPermission}
163     * @since 1.3
164     */
165    public AccessControlContext(AccessControlContext acc,
166                                DomainCombiner combiner) {
167
168        SecurityManager sm = System.getSecurityManager();
169        if (sm != null) {
170            sm.checkPermission(SecurityConstants.CREATE_ACC_PERMISSION);
171            this.isAuthorized = true;
172        }
173
174        this.context = acc.context;
175
176        // we do not need to run the combine method on the
177        // provided ACC.  it was already "combined" when the
178        // context was originally retrieved.
179        //
180        // at this point in time, we simply throw away the old
181        // combiner and use the newly provided one.
182        this.combiner = combiner;
183    }

274    /**
275     * Determines whether the access request indicated by the
276     * specified permission should be allowed or denied, based on
277     * the security policy currently in effect, and the context in
278     * this object. The request is allowed only if every ProtectionDomain
279     * in the context implies the permission. Otherwise the request is
280     * denied.
281     *
282     * <p>
283     * This method quietly returns if the access request
284     * is permitted, or throws a suitable AccessControlException otherwise.
285     *
286     * @param perm the requested permission.
287     *
288     * @exception AccessControlException if the specified permission
289     * is not permitted, based on the current security policy and the
290     * context encapsulated by this object.
291     * @exception NullPointerException if the permission to check for is null.
292     */
293    public void checkPermission(Permission perm)
294        throws AccessControlException
295    {
296        boolean dumpDebug = false;
297
298        if (perm == null) {
299            throw new NullPointerException("permission can't be null");
300        }
301        if (getDebug() != null) {
302            // If "codebase" is not specified, we dump the info by default.
303            dumpDebug = !Debug.isOn("codebase=");
304            if (!dumpDebug) {
305                // If "codebase" is specified, only dump if the specified code
306                // value is in the stack.
307                for (int i = 0; context != null && i < context.length; i++) {
308                    if (context[i].getCodeSource() != null &&
309                        context[i].getCodeSource().getLocation() != null &&
310                        Debug.isOn("codebase=" + context[i].getCodeSource().getLocation().toString())) {
311                        dumpDebug = true;
312                        break;
313                    }
314                }
315            }
316
317            dumpDebug &= !Debug.isOn("permission=") ||
318                Debug.isOn("permission=" + perm.getClass().getCanonicalName());
319
320            if (dumpDebug && Debug.isOn("stack")) {
321                Thread.currentThread().dumpStack();
322            }
323
324            if (dumpDebug && Debug.isOn("domain")) {
325                if (context == null) {
326                    debug.println("domain (context is null)");
327                } else {
328                    for (int i=0; i< context.length; i++) {
329                        debug.println("domain "+i+" "+context[i]);
330                    }
331                }
332            }
333        }
334
335        /*
336         * iterate through the ProtectionDomains in the context.
337         * Stop at the first one that doesn't allow the
338         * requested permission (throwing an exception).
339         *
340         */
341
342        /* if ctxt is null, all we had on the stack were system domains,
343           or the first domain was a Privileged system domain. This
344           is to make the common case for system code very fast */
345
346        if (context == null)
347            return;
348
349        for (int i=0; i< context.length; i++) {
350            if (context[i] != null &&  !context[i].implies(perm)) {
351                if (dumpDebug) {
352                    debug.println("access denied " + perm);
353                }
354
355                if (Debug.isOn("failure") && debug != null) {
356                    // Want to make sure this is always displayed for failure,
357                    // but do not want to display again if already displayed
358                    // above.
359                    if (!dumpDebug) {
360                        debug.println("access denied " + perm);
361                    }
362                    Thread.currentThread().dumpStack();
363                    final ProtectionDomain pd = context[i];
364                    final Debug db = debug;
365                    AccessController.doPrivileged (new PrivilegedAction<Void>() {
366                        public Void run() {
367                            db.println("domain that failed "+pd);
368                            return null;
369                        }
370                    });
371                }
372                throw new AccessControlException("access denied "+perm, perm);
373            }
374        }
375
376        // allow if all of them allowed access
377        if (dumpDebug) {
378            debug.println("access allowed "+perm);
379        }
380
381        return;
382    }

384    /**
385     * Take the stack-based context (this) and combine it with the
386     * privileged or inherited context, if need be.
387     */
388    AccessControlContext optimize() {
389        // the assigned (privileged or inherited) context
390        AccessControlContext acc;
391        if (isPrivileged) {
392            acc = privilegedContext;
393        } else {
394            acc = AccessController.getInheritedAccessControlContext();
395        }
396
397        // this.context could be null if only system code is on the stack;
398        // in that case, ignore the stack context
399        boolean skipStack = (context == null);
400
401        // acc.context could be null if only system code was involved;
402        // in that case, ignore the assigned context
403        boolean skipAssigned = (acc == null || acc.context == null);
404
405        if (acc != null && acc.combiner != null) {
406            // let the assigned acc's combiner do its thing
407            return goCombiner(context, acc);
408        }
409
410        // optimization: if neither have contexts; return acc if possible
411        // rather than this, because acc might have a combiner
412        if (skipAssigned && skipStack) {
413            return this;
414        }
415
416        // optimization: if there is no stack context; there is no reason
417        // to compress the assigned context, it already is compressed
418        if (skipStack) {
419            return acc;
420        }
421
422        int slen = context.length;
423
424        // optimization: if there is no assigned context and the stack length
425        // is less then or equal to two; there is no reason to compress the
426        // stack context, it already is
427        if (skipAssigned && slen <= 2) {
428            return this;
429        }
430
431        // optimization: if there is a single stack domain and that domain
432        // is already in the assigned context; no need to combine
433        if ((slen == 1) && (context[0] == acc.context[0])) {
434            return acc;
435        }
436
437        int n = (skipAssigned) ? 0 : acc.context.length;
438
439        // now we combine both of them, and create a new context
440        ProtectionDomain pd[] = new ProtectionDomain[slen + n];
441
442        // first copy in the assigned context domains, no need to compress
443        if (!skipAssigned) {
444            System.arraycopy(acc.context, 0, pd, 0, n);
445        }
446
447        // now add the stack context domains, discarding nulls and duplicates
448    outer:
449        for (int i = 0; i < context.length; i++) {
450            ProtectionDomain sd = context[i];
451            if (sd != null) {
452                for (int j = 0; j < n; j++) {
453                    if (sd == pd[j]) {
454                        continue outer;
455                    }
456                }
457                pd[n++] = sd;
458            }
459        }
460
461        // if length isn't equal, we need to shorten the array
462        if (n != pd.length) {
463            // optimization: if we didn't really combine anything
464            if (!skipAssigned && n == acc.context.length) {
465                return acc;
466            } else if (skipAssigned && n == slen) {
467                return this;
468            }
469            ProtectionDomain tmp[] = new ProtectionDomain[n];
470            System.arraycopy(pd, 0, tmp, 0, n);
471            pd = tmp;
472        }
473
474        //      return new AccessControlContext(pd, false);
475
476        // Reuse existing ACC
477
478        this.context = pd;
479        this.combiner = null;
480        this.isPrivileged = false;
481
482        return this;
483    }

485    private AccessControlContext goCombiner(ProtectionDomain[] current,
486                                            AccessControlContext assigned) {
487
488        // the assigned ACC's combiner is not null --
489        // let the combiner do its thing
490
491        // XXX we could add optimizations to 'current' here ...
492
493        if (getDebug() != null) {
494            debug.println("AccessControlContext invoking the Combiner");
495        }
496
497        // No need to clone current and assigned.context
498        // combine() will not update them
499        ProtectionDomain[] combinedPds = assigned.combiner.combine(
500            current, assigned.context);
501
502        // return new AccessControlContext(combinedPds, assigned.combiner);
503
504        // Reuse existing ACC
505        this.context = combinedPds;
506        this.combiner = assigned.combiner;
507        this.isPrivileged = false;
508        this.isAuthorized = assigned.isAuthorized;
509
510        return this;
511    }

513    /**
514     * Checks two AccessControlContext objects for equality.
515     * Checks that <i>obj</i> is
516     * an AccessControlContext and has the same set of ProtectionDomains
517     * as this context.
518     * <P>
519     * @param obj the object we are testing for equality with this object.
520     * @return true if <i>obj</i> is an AccessControlContext, and has the
521     * same set of ProtectionDomains as this context, false otherwise.
522     */
523    public boolean equals(Object obj) {
524        if (obj == this)
525            return true;
526
527        if (! (obj instanceof AccessControlContext))
528            return false;
529
530        AccessControlContext that = (AccessControlContext) obj;
531
532
533        if (context == null) {
534            return (that.context == null);
535        }
536
537        if (that.context == null)
538            return false;
539
540        if (!(this.containsAllPDs(that) && that.containsAllPDs(this)))
541            return false;
542
543        if (this.combiner == null)
544            return (that.combiner == null);
545
546        if (that.combiner == null)
547            return false;
548
549        if (!this.combiner.equals(that.combiner))
550            return false;
551
552        return true;
553    }
554
555    private boolean containsAllPDs(AccessControlContext that) {
556        boolean match = false;
557        //
558        // ProtectionDomains within an ACC currently cannot be null
559        // and this is enforced by the constructor and the various
560        // optimize methods. However, historically this logic made attempts
561        // to support the notion of a null PD and therefore this logic continues
562        // to support that notion.
563        ProtectionDomain thisPd;
564        for (int i = 0; i < context.length; i++) {
565            match = false;
566            if ((thisPd = context[i]) == null) {
567                for (int j = 0; (j < that.context.length) && !match; j++) {
568                    match = (that.context[j] == null);
569                }
570            } else {
571                Class thisPdClass = thisPd.getClass();
572                ProtectionDomain thatPd;
573                for (int j = 0; (j < that.context.length) && !match; j++) {
574                    thatPd = that.context[j];
575
576                    // Class check required to avoid PD exposure (4285406)
577                    match = (thatPd != null &&
578                        thisPdClass == thatPd.getClass() && thisPd.equals(thatPd));
579                }
580            }
581            if (!match) return false;
582        }
583        return match;
584    }
585    /**
586     * Returns the hash code value for this context. The hash code
587     * is computed by exclusive or-ing the hash code of all the protection
588     * domains in the context together.
589     *
590     * @return a hash code value for this context.
591     */
592
593    public int hashCode() {
594        int hashCode = 0;
595
596        if (context == null)
597            return hashCode;
598
599        for (int i =0; i < context.length; i++) {
600            if (context[i] != null)
601                hashCode ^= context[i].hashCode();
602        }
603        return hashCode;
604    }
