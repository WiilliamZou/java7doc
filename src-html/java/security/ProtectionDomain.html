042/**
043 *
044 *<p>
045 * This ProtectionDomain class encapsulates the characteristics of a domain,
046 * which encloses a set of classes whose instances are granted a set
047 * of permissions when being executed on behalf of a given set of Principals.
048 * <p>
049 * A static set of permissions can be bound to a ProtectionDomain when it is
050 * constructed; such permissions are granted to the domain regardless of the
051 * Policy in force. However, to support dynamic security policies, a
052 * ProtectionDomain can also be constructed such that it is dynamically
053 * mapped to a set of permissions by the current Policy whenever a permission
054 * is checked.
055 * <p>
056 *
057 * @author Li Gong
058 * @author Roland Schemers
059 * @author Gary Ellison
060 */
061
062public class ProtectionDomain {
063
064    static {
065        // Set up JavaSecurityAccess in SharedSecrets
066        SharedSecrets.setJavaSecurityAccess(
067            new JavaSecurityAccess() {
068                public <T> T doIntersectionPrivilege(
069                    PrivilegedAction<T> action,
070                    final AccessControlContext stack,
071                    final AccessControlContext context)
072                {
073                    if (action == null) {
074                        throw new NullPointerException();
075                    }
076                    return AccessController.doPrivileged(
077                        action,
078                        new AccessControlContext(
079                            stack.getContext(), context).optimize()
080                    );
081                }
082
083                public <T> T doIntersectionPrivilege(
084                    PrivilegedAction<T> action,
085                    AccessControlContext context)
086                {
087                    return doIntersectionPrivilege(action,
088                        AccessController.getContext(), context);
089                }
090            }
091       );
092    }

120    /**
121     * Creates a new ProtectionDomain with the given CodeSource and
122     * Permissions. If the permissions object is not null, then
123     *  <code>setReadOnly())</code> will be called on the passed in
124     * Permissions object. The only permissions granted to this domain
125     * are the ones specified; the current Policy will not be consulted.
126     *
127     * @param codesource the codesource associated with this domain
128     * @param permissions the permissions granted to this domain
129     */
130    public ProtectionDomain(CodeSource codesource,
131                            PermissionCollection permissions) {
132        this.codesource = codesource;
133        if (permissions != null) {
134            this.permissions = permissions;
135            this.permissions.setReadOnly();
136            if (permissions instanceof Permissions &&
137                ((Permissions)permissions).allPermission != null) {
138                hasAllPerm = true;
139            }
140        }
141        this.classloader = null;
142        this.principals = new Principal[0];
143        staticPermissions = true;
144    }
145
146    /**
147     * Creates a new ProtectionDomain qualified by the given CodeSource,
148     * Permissions, ClassLoader and array of Principals. If the
149     * permissions object is not null, then <code>setReadOnly()</code>
150     * will be called on the passed in Permissions object.
151     * The permissions granted to this domain are dynamic; they include
152     * both the static permissions passed to this constructor, and any
153     * permissions granted to this domain by the current Policy at the
154     * time a permission is checked.
155     * <p>
156     * This constructor is typically used by
157     * {@link SecureClassLoader ClassLoaders}
158     * and {@link DomainCombiner DomainCombiners} which delegate to
159     * <code>Policy</code> to actively associate the permissions granted to
160     * this domain. This constructor affords the
161     * Policy provider the opportunity to augment the supplied
162     * PermissionCollection to reflect policy changes.
163     * <p>
164     *
165     * @param codesource the CodeSource associated with this domain
166     * @param permissions the permissions granted to this domain
167     * @param classloader the ClassLoader associated with this domain
168     * @param principals the array of Principals associated with this
169     * domain. The contents of the array are copied to protect against
170     * subsequent modification.
171     * @see Policy#refresh
172     * @see Policy#getPermissions(ProtectionDomain)
173     * @since 1.4
174     */
175    public ProtectionDomain(CodeSource codesource,
176                            PermissionCollection permissions,
177                            ClassLoader classloader,
178                            Principal[] principals) {
179        this.codesource = codesource;
180        if (permissions != null) {
181            this.permissions = permissions;
182            this.permissions.setReadOnly();
183            if (permissions instanceof Permissions &&
184                ((Permissions)permissions).allPermission != null) {
185                hasAllPerm = true;
186            }
187        }
188        this.classloader = classloader;
189        this.principals = (principals != null ? principals.clone():
190                           new Principal[0]);
191        staticPermissions = false;
192    }

237    /**
238     * Check and see if this ProtectionDomain implies the permissions
239     * expressed in the Permission object.
240     * <p>
241     * The set of permissions evaluated is a function of whether the
242     * ProtectionDomain was constructed with a static set of permissions
243     * or it was bound to a dynamically mapped set of permissions.
244     * <p>
245     * If the ProtectionDomain was constructed to a
246     * {@link #ProtectionDomain(CodeSource, PermissionCollection)
247     * statically bound} PermissionCollection then the permission will
248     * only be checked against the PermissionCollection supplied at
249     * construction.
250     * <p>
251     * However, if the ProtectionDomain was constructed with
252     * the constructor variant which supports
253     * {@link #ProtectionDomain(CodeSource, PermissionCollection,
254     * ClassLoader, java.security.Principal[]) dynamically binding}
255     * permissions, then the permission will be checked against the
256     * combination of the PermissionCollection supplied at construction and
257     * the current Policy binding.
258     * <p>
259     *
260     * @param permission the Permission object to check.
261     *
262     * @return true if "permission" is implicit to this ProtectionDomain.
263     */
264    public boolean implies(Permission permission) {
265
266        if (hasAllPerm) {
267            // internal permission collection already has AllPermission -
268            // no need to go to policy
269            return true;
270        }
271
272        if (!staticPermissions &&
273            Policy.getPolicyNoCheck().implies(this, permission))
274            return true;
275        if (permissions != null)
276            return permissions.implies(permission);
277
278        return false;
279    }
280
281    // called by the VM -- do not remove
282    boolean impliesCreateAccessControlContext() {
283        return implies(SecurityConstants.CREATE_ACC_PERMISSION);
284    }

286    /**
287     * Convert a ProtectionDomain to a String.
288     */
289    @Override public String toString() {
290        String pals = "<no principals>";
291        if (principals != null && principals.length > 0) {
292            StringBuilder palBuf = new StringBuilder("(principals ");
293
294            for (int i = 0; i < principals.length; i++) {
295                palBuf.append(principals[i].getClass().getName() +
296                            " \"" + principals[i].getName() +
297                            "\"");
298                if (i < principals.length-1)
299                    palBuf.append(",\n");
300                else
301                    palBuf.append(")\n");
302            }
303            pals = palBuf.toString();
304        }
305
306        // Check if policy is set; we don't want to load
307        // the policy prematurely here
308        PermissionCollection pc = Policy.isSet() && seeAllp() ?
309                                      mergePermissions():
310                                      getPermissions();
311
312        return "ProtectionDomain "+
313            " "+codesource+"\n"+
314            " "+classloader+"\n"+
315            " "+pals+"\n"+
316            " "+pc+"\n";
317    }

319    /**
320     * Return true (merge policy permissions) in the following cases:
321     *
322     * . SecurityManager is null
323     *
324     * . SecurityManager is not null,
325     *          debug is not null,
326     *          SecurityManager impelmentation is in bootclasspath,
327     *          Policy implementation is in bootclasspath
328     *          (the bootclasspath restrictions avoid recursion)
329     *
330     * . SecurityManager is not null,
331     *          debug is null,
332     *          caller has Policy.getPolicy permission
333     */
334    private static boolean seeAllp() {
335        SecurityManager sm = System.getSecurityManager();
336
337        if (sm == null) {
338            return true;
339        } else {
340            if (debug != null) {
341                if (sm.getClass().getClassLoader() == null &&
342                    Policy.getPolicyNoCheck().getClass().getClassLoader()
343                                                                == null) {
344                    return true;
345                }
346            } else {
347                try {
348                    sm.checkPermission(SecurityConstants.GET_POLICY_PERMISSION);
349                    return true;
350                } catch (SecurityException se) {
351                    // fall thru and return false
352                }
353            }
354        }
355
356        return false;
357    }

359    private PermissionCollection mergePermissions() {
360        if (staticPermissions)
361            return permissions;
362
363        PermissionCollection perms =
364            java.security.AccessController.doPrivileged
365            (new java.security.PrivilegedAction<PermissionCollection>() {
366                    public PermissionCollection run() {
367                        Policy p = Policy.getPolicyNoCheck();
368                        return p.getPermissions(ProtectionDomain.this);
369                    }
370                });
371
372        Permissions mergedPerms = new Permissions();
373        int swag = 32;
374        int vcap = 8;
375        Enumeration<Permission> e;
376        List<Permission> pdVector = new ArrayList<>(vcap);
377        List<Permission> plVector = new ArrayList<>(swag);
378
379        //
380        // Build a vector of domain permissions for subsequent merge
381        if (permissions != null) {
382            synchronized (permissions) {
383                e = permissions.elements();
384                while (e.hasMoreElements()) {
385                    pdVector.add(e.nextElement());
386                }
387            }
388        }
389
390        //
391        // Build a vector of Policy permissions for subsequent merge
392        if (perms != null) {
393            synchronized (perms) {
394                e = perms.elements();
395                while (e.hasMoreElements()) {
396                    plVector.add(e.nextElement());
397                    vcap++;
398                }
399            }
400        }
401
402        if (perms != null && permissions != null) {
403            //
404            // Weed out the duplicates from the policy. Unless a refresh
405            // has occured since the pd was consed this should result in
406            // an empty vector.
407            synchronized (permissions) {
408                e = permissions.elements();   // domain vs policy
409                while (e.hasMoreElements()) {
410                    Permission pdp = e.nextElement();
411                    Class pdpClass = pdp.getClass();
412                    String pdpActions = pdp.getActions();
413                    String pdpName = pdp.getName();
414                    for (int i = 0; i < plVector.size(); i++) {
415                        Permission pp = plVector.get(i);
416                        if (pdpClass.isInstance(pp)) {
417                            // The equals() method on some permissions
418                            // have some side effects so this manual
419                            // comparison is sufficient.
420                            if (pdpName.equals(pp.getName()) &&
421                                pdpActions.equals(pp.getActions())) {
422                                plVector.remove(i);
423                                break;
424                            }
425                        }
426                    }
427                }
428            }
429        }
430
431        if (perms !=null) {
432            // the order of adding to merged perms and permissions
433            // needs to preserve the bugfix 4301064
434
435            for (int i = plVector.size()-1; i >= 0; i--) {
436                mergedPerms.add(plVector.get(i));
437            }
438        }
439        if (permissions != null) {
440            for (int i = pdVector.size()-1; i >= 0; i--) {
441                mergedPerms.add(pdVector.get(i));
442            }
443        }
444
445        return mergedPerms;
446    }

448    /**

453    static {
454        SharedSecrets.setJavaSecurityProtectionDomainAccess(
455            new JavaSecurityProtectionDomainAccess() {
456                public ProtectionDomainCache getProtectionDomainCache() {
457                    return new ProtectionDomainCache() {
458                        private final Map<Key, PermissionCollection> map =
459                            Collections.synchronizedMap
460                                (new WeakHashMap<Key, PermissionCollection>());
461                        public void put(ProtectionDomain pd,
462                            PermissionCollection pc) {
463                            map.put((pd == null ? null : pd.key), pc);
464                        }
465                        public PermissionCollection get(ProtectionDomain pd) {
466                            return pd == null ? map.get(null) : map.get(pd.key);
467                        }
468                    };
469                }
470            });
