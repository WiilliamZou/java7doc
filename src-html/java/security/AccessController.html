026package java.security;
027
028import sun.security.util.Debug;
029import sun.reflect.CallerSensitive;
030import sun.reflect.Reflection;

032/**
033 * <p> The AccessController class is used for access control operations
034 * and decisions.
035 *
036 * <p> More specifically, the AccessController class is used for
037 * three purposes:
038 *
039 * <ul>
040 * <li> to decide whether an access to a critical system
041 * resource is to be allowed or denied, based on the security policy
042 * currently in effect,<p>
043 * <li>to mark code as being "privileged", thus affecting subsequent
044 * access determinations, and<p>
045 * <li>to obtain a "snapshot" of the current calling context so
046 * access-control decisions from a different context can be made with
047 * respect to the saved context. </ul>
048 *
049 * <p> The {@link #checkPermission(Permission) checkPermission} method
050 * determines whether the access request indicated by a specified
051 * permission should be granted or denied. A sample call appears
052 * below. In this example, <code>checkPermission</code> will determine
053 * whether or not to grant "read" access to the file named "testFile" in
054 * the "/temp" directory.
055 *
056 * <pre>
057 *
058 * FilePermission perm = new FilePermission("/temp/testFile", "read");
059 * AccessController.checkPermission(perm);
060 *
061 * </pre>
062 *
063 * <p> If a requested access is allowed,
064 * <code>checkPermission</code> returns quietly. If denied, an
065 * AccessControlException is
066 * thrown. AccessControlException can also be thrown if the requested
067 * permission is of an incorrect type or contains an invalid value.
068 * Such information is given whenever possible.
069 *
070 * Suppose the current thread traversed m callers, in the order of caller 1
071 * to caller 2 to caller m. Then caller m invoked the
072 * <code>checkPermission</code> method.
073 * The <code>checkPermission </code>method determines whether access
074 * is granted or denied based on the following algorithm:
075 *
076 *  <pre> {@code
077 * for (int i = m; i > 0; i--) {
078 *
079 *     if (caller i's domain does not have the permission)
080 *         throw AccessControlException
081 *
082 *     else if (caller i is marked as privileged) {
083 *         if (a context was specified in the call to doPrivileged)
084 *             context.checkPermission(permission)
085 *         return;
086 *     }
087 * };
088 *
089 * // Next, check the context inherited when the thread was created.
090 * // Whenever a new thread is created, the AccessControlContext at
091 * // that time is stored and associated with the new thread, as the
092 * // "inherited" context.
093 *
094 * inheritedContext.checkPermission(permission);
095 * }</pre>
096 *
097 * <p> A caller can be marked as being "privileged"
098 * (see {@link #doPrivileged(PrivilegedAction) doPrivileged} and below).
099 * When making access control decisions, the <code>checkPermission</code>
100 * method stops checking if it reaches a caller that
101 * was marked as "privileged" via a <code>doPrivileged</code>
102 * call without a context argument (see below for information about a
103 * context argument). If that caller's domain has the
104 * specified permission, no further checking is done and
105 * <code>checkPermission</code>
106 * returns quietly, indicating that the requested access is allowed.
107 * If that domain does not have the specified permission, an exception
108 * is thrown, as usual.
109 *
110 * <p> The normal use of the "privileged" feature is as follows. If you
111 * don't need to return a value from within the "privileged" block, do
112 * the following:
113 *
114 *  <pre> {@code
115 * somemethod() {
116 *     ...normal code here...
117 *     AccessController.doPrivileged(new PrivilegedAction<Void>() {
118 *         public Void run() {
119 *             // privileged code goes here, for example:
120 *             System.loadLibrary("awt");
121 *             return null; // nothing to return
122 *         }
123 *     });
124 *     ...normal code here...
125 * }}</pre>
126 *
127 * <p>
128 * PrivilegedAction is an interface with a single method, named
129 * <code>run</code>.
130 * The above example shows creation of an implementation
131 * of that interface; a concrete implementation of the
132 * <code>run</code> method is supplied.
133 * When the call to <code>doPrivileged</code> is made, an
134 * instance of the PrivilegedAction implementation is passed
135 * to it. The <code>doPrivileged</code> method calls the
136 * <code>run</code> method from the PrivilegedAction
137 * implementation after enabling privileges, and returns the
138 * <code>run</code> method's return value as the
139 * <code>doPrivileged</code> return value (which is
140 * ignored in this example).
141 *
142 * <p> If you need to return a value, you can do something like the following:
143 *
144 *  <pre> {@code
145 * somemethod() {
146 *     ...normal code here...
147 *     String user = AccessController.doPrivileged(
148 *         new PrivilegedAction<String>() {
149 *         public String run() {
150 *             return System.getProperty("user.name");
151 *             }
152 *         });
153 *     ...normal code here...
154 * }}</pre>
155 *
156 * <p>If the action performed in your <code>run</code> method could
157 * throw a "checked" exception (those listed in the <code>throws</code> clause
158 * of a method), then you need to use the
159 * <code>PrivilegedExceptionAction</code> interface instead of the
160 * <code>PrivilegedAction</code> interface:
161 *
162 *  <pre> {@code
163 * somemethod() throws FileNotFoundException {
164 *     ...normal code here...
165 *     try {
166 *         FileInputStream fis = AccessController.doPrivileged(
167 *         new PrivilegedExceptionAction<FileInputStream>() {
168 *             public FileInputStream run() throws FileNotFoundException {
169 *                 return new FileInputStream("someFile");
170 *             }
171 *         });
172 *     } catch (PrivilegedActionException e) {
173 *         // e.getException() should be an instance of FileNotFoundException,
174 *         // as only "checked" exceptions will be "wrapped" in a
175 *         // PrivilegedActionException.
176 *         throw (FileNotFoundException) e.getException();
177 *     }
178 *     ...normal code here...
179 *  }}</pre>
180 *
181 * <p> Be *very* careful in your use of the "privileged" construct, and
182 * always remember to make the privileged code section as small as possible.
183 *
184 * <p> Note that <code>checkPermission</code> always performs security checks
185 * within the context of the currently executing thread.
186 * Sometimes a security check that should be made within a given context
187 * will actually need to be done from within a
188 * <i>different</i> context (for example, from within a worker thread).
189 * The {@link #getContext() getContext} method and
190 * AccessControlContext class are provided
191 * for this situation. The <code>getContext</code> method takes a "snapshot"
192 * of the current calling context, and places
193 * it in an AccessControlContext object, which it returns. A sample call is
194 * the following:
195 *
196 * <pre>
197 *
198 * AccessControlContext acc = AccessController.getContext()
199 *
200 * </pre>
201 *
202 * <p>
203 * AccessControlContext itself has a <code>checkPermission</code> method
204 * that makes access decisions based on the context <i>it</i> encapsulates,
205 * rather than that of the current execution thread.
206 * Code within a different context can thus call that method on the
207 * previously-saved AccessControlContext object. A sample call is the
208 * following:
209 *
210 * <pre>
211 *
212 * acc.checkPermission(permission)
213 *
214 * </pre>
215 *
216 * <p> There are also times where you don't know a priori which permissions
217 * to check the context against. In these cases you can use the
218 * doPrivileged method that takes a context:
219 *
220 *  <pre> {@code
221 * somemethod() {
222 *     AccessController.doPrivileged(new PrivilegedAction<Object>() {
223 *         public Object run() {
224 *             // Code goes here. Any permission checks within this
225 *             // run method will require that the intersection of the
226 *             // callers protection domain and the snapshot's
227 *             // context have the desired permission.
228 *         }
229 *     }, acc);
230 *     ...normal code here...
231 * }}</pre>
232 *
233 * @see AccessControlContext
234 *
235 * @author Li Gong
236 * @author Roland Schemers
237 */
238
239public final class AccessController {
240
241    /**
242     * Don't allow anyone to instantiate an AccessController
243     */

246    /**
247     * Performs the specified <code>PrivilegedAction</code> with privileges
248     * enabled. The action is performed with <i>all</i> of the permissions
249     * possessed by the caller's protection domain.
250     *
251     * <p> If the action's <code>run</code> method throws an (unchecked)
252     * exception, it will propagate through this method.
253     *
254     * <p> Note that any DomainCombiner associated with the current
255     * AccessControlContext will be ignored while the action is performed.
256     *
257     * @param action the action to be performed.
258     *
259     * @return the value returned by the action's <code>run</code> method.
260     *
261     * @exception NullPointerException if the action is <code>null</code>
262     *
263     * @see #doPrivileged(PrivilegedAction,AccessControlContext)
264     * @see #doPrivileged(PrivilegedExceptionAction)
265     * @see #doPrivilegedWithCombiner(PrivilegedAction)
266     * @see java.security.DomainCombiner
267     */
268
269    @CallerSensitive
270    public static native <T> T doPrivileged(PrivilegedAction<T> action);

272    /**
273     * Performs the specified <code>PrivilegedAction</code> with privileges
274     * enabled. The action is performed with <i>all</i> of the permissions
275     * possessed by the caller's protection domain.
276     *
277     * <p> If the action's <code>run</code> method throws an (unchecked)
278     * exception, it will propagate through this method.
279     *
280     * <p> This method preserves the current AccessControlContext's
281     * DomainCombiner (which may be null) while the action is performed.
282     *
283     * @param action the action to be performed.
284     *
285     * @return the value returned by the action's <code>run</code> method.
286     *
287     * @exception NullPointerException if the action is <code>null</code>
288     *
289     * @see #doPrivileged(PrivilegedAction)
290     * @see java.security.DomainCombiner
291     *
292     * @since 1.6
293     */
294    @CallerSensitive
295    public static <T> T doPrivilegedWithCombiner(PrivilegedAction<T> action) {
296        AccessControlContext acc = getStackAccessControlContext();
297        if (acc == null) {
298            return AccessController.doPrivileged(action);
299        }
300        DomainCombiner dc = acc.getAssignedCombiner();
301        return AccessController.doPrivileged(action, preserveCombiner(dc, Reflection.getCallerClass()));
302    }

367     * Performs the specified <code>PrivilegedExceptionAction</code> with
368     * privileges enabled.  The action is performed with <i>all</i> of the
369     * permissions possessed by the caller's protection domain.
370     *
371     * <p> If the action's <code>run</code> method throws an <i>unchecked</i>
372     * exception, it will propagate through this method.
373     *
374     * <p> This method preserves the current AccessControlContext's
375     * DomainCombiner (which may be null) while the action is performed.
376     *
377     * @param action the action to be performed.
378     *
379     * @return the value returned by the action's <code>run</code> method
380     *
381     * @exception PrivilegedActionException if the specified action's
382     *         <code>run</code> method threw a <i>checked</i> exception
383     * @exception NullPointerException if the action is <code>null</code>
384     *
385     * @see #doPrivileged(PrivilegedAction)
386     * @see #doPrivileged(PrivilegedExceptionAction,AccessControlContext)
387     * @see java.security.DomainCombiner
388     *
389     * @since 1.6
390     */
391    @CallerSensitive
392    public static <T> T doPrivilegedWithCombiner
393        (PrivilegedExceptionAction<T> action) throws PrivilegedActionException {
394
395        AccessControlContext acc = getStackAccessControlContext();
396        if (acc == null) {
397            return AccessController.doPrivileged(action);
398        }
399        DomainCombiner dc = acc.getAssignedCombiner();
400        return AccessController.doPrivileged(action, preserveCombiner(dc, Reflection.getCallerClass()));
401    }
402
403    /**
404     * preserve the combiner across the doPrivileged call
405     */
406    private static AccessControlContext preserveCombiner(DomainCombiner combiner,
407                                                         final Class<?> caller) {
408        ProtectionDomain callerPd = doPrivileged
409            (new PrivilegedAction<ProtectionDomain>() {
410            public ProtectionDomain run() {
411                return caller.getProtectionDomain();
412            }
413        });
414
415        // perform 'combine' on the caller of doPrivileged,
416        // even if the caller is from the bootclasspath
417        ProtectionDomain[] pds = new ProtectionDomain[] {callerPd};
418        if (combiner == null) {
419            return new AccessControlContext(pds);
420        } else {
421            return new AccessControlContext(combiner.combine(pds, null),
422                                            combiner);
423        }
424    }

462    /**
463     * Returns the AccessControl context. i.e., it gets
464     * the protection domains of all the callers on the stack,
465     * starting at the first class with a non-null
466     * ProtectionDomain.
467     *
468     * @return the access control context based on the current stack or
469     *         null if there was only privileged system code.
470     */

472    private static native AccessControlContext getStackAccessControlContext();
473
474    /**
475     * Returns the "inherited" AccessControl context. This is the context

482    /**
483     * This method takes a "snapshot" of the current calling context, which
484     * includes the current Thread's inherited AccessControlContext,
485     * and places it in an AccessControlContext object. This context may then
486     * be checked at a later point, possibly in another thread.
487     *
488     * @see AccessControlContext
489     *
490     * @return the AccessControlContext based on the current context.
491     */
492
493    public static AccessControlContext getContext()
494    {
495        AccessControlContext acc = getStackAccessControlContext();
496        if (acc == null) {
497            // all we had was privileged system code. We don't want
498            // to return null though, so we construct a real ACC.
499            return new AccessControlContext(null, true);
500        } else {
501            return acc.optimize();
502        }
503    }

505    /**
506     * Determines whether the access request indicated by the
507     * specified permission should be allowed or denied, based on
508     * the current AccessControlContext and security policy.
509     * This method quietly returns if the access request
510     * is permitted, or throws an AccessControlException otherwise. The
511     * getPermission method of the AccessControlException returns the
512     * <code>perm</code> Permission object instance.
513     *
514     * @param perm the requested permission.
515     *
516     * @exception AccessControlException if the specified permission
517     *            is not permitted, based on the current security policy.
518     * @exception NullPointerException if the specified permission
519     *            is <code>null</code> and is checked based on the
520     *            security policy currently in effect.
521     */
522
523    public static void checkPermission(Permission perm)
524                 throws AccessControlException
525    {
526        //System.err.println("checkPermission "+perm);
527        //Thread.currentThread().dumpStack();
528
529        if (perm == null) {
530            throw new NullPointerException("permission can't be null");
531        }
532
533        AccessControlContext stack = getStackAccessControlContext();
534        // if context is null, we had privileged system code on the stack.
535        if (stack == null) {
536            Debug debug = AccessControlContext.getDebug();
537            boolean dumpDebug = false;
538            if (debug != null) {
539                dumpDebug = !Debug.isOn("codebase=");
540                dumpDebug &= !Debug.isOn("permission=") ||
541                    Debug.isOn("permission=" + perm.getClass().getCanonicalName());
542            }
543
544            if (dumpDebug && Debug.isOn("stack")) {
545                Thread.currentThread().dumpStack();
546            }
547
548            if (dumpDebug && Debug.isOn("domain")) {
549                debug.println("domain (context is null)");
550            }
551
552            if (dumpDebug) {
553                debug.println("access allowed "+perm);
554            }
555            return;
556        }
557
558        AccessControlContext acc = stack.optimize();
559        acc.checkPermission(perm);
560    }
