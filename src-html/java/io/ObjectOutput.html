026package java.io;
027
028/**
029 * ObjectOutput extends the DataOutput interface to include writing of objects.
030 * DataOutput includes methods for output of primitive types, ObjectOutput
031 * extends that interface to include objects, arrays, and Strings.
032 *
033 * @author  unascribed
034 * @see java.io.InputStream
035 * @see java.io.ObjectOutputStream
036 * @see java.io.ObjectInputStream
037 * @since   JDK1.1
038 */
039public interface ObjectOutput extends DataOutput, AutoCloseable {
040    /**
041     * Write an object to the underlying storage or stream.  The
042     * class that implements this interface defines how the object is
043     * written.
044     *
045     * @param obj the object to be written
046     * @exception IOException Any of the usual Input/Output related exceptions.
047     */
048    public void writeObject(Object obj)
049      throws IOException;
050
051    /**
052     * Writes a byte. This method will block until the byte is actually
053     * written.
054     * @param b the byte
055     * @exception IOException If an I/O error has occurred.
056     */
057    public void write(int b) throws IOException;
058
059    /**
060     * Writes an array of bytes. This method will block until the bytes
061     * are actually written.
062     * @param b the data to be written
063     * @exception IOException If an I/O error has occurred.
064     */
065    public void write(byte b[]) throws IOException;
066
067    /**
068     * Writes a sub array of bytes.
069     * @param b the data to be written
070     * @param off       the start offset in the data
071     * @param len       the number of bytes that are written
072     * @exception IOException If an I/O error has occurred.
073     */
074    public void write(byte b[], int off, int len) throws IOException;
075
076    /**
077     * Flushes the stream. This will write any buffered
078     * output bytes.
079     * @exception IOException If an I/O error has occurred.
080     */
081    public void flush() throws IOException;
082
083    /**
084     * Closes the stream. This method must be called
085     * to release any resources associated with the
086     * stream.
087     * @exception IOException If an I/O error has occurred.
088     */
089    public void close() throws IOException;
090}
