026package java.io;
027
028/**
029 * Signals that an I/O operation has been interrupted. An
030 * <code>InterruptedIOException</code> is thrown to indicate that an
031 * input or output transfer has been terminated because the thread
032 * performing it was interrupted. The field {@link #bytesTransferred}
033 * indicates how many bytes were successfully transferred before
034 * the interruption occurred.
035 *
036 * @author  unascribed
037 * @see     java.io.InputStream
038 * @see     java.io.OutputStream
039 * @see     java.lang.Thread#interrupt()
040 * @since   JDK1.0
041 */
042public
043class InterruptedIOException extends IOException {
044    private static final long serialVersionUID = 4020568460727500567L;
045
046    /**
047     * Constructs an <code>InterruptedIOException</code> with
048     * <code>null</code> as its error detail message.
049     */
050    public InterruptedIOException() {
051        super();
052    }
053
054    /**
055     * Constructs an <code>InterruptedIOException</code> with the
056     * specified detail message. The string <code>s</code> can be
057     * retrieved later by the
058     * <code>{@link java.lang.Throwable#getMessage}</code>
059     * method of class <code>java.lang.Throwable</code>.
060     *
061     * @param   s   the detail message.
062     */
063    public InterruptedIOException(String s) {
064        super(s);
065    }
066
067    /**
068     * Reports how many bytes had been transferred as part of the I/O
069     * operation before it was interrupted.
070     *
071     * @serial
072     */
073    public int bytesTransferred = 0;
074}
