026package java.io;
027
028
029/**
030 * A filter for abstract pathnames.
031 *
032 * <p> Instances of this interface may be passed to the <code>{@link
033 * File#listFiles(java.io.FileFilter) listFiles(FileFilter)}</code> method
034 * of the <code>{@link java.io.File}</code> class.
035 *
036 * @since 1.2
037 */
038public interface FileFilter {
039
040    /**
041     * Tests whether or not the specified abstract pathname should be
042     * included in a pathname list.
043     *
044     * @param  pathname  The abstract pathname to be tested
045     * @return  <code>true</code> if and only if <code>pathname</code>
046     *          should be included
047     */
048    boolean accept(File pathname);
049
050}
