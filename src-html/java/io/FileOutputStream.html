026package java.io;
027
028import java.nio.channels.FileChannel;
029import sun.nio.ch.FileChannelImpl;
030import sun.misc.IoTrace;

033/**
034 * A file output stream is an output stream for writing data to a
035 * <code>File</code> or to a <code>FileDescriptor</code>. Whether or not
036 * a file is available or may be created depends upon the underlying
037 * platform.  Some platforms, in particular, allow a file to be opened
038 * for writing by only one <tt>FileOutputStream</tt> (or other
039 * file-writing object) at a time.  In such situations the constructors in
040 * this class will fail if the file involved is already open.
041 *
042 * <p><code>FileOutputStream</code> is meant for writing streams of raw bytes
043 * such as image data. For writing streams of characters, consider using
044 * <code>FileWriter</code>.
045 *
046 * @author  Arthur van Hoff
047 * @see     java.io.File
048 * @see     java.io.FileDescriptor
049 * @see     java.io.FileInputStream
050 * @see     java.nio.file.Files#newOutputStream
051 * @since   JDK1.0
052 */
053public
054class FileOutputStream extends OutputStream
055{
056    /**
057     * The system dependent file descriptor.
058     */
059    private final FileDescriptor fd;
060
061    /**
062     * The path of the referenced file (null if the stream is created with a file descriptor)
063     */
064    private final String path;
065
066    /**
067     * True if the file is opened for append.
068     */
069    private final boolean append;
070
071    /**
072     * The associated channel, initalized lazily.
073     */
074    private FileChannel channel;
075
076    private final Object closeLock = new Object();
077    private volatile boolean closed = false;
078    private static final ThreadLocal<Boolean> runningFinalize =
079        new ThreadLocal<>();
080
081    private static boolean isRunningFinalize() {
082        Boolean val;
083        if ((val = runningFinalize.get()) != null)
084            return val.booleanValue();
085        return false;
086    }

088    /**
089     * Creates a file output stream to write to the file with the
090     * specified name. A new <code>FileDescriptor</code> object is
091     * created to represent this file connection.
092     * <p>
093     * First, if there is a security manager, its <code>checkWrite</code>
094     * method is called with <code>name</code> as its argument.
095     * <p>
096     * If the file exists but is a directory rather than a regular file, does
097     * not exist but cannot be created, or cannot be opened for any other
098     * reason then a <code>FileNotFoundException</code> is thrown.
099     *
100     * @param      name   the system-dependent filename
101     * @exception  FileNotFoundException  if the file exists but is a directory
102     *                   rather than a regular file, does not exist but cannot
103     *                   be created, or cannot be opened for any other reason
104     * @exception  SecurityException  if a security manager exists and its
105     *               <code>checkWrite</code> method denies write access
106     *               to the file.
107     * @see        java.lang.SecurityManager#checkWrite(java.lang.String)
108     */
109    public FileOutputStream(String name) throws FileNotFoundException {
110        this(name != null ? new File(name) : null, false);
111    }

113    /**

116     * bytes will be written to the end of the file rather than the beginning.
117     * A new <code>FileDescriptor</code> object is created to represent this
118     * file connection.
119     * <p>
120     * First, if there is a security manager, its <code>checkWrite</code>
121     * method is called with <code>name</code> as its argument.
122     * <p>
123     * If the file exists but is a directory rather than a regular file, does
124     * not exist but cannot be created, or cannot be opened for any other
125     * reason then a <code>FileNotFoundException</code> is thrown.
126     *
127     * @param     name        the system-dependent file name
128     * @param     append      if <code>true</code>, then bytes will be written
129     *                   to the end of the file rather than the beginning
130     * @exception  FileNotFoundException  if the file exists but is a directory
131     *                   rather than a regular file, does not exist but cannot
132     *                   be created, or cannot be opened for any other reason.
133     * @exception  SecurityException  if a security manager exists and its
134     *               <code>checkWrite</code> method denies write access
135     *               to the file.

139    public FileOutputStream(String name, boolean append)
140        throws FileNotFoundException

142        this(name != null ? new File(name) : null, append);
143    }

145    /**
146     * Creates a file output stream to write to the file represented by
147     * the specified <code>File</code> object. A new
148     * <code>FileDescriptor</code> object is created to represent this
149     * file connection.
150     * <p>
151     * First, if there is a security manager, its <code>checkWrite</code>
152     * method is called with the path represented by the <code>file</code>
153     * argument as its argument.
154     * <p>
155     * If the file exists but is a directory rather than a regular file, does
156     * not exist but cannot be created, or cannot be opened for any other
157     * reason then a <code>FileNotFoundException</code> is thrown.
158     *
159     * @param      file               the file to be opened for writing.
160     * @exception  FileNotFoundException  if the file exists but is a directory
161     *                   rather than a regular file, does not exist but cannot
162     *                   be created, or cannot be opened for any other reason
163     * @exception  SecurityException  if a security manager exists and its
164     *               <code>checkWrite</code> method denies write access
165     *               to the file.
166     * @see        java.io.File#getPath()
167     * @see        java.lang.SecurityException
168     * @see        java.lang.SecurityManager#checkWrite(java.lang.String)
169     */
170    public FileOutputStream(File file) throws FileNotFoundException {
171        this(file, false);
172    }

174    /**
175     * Creates a file output stream to write to the file represented by
176     * the specified <code>File</code> object. If the second argument is
177     * <code>true</code>, then bytes will be written to the end of the file
178     * rather than the beginning. A new <code>FileDescriptor</code> object is
179     * created to represent this file connection.
180     * <p>
181     * First, if there is a security manager, its <code>checkWrite</code>
182     * method is called with the path represented by the <code>file</code>
183     * argument as its argument.
184     * <p>
185     * If the file exists but is a directory rather than a regular file, does
186     * not exist but cannot be created, or cannot be opened for any other
187     * reason then a <code>FileNotFoundException</code> is thrown.
188     *
189     * @param      file               the file to be opened for writing.
190     * @param     append      if <code>true</code>, then bytes will be written
191     *                   to the end of the file rather than the beginning
192     * @exception  FileNotFoundException  if the file exists but is a directory
193     *                   rather than a regular file, does not exist but cannot
194     *                   be created, or cannot be opened for any other reason
195     * @exception  SecurityException  if a security manager exists and its
196     *               <code>checkWrite</code> method denies write access
197     *               to the file.
198     * @see        java.io.File#getPath()
199     * @see        java.lang.SecurityException
200     * @see        java.lang.SecurityManager#checkWrite(java.lang.String)
201     * @since 1.4
202     */
203    public FileOutputStream(File file, boolean append)
204        throws FileNotFoundException
205    {
206        String name = (file != null ? file.getPath() : null);
207        SecurityManager security = System.getSecurityManager();
208        if (security != null) {
209            security.checkWrite(name);
210        }
211        if (name == null) {
212            throw new NullPointerException();
213        }
214        if (file.isInvalid()) {
215            throw new FileNotFoundException("Invalid file path");
216        }
217        this.fd = new FileDescriptor();
218        this.append = append;
219        this.path = name;
220        fd.incrementAndGetUseCount();
221        open(name, append);
222    }
223
224    /**
225     * Creates a file output stream to write to the specified file
226     * descriptor, which represents an existing connection to an actual
227     * file in the file system.
228     * <p>
229     * First, if there is a security manager, its <code>checkWrite</code>
230     * method is called with the file descriptor <code>fdObj</code>
231     * argument as its argument.
232     * <p>
233     * If <code>fdObj</code> is null then a <code>NullPointerException</code>
234     * is thrown.
235     * <p>
236     * This constructor does not throw an exception if <code>fdObj</code>
237     * is {@link java.io.FileDescriptor#valid() invalid}.
238     * However, if the methods are invoked on the resulting stream to attempt
239     * I/O on the stream, an <code>IOException</code> is thrown.
240     *
241     * @param      fdObj   the file descriptor to be opened for writing
242     * @exception  SecurityException  if a security manager exists and its
243     *               <code>checkWrite</code> method denies
244     *               write access to the file descriptor
245     * @see        java.lang.SecurityManager#checkWrite(java.io.FileDescriptor)
246     */
247    public FileOutputStream(FileDescriptor fdObj) {
248        SecurityManager security = System.getSecurityManager();
249        if (fdObj == null) {
250            throw new NullPointerException();
251        }
252        if (security != null) {
253            security.checkWrite(fdObj);
254        }
255        this.fd = fdObj;
256        this.path = null;
257        this.append = false;
258
259        /*
260         * FileDescriptor is being shared by streams.
261         * Ensure that it's GC'ed only when all the streams/channels are done
262         * using it.
263         */
264        fd.incrementAndGetUseCount();
265    }

270     * @param append whether the file is to be opened in append mode

275    /**
276     * Writes the specified byte to this file output stream.
277     *
278     * @param   b   the byte to be written.
279     * @param   append   {@code true} if the write operation first
280     *     advances the position to the end of file
281     */
282    private native void write(int b, boolean append) throws IOException;
283
284    /**
285     * Writes the specified byte to this file output stream. Implements
286     * the <code>write</code> method of <code>OutputStream</code>.
287     *
288     * @param      b   the byte to be written.
289     * @exception  IOException  if an I/O error occurs.
290     */
291    public void write(int b) throws IOException {
292        Object traceContext = IoTrace.fileWriteBegin(path);
293        int bytesWritten = 0;
294        try {
295            write(b, append);
296            bytesWritten = 1;
297        } finally {
298            IoTrace.fileWriteEnd(traceContext, bytesWritten);
299        }
300    }

302    /**
303     * Writes a sub array as a sequence of bytes.
304     * @param b the data to be written
305     * @param off the start offset in the data
306     * @param len the number of bytes that are written
307     * @param append {@code true} to first advance the position to the
308     *     end of file
309     * @exception IOException If an I/O error has occurred.
310     */
311    private native void writeBytes(byte b[], int off, int len, boolean append)
312        throws IOException;
313
314    /**
315     * Writes <code>b.length</code> bytes from the specified byte array
316     * to this file output stream.
317     *
318     * @param      b   the data.
319     * @exception  IOException  if an I/O error occurs.
320     */
321    public void write(byte b[]) throws IOException {
322        Object traceContext = IoTrace.fileWriteBegin(path);
323        int bytesWritten = 0;
324        try {
325            writeBytes(b, 0, b.length, append);
326            bytesWritten = b.length;
327        } finally {
328            IoTrace.fileWriteEnd(traceContext, bytesWritten);
329        }
330    }
331
332    /**
333     * Writes <code>len</code> bytes from the specified byte array
334     * starting at offset <code>off</code> to this file output stream.
335     *
336     * @param      b     the data.
337     * @param      off   the start offset in the data.
338     * @param      len   the number of bytes to write.
339     * @exception  IOException  if an I/O error occurs.
340     */
341    public void write(byte b[], int off, int len) throws IOException {
342        Object traceContext = IoTrace.fileWriteBegin(path);
343        int bytesWritten = 0;
344        try {
345            writeBytes(b, off, len, append);
346            bytesWritten = len;
347        } finally {
348            IoTrace.fileWriteEnd(traceContext, bytesWritten);
349        }
350    }

352    /**
353     * Closes this file output stream and releases any system resources
354     * associated with this stream. This file output stream may no longer
355     * be used for writing bytes.
356     *
357     * <p> If this stream has an associated channel then the channel is closed
358     * as well.
359     *
360     * @exception  IOException  if an I/O error occurs.
361     *
362     * @revised 1.4
363     * @spec JSR-51
364     */
365    public void close() throws IOException {
366        synchronized (closeLock) {
367            if (closed) {
368                return;
369            }
370            closed = true;
371        }
372
373        if (channel != null) {
374            /*
375             * Decrement FD use count associated with the channel
376             * The use count is incremented whenever a new channel
377             * is obtained from this stream.
378             */
379            fd.decrementAndGetUseCount();
380            channel.close();
381        }
382
383        /*
384         * Decrement FD use count associated with this stream
385         */
386        int useCount = fd.decrementAndGetUseCount();
387
388        /*
389         * If FileDescriptor is still in use by another stream, the finalizer
390         * will not close it.
391         */
392        if ((useCount <= 0) || !isRunningFinalize()) {
393            close0();
394        }
395    }

397    /**
398     * Returns the file descriptor associated with this stream.
399     *
400     * @return  the <code>FileDescriptor</code> object that represents
401     *          the connection to the file in the file system being used
402     *          by this <code>FileOutputStream</code> object.
403     *
404     * @exception  IOException  if an I/O error occurs.
405     * @see        java.io.FileDescriptor
406     */
407     public final FileDescriptor getFD()  throws IOException {
408        if (fd != null) return fd;
409        throw new IOException();
410     }
411
412    /**
413     * Returns the unique {@link java.nio.channels.FileChannel FileChannel}
414     * object associated with this file output stream. </p>
415     *
416     * <p> The initial {@link java.nio.channels.FileChannel#position()
417     * </code>position<code>} of the returned channel will be equal to the
418     * number of bytes written to the file so far unless this stream is in
419     * append mode, in which case it will be equal to the size of the file.
420     * Writing bytes to this stream will increment the channel's position
421     * accordingly.  Changing the channel's position, either explicitly or by
422     * writing, will change this stream's file position.
423     *
424     * @return  the file channel associated with this file output stream
425     *
426     * @since 1.4
427     * @spec JSR-51
428     */
429    public FileChannel getChannel() {
430        synchronized (this) {
431            if (channel == null) {
432                channel = FileChannelImpl.open(fd, path, false, true, append, this);
433
434                /*
435                 * Increment fd's use count. Invoking the channel's close()
436                 * method will result in decrementing the use count set for
437                 * the channel.
438                 */
439                fd.incrementAndGetUseCount();
440            }
441            return channel;
442        }
443    }

445    /**
446     * Cleans up the connection to the file, and ensures that the
447     * <code>close</code> method of this file output stream is
448     * called when there are no more references to this stream.
449     *
450     * @exception  IOException  if an I/O error occurs.
451     * @see        java.io.FileInputStream#close()
452     */
453    protected void finalize() throws IOException {
454        if (fd != null) {
455            if (fd == FileDescriptor.out || fd == FileDescriptor.err) {
456                flush();
457            } else {
458
459                /*
460                 * Finalizer should not release the FileDescriptor if another
461                 * stream is still using it. If the user directly invokes
462                 * close() then the FileDescriptor is also released.
463                 */
464                runningFinalize.set(Boolean.TRUE);
465                try {
466                    close();
467                } finally {
468                    runningFinalize.set(Boolean.FALSE);
469                }
470            }
471        }
472    }

474    private native void close0() throws IOException;
