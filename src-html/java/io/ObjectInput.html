026package java.io;
027
028/**
029 * ObjectInput extends the DataInput interface to include the reading of
030 * objects. DataInput includes methods for the input of primitive types,
031 * ObjectInput extends that interface to include objects, arrays, and Strings.
032 *
033 * @author  unascribed
034 * @see java.io.InputStream
035 * @see java.io.ObjectOutputStream
036 * @see java.io.ObjectInputStream
037 * @since   JDK1.1
038 */
039public interface ObjectInput extends DataInput, AutoCloseable {
040    /**
041     * Read and return an object. The class that implements this interface
042     * defines where the object is "read" from.
043     *
044     * @return the object read from the stream
045     * @exception java.lang.ClassNotFoundException If the class of a serialized
046     *      object cannot be found.
047     * @exception IOException If any of the usual Input/Output
048     * related exceptions occur.
049     */
050    public Object readObject()
051        throws ClassNotFoundException, IOException;
052
053    /**
054     * Reads a byte of data. This method will block if no input is
055     * available.
056     * @return  the byte read, or -1 if the end of the
057     *          stream is reached.
058     * @exception IOException If an I/O error has occurred.
059     */
060    public int read() throws IOException;
061
062    /**
063     * Reads into an array of bytes.  This method will
064     * block until some input is available.
065     * @param b the buffer into which the data is read
066     * @return  the actual number of bytes read, -1 is
067     *          returned when the end of the stream is reached.
068     * @exception IOException If an I/O error has occurred.
069     */
070    public int read(byte b[]) throws IOException;
071
072    /**
073     * Reads into an array of bytes.  This method will
074     * block until some input is available.
075     * @param b the buffer into which the data is read
076     * @param off the start offset of the data
077     * @param len the maximum number of bytes read
078     * @return  the actual number of bytes read, -1 is
079     *          returned when the end of the stream is reached.
080     * @exception IOException If an I/O error has occurred.
081     */
082    public int read(byte b[], int off, int len) throws IOException;
083
084    /**
085     * Skips n bytes of input.
086     * @param n the number of bytes to be skipped
087     * @return  the actual number of bytes skipped.
088     * @exception IOException If an I/O error has occurred.
089     */
090    public long skip(long n) throws IOException;
091
092    /**
093     * Returns the number of bytes that can be read
094     * without blocking.
095     * @return the number of available bytes.
096     * @exception IOException If an I/O error has occurred.
097     */
098    public int available() throws IOException;
099
100    /**
101     * Closes the input stream. Must be called
102     * to release any resources associated with
103     * the stream.
104     * @exception IOException If an I/O error has occurred.
105     */
106    public void close() throws IOException;
107}
