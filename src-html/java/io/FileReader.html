026package java.io;
027
028
029/**
030 * Convenience class for reading character files.  The constructors of this
031 * class assume that the default character encoding and the default byte-buffer
032 * size are appropriate.  To specify these values yourself, construct an
033 * InputStreamReader on a FileInputStream.
034 *
035 * <p><code>FileReader</code> is meant for reading streams of characters.
036 * For reading streams of raw bytes, consider using a
037 * <code>FileInputStream</code>.
038 *
039 * @see InputStreamReader
040 * @see FileInputStream
041 *
042 * @author      Mark Reinhold
043 * @since       JDK1.1
044 */
045public class FileReader extends InputStreamReader {
046
047   /**
048    * Creates a new <tt>FileReader</tt>, given the name of the
049    * file to read from.
050    *
051    * @param fileName the name of the file to read from
052    * @exception  FileNotFoundException  if the named file does not exist,
053    *                   is a directory rather than a regular file,
054    *                   or for some other reason cannot be opened for
055    *                   reading.
056    */
057    public FileReader(String fileName) throws FileNotFoundException {
058        super(new FileInputStream(fileName));
059    }
060
061   /**
062    * Creates a new <tt>FileReader</tt>, given the <tt>File</tt>
063    * to read from.
064    *
065    * @param file the <tt>File</tt> to read from
066    * @exception  FileNotFoundException  if the file does not exist,
067    *                   is a directory rather than a regular file,
068    *                   or for some other reason cannot be opened for
069    *                   reading.
070    */
071    public FileReader(File file) throws FileNotFoundException {
072        super(new FileInputStream(file));
073    }
074
075   /**
076    * Creates a new <tt>FileReader</tt>, given the
077    * <tt>FileDescriptor</tt> to read from.
078    *
079    * @param fd the FileDescriptor to read from
080    */
081    public FileReader(FileDescriptor fd) {
082        super(new FileInputStream(fd));
083    }
084
085}
