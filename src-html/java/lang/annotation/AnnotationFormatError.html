026package java.lang.annotation;
027
028/**
029 * Thrown when the annotation parser attempts to read an annotation
030 * from a class file and determines that the annotation is malformed.
031 * This error can be thrown by the {@linkplain
032 * java.lang.reflect.AnnotatedElement API used to read annotations
033 * reflectively}.
034 *
035 * @author  Josh Bloch
036 * @see     java.lang.reflect.AnnotatedElement
037 * @since   1.5
038 */
039public class AnnotationFormatError extends Error {
040    private static final long serialVersionUID = -4256701562333669892L;
041
042    /**
043     * Constructs a new <tt>AnnotationFormatError</tt> with the specified
044     * detail message.
045     *
046     * @param   message   the detail message.
047     */
048    public AnnotationFormatError(String message) {
049        super(message);
050    }
051
052    /**
053     * Constructs a new <tt>AnnotationFormatError</tt> with the specified
054     * detail message and cause.  Note that the detail message associated
055     * with <code>cause</code> is <i>not</i> automatically incorporated in
056     * this error's detail message.
057     *
058     * @param  message the detail message
059     * @param  cause the cause (A <tt>null</tt> value is permitted, and
060     *     indicates that the cause is nonexistent or unknown.)
061     */
062    public AnnotationFormatError(String message, Throwable cause) {
063        super(message, cause);
064    }
065
066
067    /**
068     * Constructs a new <tt>AnnotationFormatError</tt> with the specified
069     * cause and a detail message of
070     * <tt>(cause == null ? null : cause.toString())</tt> (which
071     * typically contains the class and detail message of <tt>cause</tt>).
072     *
073     * @param  cause the cause (A <tt>null</tt> value is permitted, and
074     *     indicates that the cause is nonexistent or unknown.)
075     */
076    public AnnotationFormatError(Throwable cause) {
077        super(cause);
078    }
079}
