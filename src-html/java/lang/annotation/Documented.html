026package java.lang.annotation;
027
028/**
029 * Indicates that annotations with a type are to be documented by javadoc
030 * and similar tools by default.  This type should be used to annotate the
031 * declarations of types whose annotations affect the use of annotated
032 * elements by their clients.  If a type declaration is annotated with
033 * Documented, its annotations become part of the public API
034 * of the annotated elements.
035 *
036 * @author  Joshua Bloch
037 * @since 1.5
038 */
039@Documented
040@Retention(RetentionPolicy.RUNTIME)
041@Target(ElementType.ANNOTATION_TYPE)
042public @interface Documented {
043}
