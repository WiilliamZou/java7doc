026package java.lang.annotation;
027
028/**
029 * Indicates that an annotation type is automatically inherited.  If
030 * an Inherited meta-annotation is present on an annotation type
031 * declaration, and the user queries the annotation type on a class
032 * declaration, and the class declaration has no annotation for this type,
033 * then the class's superclass will automatically be queried for the
034 * annotation type.  This process will be repeated until an annotation for this
035 * type is found, or the top of the class hierarchy (Object)
036 * is reached.  If no superclass has an annotation for this type, then
037 * the query will indicate that the class in question has no such annotation.
038 *
039 * <p>Note that this meta-annotation type has no effect if the annotated
040 * type is used to annotate anything other than a class.  Note also
041 * that this meta-annotation only causes annotations to be inherited
042 * from superclasses; annotations on implemented interfaces have no
043 * effect.
044 *
045 * @author  Joshua Bloch
046 * @since 1.5
047 */
048@Documented
049@Retention(RetentionPolicy.RUNTIME)
050@Target(ElementType.ANNOTATION_TYPE)
051public @interface Inherited {
052}
