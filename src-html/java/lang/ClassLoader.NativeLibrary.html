064/**
065 * A class loader is an object that is responsible for loading classes. The
066 * class <tt>ClassLoader</tt> is an abstract class.  Given the <a
067 * href="#name">binary name</a> of a class, a class loader should attempt to
068 * locate or generate data that constitutes a definition for the class.  A
069 * typical strategy is to transform the name into a file name and then read a
070 * "class file" of that name from a file system.
071 *
072 * <p> Every {@link Class <tt>Class</tt>} object contains a {@link
073 * Class#getClassLoader() reference} to the <tt>ClassLoader</tt> that defined
074 * it.
075 *
076 * <p> <tt>Class</tt> objects for array classes are not created by class
077 * loaders, but are created automatically as required by the Java runtime.
078 * The class loader for an array class, as returned by {@link
079 * Class#getClassLoader()} is the same as the class loader for its element
080 * type; if the element type is a primitive type, then the array class has no
081 * class loader.
082 *
083 * <p> Applications implement subclasses of <tt>ClassLoader</tt> in order to
084 * extend the manner in which the Java virtual machine dynamically loads
085 * classes.
086 *
087 * <p> Class loaders may typically be used by security managers to indicate
088 * security domains.
089 *
090 * <p> The <tt>ClassLoader</tt> class uses a delegation model to search for
091 * classes and resources.  Each instance of <tt>ClassLoader</tt> has an
092 * associated parent class loader.  When requested to find a class or
093 * resource, a <tt>ClassLoader</tt> instance will delegate the search for the
094 * class or resource to its parent class loader before attempting to find the
095 * class or resource itself.  The virtual machine's built-in class loader,
096 * called the "bootstrap class loader", does not itself have a parent but may
097 * serve as the parent of a <tt>ClassLoader</tt> instance.
098 *
099 * <p> Class loaders that support concurrent loading of classes are known as
100 * <em>parallel capable</em> class loaders and are required to register
101 * themselves at their class initialization time by invoking the
102 * {@link
103 * #registerAsParallelCapable <tt>ClassLoader.registerAsParallelCapable</tt>}
104 * method. Note that the <tt>ClassLoader</tt> class is registered as parallel
105 * capable by default. However, its subclasses still need to register themselves
106 * if they are parallel capable. <br>
107 * In environments in which the delegation model is not strictly
108 * hierarchical, class loaders need to be parallel capable, otherwise class
109 * loading can lead to deadlocks because the loader lock is held for the
110 * duration of the class loading process (see {@link #loadClass
111 * <tt>loadClass</tt>} methods).
112 *
113 * <p> Normally, the Java virtual machine loads classes from the local file
114 * system in a platform-dependent manner.  For example, on UNIX systems, the
115 * virtual machine loads classes from the directory defined by the
116 * <tt>CLASSPATH</tt> environment variable.
117 *
118 * <p> However, some classes may not originate from a file; they may originate
119 * from other sources, such as the network, or they could be constructed by an
120 * application.  The method {@link #defineClass(String, byte[], int, int)
121 * <tt>defineClass</tt>} converts an array of bytes into an instance of class
122 * <tt>Class</tt>. Instances of this newly defined class can be created using
123 * {@link Class#newInstance <tt>Class.newInstance</tt>}.
124 *
125 * <p> The methods and constructors of objects created by a class loader may
126 * reference other classes.  To determine the class(es) referred to, the Java
127 * virtual machine invokes the {@link #loadClass <tt>loadClass</tt>} method of
128 * the class loader that originally created the class.
129 *
130 * <p> For example, an application could create a network class loader to
131 * download class files from a server.  Sample code might look like:
132 *
133 * <blockquote><pre>
134 *   ClassLoader loader&nbsp;= new NetworkClassLoader(host,&nbsp;port);
135 *   Object main&nbsp;= loader.loadClass("Main", true).newInstance();
136 *       &nbsp;.&nbsp;.&nbsp;.
137 * </pre></blockquote>
138 *
139 * <p> The network class loader subclass must define the methods {@link
140 * #findClass <tt>findClass</tt>} and <tt>loadClassData</tt> to load a class
141 * from the network.  Once it has downloaded the bytes that make up the class,
142 * it should use the method {@link #defineClass <tt>defineClass</tt>} to
143 * create a class instance.  A sample implementation is:
144 *
145 * <blockquote><pre>
146 *     class NetworkClassLoader extends ClassLoader {
147 *         String host;
148 *         int port;
149 *
150 *         public Class findClass(String name) {
151 *             byte[] b = loadClassData(name);
152 *             return defineClass(name, b, 0, b.length);
153 *         }
154 *
155 *         private byte[] loadClassData(String name) {
156 *             // load the class data from the connection
157 *             &nbsp;.&nbsp;.&nbsp;.
158 *         }
159 *     }
160 * </pre></blockquote>
161 *
162 * <h4> <a name="name">Binary names</a> </h4>
163 *
164 * <p> Any class name provided as a {@link String} parameter to methods in
165 * <tt>ClassLoader</tt> must be a binary name as defined by
166 * <cite>The Java&trade; Language Specification</cite>.
167 *
168 * <p> Examples of valid class names include:
169 * <blockquote><pre>
170 *   "java.lang.String"
171 *   "javax.swing.JSpinner$DefaultEditor"
172 *   "java.security.KeyStore$Builder$FileBuilder$1"
173 *   "java.net.URLClassLoader$3$1"
174 * </pre></blockquote>
175 *
176 * @see      #resolveClass(Class)
177 * @since 1.0
178 */

191    /**
192     * Encapsulates the set of parallel capable loader types.
193     */
194    private static class ParallelLoaders {
195        private ParallelLoaders() {}
196
197        // the set of parallel capable loader types
198        private static final Set<Class<? extends ClassLoader>> loaderTypes =
199            Collections.newSetFromMap(
200                new WeakHashMap<Class<? extends ClassLoader>, Boolean>());
201        static {
202            synchronized (loaderTypes) { loaderTypes.add(ClassLoader.class); }
203        }
204
205        /**
206         * Registers the given class loader type as parallel capabale.
207         * Returns {@code true} is successfully registered; {@code false} if
208         * loader's super class is not registered.
209         */
210        static boolean register(Class<? extends ClassLoader> c) {
211            synchronized (loaderTypes) {
212                if (loaderTypes.contains(c.getSuperclass())) {
213                    // register the class loader as parallel capable
214                    // if and only if all of its super classes are.
215                    // Note: given current classloading sequence, if
216                    // the immediate super class is parallel capable,
217                    // all the super classes higher up must be too.
218                    loaderTypes.add(c);
219                    return true;
220                } else {
221                    return false;
222                }
223            }
224        }
225
226        /**
227         * Returns {@code true} if the given class loader type is
228         * registered as parallel capable.
229         */
230        static boolean isRegistered(Class<? extends ClassLoader> c) {
231            synchronized (loaderTypes) {
232                return loaderTypes.contains(c);
233            }
234        }
235    }

361    /**
362     * Loads the class with the specified <a href="#name">binary name</a>.  The
363     * default implementation of this method searches for classes in the
364     * following order:
365     *
366     * <p><ol>
367     *
368     *   <li><p> Invoke {@link #findLoadedClass(String)} to check if the class
369     *   has already been loaded.  </p></li>
370     *
371     *   <li><p> Invoke the {@link #loadClass(String) <tt>loadClass</tt>} method
372     *   on the parent class loader.  If the parent is <tt>null</tt> the class
373     *   loader built-in to the virtual machine is used, instead.  </p></li>
374     *
375     *   <li><p> Invoke the {@link #findClass(String)} method to find the
376     *   class.  </p></li>
377     *
378     * </ol>
379     *
380     * <p> If the class was found using the above steps, and the
381     * <tt>resolve</tt> flag is true, this method will then invoke the {@link
382     * #resolveClass(Class)} method on the resulting <tt>Class</tt> object.
383     *
384     * <p> Subclasses of <tt>ClassLoader</tt> are encouraged to override {@link
385     * #findClass(String)}, rather than this method.  </p>
386     *
387     * <p> Unless overridden, this method synchronizes on the result of
388     * {@link #getClassLoadingLock <tt>getClassLoadingLock</tt>} method
389     * during the entire class loading process.
390     *
391     * @param  name
392     *         The <a href="#name">binary name</a> of the class
393     *
394     * @param  resolve
395     *         If <tt>true</tt> then resolve the class
396     *
397     * @return  The resulting <tt>Class</tt> object
398     *
399     * @throws  ClassNotFoundException
400     *          If the class could not be found
401     */
402    protected Class<?> loadClass(String name, boolean resolve)
403        throws ClassNotFoundException
404    {
405        synchronized (getClassLoadingLock(name)) {
406            // First, check if the class has already been loaded
407            Class c = findLoadedClass(name);
408            if (c == null) {
409                long t0 = System.nanoTime();
410                try {
411                    if (parent != null) {
412                        c = parent.loadClass(name, false);
413                    } else {
414                        c = findBootstrapClassOrNull(name);
415                    }
416                } catch (ClassNotFoundException e) {
417                    // ClassNotFoundException thrown if class not found
418                    // from the non-null parent class loader
419                }
420
421                if (c == null) {
422                    // If still not found, then invoke findClass in order
423                    // to find the class.
424                    long t1 = System.nanoTime();
425                    c = findClass(name);
426
427                    // this is the defining class loader; record the stats
428                    sun.misc.PerfCounter.getParentDelegationTime().addTime(t1 - t0);
429                    sun.misc.PerfCounter.getFindClassTime().addElapsedTimeFrom(t1);
430                    sun.misc.PerfCounter.getFindClasses().increment();
431                }
432            }
433            if (resolve) {
434                resolveClass(c);
435            }
436            return c;
437        }
438    }

442     * For backward compatibility, the default implementation of this method
443     * behaves as follows. If this ClassLoader object is registered as
444     * parallel capable, the method returns a dedicated object associated
445     * with the specified class name. Otherwise, the method returns this
446     * ClassLoader object. </p>

448     * @param  className
449     *         The name of the to-be-loaded class

460    protected Object getClassLoadingLock(String className) {
461        Object lock = this;
462        if (parallelLockMap != null) {
463            Object newLock = new Object();
464            lock = parallelLockMap.putIfAbsent(className, newLock);
465            if (lock == null) {
466                lock = newLock;
467            }
468        }
469        return lock;
470    }

477        // the current class loader is not parallel capable.
478        if (parallelLockMap == null) {

481            }
482        } else {

487    // Invoked by the VM after loading class with this loader.
488    private void checkPackageAccess(Class cls, ProtectionDomain pd) {
489        final SecurityManager sm = System.getSecurityManager();
490        if (sm != null) {
491            if (ReflectUtil.isNonPublicProxyClass(cls)) {
492                for (Class intf: cls.getInterfaces()) {
493                    checkPackageAccess(intf, pd);
494                }
495                return;
496            }
497
498            final String name = cls.getName();
499            final int i = name.lastIndexOf('.');
500            if (i != -1) {
501                AccessController.doPrivileged(new PrivilegedAction<Void>() {
502                    public Void run() {
503                        sm.checkPackageAccess(name.substring(0, i));
504                        return null;
505                    }
506                }, new AccessControlContext(new ProtectionDomain[] {pd}));
507            }
508        }
509        domains.add(pd);
510    }

646    /* Determine protection domain, and check that:
647        - not define java.* class,
648        - signer of this class matches signers for the rest of the classes in
649          package.
650    */
651    private ProtectionDomain preDefineClass(String name,
652                                            ProtectionDomain pd)
653    {
654        if (!checkName(name))
655            throw new NoClassDefFoundError("IllegalName: " + name);
656
657        if ((name != null) && name.startsWith("java.")) {
658            throw new SecurityException
659                ("Prohibited package name: " +
660                 name.substring(0, name.lastIndexOf('.')));
661        }
662        if (pd == null) {
663            pd = defaultDomain;
664        }
665
666        if (name != null) checkCerts(name, pd.getCodeSource());
667
668        return pd;
669    }

671    private String defineClassSourceLocation(ProtectionDomain pd)
672    {

931    private void checkCerts(String name, CodeSource cs) {
932        int i = name.lastIndexOf('.');
933        String pname = (i == -1) ? "" : name.substring(0, i);
934
935        Certificate[] certs = null;
936        if (cs != null) {
937            certs = cs.getCertificates();
938        }
939        Certificate[] pcerts = null;
940        if (parallelLockMap == null) {
941            synchronized (this) {
942                pcerts = package2certs.get(pname);
943                if (pcerts == null) {
944                    package2certs.put(pname, (certs == null? nocerts:certs));
945                }
946            }
947        } else {
948            pcerts = ((ConcurrentHashMap<String, Certificate[]>)package2certs).
949                putIfAbsent(pname, (certs == null? nocerts:certs));
950        }
951        if (pcerts != null && !compareCerts(pcerts, certs)) {

957    /**
958     * check to make sure the certs for the new class (certs) are the same as
959     * the certs for the first class inserted in the package (pcerts)
960     */
961    private boolean compareCerts(Certificate[] pcerts,
962                                 Certificate[] certs)
963    {
964        // certs can be null, indicating no certs.
965        if ((certs == null) || (certs.length == 0)) {
966            return pcerts.length == 0;
967        }
968
969        // the length must be the same at this point
970        if (certs.length != pcerts.length)
971            return false;
972
973        // go through and make sure all the certs in one array
974        // are in the other and vice-versa.
975        boolean match;
976        for (int i = 0; i < certs.length; i++) {
977            match = false;
978            for (int j = 0; j < pcerts.length; j++) {
979                if (certs[i].equals(pcerts[j])) {
980                    match = true;
981                    break;
982                }
983            }
984            if (!match) return false;
985        }
986
987        // now do the same for pcerts
988        for (int i = 0; i < pcerts.length; i++) {
989            match = false;
990            for (int j = 0; j < certs.length; j++) {
991                if (pcerts[i].equals(certs[j])) {
992                    match = true;
993                    break;
994                }
995            }
996            if (!match) return false;
997        }
998
999        return true;
1000    }

1730     * The inner class NativeLibrary denotes a loaded native library instance.
1731     * Every classloader contains a vector of loaded native libraries in the
1732     * private field <tt>nativeLibraries</tt>.  The native libraries loaded
1733     * into the system are entered into the <tt>systemNativeLibraries</tt>
1734     * vector.

1736     * <p> Every native library requires a particular version of JNI. This is
1737     * denoted by the private <tt>jniVersion</tt> field.  This field is set by
1738     * the VM when it loads the library, and used by the VM to pass the correct
1739     * version of JNI to the native methods.  </p>

1741     * @see      ClassLoader
1742     * @since    1.2
1743     */
1744    static class NativeLibrary {
1745        // opaque handle to native library, used in native code.
1746        long handle;
1747        // the version of JNI environment the native library requires.
1748        private int jniVersion;
1749        // the class from which the library is loaded, also indicates
1750        // the loader this native library belongs.
1751        private Class fromClass;
1752        // the canonicalized name of the native library.
1753        String name;
1754
1755        native void load(String name);
1756        native long find(String name);
1757        native void unload();

1765            synchronized (loadedLibraryNames) {
1766                if (fromClass.getClassLoader() != null && handle != 0) {
1767                    /* remove the native library name */
1768                    int size = loadedLibraryNames.size();
1769                    for (int i = 0; i < size; i++) {
1770                        if (name.equals(loadedLibraryNames.elementAt(i))) {
1771                            loadedLibraryNames.removeElementAt(i);
1772                            break;
1773                        }
1774                    }
1775                    /* unload the library. */
1776                    ClassLoader.nativeLibraryContext.push(this);
1777                    try {
1778                        unload();
1779                    } finally {
1780                        ClassLoader.nativeLibraryContext.pop();
1781                    }
1782                }
1783            }
1784        }

1786        // JNI_Load/JNI_Unload
1787        static Class getFromClass() {
1788            return ClassLoader.nativeLibraryContext.peek().fromClass;

1809    private static String[] initializePath(String propname) {
1810        String ldpath = System.getProperty(propname, "");
1811        String ps = File.pathSeparator;
1812        int ldlen = ldpath.length();
1813        int i, j, n;
1814        // Count the separators in the path
1815        i = ldpath.indexOf(ps);
1816        n = 0;
1817        while (i >= 0) {
1818            n++;
1819            i = ldpath.indexOf(ps, i + 1);
1820        }
1821
1822        // allocate the array of paths - n :'s = n + 1 path elements
1823        String[] paths = new String[n + 1];
1824
1825        // Fill the array with paths from the ldpath
1826        n = i = 0;
1827        j = ldpath.indexOf(ps);
1828        while (j >= 0) {
1829            if (j - i > 0) {
1830                paths[n++] = ldpath.substring(i, j);
1831            } else if (j - i == 0) {
1832                paths[n++] = ".";
1833            }
1834            i = j + 1;
1835            j = ldpath.indexOf(ps, i);
1836        }
1837        paths[n] = ldpath.substring(i, ldlen);
1838        return paths;
1839    }

1841    // Invoked in the java.lang.Runtime class to implement load and loadLibrary.
1842    static void loadLibrary(Class fromClass, String name,
1843                            boolean isAbsolute) {
1844        ClassLoader loader =
1845            (fromClass == null) ? null : fromClass.getClassLoader();
1846        if (sys_paths == null) {
1847            usr_paths = initializePath("java.library.path");
1848            sys_paths = initializePath("sun.boot.library.path");
1849        }
1850        if (isAbsolute) {
1851            if (loadLibrary0(fromClass, new File(name))) {
1852                return;
1853            }
1854            throw new UnsatisfiedLinkError("Can't load library: " + name);
1855        }
1856        if (loader != null) {
1857            String libfilename = loader.findLibrary(name);
1858            if (libfilename != null) {
1859                File libfile = new File(libfilename);
1860                if (!libfile.isAbsolute()) {
1861                    throw new UnsatisfiedLinkError(
1862    "ClassLoader.findLibrary failed to return an absolute path: " + libfilename);
1863                }
1864                if (loadLibrary0(fromClass, libfile)) {
1865                    return;
1866                }
1867                throw new UnsatisfiedLinkError("Can't load " + libfilename);
1868            }
1869        }
1870        for (int i = 0 ; i < sys_paths.length ; i++) {
1871            File libfile = new File(sys_paths[i], System.mapLibraryName(name));
1872            if (loadLibrary0(fromClass, libfile)) {
1873                return;
1874            }
1875        }
1876        if (loader != null) {
1877            for (int i = 0 ; i < usr_paths.length ; i++) {
1878                File libfile = new File(usr_paths[i],
1879                                        System.mapLibraryName(name));
1880                if (loadLibrary0(fromClass, libfile)) {
1881                    return;
1882                }
1883            }
1884        }
1885        // Oops, it failed
1886        throw new UnsatisfiedLinkError("no " + name + " in java.library.path");
1887    }

1900    private static boolean loadLibrary1(Class fromClass, final File file) {
1901        boolean exists = AccessController.doPrivileged(
1902            new PrivilegedAction<Object>() {
1903                public Object run() {
1904                    return file.exists() ? Boolean.TRUE : null;
1905                }})
1906            != null;
1907        if (!exists) {
1908            return false;
1909        }
1910        String name;
1911        try {
1912            name = file.getCanonicalPath();
1913        } catch (IOException e) {
1914            return false;
1915        }
1916        ClassLoader loader =
1917            (fromClass == null) ? null : fromClass.getClassLoader();
1918        Vector<NativeLibrary> libs =
1919            loader != null ? loader.nativeLibraries : systemNativeLibraries;
1920        synchronized (libs) {
1921            int size = libs.size();
1922            for (int i = 0; i < size; i++) {
1923                NativeLibrary lib = libs.elementAt(i);
1924                if (name.equals(lib.name)) {
1925                    return true;
1926                }
1927            }
1928
1929            synchronized (loadedLibraryNames) {
1930                if (loadedLibraryNames.contains(name)) {
1931                    throw new UnsatisfiedLinkError
1932                        ("Native Library " +
1933                         name +
1934                         " already loaded in another classloader");
1935                }
1936                /* If the library is being loaded (must be by the same thread,
1937                 * because Runtime.load and Runtime.loadLibrary are
1938                 * synchronous). The reason is can occur is that the JNI_OnLoad
1939                 * function can cause another loadLibrary invocation.
1940                 *
1941                 * Thus we can use a static stack to hold the list of libraries
1942                 * we are loading.
1943                 *
1944                 * If there is a pending load operation for the library, we
1945                 * immediately return success; otherwise, we raise
1946                 * UnsatisfiedLinkError.
1947                 */
1948                int n = nativeLibraryContext.size();
1949                for (int i = 0; i < n; i++) {
1950                    NativeLibrary lib = nativeLibraryContext.elementAt(i);
1951                    if (name.equals(lib.name)) {
1952                        if (loader == lib.fromClass.getClassLoader()) {
1953                            return true;
1954                        } else {
1955                            throw new UnsatisfiedLinkError
1956                                ("Native Library " +
1957                                 name +
1958                                 " is being loaded in another classloader");
1959                        }
1960                    }
1961                }
1962                NativeLibrary lib = new NativeLibrary(fromClass, name);
1963                nativeLibraryContext.push(lib);
1964                try {
1965                    lib.load(name);
1966                } finally {
1967                    nativeLibraryContext.pop();
1968                }
1969                if (lib.handle != 0) {
1970                    loadedLibraryNames.addElement(name);
1971                    libs.addElement(lib);
1972                    return true;
1973                }
1974                return false;
1975            }
1976        }
1977    }

1981        Vector<NativeLibrary> libs =
1982            loader != null ? loader.nativeLibraries : systemNativeLibraries;

2147     * setting will be returned; otherwise, if any package default assertion

2161     * @see  #setDefaultAssertionStatus(boolean)
2162     *
2163     * @since  1.4
2164     */
2165    boolean desiredAssertionStatus(String className) {
2166        synchronized (assertionLock) {
2167            // assert classAssertionStatus   != null;
2168            // assert packageAssertionStatus != null;
2169
2170            // Check for a class entry
2171            Boolean result = classAssertionStatus.get(className);
2172            if (result != null)
2173                return result.booleanValue();
2174
2175            // Check for most specific package entry
2176            int dotIndex = className.lastIndexOf(".");
2177            if (dotIndex < 0) { // default package
2178                result = packageAssertionStatus.get(null);
2179                if (result != null)
2180                    return result.booleanValue();
2181            }
2182            while(dotIndex > 0) {
2183                className = className.substring(0, dotIndex);
2184                result = packageAssertionStatus.get(className);
2185                if (result != null)
2186                    return result.booleanValue();
2187                dotIndex = className.lastIndexOf(".", dotIndex-1);
2188            }
2189
2190            // Return the classloader default
2191            return defaultAssertionStatus;
2192        }
