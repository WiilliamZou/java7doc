026package java.lang;
027
028/**
029 * Common superclass of exceptions thrown by reflective operations in
030 * core reflection.
031 *
032 * @see LinkageError
033 * @since 1.7
034 */
035public class ReflectiveOperationException extends Exception {
036    static final long serialVersionUID = 123456789L;
037
038    /**
039     * Constructs a new exception with {@code null} as its detail
040     * message.  The cause is not initialized, and may subsequently be
041     * initialized by a call to {@link #initCause}.
042     */
043    public ReflectiveOperationException() {
044        super();
045    }
046
047    /**
048     * Constructs a new exception with the specified detail message.
049     * The cause is not initialized, and may subsequently be
050     * initialized by a call to {@link #initCause}.
051     *
052     * @param   message   the detail message. The detail message is saved for
053     *          later retrieval by the {@link #getMessage()} method.
054     */
055    public ReflectiveOperationException(String message) {
056        super(message);
057    }
058
059    /**
060     * Constructs a new exception with the specified detail message
061     * and cause.
062     *
063     * <p>Note that the detail message associated with
064     * {@code cause} is <em>not</em> automatically incorporated in
065     * this exception's detail message.
066     *
067     * @param  message the detail message (which is saved for later retrieval
068     *         by the {@link #getMessage()} method).
069     * @param  cause the cause (which is saved for later retrieval by the
070     *         {@link #getCause()} method).  (A {@code null} value is
071     *         permitted, and indicates that the cause is nonexistent or
072     *         unknown.)
073     */
074    public ReflectiveOperationException(String message, Throwable cause) {
075        super(message, cause);
076    }
077
078    /**
079     * Constructs a new exception with the specified cause and a detail
080     * message of {@code (cause==null ? null : cause.toString())} (which
081     * typically contains the class and detail message of {@code cause}).
082     *
083     * @param  cause the cause (which is saved for later retrieval by the
084     *         {@link #getCause()} method).  (A {@code null} value is
085     *         permitted, and indicates that the cause is nonexistent or
086     *         unknown.)
087     */
088    public ReflectiveOperationException(Throwable cause) {
089        super(cause);
090    }
091}
