026package java.lang;
027
028import java.lang.annotation.*;
029import static java.lang.annotation.ElementType.*;
030
031/**
032 * Indicates that the named compiler warnings should be suppressed in the
033 * annotated element (and in all program elements contained in the annotated
034 * element).  Note that the set of warnings suppressed in a given element is
035 * a superset of the warnings suppressed in all containing elements.  For
036 * example, if you annotate a class to suppress one warning and annotate a
037 * method to suppress another, both warnings will be suppressed in the method.
038 *
039 * <p>As a matter of style, programmers should always use this annotation
040 * on the most deeply nested element where it is effective.  If you want to
041 * suppress a warning in a particular method, you should annotate that
042 * method rather than its class.
043 *
044 * @since 1.5
045 * @author Josh Bloch
046 */
047@Target({TYPE, FIELD, METHOD, PARAMETER, CONSTRUCTOR, LOCAL_VARIABLE})
048@Retention(RetentionPolicy.SOURCE)
049public @interface SuppressWarnings {
050    /**
051     * The set of warnings that are to be suppressed by the compiler in the
052     * annotated element.  Duplicate names are permitted.  The second and
053     * successive occurrences of a name are ignored.  The presence of
054     * unrecognized warning names is <i>not</i> an error: Compilers must
055     * ignore any warning names they do not recognize.  They are, however,
056     * free to emit a warning if an annotation contains an unrecognized
057     * warning name.
058     *
059     * <p>Compiler vendors should document the warning names they support in
060     * conjunction with this annotation type. They are encouraged to cooperate
061     * to ensure that the same names work across multiple compilers.
062     */
063    String[] value();
064}
