045/**
046 * This class consists exclusively of static methods that operate on or return
047 * method handles. They fall into several categories:
048 * <ul>
049 * <li>Lookup methods which help create method handles for methods and fields.

052 * <li>Wrapper methods which can convert between method handles and interface types.
053 * </ul>

096    /**
097     * A <em>lookup object</em> is a factory for creating method handles,
098     * when the creation requires access checking.
099     * Method handles do not perform
100     * access checks when they are called, but rather when they are created.
101     * Therefore, method handle access
102     * restrictions must be enforced when a method handle is created.
103     * The caller class against which those restrictions are enforced
104     * is known as the {@linkplain #lookupClass lookup class}.
105     * <p>
106     * A lookup class which needs to create method handles will call
107     * {@link MethodHandles#lookup MethodHandles.lookup} to create a factory for itself.
108     * When the {@code Lookup} factory object is created, the identity of the lookup class is
109     * determined, and securely stored in the {@code Lookup} object.
110     * The lookup class (or its delegates) may then use factory methods
111     * on the {@code Lookup} object to create method handles for access-checked members.
112     * This includes all methods, constructors, and fields which are allowed to the lookup class,
113     * even private ones.
114     * <p>
115     * The factory methods on a {@code Lookup} object correspond to all major
116     * use cases for methods, constructors, and fields.
117     * Here is a summary of the correspondence between these factory methods and
118     * the behavior the resulting method handles:
119     * <code>
120     * <table border=1 cellpadding=5 summary="lookup method behaviors">
121     * <tr><th>lookup expression</th><th>member</th><th>behavior</th></tr>
122     * <tr>
123     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findGetter lookup.findGetter(C.class,"f",FT.class)}</td>
124     *     <td>FT f;</td><td>(T) this.f;</td>
125     * </tr>
126     * <tr>
127     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findStaticGetter lookup.findStaticGetter(C.class,"f",FT.class)}</td>
128     *     <td>static<br>FT f;</td><td>(T) C.f;</td>
129     * </tr>
130     * <tr>
131     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findSetter lookup.findSetter(C.class,"f",FT.class)}</td>
132     *     <td>FT f;</td><td>this.f = x;</td>
133     * </tr>
134     * <tr>
135     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findStaticSetter lookup.findStaticSetter(C.class,"f",FT.class)}</td>
136     *     <td>static<br>FT f;</td><td>C.f = arg;</td>
137     * </tr>
138     * <tr>
139     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findVirtual lookup.findVirtual(C.class,"m",MT)}</td>
140     *     <td>T m(A*);</td><td>(T) this.m(arg*);</td>
141     * </tr>
142     * <tr>
143     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findStatic lookup.findStatic(C.class,"m",MT)}</td>
144     *     <td>static<br>T m(A*);</td><td>(T) C.m(arg*);</td>
145     * </tr>
146     * <tr>
147     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findSpecial lookup.findSpecial(C.class,"m",MT,this.class)}</td>
148     *     <td>T m(A*);</td><td>(T) super.m(arg*);</td>
149     * </tr>
150     * <tr>
151     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#findConstructor lookup.findConstructor(C.class,MT)}</td>
152     *     <td>C(A*);</td><td>(T) new C(arg*);</td>
153     * </tr>
154     * <tr>
155     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#unreflectGetter lookup.unreflectGetter(aField)}</td>
156     *     <td>(static)?<br>FT f;</td><td>(FT) aField.get(thisOrNull);</td>
157     * </tr>
158     * <tr>
159     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#unreflectSetter lookup.unreflectSetter(aField)}</td>
160     *     <td>(static)?<br>FT f;</td><td>aField.set(thisOrNull, arg);</td>
161     * </tr>
162     * <tr>
163     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#unreflect lookup.unreflect(aMethod)}</td>
164     *     <td>(static)?<br>T m(A*);</td><td>(T) aMethod.invoke(thisOrNull, arg*);</td>
165     * </tr>
166     * <tr>
167     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#unreflectConstructor lookup.unreflectConstructor(aConstructor)}</td>
168     *     <td>C(A*);</td><td>(C) aConstructor.newInstance(arg*);</td>
169     * </tr>
170     * <tr>
171     *     <td>{@linkplain java.lang.invoke.MethodHandles.Lookup#unreflect lookup.unreflect(aMethod)}</td>
172     *     <td>(static)?<br>T m(A*);</td><td>(T) aMethod.invoke(thisOrNull, arg*);</td>
173     * </tr>
174     * </table>
175     * </code>
176     * Here, the type {@code C} is the class or interface being searched for a member,
177     * documented as a parameter named {@code refc} in the lookup methods.
178     * The method or constructor type {@code MT} is composed from the return type {@code T}
179     * and the sequence of argument types {@code A*}.
180     * Both {@code MT} and the field type {@code FT} are documented as a parameter named {@code type}.
181     * The formal parameter {@code this} stands for the self-reference of type {@code C};
182     * if it is present, it is always the leading argument to the method handle invocation.
183     * (In the case of some {@code protected} members, {@code this} may be
184     * restricted in type to the lookup class; see below.)
185     * The name {@code arg} stands for all the other method handle arguments.
186     * In the code examples for the Core Reflection API, the name {@code thisOrNull}
187     * stands for a null reference if the accessed method or field is static,
188     * and {@code this} otherwise.
189     * The names {@code aMethod}, {@code aField}, and {@code aConstructor} stand
190     * for reflective objects corresponding to the given members.
191     * <p>
192     * In cases where the given member is of variable arity (i.e., a method or constructor)
193     * the returned method handle will also be of {@linkplain MethodHandle#asVarargsCollector variable arity}.
194     * In all other cases, the returned method handle will be of fixed arity.
195     * <p>
196     * The equivalence between looked-up method handles and underlying
197     * class members can break down in a few ways:
198     * <ul>
199     * <li>If {@code C} is not symbolically accessible from the lookup class's loader,
200     * the lookup can still succeed, even when there is no equivalent
201     * Java expression or bytecoded constant.
202     * <li>Likewise, if {@code T} or {@code MT}
203     * is not symbolically accessible from the lookup class's loader,
204     * the lookup can still succeed.
205     * For example, lookups for {@code MethodHandle.invokeExact} and
206     * {@code MethodHandle.invoke} will always succeed, regardless of requested type.
207     * <li>If there is a security manager installed, it can forbid the lookup
208     * on various grounds (<a href="#secmgr">see below</a>).
209     * By contrast, the {@code ldc} instruction is not subject to
210     * security manager checks.
211     * </ul>
212     *
213     * <h3><a name="access"></a>Access checking</h3>
214     * Access checks are applied in the factory methods of {@code Lookup},
215     * when a method handle is created.
216     * This is a key difference from the Core Reflection API, since
217     * {@link java.lang.reflect.Method#invoke java.lang.reflect.Method.invoke}
218     * performs access checking against every caller, on every call.
219     * <p>
220     * All access checks start from a {@code Lookup} object, which
221     * compares its recorded lookup class against all requests to
222     * create method handles.
223     * A single {@code Lookup} object can be used to create any number
224     * of access-checked method handles, all checked against a single
225     * lookup class.
226     * <p>
227     * A {@code Lookup} object can be shared with other trusted code,
228     * such as a metaobject protocol.
229     * A shared {@code Lookup} object delegates the capability
230     * to create method handles on private members of the lookup class.
231     * Even if privileged code uses the {@code Lookup} object,
232     * the access checking is confined to the privileges of the
233     * original lookup class.
234     * <p>
235     * A lookup can fail, because
236     * the containing class is not accessible to the lookup class, or
237     * because the desired class member is missing, or because the
238     * desired class member is not accessible to the lookup class.
239     * In any of these cases, a {@code ReflectiveOperationException} will be
240     * thrown from the attempted lookup.  The exact class will be one of
241     * the following:
242     * <ul>
243     * <li>NoSuchMethodException &mdash; if a method is requested but does not exist
244     * <li>NoSuchFieldException &mdash; if a field is requested but does not exist
245     * <li>IllegalAccessException &mdash; if the member exists but an access check fails
246     * </ul>
247     * <p>
248     * In general, the conditions under which a method handle may be
249     * looked up for a method {@code M} are exactly equivalent to the conditions
250     * under which the lookup class could have compiled and resolved a call to {@code M}.
251     * Where the JVM would raise exceptions like {@code NoSuchMethodError},
252     * a method handle lookup will generally raise a corresponding
253     * checked exception, such as {@code NoSuchMethodException}.
254     * And the effect of invoking the method handle resulting from the lookup
255     * is exactly equivalent to executing the compiled and resolved call to {@code M}.
256     * The same point is true of fields and constructors.
257     * <p>
258     * If the desired member is {@code protected}, the usual JVM rules apply,
259     * including the requirement that the lookup class must be either be in the
260     * same package as the desired member, or must inherit that member.
261     * (See the Java Virtual Machine Specification, sections 4.9.2, 5.4.3.5, and 6.4.)
262     * In addition, if the desired member is a non-static field or method
263     * in a different package, the resulting method handle may only be applied
264     * to objects of the lookup class or one of its subclasses.
265     * This requirement is enforced by narrowing the type of the leading
266     * {@code this} parameter from {@code C}
267     * (which will necessarily be a superclass of the lookup class)
268     * to the lookup class itself.
269     * <p>
270     * The JVM represents constructors and static initializer blocks as internal methods
271     * with special names ({@code "<init>"} and {@code "<clinit>"}).
272     * The internal syntax of invocation instructions allows them to refer to such internal
273     * methods as if they were normal methods, but the JVM verifier rejects them.
274     * A lookup of such an internal method will produce a {@code NoSuchMethodException}.
275     * <p>
276     * In some cases, access between nested classes is obtained by the Java compiler by creating
277     * an wrapper method to access a private method of another class
278     * in the same top-level declaration.
279     * For example, a nested class {@code C.D}
280     * can access private members within other related classes such as
281     * {@code C}, {@code C.D.E}, or {@code C.B},
282     * but the Java compiler may need to generate wrapper methods in
283     * those related classes.  In such cases, a {@code Lookup} object on
284     * {@code C.E} would be unable to those private members.
285     * A workaround for this limitation is the {@link Lookup#in Lookup.in} method,
286     * which can transform a lookup on {@code C.E} into one on any of those other
287     * classes, without special elevation of privilege.
288     * <p>
289     * Although bytecode instructions can only refer to classes in
290     * a related class loader, this API can search for methods in any
291     * class, as long as a reference to its {@code Class} object is
292     * available.  Such cross-loader references are also possible with the
293     * Core Reflection API, and are impossible to bytecode instructions
294     * such as {@code invokestatic} or {@code getfield}.
295     * There is a {@linkplain java.lang.SecurityManager security manager API}
296     * to allow applications to check such cross-loader references.
297     * These checks apply to both the {@code MethodHandles.Lookup} API
298     * and the Core Reflection API
299     * (as found on {@link java.lang.Class Class}).
300     * <p>
301     * Access checks only apply to named and reflected methods,
302     * constructors, and fields.
303     * Other method handle creation methods, such as
304     * {@link MethodHandle#asType MethodHandle.asType},
305     * do not require any access checks, and are done
306     * with static methods of {@link MethodHandles},
307     * independently of any {@code Lookup} object.
308     *
309     * <h3>Security manager interactions</h3>
310     * <a name="secmgr"></a>
311     * If a security manager is present, member lookups are subject to
312     * additional checks.
313     * From one to four calls are made to the security manager.
314     * Any of these calls can refuse access by throwing a
315     * {@link java.lang.SecurityException SecurityException}.
316     * Define {@code smgr} as the security manager,
317     * {@code refc} as the containing class in which the member
318     * is being sought, and {@code defc} as the class in which the
319     * member is actually defined.
320     * The calls are made according to the following rules:
321     * <ul>
322     * <li>In all cases, {@link SecurityManager#checkMemberAccess
323     *     smgr.checkMemberAccess(refc, Member.PUBLIC)} is called.
324     * <li>If the class loader of the lookup class is not
325     *     the same as or an ancestor of the class loader of {@code refc},
326     *     then {@link SecurityManager#checkPackageAccess
327     *     smgr.checkPackageAccess(refcPkg)} is called,
328     *     where {@code refcPkg} is the package of {@code refc}.
329     * <li>If the retrieved member is not public,
330     *     {@link SecurityManager#checkMemberAccess
331     *     smgr.checkMemberAccess(defc, Member.DECLARED)} is called.
332     *     (Note that {@code defc} might be the same as {@code refc}.)
333     *     The default implementation of this security manager method
334     *     inspects the stack to determine the original caller of
335     *     the reflective request (such as {@code findStatic}),
336     *     and performs additional permission checks if the
337     *     class loader of {@code defc} differs from the class
338     *     loader of the class from which the reflective request came.
339     * <li>If the retrieved member is not public,
340     *     and if {@code defc} and {@code refc} are in different class loaders,
341     *     and if the class loader of the lookup class is not
342     *     the same as or an ancestor of the class loader of {@code defc},
343     *     then {@link SecurityManager#checkPackageAccess
344     *     smgr.checkPackageAccess(defcPkg)} is called,
345     *     where {@code defcPkg} is the package of {@code defc}.
346     * </ul>
347     */
348    public static final
349    class Lookup {
350        /** The class on behalf of whom the lookup is being performed. */
351        private final Class<?> lookupClass;
352
353        /** The allowed sorts of members which may be looked up (PUBLIC, etc.). */
354        private final int allowedModes;
355
356        /** A single-bit mask representing {@code public} access,
357         *  which may contribute to the result of {@link #lookupModes lookupModes}.
358         *  The value, {@code 0x01}, happens to be the same as the value of the
359         *  {@code public} {@linkplain java.lang.reflect.Modifier#PUBLIC modifier bit}.
360         */
361        public static final int PUBLIC = Modifier.PUBLIC;
362
363        /** A single-bit mask representing {@code private} access,
364         *  which may contribute to the result of {@link #lookupModes lookupModes}.
365         *  The value, {@code 0x02}, happens to be the same as the value of the
366         *  {@code private} {@linkplain java.lang.reflect.Modifier#PRIVATE modifier bit}.
367         */
368        public static final int PRIVATE = Modifier.PRIVATE;
369
370        /** A single-bit mask representing {@code protected} access,
371         *  which may contribute to the result of {@link #lookupModes lookupModes}.
372         *  The value, {@code 0x04}, happens to be the same as the value of the
373         *  {@code protected} {@linkplain java.lang.reflect.Modifier#PROTECTED modifier bit}.
374         */
375        public static final int PROTECTED = Modifier.PROTECTED;
376
377        /** A single-bit mask representing {@code package} access (default access),
378         *  which may contribute to the result of {@link #lookupModes lookupModes}.
379         *  The value is {@code 0x08}, which does not correspond meaningfully to
380         *  any particular {@linkplain java.lang.reflect.Modifier modifier bit}.
381         */
382        public static final int PACKAGE = Modifier.STATIC;
383
384        private static final int ALL_MODES = (PUBLIC | PRIVATE | PROTECTED | PACKAGE);
385        private static final int TRUSTED   = -1;
386
387        private static int fixmods(int mods) {
388            mods &= (ALL_MODES - PACKAGE);
389            return (mods != 0) ? mods : PACKAGE;
390        }
391
392        /** Tells which class is performing the lookup.  It is this class against
393         *  which checks are performed for visibility and access permissions.
394         *  <p>
395         *  The class implies a maximum level of access permission,
396         *  but the permissions may be additionally limited by the bitmask
397         *  {@link #lookupModes lookupModes}, which controls whether non-public members
398         *  can be accessed.
399         */
400        public Class<?> lookupClass() {
401            return lookupClass;
402        }
403
404        // This is just for calling out to MethodHandleImpl.
405        private Class<?> lookupClassOrNull() {
406            return (allowedModes == TRUSTED) ? null : lookupClass;
407        }
408
409        /** Tells which access-protection classes of members this lookup object can produce.
410         *  The result is a bit-mask of the bits
411         *  {@linkplain #PUBLIC PUBLIC (0x01)},
412         *  {@linkplain #PRIVATE PRIVATE (0x02)},
413         *  {@linkplain #PROTECTED PROTECTED (0x04)},
414         *  and {@linkplain #PACKAGE PACKAGE (0x08)}.
415         *  <p>
416         *  A freshly-created lookup object
417         *  on the {@linkplain java.lang.invoke.MethodHandles#lookup() caller's class}
418         *  has all possible bits set, since the caller class can access all its own members.
419         *  A lookup object on a new lookup class
420         *  {@linkplain java.lang.invoke.MethodHandles.Lookup#in created from a previous lookup object}
421         *  may have some mode bits set to zero.
422         *  The purpose of this is to restrict access via the new lookup object,
423         *  so that it can access only names which can be reached by the original
424         *  lookup object, and also by the new lookup class.
425         */
426        public int lookupModes() {
427            return allowedModes & ALL_MODES;
428        }
429
430        /** Embody the current class (the lookupClass) as a lookup class
431         * for method handle creation.
432         * Must be called by from a method in this package,
433         * which in turn is called by a method not in this package.
434         * <p>
435         * Also, don't make it private, lest javac interpose
436         * an access$N method.
437         */
438
439        Lookup(Class<?> lookupClass) {
440            this(lookupClass, ALL_MODES);
441            checkUnprivilegedlookupClass(lookupClass, ALL_MODES);
442        }
443
444        private Lookup(Class<?> lookupClass, int allowedModes) {
445            this.lookupClass = lookupClass;
446            this.allowedModes = allowedModes;
447        }
448
449        /**
450         * Creates a lookup on the specified new lookup class.
451         * The resulting object will report the specified
452         * class as its own {@link #lookupClass lookupClass}.
453         * <p>
454         * However, the resulting {@code Lookup} object is guaranteed
455         * to have no more access capabilities than the original.
456         * In particular, access capabilities can be lost as follows:<ul>
457         * <li>If the new lookup class differs from the old one,
458         * protected members will not be accessible by virtue of inheritance.
459         * (Protected members may continue to be accessible because of package sharing.)
460         * <li>If the new lookup class is in a different package
461         * than the old one, protected and default (package) members will not be accessible.
462         * <li>If the new lookup class is not within the same package member
463         * as the old one, private members will not be accessible.
464         * <li>If the new lookup class is not accessible to the old lookup class,
465         * then no members, not even public members, will be accessible.
466         * (In all other cases, public members will continue to be accessible.)
467         * </ul>
468         *
469         * @param requestedLookupClass the desired lookup class for the new lookup object
470         * @return a lookup object which reports the desired lookup class
471         * @throws NullPointerException if the argument is null
472         */
473        public Lookup in(Class<?> requestedLookupClass) {
474            requestedLookupClass.getClass();  // null check
475            if (allowedModes == TRUSTED)  // IMPL_LOOKUP can make any lookup at all
476                return new Lookup(requestedLookupClass, ALL_MODES);
477            if (requestedLookupClass == this.lookupClass)
478                return this;  // keep same capabilities
479            int newModes = (allowedModes & (ALL_MODES & ~PROTECTED));
480            if ((newModes & PACKAGE) != 0
481                && !VerifyAccess.isSamePackage(this.lookupClass, requestedLookupClass)) {
482                newModes &= ~(PACKAGE|PRIVATE);
483            }
484            // Allow nestmate lookups to be created without special privilege:
485            if ((newModes & PRIVATE) != 0
486                && !VerifyAccess.isSamePackageMember(this.lookupClass, requestedLookupClass)) {
487                newModes &= ~PRIVATE;
488            }
489            if ((newModes & PUBLIC) != 0
490                && !VerifyAccess.isClassAccessible(requestedLookupClass, this.lookupClass, allowedModes)) {
491                // The requested class it not accessible from the lookup class.
492                // No permissions.
493                newModes = 0;
494            }
495            checkUnprivilegedlookupClass(requestedLookupClass, newModes);
496            return new Lookup(requestedLookupClass, newModes);
497        }

502        /** Version of lookup which is trusted minimally.
503         *  It can only be used to create method handles to
504         *  publicly accessible members.
505         */
506        static final Lookup PUBLIC_LOOKUP = new Lookup(Object.class, PUBLIC);

508        /** Package-private version of lookup which is trusted. */
509        static final Lookup IMPL_LOOKUP = new Lookup(Object.class, TRUSTED);

511        private static void checkUnprivilegedlookupClass(Class<?> lookupClass, int allowedModes) {
512            String name = lookupClass.getName();
513            if (name.startsWith("java.lang.invoke."))
514                throw newIllegalArgumentException("illegal lookupClass: "+lookupClass);
515
516            // For caller-sensitive MethodHandles.lookup()
517            // disallow lookup more restricted packages
518            if (allowedModes == ALL_MODES && lookupClass.getClassLoader() == null) {
519                if (name.startsWith("java.") ||
520                        (name.startsWith("sun.") && !name.startsWith("sun.invoke."))) {
521                    throw newIllegalArgumentException("illegal lookupClass: " + lookupClass);
522                }
523            }
524        }

526        /**
527         * Displays the name of the class from which lookups are to be made.
528         * (The name is the one reported by {@link java.lang.Class#getName() Class.getName}.)
529         * If there are restrictions on the access permitted to this lookup,
530         * this is indicated by adding a suffix to the class name, consisting
531         * of a slash and a keyword.  The keyword represents the strongest
532         * allowed access, and is chosen as follows:
533         * <ul>
534         * <li>If no access is allowed, the suffix is "/noaccess".
535         * <li>If only public access is allowed, the suffix is "/public".
536         * <li>If only public and package access are allowed, the suffix is "/package".
537         * <li>If only public, package, and private access are allowed, the suffix is "/private".
538         * </ul>
539         * If none of the above cases apply, it is the case that full
540         * access (public, package, private, and protected) is allowed.
541         * In this case, no suffix is added.
542         * This is true only of an object obtained originally from
543         * {@link java.lang.invoke.MethodHandles#lookup MethodHandles.lookup}.
544         * Objects created by {@link java.lang.invoke.MethodHandles.Lookup#in Lookup.in}
545         * always have restricted access, and will display a suffix.
546         * <p>
547         * (It may seem strange that protected access should be
548         * stronger than private access.  Viewed independently from
549         * package access, protected access is the first to be lost,
550         * because it requires a direct subclass relationship between
551         * caller and callee.)
552         * @see #in
553         */
554        @Override
555        public String toString() {
556            String cname = lookupClass.getName();
557            switch (allowedModes) {
558            case 0:  // no privileges
559                return cname + "/noaccess";
560            case PUBLIC:
561                return cname + "/public";
562            case PUBLIC|PACKAGE:
563                return cname + "/package";
564            case ALL_MODES & ~PROTECTED:
565                return cname + "/private";
566            case ALL_MODES:
567                return cname;
568            case TRUSTED:
569                return "/trusted";  // internal only; not exported
570            default:  // Should not happen, but it's a bitfield...
571                cname = cname + "/" + Integer.toHexString(allowedModes);
572                assert(false) : cname;
573                return cname;
574            }
575        }

621         * The type of the method handle will be that of the method,
622         * with the receiver type (usually {@code refc}) prepended.
623         * The method and all its argument types must be accessible to the lookup class.
624         * <p>
625         * When called, the handle will treat the first argument as a receiver

627         * implementation to enter.

637         * {@linkplain MethodHandle#asVarargsCollector variable arity} if and only if
638         * the method's variable arity modifier bit ({@code 0x0080}) is set.
639         * <p>
640         * Because of the general equivalence between {@code invokevirtual}
641         * instructions and method handles produced by {@code findVirtual},
642         * if the class is {@code MethodHandle} and the name string is
643         * {@code invokeExact} or {@code invoke}, the resulting
644         * method handle is equivalent to one produced by
645         * {@link java.lang.invoke.MethodHandles#exactInvoker MethodHandles.exactInvoker} or
646         * {@link java.lang.invoke.MethodHandles#invoker MethodHandles.invoker}
647         * with the same {@code type} argument.

649         * <b>Example:</b>
650         * <p><blockquote><pre>{@code
651import static java.lang.invoke.MethodHandles.*;
652import static java.lang.invoke.MethodType.*;
653...
654MethodHandle MH_concat = publicLookup().findVirtual(String.class,
655  "concat", methodType(String.class, String.class));
656MethodHandle MH_hashCode = publicLookup().findVirtual(Object.class,
657  "hashCode", methodType(int.class));
658MethodHandle MH_hashCode_String = publicLookup().findVirtual(String.class,
659  "hashCode", methodType(int.class));
660assertEquals("xy", (String) MH_concat.invokeExact("x", "y"));
661assertEquals("xy".hashCode(), (int) MH_hashCode.invokeExact((Object)"xy"));
662assertEquals("xy".hashCode(), (int) MH_hashCode_String.invokeExact("xy"));
663// interface method:
664MethodHandle MH_subSequence = publicLookup().findVirtual(CharSequence.class,
665  "subSequence", methodType(CharSequence.class, int.class, int.class));
666assertEquals("def", MH_subSequence.invoke("abcdefghi", 3, 6).toString());
667// constructor "internal method" must be accessed differently:
668MethodType MT_newString = methodType(void.class); //()V for new String()
669try { assertEquals("impossible", lookup()
670        .findVirtual(String.class, "<init>", MT_newString));
671 } catch (NoSuchMethodException ex) { } // OK
672MethodHandle MH_newString = publicLookup()
673  .findConstructor(String.class, MT_newString);
674assertEquals("", (String) MH_newString.invokeExact());
675         * }</pre></blockquote>
676         *
677         * @param refc the class or interface from which the method is accessed
678         * @param name the name of the method
679         * @param type the type of the method, with the receiver argument omitted
680         * @return the desired method handle
681         * @throws NoSuchMethodException if the method does not exist
682         * @throws IllegalAccessException if access checking fails,
683         *                                or if the method is {@code static}
684         *                                or if the method's variable arity modifier bit
685         *                                is set and {@code asVarargsCollector} fails
686         * @exception SecurityException if a security manager is present and it
687         *                              <a href="MethodHandles.Lookup.html#secmgr">refuses access</a>
688         * @throws NullPointerException if any argument is null
689         */
690        public MethodHandle findVirtual(Class<?> refc, String name, MethodType type) throws NoSuchMethodException, IllegalAccessException {
691            if (refc == MethodHandle.class) {
692                MethodHandle mh = findVirtualForMH(name, type);
693                if (mh != null)  return mh;
694            }
695            byte refKind = (refc.isInterface() ? REF_invokeInterface : REF_invokeVirtual);
696            MemberName method = resolveOrFail(refKind, refc, name, type);
697            checkSecurityManager(refc, method);
698            return getDirectMethod(refKind, refc, method, findBoundCallerClass(method));
699        }

1188         * Perform necessary <a href="MethodHandles.Lookup.html#secmgr">access checks</a>.
1189         * This function performs stack walk magic: do not refactor it.
1190         */
1191        void checkSecurityManager(Class<?> refc, MemberName m) {
1192            SecurityManager smgr = System.getSecurityManager();
1193            if (smgr == null)  return;
1194            if (allowedModes == TRUSTED)  return;
1195
1196            final boolean overridden = isCheckMemberAccessOverridden(smgr);
1197            // Step 1:
1198            {
1199                // Default policy is to allow Member.PUBLIC; no need to check
1200                // permission if SecurityManager is the default implementation
1201                final int which = Member.PUBLIC;
1202                final Class<?> clazz = refc;
1203                if (overridden) {
1204                    // Don't refactor; otherwise break the stack depth for
1205                    // checkMemberAccess of subclasses of SecurityManager as specified.
1206                    smgr.checkMemberAccess(clazz, which);
1207                }
1208            }
1209
1210            // Step 2:
1211            if (!isFullPowerLookup() ||
1212                !VerifyAccess.classLoaderIsAncestor(lookupClass, refc)) {
1213                ReflectUtil.checkPackageAccess(refc);
1214            }
1215
1216            // Step 3:
1217            if (m.isPublic()) return;
1218            Class<?> defc = m.getDeclaringClass();
1219            {
1220                // Inline SecurityManager.checkMemberAccess
1221                final int which = Member.DECLARED;
1222                final Class<?> clazz = defc;
1223                if (!overridden) {
1224                    if (!isFullPowerLookup()) {
1225                        smgr.checkPermission(SecurityConstants.CHECK_MEMBER_ACCESS_PERMISSION);
1226                    }
1227                } else {
1228                    // Don't refactor; otherwise break the stack depth for
1229                    // checkMemberAccess of subclasses of SecurityManager as specified.
1230                    smgr.checkMemberAccess(clazz, which);
1231                }
1232            }
1233
1234            // Step 4:
1235            if (defc != refc) {
1236                ReflectUtil.checkPackageAccess(defc);
1237            }
1238        }

1264        void checkAccess(byte refKind, Class<?> refc, MemberName m) throws IllegalAccessException {
1265            assert(m.referenceKindIsConsistentWith(refKind) &&
1266                   MethodHandleNatives.refKindIsValid(refKind) &&
1267                   (MethodHandleNatives.refKindIsField(refKind) == m.isField()));
1268            int allowedModes = this.allowedModes;
1269            if (allowedModes == TRUSTED)  return;
1270            int mods = m.getModifiers();
1271            if (Modifier.isProtected(mods)) {
1272                if (refKind == REF_invokeVirtual &&
1273                        m.getDeclaringClass() == Object.class &&
1274                        m.getName().equals("clone") &&
1275                        refc.isArray()) {
1276                    // The JVM does this hack also.
1277                    // (See ClassVerifier::verify_invoke_instructions
1278                    // and LinkResolver::check_method_accessability.)
1279                    // Because the JVM does not allow separate methods on array types,
1280                    // there is no separate method for int[].clone.
1281                    // All arrays simply inherit Object.clone.
1282                    // But for access checking logic, we make Object.clone
1283                    // (normally protected) appear to be public.
1284                    // Later on, when the DirectMethodHandle is created,
1285                    // its leading argument will be restricted to the
1286                    // requested array type.
1287                    // N.B. The return type is not adjusted, because
1288                    // that is *not* the bytecode behavior.
1289                    mods ^= Modifier.PROTECTED | Modifier.PUBLIC;
1290                }
1291                if (refKind == REF_newInvokeSpecial) {
1292                    // cannot "new" a protected ctor in a different package
1293                    mods ^= Modifier.PROTECTED;
1294                }
1295            }
1296            if (Modifier.isFinal(mods) &&
1297                    MethodHandleNatives.refKindIsSetter(refKind))
1298                throw m.makeAccessException("unexpected set of a final field", this);
1299            if (Modifier.isPublic(mods) && Modifier.isPublic(refc.getModifiers()) && allowedModes != 0)
1300                return;  // common case
1301            int requestedModes = fixmods(mods);  // adjust 0 => PACKAGE
1302            if ((requestedModes & allowedModes) != 0) {
1303                if (VerifyAccess.isMemberAccessible(refc, m.getDeclaringClass(),
1304                                                    mods, lookupClass(), allowedModes))
1305                    return;
1306            } else {
1307                // Protected members can also be checked as if they were package-private.
1308                if ((requestedModes & PROTECTED) != 0 && (allowedModes & PACKAGE) != 0
1309                        && VerifyAccess.isSamePackage(m.getDeclaringClass(), lookupClass()))
1310                    return;
1311            }
1312            throw m.makeAccessException(accessFailedMessage(refc, m), this);
1313        }

1315        String accessFailedMessage(Class<?> refc, MemberName m) {
1316            Class<?> defc = m.getDeclaringClass();
1317            int mods = m.getModifiers();
1318            // check the class first:
1319            boolean classOK = (Modifier.isPublic(defc.getModifiers()) &&
1320                               (defc == refc ||
1321                                Modifier.isPublic(refc.getModifiers())));
1322            if (!classOK && (allowedModes & PACKAGE) != 0) {
1323                classOK = (VerifyAccess.isClassAccessible(defc, lookupClass(), ALL_MODES) &&
1324                           (defc == refc ||
1325                            VerifyAccess.isClassAccessible(refc, lookupClass(), ALL_MODES)));
1326            }
1327            if (!classOK)
1328                return "class is not public";
1329            if (Modifier.isPublic(mods))
1330                return "access to public member failed";  // (how?)
1331            if (Modifier.isPrivate(mods))
1332                return "member is private";
1333            if (Modifier.isProtected(mods))
1334                return "member is protected";
1335            return "member is private to package";
1336        }

1352            // The accessing class only has the right to use a protected member
1353            // on itself or a subclass.  Enforce that restriction, from JVMS 5.4.4, etc.
1354            if (!method.isProtected() || method.isStatic()
1355                || allowedModes == TRUSTED
1356                || method.getDeclaringClass() == lookupClass()

1384        private MethodHandle getDirectMethodCommon(byte refKind, Class<?> refc, MemberName method,
1385                                                   boolean doRestrict, Class<?> callerClass) throws IllegalAccessException {
1386            checkMethod(refKind, refc, method);
1387            if (method.isMethodHandleInvoke())
1388                return fakeMethodHandleInvoke(method);
1389
1390            Class<?> refcAsSuper;
1391            if (refKind == REF_invokeSpecial &&
1392                refc != lookupClass() &&
1393                refc != (refcAsSuper = lookupClass().getSuperclass()) &&
1394                refc.isAssignableFrom(lookupClass())) {
1395                assert(!method.getName().equals("<init>"));  // not this code path
1396                // Per JVMS 6.5, desc. of invokespecial instruction:
1397                // If the method is in a superclass of the LC,
1398                // and if our original search was above LC.super,
1399                // repeat the search (symbolic lookup) from LC.super.
1400                // FIXME: MemberName.resolve should handle this instead.
1401                MemberName m2 = new MemberName(refcAsSuper,
1402                                               method.getName(),
1403                                               method.getMethodType(),
1404                                               REF_invokeSpecial);
1405                m2 = IMPL_NAMES.resolveOrNull(refKind, m2, lookupClassOrNull());
1406                if (m2 == null)  throw new InternalError(method.toString());
1407                method = m2;
1408                refc = refcAsSuper;
1409                // redo basic checks
1410                checkMethod(refKind, refc, method);
1411            }
1412
1413            MethodHandle mh = DirectMethodHandle.make(refc, method);
1414            mh = maybeBindCaller(method, mh, callerClass);
1415            mh = mh.setVarargs(method);
1416            if (doRestrict)
1417                mh = restrictReceiver(method, mh, lookupClass());
1418            return mh;
1419        }

1517     * given {@code type}, with a given number of trailing arguments replaced by
1518     * a single trailing {@code Object[]} array.
1519     * The resulting invoker will be a method handle with the following
1520     * arguments:
1521     * <ul>
1522     * <li>a single {@code MethodHandle} target
1523     * <li>zero or more leading values (counted by {@code leadingArgCount})

1525     * </ul>
1526     * <p>
1527     * The invoker will invoke its target like a call to {@link MethodHandle#invoke invoke} with
1528     * the indicated {@code type}.
1529     * That is, if the target is exactly of the given {@code type}, it will behave

1533     * The type of the returned invoker will not be the given {@code type}, but rather

1536     * the final parameter.
1537     * <p>
1538     * Before invoking its target, the invoker will spread the final array, apply
1539     * reference casts as necessary, and unbox and widen primitive arguments.

1548invoker = invoker.asSpreader(Object[].class, spreadArgCount);
1549return invoker;

1551     * <p>

1553     * @param type the desired target type
1554     * @param leadingArgCount number of fixed arguments, to be passed unchanged to the target
1555     * @return a method handle suitable for invoking any method handle of the given type
1556     * @throws NullPointerException if {@code type} is null
1557     * @throws IllegalArgumentException if {@code leadingArgCount} is not in
1558     *                  the range from 0 to {@code type.parameterCount()} inclusive
1559     */
1560    static public
1561    MethodHandle spreadInvoker(MethodType type, int leadingArgCount) {
1562        if (leadingArgCount < 0 || leadingArgCount > type.parameterCount())
1563            throw new IllegalArgumentException("bad argument count "+leadingArgCount);
1564        return type.invokers().spreadInvoker(leadingArgCount);
1565    }

1648     * Produces a method handle which adapts the type of the
1649     * given method handle to a new type by pairwise argument and return type conversion.
1650     * The original type and new type must have the same number of arguments.
1651     * The resulting method handle is guaranteed to report a type
1652     * which is equal to the desired new type.
1653     * <p>
1654     * If the original type and new type are equal, returns target.
1655     * <p>
1656     * The same conversions are allowed as for {@link MethodHandle#asType MethodHandle.asType},
1657     * and some additional conversions are also applied if those conversions fail.
1658     * Given types <em>T0</em>, <em>T1</em>, one of the following conversions is applied
1659     * if possible, before or instead of any conversions done by {@code asType}:
1660     * <ul>
1661     * <li>If <em>T0</em> and <em>T1</em> are references, and <em>T1</em> is an interface type,
1662     *     then the value of type <em>T0</em> is passed as a <em>T1</em> without a cast.

1664     * <li>If <em>T0</em> is boolean and <em>T1</em> is another primitive,
1665     *     the boolean is converted to a byte value, 1 for true, 0 for false.
1666     *     (This treatment follows the usage of the bytecode verifier.)

1668     *     <em>T0</em> is converted to byte via Java casting conversion (JLS 5.5),

1679     * <li>If <em>T0</em> is a reference and <em>T1</em> a primitive,
1680     *     and if the reference is null at runtime, a zero value is introduced.
1681     * </ul>

1683     * @param newType the expected type of the new method handle

1699    /**
1700     * Produces a method handle which adapts the calling sequence of the
1701     * given method handle to a new type, by reordering the arguments.
1702     * The resulting method handle is guaranteed to report a type
1703     * which is equal to the desired new type.
1704     * <p>
1705     * The given array controls the reordering.
1706     * Call {@code #I} the number of incoming parameters (the value
1707     * {@code newType.parameterCount()}, and call {@code #O} the number
1708     * of outgoing parameters (the value {@code target.type().parameterCount()}).
1709     * Then the length of the reordering array must be {@code #O},
1710     * and each element must be a non-negative number less than {@code #I}.
1711     * For every {@code N} less than {@code #O}, the {@code N}-th
1712     * outgoing argument will be taken from the {@code I}-th incoming
1713     * argument, where {@code I} is {@code reorder[N]}.
1714     * <p>
1715     * No argument or return value conversions are applied.
1716     * The type of each incoming argument, as determined by {@code newType},
1717     * must be identical to the type of the corresponding outgoing parameter
1718     * or parameters in the target method handle.
1719     * The return type of {@code newType} must be identical to the return
1720     * type of the original target.
1721     * <p>
1722     * The reordering array need not specify an actual permutation.
1723     * An incoming argument will be duplicated if its index appears
1724     * more than once in the array, and an incoming argument will be dropped
1725     * if its index does not appear in the array.
1726     * As in the case of {@link #dropArguments(MethodHandle,int,List) dropArguments},
1727     * incoming arguments which are not mentioned in the reordering array
1728     * are may be any type, as determined only by {@code newType}.
1729     * <blockquote><pre>
1730import static java.lang.invoke.MethodHandles.*;
1731import static java.lang.invoke.MethodType.*;
1732...
1733MethodType intfn1 = methodType(int.class, int.class);
1734MethodType intfn2 = methodType(int.class, int.class, int.class);
1735MethodHandle sub = ... {int x, int y => x-y} ...;
1736assert(sub.type().equals(intfn2));
1737MethodHandle sub1 = permuteArguments(sub, intfn2, 0, 1);
1738MethodHandle rsub = permuteArguments(sub, intfn2, 1, 0);
1739assert((int)rsub.invokeExact(1, 100) == 99);
1740MethodHandle add = ... {int x, int y => x+y} ...;
1741assert(add.type().equals(intfn2));
1742MethodHandle twice = permuteArguments(add, intfn1, 0, 0);
1743assert(twice.type().equals(intfn1));
1744assert((int)twice.invokeExact(21) == 42);
1745     * </pre></blockquote>
1746     * @param target the method handle to invoke after arguments are reordered
1747     * @param newType the expected type of the new method handle
1748     * @param reorder an index array which controls the reordering
1749     * @return a method handle which delegates to the target after it
1750     *           drops unused arguments and moves and/or duplicates the other arguments
1751     * @throws NullPointerException if any argument is null
1752     * @throws IllegalArgumentException if the index array length is not equal to
1753     *                  the arity of the target, or if any index array element
1754     *                  not a valid index for a parameter of {@code newType},
1755     *                  or if two corresponding parameter types in
1756     *                  {@code target.type()} and {@code newType} are not identical,
1757     */
1758    public static
1759    MethodHandle permuteArguments(MethodHandle target, MethodType newType, int... reorder) {
1760        reorder = reorder.clone();
1761        checkReorder(reorder, newType, target.type());
1762        return target.permuteArguments(newType, reorder);
1763    }
1764
1765    private static void checkReorder(int[] reorder, MethodType newType, MethodType oldType) {
1766        if (newType.returnType() != oldType.returnType())
1767            throw newIllegalArgumentException("return types do not match",
1768                    oldType, newType);
1769        if (reorder.length == oldType.parameterCount()) {
1770            int limit = newType.parameterCount();
1771            boolean bad = false;
1772            for (int j = 0; j < reorder.length; j++) {
1773                int i = reorder[j];
1774                if (i < 0 || i >= limit) {
1775                    bad = true; break;
1776                }
1777                Class<?> src = newType.parameterType(i);
1778                Class<?> dst = oldType.parameterType(j);
1779                if (src != dst)
1780                    throw newIllegalArgumentException("parameter types do not match after reorder",
1781                            oldType, newType);
1782            }
1783            if (!bad)  return;
1784        }
1785        throw newIllegalArgumentException("bad reorder array: "+Arrays.toString(reorder));
1786    }

1834    /**
1835     * Provides a target method handle with one or more <em>bound arguments</em>
1836     * in advance of the method handle's invocation.
1837     * The formal parameters to the target corresponding to the bound
1838     * arguments are called <em>bound parameters</em>.
1839     * Returns a new method handle which saves away the bound arguments.
1840     * When it is invoked, it receives arguments for any non-bound parameters,
1841     * binds the saved arguments to their corresponding parameters,
1842     * and calls the original target.
1843     * <p>
1844     * The type of the new method handle will drop the types for the bound
1845     * parameters from the original target type, since the new method handle
1846     * will no longer require those arguments to be supplied by its callers.
1847     * <p>
1848     * Each given argument object must match the corresponding bound parameter type.
1849     * If a bound parameter type is a primitive, the argument object
1850     * must be a wrapper, and will be unboxed to produce the primitive value.
1851     * <p>
1852     * The {@code pos} argument selects which parameters are to be bound.
1853     * It may range between zero and <i>N-L</i> (inclusively),
1854     * where <i>N</i> is the arity of the target method handle
1855     * and <i>L</i> is the length of the values array.
1856     * @param target the method handle to invoke after the argument is inserted
1857     * @param pos where to insert the argument (zero for the first)
1858     * @param values the series of arguments to insert
1859     * @return a method handle which inserts an additional argument,
1860     *         before calling the original method handle
1861     * @throws NullPointerException if the target or the {@code values} array is null
1862     * @see MethodHandle#bindTo
1863     */
1864    public static
1865    MethodHandle insertArguments(MethodHandle target, int pos, Object... values) {
1866        int insCount = values.length;
1867        MethodType oldType = target.type();
1868        int outargs = oldType.parameterCount();
1869        int inargs  = outargs - insCount;
1870        if (inargs < 0)
1871            throw newIllegalArgumentException("too many values to insert");
1872        if (pos < 0 || pos > inargs)
1873            throw newIllegalArgumentException("no argument type to append");
1874        MethodHandle result = target;
1875        for (int i = 0; i < insCount; i++) {
1876            Object value = values[i];
1877            Class<?> ptype = oldType.parameterType(pos+i);
1878            if (ptype.isPrimitive()) {
1879                char btype = 'I';
1880                Wrapper w = Wrapper.forPrimitiveType(ptype);
1881                switch (w) {
1882                case LONG:    btype = 'J'; break;
1883                case FLOAT:   btype = 'F'; break;
1884                case DOUBLE:  btype = 'D'; break;
1885                }
1886                // perform unboxing and/or primitive conversion
1887                value = w.convert(value, ptype);
1888                result = result.bindArgument(pos, btype, value);
1889                continue;
1890            }
1891            value = ptype.cast(value);  // throw CCE if needed
1892            if (pos == 0) {
1893                result = result.bindReceiver(value);
1894            } else {
1895                result = result.bindArgument(pos, 'L', value);
1896            }
1897        }
1898        return result;
1899    }

2011    /**
2012     * Adapts a target method handle by pre-processing
2013     * one or more of its arguments, each with its own unary filter function,
2014     * and then calling the target with each pre-processed argument
2015     * replaced by the result of its corresponding filter function.
2016     * <p>
2017     * The pre-processing is performed by one or more method handles,
2018     * specified in the elements of the {@code filters} array.
2019     * The first element of the filter array corresponds to the {@code pos}
2020     * argument of the target, and so on in sequence.
2021     * <p>
2022     * Null arguments in the array are treated as identity functions,
2023     * and the corresponding arguments left unchanged.
2024     * (If there are no non-null elements in the array, the original target is returned.)
2025     * Each filter is applied to the corresponding argument of the adapter.
2026     * <p>
2027     * If a filter {@code F} applies to the {@code N}th argument of
2028     * the target, then {@code F} must be a method handle which
2029     * takes exactly one argument.  The type of {@code F}'s sole argument
2030     * replaces the corresponding argument type of the target
2031     * in the resulting adapted method handle.
2032     * The return type of {@code F} must be identical to the corresponding
2033     * parameter type of the target.
2034     * <p>
2035     * It is an error if there are elements of {@code filters}
2036     * (null or not)
2037     * which do not correspond to argument positions in the target.
2038     * <b>Example:</b>
2039     * <p><blockquote><pre>
2040import static java.lang.invoke.MethodHandles.*;
2041import static java.lang.invoke.MethodType.*;
2042...
2043MethodHandle cat = lookup().findVirtual(String.class,
2044  "concat", methodType(String.class, String.class));
2045MethodHandle upcase = lookup().findVirtual(String.class,
2046  "toUpperCase", methodType(String.class));
2047assertEquals("xy", (String) cat.invokeExact("x", "y"));
2048MethodHandle f0 = filterArguments(cat, 0, upcase);
2049assertEquals("Xy", (String) f0.invokeExact("x", "y")); // Xy
2050MethodHandle f1 = filterArguments(cat, 1, upcase);
2051assertEquals("xY", (String) f1.invokeExact("x", "y")); // xY
2052MethodHandle f2 = filterArguments(cat, 0, upcase, upcase);
2053assertEquals("XY", (String) f2.invokeExact("x", "y")); // XY
2054     * </pre></blockquote>
2055     * <p> Here is pseudocode for the resulting adapter:
2056     * <blockquote><pre>
2057     * V target(P... p, A[i]... a[i], B... b);
2058     * A[i] filter[i](V[i]);
2059     * T adapter(P... p, V[i]... v[i], B... b) {
2060     *   return target(p..., f[i](v[i])..., b...);
2061     * }
2062     * </pre></blockquote>
2063     *
2064     * @param target the method handle to invoke after arguments are filtered
2065     * @param pos the position of the first argument to filter
2066     * @param filters method handles to call initially on filtered arguments
2067     * @return method handle which incorporates the specified argument filtering logic
2068     * @throws NullPointerException if the target is null
2069     *                              or if the {@code filters} array is null
2070     * @throws IllegalArgumentException if a non-null element of {@code filters}
2071     *          does not match a corresponding argument type of target as described above,
2072     *          or if the {@code pos+filters.length} is greater than {@code target.type().parameterCount()}
2073     */
2074    public static
2075    MethodHandle filterArguments(MethodHandle target, int pos, MethodHandle... filters) {
2076        MethodType targetType = target.type();
2077        MethodHandle adapter = target;
2078        MethodType adapterType = null;
2079        assert((adapterType = targetType) != null);
2080        int maxPos = targetType.parameterCount();
2081        if (pos + filters.length > maxPos)
2082            throw newIllegalArgumentException("too many filters");
2083        int curPos = pos-1;  // pre-incremented
2084        for (MethodHandle filter : filters) {
2085            curPos += 1;
2086            if (filter == null)  continue;  // ignore null elements of filters
2087            adapter = filterArgument(adapter, curPos, filter);
2088            assert((adapterType = adapterType.changeParameterType(curPos, filter.type().parameterType(0))) != null);
2089        }
2090        assert(adapterType.equals(adapter.type()));
2091        return adapter;
2092    }

2115    /**
2116     * Adapts a target method handle by post-processing
2117     * its return value (if any) with a filter (another method handle).
2118     * The result of the filter is returned from the adapter.
2119     * <p>
2120     * If the target returns a value, the filter must accept that value as
2121     * its only argument.
2122     * If the target returns void, the filter must accept no arguments.
2123     * <p>
2124     * The return type of the filter
2125     * replaces the return type of the target
2126     * in the resulting adapted method handle.
2127     * The argument type of the filter (if any) must be identical to the
2128     * return type of the target.
2129     * <b>Example:</b>
2130     * <p><blockquote><pre>
2131import static java.lang.invoke.MethodHandles.*;
2132import static java.lang.invoke.MethodType.*;
2133...
2134MethodHandle cat = lookup().findVirtual(String.class,
2135  "concat", methodType(String.class, String.class));
2136MethodHandle length = lookup().findVirtual(String.class,
2137  "length", methodType(int.class));
2138System.out.println((String) cat.invokeExact("x", "y")); // xy
2139MethodHandle f0 = filterReturnValue(cat, length);
2140System.out.println((int) f0.invokeExact("x", "y")); // 2
2141     * </pre></blockquote>
2142     * <p> Here is pseudocode for the resulting adapter:
2143     * <blockquote><pre>
2144     * V target(A...);
2145     * T filter(V);
2146     * T adapter(A... a) {
2147     *   V v = target(a...);
2148     *   return filter(v);
2149     * }
2150     * // and if the target has a void return:
2151     * void target2(A...);
2152     * T filter2();
2153     * T adapter2(A... a) {
2154     *   target2(a...);
2155     *   return filter2();
2156     * }
2157     * // and if the filter has a void return:
2158     * V target3(A...);
2159     * void filter3(V);
2160     * void adapter3(A... a) {
2161     *   V v = target3(a...);
2162     *   filter3(v);
2163     * }
2164     * </pre></blockquote>
2165     * @param target the method handle to invoke before filtering the return value
2166     * @param filter method handle to call on the return value
2167     * @return method handle which incorporates the specified return value filtering logic
2168     * @throws NullPointerException if either argument is null
2169     * @throws IllegalArgumentException if the argument list of {@code filter}
2170     *          does not match the return type of target as described above
2171     */
2172    public static
2173    MethodHandle filterReturnValue(MethodHandle target, MethodHandle filter) {
2174        MethodType targetType = target.type();
2175        MethodType filterType = filter.type();
2176        Class<?> rtype = targetType.returnType();
2177        int filterValues = filterType.parameterCount();
2178        if (filterValues == 0
2179                ? (rtype != void.class)
2180                : (rtype != filterType.parameterType(0)))
2181            throw newIllegalArgumentException("target and filter types do not match", target, filter);
2182        // result = fold( lambda(retval, arg...) { filter(retval) },
2183        //                lambda(        arg...) { target(arg...) } )
2184        return MethodHandleImpl.makeCollectArguments(filter, target, 0, false);

2187    /**
2188     * Adapts a target method handle by pre-processing
2189     * some of its arguments, and then calling the target with
2190     * the result of the pre-processing, inserted into the original
2191     * sequence of arguments.
2192     * <p>
2193     * The pre-processing is performed by {@code combiner}, a second method handle.
2194     * Of the arguments passed to the adapter, the first {@code N} arguments
2195     * are copied to the combiner, which is then called.
2196     * (Here, {@code N} is defined as the parameter count of the combiner.)
2197     * After this, control passes to the target, with any result
2198     * from the combiner inserted before the original {@code N} incoming
2199     * arguments.
2200     * <p>
2201     * If the combiner returns a value, the first parameter type of the target
2202     * must be identical with the return type of the combiner, and the next
2203     * {@code N} parameter types of the target must exactly match the parameters
2204     * of the combiner.
2205     * <p>
2206     * If the combiner has a void return, no result will be inserted,
2207     * and the first {@code N} parameter types of the target
2208     * must exactly match the parameters of the combiner.
2209     * <p>
2210     * The resulting adapter is the same type as the target, except that the
2211     * first parameter type is dropped,
2212     * if it corresponds to the result of the combiner.
2213     * <p>
2214     * (Note that {@link #dropArguments(MethodHandle,int,List) dropArguments} can be used to remove any arguments
2215     * that either the combiner or the target does not wish to receive.
2216     * If some of the incoming arguments are destined only for the combiner,
2217     * consider using {@link MethodHandle#asCollector asCollector} instead, since those
2218     * arguments will not need to be live on the stack on entry to the
2219     * target.)
2220     * <b>Example:</b>
2221     * <p><blockquote><pre>
2222import static java.lang.invoke.MethodHandles.*;
2223import static java.lang.invoke.MethodType.*;
2224...
2225MethodHandle trace = publicLookup().findVirtual(java.io.PrintStream.class,
2226  "println", methodType(void.class, String.class))
2227    .bindTo(System.out);
2228MethodHandle cat = lookup().findVirtual(String.class,
2229  "concat", methodType(String.class, String.class));
2230assertEquals("boojum", (String) cat.invokeExact("boo", "jum"));
2231MethodHandle catTrace = foldArguments(cat, trace);
2232// also prints "boo":
2233assertEquals("boojum", (String) catTrace.invokeExact("boo", "jum"));
2234     * </pre></blockquote>
2235     * <p> Here is pseudocode for the resulting adapter:
2236     * <blockquote><pre>
2237     * // there are N arguments in A...
2238     * T target(V, A[N]..., B...);
2239     * V combiner(A...);
2240     * T adapter(A... a, B... b) {
2241     *   V v = combiner(a...);
2242     *   return target(v, a..., b...);
2243     * }
2244     * // and if the combiner has a void return:
2245     * T target2(A[N]..., B...);
2246     * void combiner2(A...);
2247     * T adapter2(A... a, B... b) {
2248     *   combiner2(a...);
2249     *   return target2(a..., b...);
2250     * }
2251     * </pre></blockquote>
2252     * @param target the method handle to invoke after arguments are combined
2253     * @param combiner method handle to call initially on the incoming arguments
2254     * @return method handle which incorporates the specified argument folding logic
2255     * @throws NullPointerException if either argument is null
2256     * @throws IllegalArgumentException if {@code combiner}'s return type
2257     *          is non-void and not the same as the first argument type of
2258     *          the target, or if the initial {@code N} argument types
2259     *          of the target
2260     *          (skipping one matching the {@code combiner}'s return type)
2261     *          are not identical with the argument types of {@code combiner}
2262     */
2263    public static
2264    MethodHandle foldArguments(MethodHandle target, MethodHandle combiner) {
2265        int pos = 0;
2266        MethodType targetType = target.type();
2267        MethodType combinerType = combiner.type();
2268        int foldPos = pos;
2269        int foldArgs = combinerType.parameterCount();
2270        int foldVals = combinerType.returnType() == void.class ? 0 : 1;
2271        int afterInsertPos = foldPos + foldVals;
2272        boolean ok = (targetType.parameterCount() >= afterInsertPos + foldArgs);
2273        if (ok && !(combinerType.parameterList()
2274                    .equals(targetType.parameterList().subList(afterInsertPos,
2275                                                               afterInsertPos + foldArgs))))
2276            ok = false;
2277        if (ok && foldVals != 0 && !combinerType.returnType().equals(targetType.parameterType(0)))
2278            ok = false;
2279        if (!ok)
2280            throw misMatchedTypes("target and combiner types", targetType, combinerType);
2281        MethodType newType = targetType.dropParameterTypes(foldPos, afterInsertPos);
2282        return MethodHandleImpl.makeCollectArguments(target, combiner, foldPos, true);
2283    }

2286     * Makes a method handle which adapts a target method handle,
2287     * by guarding it with a test, a boolean-valued method handle.
2288     * If the guard fails, a fallback handle is called instead.
2289     * All three method handles must have the same corresponding
2290     * argument and return types, except that the return type
2291     * of the test must be boolean, and the test is allowed
2292     * to have fewer arguments than the other two method handles.
2293     * <p> Here is pseudocode for the resulting adapter:
2294     * <blockquote><pre>
2295     * boolean test(A...);
2296     * T target(A...,B...);
2297     * T fallback(A...,B...);
2298     * T adapter(A... a,B... b) {
2299     *   if (test(a...))
2300     *     return target(a..., b...);
2301     *   else
2302     *     return fallback(a..., b...);
2303     * }
2304     * </pre></blockquote>
2305     * Note that the test arguments ({@code a...} in the pseudocode) cannot
2306     * be modified by execution of the test, and so are passed unchanged
2307     * from the caller to the target or fallback as appropriate.
2308     * @param test method handle used for test, must return boolean
2309     * @param target method handle to call if test passes
2310     * @param fallback method handle to call if test fails
2311     * @return method handle which incorporates the specified if/then/else logic
2312     * @throws NullPointerException if any argument is null
2313     * @throws IllegalArgumentException if {@code test} does not return boolean,
2314     *          or if all three method types do not match (with the return
2315     *          type of {@code test} changed to match that of the target).
2316     */
2317    public static
2318    MethodHandle guardWithTest(MethodHandle test,
2319                               MethodHandle target,
2320                               MethodHandle fallback) {
2321        MethodType gtype = test.type();
2322        MethodType ttype = target.type();
2323        MethodType ftype = fallback.type();
2324        if (!ttype.equals(ftype))
2325            throw misMatchedTypes("target and fallback types", ttype, ftype);
2326        if (gtype.returnType() != boolean.class)
2327            throw newIllegalArgumentException("guard type is not a predicate "+gtype);
2328        List<Class<?>> targs = ttype.parameterList();
2329        List<Class<?>> gargs = gtype.parameterList();
2330        if (!targs.equals(gargs)) {
2331            int gpc = gargs.size(), tpc = targs.size();
2332            if (gpc >= tpc || !targs.subList(0, gpc).equals(gargs))
2333                throw misMatchedTypes("target and test types", ttype, gtype);
2334            test = dropArguments(test, gpc, targs.subList(gpc, tpc));
2335            gtype = test.type();
2336        }
2337        return MethodHandleImpl.makeGuardWithTest(test, target, fallback);
2338    }

2344    /**
2345     * Makes a method handle which adapts a target method handle,
2346     * by running it inside an exception handler.
2347     * If the target returns normally, the adapter returns that value.
2348     * If an exception matching the specified type is thrown, the fallback
2349     * handle is called instead on the exception, plus the original arguments.
2350     * <p>
2351     * The target and handler must have the same corresponding
2352     * argument and return types, except that handler may omit trailing arguments
2353     * (similarly to the predicate in {@link #guardWithTest guardWithTest}).
2354     * Also, the handler must have an extra leading parameter of {@code exType} or a supertype.
2355     * <p> Here is pseudocode for the resulting adapter:
2356     * <blockquote><pre>
2357     * T target(A..., B...);
2358     * T handler(ExType, A...);
2359     * T adapter(A... a, B... b) {
2360     *   try {
2361     *     return target(a..., b...);
2362     *   } catch (ExType ex) {
2363     *     return handler(ex, a...);
2364     *   }
2365     * }
2366     * </pre></blockquote>
2367     * Note that the saved arguments ({@code a...} in the pseudocode) cannot
2368     * be modified by execution of the target, and so are passed unchanged
2369     * from the caller to the handler, if the handler is invoked.
2370     * <p>
2371     * The target and handler must return the same type, even if the handler
2372     * always throws.  (This might happen, for instance, because the handler
2373     * is simulating a {@code finally} clause).
2374     * To create such a throwing handler, compose the handler creation logic
2375     * with {@link #throwException throwException},
2376     * in order to create a method handle of the correct return type.
2377     * @param target method handle to call
2378     * @param exType the type of exception which the handler will catch
2379     * @param handler method handle to call if a matching exception is thrown
2380     * @return method handle which incorporates the specified try/catch logic
2381     * @throws NullPointerException if any argument is null
2382     * @throws IllegalArgumentException if {@code handler} does not accept
2383     *          the given exception type, or if the method handle types do
2384     *          not match in their return types and their
2385     *          corresponding parameters
2386     */
2387    public static
2388    MethodHandle catchException(MethodHandle target,
2389                                Class<? extends Throwable> exType,
2390                                MethodHandle handler) {
2391        MethodType ttype = target.type();
2392        MethodType htype = handler.type();
2393        if (htype.parameterCount() < 1 ||
2394            !htype.parameterType(0).isAssignableFrom(exType))
2395            throw newIllegalArgumentException("handler does not accept exception type "+exType);
2396        if (htype.returnType() != ttype.returnType())
2397            throw misMatchedTypes("target and handler return types", ttype, htype);
2398        List<Class<?>> targs = ttype.parameterList();
2399        List<Class<?>> hargs = htype.parameterList();
2400        hargs = hargs.subList(1, hargs.size());  // omit leading parameter from handler
2401        if (!targs.equals(hargs)) {
2402            int hpc = hargs.size(), tpc = targs.size();
2403            if (hpc >= tpc || !targs.subList(0, hpc).equals(hargs))
2404                throw misMatchedTypes("target and handler types", ttype, htype);
2405            handler = dropArguments(handler, 1+hpc, targs.subList(hpc, tpc));
2406            htype = handler.type();
2407        }
2408        return MethodHandleImpl.makeGuardWithCatch(target, exType, handler);
2409    }
