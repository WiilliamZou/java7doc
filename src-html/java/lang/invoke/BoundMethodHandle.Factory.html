047/**
048 * The flavor of method handle which emulates an invoke instruction
049 * on a predetermined argument.  The JVM dispatches to the correct method
050 * when the handle is created, not when it is invoked.
051 *
052 * All bound arguments are encapsulated in dedicated species.
053 */
054/* non-public */ abstract class BoundMethodHandle extends MethodHandle {
055
056    /* non-public */ BoundMethodHandle(MethodType type, LambdaForm form) {
057        super(type, form);
058    }

064    static MethodHandle bindSingle(MethodType type, LambdaForm form, char xtype, Object x) {
065        // for some type signatures, there exist pre-defined concrete BMH classes
066        try {
067            switch (xtype) {
068            case 'L':
069                if (true)  return bindSingle(type, form, x);  // Use known fast path.
070                return (BoundMethodHandle) SpeciesData.EMPTY.extendWithType('L').constructor[0].invokeBasic(type, form, x);
071            case 'I':
072                return (BoundMethodHandle) SpeciesData.EMPTY.extendWithType('I').constructor[0].invokeBasic(type, form, ValueConversions.widenSubword(x));
073            case 'J':
074                return (BoundMethodHandle) SpeciesData.EMPTY.extendWithType('J').constructor[0].invokeBasic(type, form, (long) x);
075            case 'F':
076                return (BoundMethodHandle) SpeciesData.EMPTY.extendWithType('F').constructor[0].invokeBasic(type, form, (float) x);
077            case 'D':
078                return (BoundMethodHandle) SpeciesData.EMPTY.extendWithType('D').constructor[0].invokeBasic(type, form, (double) x);
079            default : throw new InternalError("unexpected xtype: " + xtype);
080            }
081        } catch (Throwable t) {
082            throw newInternalError(t);
083        }
084    }
085
086    static MethodHandle bindSingle(MethodType type, LambdaForm form, Object x) {
087            return new Species_L(type, form, x);
088    }

090    MethodHandle cloneExtend(MethodType type, LambdaForm form, char xtype, Object x) {
091        try {
092            switch (xtype) {
093            case 'L': return cloneExtendL(type, form, x);
094            case 'I': return cloneExtendI(type, form, ValueConversions.widenSubword(x));
095            case 'J': return cloneExtendJ(type, form, (long) x);
096            case 'F': return cloneExtendF(type, form, (float) x);
097            case 'D': return cloneExtendD(type, form, (double) x);
098            }
099        } catch (Throwable t) {
100            throw newInternalError(t);
101        }
102        throw new InternalError("unexpected type: " + xtype);
103    }

105    @Override
106    MethodHandle bindArgument(int pos, char basicType, Object value) {
107        MethodType type = type().dropParameterTypes(pos, pos+1);
108        LambdaForm form = internalForm().bind(1+pos, speciesData());
109        return cloneExtend(type, form, basicType, value);
110    }
111
112    @Override
113    MethodHandle dropArguments(MethodType srcType, int pos, int drops) {
114        LambdaForm form = internalForm().addArguments(pos, srcType.parameterList().subList(pos, pos+drops));
115        try {
116             return clone(srcType, form);
117         } catch (Throwable t) {
118             throw newInternalError(t);
119         }
120    }
121
122    @Override
123    MethodHandle permuteArguments(MethodType newType, int[] reorder) {
124        try {
125             return clone(newType, form.permuteArguments(1, reorder, basicTypes(newType.parameterList())));
126         } catch (Throwable t) {
127             throw newInternalError(t);
128         }
129    }

160        try {

332    static class SpeciesData {
333        final String                             types;
334        final Class<? extends BoundMethodHandle> clazz;
335        // Bootstrapping requires circular relations MH -> BMH -> SpeciesData -> MH
336        // Therefore, we need a non-final link in the chain.  Use array elements.
337        final MethodHandle[]                     constructor;
338        final MethodHandle[]                     getters;
339        final SpeciesData[]                      extensions;
340
341        /*non-public*/ int fieldCount() {
342            return types.length();
343        }
344        /*non-public*/ char fieldType(int i) {
345            return types.charAt(i);
346        }
347
348        public String toString() {
349            return "SpeciesData["+(isPlaceholder() ? "<placeholder>" : clazz.getSimpleName())+":"+types+"]";
350        }

352        /**
353         * Return a {@link LambdaForm.Name} containing a {@link LambdaForm.NamedFunction} that
354         * represents a MH bound to a generic invoker, which in turn forwards to the corresponding
355         * getter.
356         */
357        Name getterName(Name mhName, int i) {
358            MethodHandle mh = getters[i];
359            assert(mh != null) : this+"."+i;
360            return new Name(mh, mhName);
361        }

363        static final SpeciesData EMPTY = new SpeciesData("", BoundMethodHandle.class);
364
365        private SpeciesData(String types, Class<? extends BoundMethodHandle> clazz) {
366            this.types = types;
367            this.clazz = clazz;
368            if (!INIT_DONE) {
369                this.constructor = new MethodHandle[1];
370                this.getters = new MethodHandle[types.length()];
371            } else {
372                this.constructor = Factory.makeCtors(clazz, types, null);
373                this.getters = Factory.makeGetters(clazz, types, null);
374            }
375            this.extensions = new SpeciesData[EXTENSION_TYPES.length()];
376        }
377
378        private void initForBootstrap() {
379            assert(!INIT_DONE);
380            if (constructor[0] == null) {
381                Factory.makeCtors(clazz, types, this.constructor);
382                Factory.makeGetters(clazz, types, this.getters);
383            }
384        }

386        private SpeciesData(String types) {
387            // Placeholder only.
388            this.types = types;
389            this.clazz = null;
390            this.constructor = null;
391            this.getters = null;
392            this.extensions = null;
393        }
394        private boolean isPlaceholder() { return clazz == null; }

396        private static final HashMap<String, SpeciesData> CACHE = new HashMap<>();
397        private static final boolean INIT_DONE;  // set after <clinit> finishes...

399        SpeciesData extendWithType(char type) {
400            int i = extensionIndex(type);
401            SpeciesData d = extensions[i];
402            if (d != null)  return d;
403            extensions[i] = d = get(types+type);
404            return d;
405        }
406
407        SpeciesData extendWithIndex(byte index) {
408            SpeciesData d = extensions[index];
409            if (d != null)  return d;
410            extensions[index] = d = get(types+EXTENSION_TYPES.charAt(index));
411            return d;
412        }

414        private static SpeciesData get(String types) {
415            // Acquire cache lock for query.
416            SpeciesData d = lookupCache(types);
417            if (!d.isPlaceholder())
418                return d;
419            synchronized (d) {
420                // Use synch. on the placeholder to prevent multiple instantiation of one species.
421                // Creating this class forces a recursive call to getForClass.
422                if (lookupCache(types).isPlaceholder())
423                    Factory.generateConcreteBMHClass(types);
424            }
425            // Reacquire cache lock.
426            d = lookupCache(types);
427            // Class loading must have upgraded the cache.
428            assert(d != null && !d.isPlaceholder());
429            return d;
430        }
431        static SpeciesData getForClass(String types, Class<? extends BoundMethodHandle> clazz) {
432            // clazz is a new class which is initializing its SPECIES_DATA field
433            return updateCache(types, new SpeciesData(types, clazz));
434        }
435        private static synchronized SpeciesData lookupCache(String types) {
436            SpeciesData d = CACHE.get(types);
437            if (d != null)  return d;
438            d = new SpeciesData(types);
439            assert(d.isPlaceholder());
440            CACHE.put(types, d);
441            return d;
442        }
443        private static synchronized SpeciesData updateCache(String types, SpeciesData d) {
444            SpeciesData d2;
445            assert((d2 = CACHE.get(types)) == null || d2.isPlaceholder());
446            assert(!d.isPlaceholder());
447            CACHE.put(types, d);
448            return d;
449        }

451        static {
452            // pre-fill the BMH speciesdata cache with BMH's inner classes
453            final Class<BoundMethodHandle> rootCls = BoundMethodHandle.class;
454            SpeciesData d0 = BoundMethodHandle.SPECIES_DATA;  // trigger class init
455            assert(d0 == null || d0 == lookupCache("")) : d0;
456            try {
457                for (Class<?> c : rootCls.getDeclaredClasses()) {
458                    if (rootCls.isAssignableFrom(c)) {
459                        final Class<? extends BoundMethodHandle> cbmh = c.asSubclass(BoundMethodHandle.class);
460                        SpeciesData d = Factory.speciesDataFromConcreteBMHClass(cbmh);
461                        assert(d != null) : cbmh.getName();
462                        assert(d.clazz == cbmh);
463                        assert(d == lookupCache(d.types));
464                    }
465                }
466            } catch (Throwable e) {
467                throw newInternalError(e);
468            }
469
470            for (SpeciesData d : CACHE.values()) {
471                d.initForBootstrap();
472            }
473            // Note:  Do not simplify this, because INIT_DONE must not be
474            // a compile-time constant during bootstrapping.
475            INIT_DONE = Boolean.TRUE;
476        }

521        /**
522         * Generate a concrete subclass of BMH for a given combination of bound types.
523         *
524         * A concrete BMH species adheres to the following schema:
525         *
526         * <pre>
527         * class Species_<<types>> extends BoundMethodHandle {
528         *     <<fields>>
529         *     final SpeciesData speciesData() { return SpeciesData.get("<<types>>"); }
530         * }
531         * </pre>
532         *
533         * The {@code <<types>>} signature is precisely the string that is passed to this
534         * method.
535         *
536         * The {@code <<fields>>} section consists of one field definition per character in
537         * the type signature, adhering to the naming schema described in the definition of
538         * {@link #makeFieldName()}.
539         *
540         * For example, a concrete BMH species for two reference and one integral bound values
541         * would have the following shape:
542         *
543         * <pre>
544         * class BoundMethodHandle { ... private static
545         * final class Species_LLI extends BoundMethodHandle {
546         *     final Object argL0;
547         *     final Object argL1;
548         *     final int argI2;
549         *     Species_LLI(MethodType mt, LambdaForm lf, Object argL0, Object argL1, int argI2) {
550         *         super(mt, lf);
551         *         this.argL0 = argL0;
552         *         this.argL1 = argL1;
553         *         this.argI2 = argI2;
554         *     }
555         *     final SpeciesData speciesData() { return SPECIES_DATA; }
556         *     static final SpeciesData SPECIES_DATA = SpeciesData.getForClass("LLI", Species_LLI.class);
557         *     final BoundMethodHandle clone(MethodType mt, LambdaForm lf) {
558         *         return SPECIES_DATA.constructor[0].invokeBasic(mt, lf, argL0, argL1, argI2);
559         *     }
560         *     final BoundMethodHandle cloneExtendL(MethodType mt, LambdaForm lf, Object narg) {
561         *         return SPECIES_DATA.extendWithIndex(INDEX_L).constructor[0].invokeBasic(mt, lf, argL0, argL1, argI2, narg);
562         *     }
563         *     final BoundMethodHandle cloneExtendI(MethodType mt, LambdaForm lf, int narg) {
564         *         return SPECIES_DATA.extendWithIndex(INDEX_I).constructor[0].invokeBasic(mt, lf, argL0, argL1, argI2, narg);
565         *     }
566         *     final BoundMethodHandle cloneExtendJ(MethodType mt, LambdaForm lf, long narg) {
567         *         return SPECIES_DATA.extendWithIndex(INDEX_J).constructor[0].invokeBasic(mt, lf, argL0, argL1, argI2, narg);
568         *     }
569         *     final BoundMethodHandle cloneExtendF(MethodType mt, LambdaForm lf, float narg) {
570         *         return SPECIES_DATA.extendWithIndex(INDEX_F).constructor[0].invokeBasic(mt, lf, argL0, argL1, argI2, narg);
571         *     }
572         *     final BoundMethodHandle cloneExtendD(MethodType mt, LambdaForm lf, double narg) {
573         *         return SPECIES_DATA.extendWithIndex(INDEX_D).constructor[0].invokeBasic(mt, lf, argL0, argL1, argI2, narg);
574         *     }
575         * }
576         * </pre>
577         *
578         * @param types the type signature, wherein reference types are erased to 'L'
579         * @return the generated concrete BMH class
580         */
581        static Class<? extends BoundMethodHandle> generateConcreteBMHClass(String types) {
582            final ClassWriter cw = new ClassWriter(ClassWriter.COMPUTE_MAXS + ClassWriter.COMPUTE_FRAMES);
583
584            final String className  = SPECIES_PREFIX_PATH + types;
585            final String sourceFile = SPECIES_PREFIX_NAME + types;
586            final int NOT_ACC_PUBLIC = 0;  // not ACC_PUBLIC
587            cw.visit(V1_6, NOT_ACC_PUBLIC + ACC_FINAL + ACC_SUPER, className, null, BMH, null);
588            cw.visitSource(sourceFile, null);
589
590            // emit static types and SPECIES_DATA fields
591            cw.visitField(NOT_ACC_PUBLIC + ACC_STATIC, "SPECIES_DATA", SPECIES_DATA_SIG, null, null).visitEnd();
592
593            // emit bound argument fields
594            for (int i = 0; i < types.length(); ++i) {
595                final char t = types.charAt(i);
596                final String fieldName = makeFieldName(types, i);
597                final String fieldDesc = t == 'L' ? JLO_SIG : String.valueOf(t);
598                cw.visitField(ACC_FINAL, fieldName, fieldDesc, null, null).visitEnd();
599            }
600
601            MethodVisitor mv;
602
603            // emit constructor
604            mv = cw.visitMethod(NOT_ACC_PUBLIC, "<init>", makeSignature(types, true), null, null);
605            mv.visitCode();
606            mv.visitVarInsn(ALOAD, 0);
607            mv.visitVarInsn(ALOAD, 1);
608            mv.visitVarInsn(ALOAD, 2);
609
610            mv.visitMethodInsn(INVOKESPECIAL, BMH, "<init>", makeSignature("", true));
611
612            for (int i = 0, j = 0; i < types.length(); ++i, ++j) {
613                // i counts the arguments, j counts corresponding argument slots
614                char t = types.charAt(i);
615                mv.visitVarInsn(ALOAD, 0);
616                mv.visitVarInsn(typeLoadOp(t), j + 3); // parameters start at 3
617                mv.visitFieldInsn(PUTFIELD, className, makeFieldName(types, i), typeSig(t));
618                if (t == 'J' || t == 'D') {
619                    ++j; // adjust argument register access
620                }
621            }
622
623            mv.visitInsn(RETURN);
624            mv.visitMaxs(0, 0);
625            mv.visitEnd();
626
627            // emit implementation of reinvokerTarget()
628            mv = cw.visitMethod(NOT_ACC_PUBLIC + ACC_FINAL, "reinvokerTarget", "()" + MH_SIG, null, null);
629            mv.visitCode();
630            mv.visitVarInsn(ALOAD, 0);
631            mv.visitFieldInsn(GETFIELD, className, "argL0", JLO_SIG);
632            mv.visitTypeInsn(CHECKCAST, MH);
633            mv.visitInsn(ARETURN);
634            mv.visitMaxs(0, 0);
635            mv.visitEnd();
636
637            // emit implementation of speciesData()
638            mv = cw.visitMethod(NOT_ACC_PUBLIC + ACC_FINAL, "speciesData", MYSPECIES_DATA_SIG, null, null);
639            mv.visitCode();
640            mv.visitFieldInsn(GETSTATIC, className, "SPECIES_DATA", SPECIES_DATA_SIG);
641            mv.visitInsn(ARETURN);
642            mv.visitMaxs(0, 0);
643            mv.visitEnd();
644
645            // emit clone()
646            mv = cw.visitMethod(NOT_ACC_PUBLIC + ACC_FINAL, "clone", makeSignature("", false), null, E_THROWABLE);
647            mv.visitCode();
648            // return speciesData().constructor[0].invokeBasic(mt, lf, argL0, ...)
649            // obtain constructor
650            mv.visitVarInsn(ALOAD, 0);
651            mv.visitFieldInsn(GETSTATIC, className, "SPECIES_DATA", SPECIES_DATA_SIG);
652            mv.visitFieldInsn(GETFIELD, SPECIES_DATA, "constructor", "[" + MH_SIG);
653            mv.visitInsn(ICONST_0);
654            mv.visitInsn(AALOAD);
655            // load mt, lf
656            mv.visitVarInsn(ALOAD, 1);
657            mv.visitVarInsn(ALOAD, 2);
658            // put fields on the stack
659            emitPushFields(types, className, mv);
660            // finally, invoke the constructor and return
661            mv.visitMethodInsn(INVOKEVIRTUAL, MH, "invokeBasic", makeSignature(types, false));
662            mv.visitInsn(ARETURN);
663            mv.visitMaxs(0, 0);
664            mv.visitEnd();
665
666            // for each type, emit cloneExtendT()
667            for (Class<?> c : TYPES) {
668                char t = Wrapper.basicTypeChar(c);
669                mv = cw.visitMethod(NOT_ACC_PUBLIC + ACC_FINAL, "cloneExtend" + t, makeSignature(String.valueOf(t), false), null, E_THROWABLE);
670                mv.visitCode();
671                // return SPECIES_DATA.extendWithIndex(extensionIndex(t)).constructor[0].invokeBasic(mt, lf, argL0, ..., narg)
672                // obtain constructor
673                mv.visitFieldInsn(GETSTATIC, className, "SPECIES_DATA", SPECIES_DATA_SIG);
674                int iconstInsn = ICONST_0 + extensionIndex(t);
675                assert(iconstInsn <= ICONST_5);
676                mv.visitInsn(iconstInsn);
677                mv.visitMethodInsn(INVOKEVIRTUAL, SPECIES_DATA, "extendWithIndex", BMHSPECIES_DATA_EWI_SIG);
678                mv.visitFieldInsn(GETFIELD, SPECIES_DATA, "constructor", "[" + MH_SIG);
679                mv.visitInsn(ICONST_0);
680                mv.visitInsn(AALOAD);
681                // load mt, lf
682                mv.visitVarInsn(ALOAD, 1);
683                mv.visitVarInsn(ALOAD, 2);
684                // put fields on the stack
685                emitPushFields(types, className, mv);
686                // put narg on stack
687                mv.visitVarInsn(typeLoadOp(t), 3);
688                // finally, invoke the constructor and return
689                mv.visitMethodInsn(INVOKEVIRTUAL, MH, "invokeBasic", makeSignature(types + t, false));
690                mv.visitInsn(ARETURN);
691                mv.visitMaxs(0, 0);
692                mv.visitEnd();
693            }
694
695            // emit class initializer
696            mv = cw.visitMethod(NOT_ACC_PUBLIC | ACC_STATIC, "<clinit>", VOID_SIG, null, null);
697            mv.visitCode();
698            mv.visitLdcInsn(types);
699            mv.visitLdcInsn(Type.getObjectType(className));
700            mv.visitMethodInsn(INVOKESTATIC, SPECIES_DATA, "getForClass", BMHSPECIES_DATA_GFC_SIG);
701            mv.visitFieldInsn(PUTSTATIC, className, "SPECIES_DATA", SPECIES_DATA_SIG);
702            mv.visitInsn(RETURN);
703            mv.visitMaxs(0, 0);
704            mv.visitEnd();
705
706            cw.visitEnd();
707
708            // load class
709            final byte[] classFile = cw.toByteArray();
710            InvokerBytecodeGenerator.maybeDump(className, classFile);
711            Class<? extends BoundMethodHandle> bmhClass =
712                //UNSAFE.defineAnonymousClass(BoundMethodHandle.class, classFile, null).asSubclass(BoundMethodHandle.class);
713                UNSAFE.defineClass(className, classFile, 0, classFile.length).asSubclass(BoundMethodHandle.class);
714            UNSAFE.ensureClassInitialized(bmhClass);
715
716            return bmhClass;
717        }

719        private static int typeLoadOp(char t) {
720            switch (t) {
721            case 'L': return ALOAD;
722            case 'I': return ILOAD;
723            case 'J': return LLOAD;
724            case 'F': return FLOAD;
725            case 'D': return DLOAD;
726            default : throw new InternalError("unrecognized type " + t);
727            }
728        }
729
730        private static void emitPushFields(String types, String className, MethodVisitor mv) {
731            for (int i = 0; i < types.length(); ++i) {
732                char tc = types.charAt(i);
733                mv.visitVarInsn(ALOAD, 0);
734                mv.visitFieldInsn(GETFIELD, className, makeFieldName(types, i), typeSig(tc));
735            }
736        }
737
738        static String typeSig(char t) {
739            return t == 'L' ? JLO_SIG : String.valueOf(t);
740        }

746        private static MethodHandle makeGetter(Class<?> cbmhClass, String types, int index) {
747            String fieldName = makeFieldName(types, index);
748            Class<?> fieldType = Wrapper.forBasicType(types.charAt(index)).primitiveType();
749            try {
750                return LOOKUP.findGetter(cbmhClass, fieldName, fieldType);
751            } catch (NoSuchFieldException | IllegalAccessException e) {
752                throw newInternalError(e);
753            }
754        }

756        static MethodHandle[] makeGetters(Class<?> cbmhClass, String types, MethodHandle[] mhs) {
757            if (mhs == null)  mhs = new MethodHandle[types.length()];
758            for (int i = 0; i < mhs.length; ++i) {
759                mhs[i] = makeGetter(cbmhClass, types, i);
760                assert(mhs[i].internalMemberName().getDeclaringClass() == cbmhClass);
761            }
762            return mhs;
763        }

765        static MethodHandle[] makeCtors(Class<? extends BoundMethodHandle> cbmh, String types, MethodHandle mhs[]) {
766            if (mhs == null)  mhs = new MethodHandle[1];
767            mhs[0] = makeCbmhCtor(cbmh, types);
768            return mhs;
769        }

784        /**
785         * Field names in concrete BMHs adhere to this pattern:
786         * arg + type + index
787         * where type is a single character (L, I, J, F, D).
788         */
789        private static String makeFieldName(String types, int index) {
790            assert index >= 0 && index < types.length();
791            return "arg" + types.charAt(index) + index;
792        }

794        private static String makeSignature(String types, boolean ctor) {
795            StringBuilder buf = new StringBuilder(SIG_INCIPIT);
796            for (char c : types.toCharArray()) {
797                buf.append(typeSig(c));
798            }
799            return buf.append(')').append(ctor ? "V" : BMH_SIG).toString();
800        }

802        static MethodHandle makeCbmhCtor(Class<? extends BoundMethodHandle> cbmh, String types) {
803            try {
804                return linkConstructor(LOOKUP.findConstructor(cbmh, MethodType.fromMethodDescriptorString(makeSignature(types, true), null)));
805            } catch (NoSuchMethodException | IllegalAccessException | IllegalArgumentException | TypeNotPresentException e) {
806                throw newInternalError(e);
807            }
808        }

810        /**
811         * Wrap a constructor call in a {@link LambdaForm}.
812         *
813         * If constructors ({@code <init>} methods) are called in LFs, problems might arise if the LFs
814         * are turned into bytecode, because the call to the allocator is routed through an MH, and the
815         * verifier cannot find a {@code NEW} instruction preceding the {@code INVOKESPECIAL} to
816         * {@code <init>}. To avoid this, we add an indirection by invoking {@code <init>} through
817         * {@link MethodHandle#linkToSpecial}.
818         *
819         * The last {@link LambdaForm#Name Name} in the argument's form is expected to be the {@code void}
820         * result of the {@code <init>} invocation. This entry is replaced.
821         */
822        private static MethodHandle linkConstructor(MethodHandle cmh) {
823            final LambdaForm lf = cmh.form;
824            final int initNameIndex = lf.names.length - 1;
825            final Name initName = lf.names[initNameIndex];
826            final MemberName ctorMN = initName.function.member;
827            final MethodType ctorMT = ctorMN.getInvocationType();
828
829            // obtain function member (call target)
830            // linker method type replaces initial parameter (BMH species) with BMH to avoid naming a species (anonymous class!)
831            final MethodType linkerMT = ctorMT.changeParameterType(0, BoundMethodHandle.class).appendParameterTypes(MemberName.class);
832            MemberName linkerMN = new MemberName(MethodHandle.class, "linkToSpecial", linkerMT, REF_invokeStatic);
833            try {
834                linkerMN = MemberName.getFactory().resolveOrFail(REF_invokeStatic, linkerMN, null, NoSuchMethodException.class);
835                assert(linkerMN.isStatic());
836            } catch (ReflectiveOperationException ex) {
837                throw newInternalError(ex);
838            }
839            // extend arguments array
840            Object[] newArgs = Arrays.copyOf(initName.arguments, initName.arguments.length + 1);
841            newArgs[newArgs.length - 1] = ctorMN;
842            // replace function
843            final NamedFunction nf = new NamedFunction(linkerMN);
844            final Name linkedCtor = new Name(nf, newArgs);
845            linkedCtor.initIndex(initNameIndex);
846            lf.names[initNameIndex] = linkedCtor;
847            return cmh;
848        }
