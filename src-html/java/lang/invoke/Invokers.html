066    /** Compute and cache information common to all collecting adapters
067     *  that implement members of the erasure-family of the given erased type.
068     */
069    /*non-public*/ Invokers(MethodType targetType) {
070        this.targetType = targetType;
071        this.spreadInvokers = new MethodHandle[targetType.parameterCount()+1];
072    }

074    /*non-public*/ MethodHandle exactInvoker() {
075        MethodHandle invoker = exactInvoker;
076        if (invoker != null)  return invoker;
077        MethodType mtype = targetType;
078        MethodType invokerType = mtype.invokerType();
079        LambdaForm lform;
080        final int MTYPE_ARG_APPENDED = 1;  // argument count for appended mtype value
081        if (mtype.parameterSlotCount() <= MethodType.MAX_MH_INVOKER_ARITY - MTYPE_ARG_APPENDED) {
082            lform = invokeForm(mtype, false, MethodTypeForm.LF_EX_INVOKER);
083            invoker = BoundMethodHandle.bindSingle(invokerType, lform, mtype);
084        } else {
085            // At maximum arity, we cannot afford an extra mtype argument,
086            // so build a fully customized (non-cached) invoker form.
087            lform = invokeForm(mtype, true, MethodTypeForm.LF_EX_INVOKER);
088            invoker = SimpleMethodHandle.make(invokerType, lform);
089        }
090        assert(checkInvoker(invoker));
091        exactInvoker = invoker;
092        return invoker;
093    }

095    /*non-public*/ MethodHandle generalInvoker() {
096        MethodHandle invoker = generalInvoker;
097        if (invoker != null)  return invoker;
098        MethodType mtype = targetType;
099        MethodType invokerType = mtype.invokerType();
100        LambdaForm lform;
101        final int MTYPE_ARG_APPENDED = 1;  // argument count for appended mtype value
102        assert(GENERIC_INVOKER_SLOP >= MTYPE_ARG_APPENDED);
103        if (mtype.parameterSlotCount() <= MethodType.MAX_MH_INVOKER_ARITY - GENERIC_INVOKER_SLOP) {
104            prepareForGenericCall(mtype);
105            lform = invokeForm(mtype, false, MethodTypeForm.LF_GEN_INVOKER);
106            invoker = BoundMethodHandle.bindSingle(invokerType, lform, mtype);
107        } else {
108            // At maximum arity, we cannot afford an extra mtype argument,
109            // so build a fully customized (non-cached) invoker form.
110            lform = invokeForm(mtype, true, MethodTypeForm.LF_GEN_INVOKER);
111            invoker = SimpleMethodHandle.make(invokerType, lform);
112        }
113        assert(checkInvoker(invoker));
114        generalInvoker = invoker;
115        return invoker;
116    }

118    /*non-public*/ MethodHandle makeBasicInvoker() {
119        MethodHandle invoker = DirectMethodHandle.make(invokeBasicMethod(targetType));
120        assert(targetType == targetType.basicType());
121        // Note:  This is not cached here.  It is cached by the calling MethodTypeForm.
122        assert(checkInvoker(invoker));
123        return invoker;
124    }

126    static MemberName invokeBasicMethod(MethodType type) {
127        type = type.basicType();
128        String name = "invokeBasic";
129        try {
130            //Lookup.findVirtual(MethodHandle.class, name, type);
131            return IMPL_LOOKUP.resolveOrFail(REF_invokeVirtual, MethodHandle.class, name, type);
132        } catch (ReflectiveOperationException ex) {
133            throw newInternalError("JVM cannot find invoker for "+type, ex);
134        }
135    }

137    private boolean checkInvoker(MethodHandle invoker) {
138        assert(targetType.invokerType().equals(invoker.type()))
139                : java.util.Arrays.asList(targetType, targetType.invokerType(), invoker);
140        assert(invoker.internalMemberName() == null ||
141               invoker.internalMemberName().getMethodType().equals(targetType));
142        assert(!invoker.isVarargsCollector());
143        return true;
144    }

157    /*non-public*/ MethodHandle spreadInvoker(int leadingArgCount) {
158        MethodHandle vaInvoker = spreadInvokers[leadingArgCount];
159        if (vaInvoker != null)  return vaInvoker;
160        int spreadArgCount = targetType.parameterCount() - leadingArgCount;
161        MethodType spreadInvokerType = targetType
162            .replaceParameterTypes(leadingArgCount, targetType.parameterCount(), Object[].class);
163        if (targetType.parameterSlotCount() <= MethodType.MAX_MH_INVOKER_ARITY) {
164            // Factor sinvoker.invoke(mh, a) into ginvoker.asSpreader().invoke(mh, a)
165            // where ginvoker.invoke(mh, a*) => mh.invoke(a*).
166            MethodHandle genInvoker = generalInvoker();
167            vaInvoker = genInvoker.asSpreader(Object[].class, spreadArgCount);
168        } else {
169            // Cannot build a general invoker here of type ginvoker.invoke(mh, a*[254]).
170            // Instead, factor sinvoker.invoke(mh, a) into ainvoker.invoke(filter(mh), a)
171            // where filter(mh) == mh.asSpreader(Object[], spreadArgCount)
172            MethodHandle arrayInvoker = MethodHandles.exactInvoker(spreadInvokerType);
173            MethodHandle makeSpreader;
174            try {
175                makeSpreader = IMPL_LOOKUP
176                    .findVirtual(MethodHandle.class, "asSpreader",
177                        MethodType.methodType(MethodHandle.class, Class.class, int.class));
178            } catch (ReflectiveOperationException ex) {
179                throw newInternalError(ex);
180            }
181            makeSpreader = MethodHandles.insertArguments(makeSpreader, 1, Object[].class, spreadArgCount);
182            vaInvoker = MethodHandles.filterArgument(arrayInvoker, 0, makeSpreader);
183        }
184        assert(vaInvoker.type().equals(spreadInvokerType.invokerType()));
185        spreadInvokers[leadingArgCount] = vaInvoker;
186        return vaInvoker;
187    }

189    /*non-public*/ MethodHandle varargsInvoker() {
190        MethodHandle vaInvoker = varargsInvoker;
191        if (vaInvoker != null)  return vaInvoker;
192        vaInvoker = spreadInvoker(0).asType(MethodType.genericMethodType(0, true).invokerType());
193        varargsInvoker = vaInvoker;
194        return vaInvoker;
195    }

199    /*non-public*/ MethodHandle uninitializedCallSite() {
200        MethodHandle invoker = uninitializedCallSite;
201        if (invoker != null)  return invoker;
202        if (targetType.parameterCount() > 0) {
203            MethodType type0 = targetType.dropParameterTypes(0, targetType.parameterCount());
204            Invokers invokers0 = type0.invokers();
205            invoker = MethodHandles.dropArguments(invokers0.uninitializedCallSite(),
206                                                  0, targetType.parameterList());
207            assert(invoker.type().equals(targetType));
208            uninitializedCallSite = invoker;
209            return invoker;
210        }
211        invoker = THROW_UCS;
212        if (invoker == null) {
213            try {
214                THROW_UCS = invoker = IMPL_LOOKUP
215                    .findStatic(CallSite.class, "uninitializedCallSite",
216                                MethodType.methodType(Empty.class));
217            } catch (ReflectiveOperationException ex) {
218                throw newInternalError(ex);
219            }
220        }
221        invoker = MethodHandles.explicitCastArguments(invoker, MethodType.methodType(targetType.returnType()));
222        invoker = invoker.dropArguments(targetType, 0, targetType.parameterCount());
223        assert(invoker.type().equals(targetType));
224        uninitializedCallSite = invoker;
225        return invoker;
226    }

232    static MemberName exactInvokerMethod(MethodType mtype, Object[] appendixResult) {
233        LambdaForm lform;
234        final int MTYPE_ARG_APPENDED = 1;  // argument count for appended mtype value
235        if (mtype.parameterSlotCount() <= MethodType.MAX_MH_ARITY - MTYPE_ARG_APPENDED) {
236            lform = invokeForm(mtype, false, MethodTypeForm.LF_EX_LINKER);
237            appendixResult[0] = mtype;
238        } else {
239            lform = invokeForm(mtype, true, MethodTypeForm.LF_EX_LINKER);
240        }
241        return lform.vmentry;
242    }
243
244    static MemberName genericInvokerMethod(MethodType mtype, Object[] appendixResult) {
245        LambdaForm lform;
246        final int MTYPE_ARG_APPENDED = 1;  // argument count for appended mtype value
247        if (mtype.parameterSlotCount() <= MethodType.MAX_MH_ARITY - (MTYPE_ARG_APPENDED + GENERIC_INVOKER_SLOP)) {
248            lform = invokeForm(mtype, false, MethodTypeForm.LF_GEN_LINKER);
249            appendixResult[0] = mtype;
250            prepareForGenericCall(mtype);
251        } else {
252            lform = invokeForm(mtype, true, MethodTypeForm.LF_GEN_LINKER);
253        }
254        return lform.vmentry;
255    }

257    private static LambdaForm invokeForm(MethodType mtype, boolean customized, int which) {
258        boolean isCached;
259        if (!customized) {
260            mtype = mtype.basicType();  // normalize Z to I, String to Object, etc.
261            isCached = true;
262        } else {
263            isCached = false;  // maybe cache if mtype == mtype.basicType()
264        }
265        boolean isLinker, isGeneric;
266        String debugName;
267        switch (which) {
268        case MethodTypeForm.LF_EX_LINKER:   isLinker = true;  isGeneric = false; debugName = "invokeExact_MT"; break;
269        case MethodTypeForm.LF_EX_INVOKER:  isLinker = false; isGeneric = false; debugName = "exactInvoker"; break;
270        case MethodTypeForm.LF_GEN_LINKER:  isLinker = true;  isGeneric = true;  debugName = "invoke_MT"; break;
271        case MethodTypeForm.LF_GEN_INVOKER: isLinker = false; isGeneric = true;  debugName = "invoker"; break;
272        default: throw new InternalError();
273        }
274        LambdaForm lform;
275        if (isCached) {
276            lform = mtype.form().cachedLambdaForm(which);
277            if (lform != null)  return lform;
278        }
279        // exactInvokerForm (Object,Object)Object
280        //   link with java.lang.invoke.MethodHandle.invokeBasic(MethodHandle,Object,Object)Object/invokeSpecial
281        final int THIS_MH      = 0;
282        final int CALL_MH      = THIS_MH + (isLinker ? 0 : 1);
283        final int ARG_BASE     = CALL_MH + 1;
284        final int OUTARG_LIMIT = ARG_BASE + mtype.parameterCount();
285        final int INARG_LIMIT  = OUTARG_LIMIT + (isLinker && !customized ? 1 : 0);
286        int nameCursor = OUTARG_LIMIT;
287        final int MTYPE_ARG    = customized ? -1 : nameCursor++;  // might be last in-argument
288        final int CHECK_TYPE   = nameCursor++;
289        final int LINKER_CALL  = nameCursor++;
290        MethodType invokerFormType = mtype.invokerType();
291        if (isLinker) {
292            if (!customized)
293                invokerFormType = invokerFormType.appendParameterTypes(MemberName.class);
294        } else {
295            invokerFormType = invokerFormType.invokerType();
296        }
297        Name[] names = arguments(nameCursor - INARG_LIMIT, invokerFormType);
298        assert(names.length == nameCursor)
299                : Arrays.asList(mtype, customized, which, nameCursor, names.length);
300        if (MTYPE_ARG >= INARG_LIMIT) {
301            assert(names[MTYPE_ARG] == null);
302            names[MTYPE_ARG] = BoundMethodHandle.getSpeciesData("L").getterName(names[THIS_MH], 0);
303            // else if isLinker, then MTYPE is passed in from the caller (e.g., the JVM)
304        }
305
306        // Make the final call.  If isGeneric, then prepend the result of type checking.
307        MethodType outCallType;
308        Object[] outArgs;
309        Object mtypeArg = (customized ? mtype : names[MTYPE_ARG]);
310        if (!isGeneric) {
311            names[CHECK_TYPE] = new Name(NF_checkExactType, names[CALL_MH], mtypeArg);
312            // mh.invokeExact(a*):R => checkExactType(mh, TYPEOF(a*:R)); mh.invokeBasic(a*)
313            outArgs = Arrays.copyOfRange(names, CALL_MH, OUTARG_LIMIT, Object[].class);
314            outCallType = mtype;
315        } else if (customized) {
316            names[CHECK_TYPE] = new Name(NF_asType, names[CALL_MH], mtypeArg);
317            // mh.invokeGeneric(a*):R =>
318            //  let mt=TYPEOF(a*:R), tmh=asType(mh, mt);
319            //    tmh.invokeBasic(a*)
320            outArgs = Arrays.copyOfRange(names, CALL_MH, OUTARG_LIMIT, Object[].class);
321            outArgs[0] = names[CHECK_TYPE];
322            outCallType = mtype;
323        } else {
324            names[CHECK_TYPE] = new Name(NF_checkGenericType, names[CALL_MH], mtypeArg);
325            // mh.invokeGeneric(a*):R =>
326            //  let mt=TYPEOF(a*:R), gamh=checkGenericType(mh, mt);
327            //    gamh.invokeBasic(mt, mh, a*)
328            final int PREPEND_GAMH = 0, PREPEND_MT = 1, PREPEND_COUNT = 2;
329            assert(GENERIC_INVOKER_SLOP == PREPEND_COUNT);
330            outArgs = Arrays.copyOfRange(names, CALL_MH, OUTARG_LIMIT + PREPEND_COUNT, Object[].class);
331            // prepend arguments:
332            System.arraycopy(outArgs, 0, outArgs, PREPEND_COUNT, outArgs.length - PREPEND_COUNT);
333            outArgs[PREPEND_GAMH] = names[CHECK_TYPE];
334            outArgs[PREPEND_MT] = mtypeArg;
335            outCallType = mtype.insertParameterTypes(0, MethodType.class, MethodHandle.class);
336        }
337        names[LINKER_CALL] = new Name(invokeBasicMethod(outCallType), outArgs);
338        lform = new LambdaForm(debugName, INARG_LIMIT, names);
339        if (isLinker)
340            lform.compileToBytecode();  // JVM needs a real methodOop
341        if (isCached)
342            lform = mtype.form().setCachedLambdaForm(which, lform);
343        return lform;
344    }

354    /*non-public*/ static
355    @ForceInline
356    void checkExactType(Object mhObj, Object expectedObj) {
357        MethodHandle mh = (MethodHandle) mhObj;
358        MethodType expected = (MethodType) expectedObj;
359        MethodType actual = mh.type();
360        if (actual != expected)
361            throw newWrongMethodTypeException(expected, actual);
362    }

364    /** Static definition of MethodHandle.invokeGeneric checking code. */
365    /*non-public*/ static
366    @ForceInline
367    Object checkGenericType(Object mhObj, Object expectedObj) {
368        MethodHandle mh = (MethodHandle) mhObj;
369        MethodType expected = (MethodType) expectedObj;
370        //MethodType actual = mh.type();
371        MethodHandle gamh = expected.form().genericInvoker;
372        if (gamh != null)  return gamh;
373        return prepareForGenericCall(expected);
374    }

376    /**
377     * Returns an adapter GA for invoking a MH with type adjustments.
378     * The MethodType of the generic invocation site is prepended to MH
379     * and its arguments as follows:
380     * {@code (R)MH.invoke(A*) => GA.invokeBasic(TYPEOF<A*,R>, MH, A*)}
381     */
382    /*non-public*/ static MethodHandle prepareForGenericCall(MethodType mtype) {
383        // force any needed adapters to be preconstructed
384        MethodTypeForm form = mtype.form();
385        MethodHandle gamh = form.genericInvoker;
386        if (gamh != null)  return gamh;
387        try {
388            // Trigger adapter creation.
389            gamh = InvokeGeneric.generalInvokerOf(form.erasedType);
390            form.genericInvoker = gamh;
391            return gamh;
392        } catch (Exception ex) {
393            throw newInternalError("Exception while resolving inexact invoke", ex);
394        }
395    }

402    private static LambdaForm callSiteForm(MethodType mtype) {
403        mtype = mtype.basicType();  // normalize Z to I, String to Object, etc.
404        LambdaForm lform = mtype.form().cachedLambdaForm(MethodTypeForm.LF_CS_LINKER);
405        if (lform != null)  return lform;
406        // exactInvokerForm (Object,Object)Object
407        //   link with java.lang.invoke.MethodHandle.invokeBasic(MethodHandle,Object,Object)Object/invokeSpecial
408        final int ARG_BASE     = 0;
409        final int OUTARG_LIMIT = ARG_BASE + mtype.parameterCount();
410        final int INARG_LIMIT  = OUTARG_LIMIT + 1;
411        int nameCursor = OUTARG_LIMIT;
412        final int CSITE_ARG    = nameCursor++;  // the last in-argument
413        final int CALL_MH      = nameCursor++;  // result of getTarget
414        final int LINKER_CALL  = nameCursor++;
415        MethodType invokerFormType = mtype.appendParameterTypes(CallSite.class);
416        Name[] names = arguments(nameCursor - INARG_LIMIT, invokerFormType);
417        assert(names.length == nameCursor);
418        assert(names[CSITE_ARG] != null);
419        names[CALL_MH] = new Name(NF_getCallSiteTarget, names[CSITE_ARG]);
420        // (site.)invokedynamic(a*):R => mh = site.getTarget(); mh.invokeBasic(a*)
421        final int PREPEND_MH = 0, PREPEND_COUNT = 1;
422        Object[] outArgs = Arrays.copyOfRange(names, ARG_BASE, OUTARG_LIMIT + PREPEND_COUNT, Object[].class);
423        // prepend MH argument:
424        System.arraycopy(outArgs, 0, outArgs, PREPEND_COUNT, outArgs.length - PREPEND_COUNT);
425        outArgs[PREPEND_MH] = names[CALL_MH];
426        names[LINKER_CALL] = new Name(invokeBasicMethod(mtype), outArgs);
427        lform = new LambdaForm("linkToCallSite", INARG_LIMIT, names);
428        lform.compileToBytecode();  // JVM needs a real methodOop
429        lform = mtype.form().setCachedLambdaForm(MethodTypeForm.LF_CS_LINKER, lform);
430        return lform;
431    }
