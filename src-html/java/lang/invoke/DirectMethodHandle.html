042/**
043 * The flavor of method handle which implements a constant reference
044 * to a class member.
045 * @author jrose
046 */
047class DirectMethodHandle extends MethodHandle {
048    final MemberName member;
049
050    // Constructors and factory methods in this class *must* be package scoped or private.
051    private DirectMethodHandle(MethodType mtype, LambdaForm form, MemberName member) {
052        super(mtype, form);
053        if (!member.isResolved())
054            throw new InternalError();
055
056        if (member.getDeclaringClass().isInterface() &&
057                member.isMethod() && !member.isAbstract()) {
058           // Check for corner case: invokeinterface of Object method
059            MemberName m = new MemberName(Object.class, member.getName(), member.getMethodType(), member.getReferenceKind());
060            m = MemberName.getFactory().resolveOrNull(m.getReferenceKind(), m, null);
061            if (m != null && m.isPublic()) {
062                member = m;
063            }
064        }
065
066        this.member = member;
067    }

069    // Factory methods:

071    static DirectMethodHandle make(Class<?> receiver, MemberName member) {
072        MethodType mtype = member.getMethodOrFieldType();
073        if (!member.isStatic()) {
074            if (!member.getDeclaringClass().isAssignableFrom(receiver) || member.isConstructor())
075                throw new InternalError(member.toString());
076            mtype = mtype.insertParameterTypes(0, receiver);
077        }
078        if (!member.isField()) {
079            LambdaForm lform = preparedLambdaForm(member);
080            return new DirectMethodHandle(mtype, lform, member);
081        } else {
082            LambdaForm lform = preparedFieldLambdaForm(member);
083            if (member.isStatic()) {
084                long offset = MethodHandleNatives.staticFieldOffset(member);
085                Object base = MethodHandleNatives.staticFieldBase(member);
086                return new StaticAccessor(mtype, lform, member, base, offset);
087            } else {
088                long offset = MethodHandleNatives.objectFieldOffset(member);
089                assert(offset == (int)offset);
090                return new Accessor(mtype, lform, member, (int)offset);
091            }
092        }
093    }
094    static DirectMethodHandle make(MemberName member) {
095        if (member.isConstructor())
096            return makeAllocator(member);
097        return make(member.getDeclaringClass(), member);
098    }
099    static DirectMethodHandle make(Method method) {
100        return make(method.getDeclaringClass(), new MemberName(method));
101    }
102    static DirectMethodHandle make(Field field) {
103        return make(field.getDeclaringClass(), new MemberName(field));
104    }
105    private static DirectMethodHandle makeAllocator(MemberName ctor) {
106        assert(ctor.isConstructor() && ctor.getName().equals("<init>"));
107        Class<?> instanceClass = ctor.getDeclaringClass();
108        ctor = ctor.asConstructor();
109        assert(ctor.isConstructor() && ctor.getReferenceKind() == REF_newInvokeSpecial) : ctor;
110        MethodType mtype = ctor.getMethodType().changeReturnType(instanceClass);
111        LambdaForm lform = preparedLambdaForm(ctor);
112        MemberName init = ctor.asSpecial();
113        assert(init.getMethodType().returnType() == void.class);
114        return new Constructor(mtype, lform, ctor, init, instanceClass);
115    }

134    @Override
135    MethodHandle bindArgument(int pos, char basicType, Object value) {
136        // If the member needs dispatching, do so.
137        if (pos == 0 && basicType == 'L') {
138            DirectMethodHandle concrete = maybeRebind(value);
139            if (concrete != null)
140                return concrete.bindReceiver(value);
141        }
142        return super.bindArgument(pos, basicType, value);
143    }

145    @Override
146    MethodHandle bindReceiver(Object receiver) {
147        // If the member needs dispatching, do so.
148        DirectMethodHandle concrete = maybeRebind(receiver);
149        if (concrete != null)
150            return concrete.bindReceiver(receiver);
151        return super.bindReceiver(receiver);
152    }

154    private static final MemberName.Factory IMPL_NAMES = MemberName.getFactory();

156    private DirectMethodHandle maybeRebind(Object receiver) {
157        if (receiver != null) {
158            switch (member.getReferenceKind()) {
159            case REF_invokeInterface:
160            case REF_invokeVirtual:
161                // Pre-dispatch the member.
162                Class<?> concreteClass = receiver.getClass();
163                MemberName concrete = new MemberName(concreteClass, member.getName(), member.getMethodType(), REF_invokeSpecial);
164                concrete = IMPL_NAMES.resolveOrNull(REF_invokeSpecial, concrete, concreteClass);
165                if (concrete != null)
166                    return new DirectMethodHandle(type(), preparedLambdaForm(concrete), concrete);
167                break;
168            }
169        }
170        return null;
171    }

173    /**
174     * Create a LF which can invoke the given method.
175     * Cache and share this structure among all methods with
176     * the same basicType and refKind.
177     */
178    private static LambdaForm preparedLambdaForm(MemberName m) {
179        assert(m.isInvocable()) : m;  // call preparedFieldLambdaForm instead
180        MethodType mtype = m.getInvocationType().basicType();
181        assert(!m.isMethodHandleInvoke() || "invokeBasic".equals(m.getName())) : m;
182        int which;
183        switch (m.getReferenceKind()) {
184        case REF_invokeVirtual:    which = LF_INVVIRTUAL;    break;
185        case REF_invokeStatic:     which = LF_INVSTATIC;     break;
186        case REF_invokeSpecial:    which = LF_INVSPECIAL;    break;
187        case REF_invokeInterface:  which = LF_INVINTERFACE;  break;
188        case REF_newInvokeSpecial: which = LF_NEWINVSPECIAL; break;
189        default:  throw new InternalError(m.toString());
190        }
191        if (which == LF_INVSTATIC && shouldBeInitialized(m)) {
192            // precompute the barrier-free version:
193            preparedLambdaForm(mtype, which);
194            which = LF_INVSTATIC_INIT;
195        }
196        LambdaForm lform = preparedLambdaForm(mtype, which);
197        maybeCompile(lform, m);
198        assert(lform.methodType().dropParameterTypes(0, 1)
199                .equals(m.getInvocationType().basicType()))
200                : Arrays.asList(m, m.getInvocationType().basicType(), lform, lform.methodType());
201        return lform;
202    }
203
204    private static LambdaForm preparedLambdaForm(MethodType mtype, int which) {
205        LambdaForm lform = mtype.form().cachedLambdaForm(which);
206        if (lform != null)  return lform;
207        lform = makePreparedLambdaForm(mtype, which);
208        return mtype.form().setCachedLambdaForm(which, lform);
209    }
210
211    private static LambdaForm makePreparedLambdaForm(MethodType mtype, int which) {
212        boolean needsInit = (which == LF_INVSTATIC_INIT);
213        boolean doesAlloc = (which == LF_NEWINVSPECIAL);
214        String linkerName, lambdaName;
215        switch (which) {
216        case LF_INVVIRTUAL:    linkerName = "linkToVirtual";    lambdaName = "DMH.invokeVirtual";    break;
217        case LF_INVSTATIC:     linkerName = "linkToStatic";     lambdaName = "DMH.invokeStatic";     break;
218        case LF_INVSTATIC_INIT:linkerName = "linkToStatic";     lambdaName = "DMH.invokeStaticInit"; break;
219        case LF_INVSPECIAL:    linkerName = "linkToSpecial";    lambdaName = "DMH.invokeSpecial";    break;
220        case LF_INVINTERFACE:  linkerName = "linkToInterface";  lambdaName = "DMH.invokeInterface";  break;
221        case LF_NEWINVSPECIAL: linkerName = "linkToSpecial";    lambdaName = "DMH.newInvokeSpecial"; break;
222        default:  throw new InternalError("which="+which);
223        }
224        MethodType mtypeWithArg = mtype.appendParameterTypes(MemberName.class);
225        if (doesAlloc)
226            mtypeWithArg = mtypeWithArg
227                    .insertParameterTypes(0, Object.class)  // insert newly allocated obj
228                    .changeReturnType(void.class);          // <init> returns void
229        MemberName linker = new MemberName(MethodHandle.class, linkerName, mtypeWithArg, REF_invokeStatic);
230        try {
231            linker = IMPL_NAMES.resolveOrFail(REF_invokeStatic, linker, null, NoSuchMethodException.class);
232        } catch (ReflectiveOperationException ex) {
233            throw newInternalError(ex);
234        }
235        final int DMH_THIS    = 0;
236        final int ARG_BASE    = 1;
237        final int ARG_LIMIT   = ARG_BASE + mtype.parameterCount();
238        int nameCursor = ARG_LIMIT;
239        final int NEW_OBJ     = (doesAlloc ? nameCursor++ : -1);
240        final int GET_MEMBER  = nameCursor++;
241        final int LINKER_CALL = nameCursor++;
242        Name[] names = arguments(nameCursor - ARG_LIMIT, mtype.invokerType());
243        assert(names.length == nameCursor);
244        if (doesAlloc) {
245            // names = { argx,y,z,... new C, init method }
246            names[NEW_OBJ] = new Name(Lazy.NF_allocateInstance, names[DMH_THIS]);
247            names[GET_MEMBER] = new Name(Lazy.NF_constructorMethod, names[DMH_THIS]);
248        } else if (needsInit) {
249            names[GET_MEMBER] = new Name(Lazy.NF_internalMemberNameEnsureInit, names[DMH_THIS]);
250        } else {
251            names[GET_MEMBER] = new Name(Lazy.NF_internalMemberName, names[DMH_THIS]);
252        }
253        Object[] outArgs = Arrays.copyOfRange(names, ARG_BASE, GET_MEMBER+1, Object[].class);
254        assert(outArgs[outArgs.length-1] == names[GET_MEMBER]);  // look, shifted args!
255        int result = LambdaForm.LAST_RESULT;
256        if (doesAlloc) {
257            assert(outArgs[outArgs.length-2] == names[NEW_OBJ]);  // got to move this one
258            System.arraycopy(outArgs, 0, outArgs, 1, outArgs.length-2);
259            outArgs[0] = names[NEW_OBJ];
260            result = NEW_OBJ;
261        }
262        names[LINKER_CALL] = new Name(linker, outArgs);
263        lambdaName += "_" + LambdaForm.basicTypeSignature(mtype);
264        LambdaForm lform = new LambdaForm(lambdaName, ARG_LIMIT, names, result);
265        // This is a tricky bit of code.  Don't send it through the LF interpreter.
266        lform.compileToBytecode();
267        return lform;
268    }

270    private static void maybeCompile(LambdaForm lform, MemberName m) {
271        if (VerifyAccess.isSamePackage(m.getDeclaringClass(), MethodHandle.class))
272            // Help along bootstrapping...
273            lform.compileToBytecode();
274    }

291    /*non-public*/ static
292    boolean shouldBeInitialized(MemberName member) {
293        switch (member.getReferenceKind()) {
294        case REF_invokeStatic:
295        case REF_getStatic:
296        case REF_putStatic:
297        case REF_newInvokeSpecial:
298            break;
299        default:
300            // No need to initialize the class on this kind of member.
301            return false;
302        }
303        Class<?> cls = member.getDeclaringClass();
304        if (cls == ValueConversions.class ||
305            cls == MethodHandleImpl.class ||
306            cls == Invokers.class) {
307            // These guys have lots of <clinit> DMH creation but we know
308            // the MHs will not be used until the system is booted.
309            return false;
310        }
311        if (VerifyAccess.isSamePackage(MethodHandle.class, cls) ||
312            VerifyAccess.isSamePackage(ValueConversions.class, cls)) {
313            // It is a system class.  It is probably in the process of
314            // being initialized, but we will help it along just to be safe.
315            if (UNSAFE.shouldBeInitialized(cls)) {
316                UNSAFE.ensureClassInitialized(cls);
317            }
318            return false;
319        }
320        return UNSAFE.shouldBeInitialized(cls);
321    }
322
323    private static class EnsureInitialized extends ClassValue<WeakReference<Thread>> {
324        @Override
325        protected WeakReference<Thread> computeValue(Class<?> type) {
326            UNSAFE.ensureClassInitialized(type);
327            if (UNSAFE.shouldBeInitialized(type))
328                // If the previous call didn't block, this can happen.
329                // We are executing inside <clinit>.
330                return new WeakReference<>(Thread.currentThread());
331            return null;
332        }
333        static final EnsureInitialized INSTANCE = new EnsureInitialized();
334    }
335
336    private void ensureInitialized() {
337        if (checkInitialized(member)) {
338            // The coast is clear.  Delete the <clinit> barrier.
339            if (member.isField())
340                updateForm(preparedFieldLambdaForm(member));
341            else
342                updateForm(preparedLambdaForm(member));
343        }
344    }
345    private static boolean checkInitialized(MemberName member) {
346        Class<?> defc = member.getDeclaringClass();
347        WeakReference<Thread> ref = EnsureInitialized.INSTANCE.get(defc);
348        if (ref == null) {
349            return true;  // the final state
350        }
351        Thread clinitThread = ref.get();
352        // Somebody may still be running defc.<clinit>.
353        if (clinitThread == Thread.currentThread()) {
354            // If anybody is running defc.<clinit>, it is this thread.
355            if (UNSAFE.shouldBeInitialized(defc))
356                // Yes, we are running it; keep the barrier for now.
357                return false;
358        } else {
359            // We are in a random thread.  Block.
360            UNSAFE.ensureClassInitialized(defc);
361        }
362        assert(!UNSAFE.shouldBeInitialized(defc));
363        // put it into the final state
364        EnsureInitialized.INSTANCE.remove(defc);
365        return true;
366    }

476    // Caching machinery for field accessors:
477    private static byte
478            AF_GETFIELD        = 0,
479            AF_PUTFIELD        = 1,
480            AF_GETSTATIC       = 2,
481            AF_PUTSTATIC       = 3,
482            AF_GETSTATIC_INIT  = 4,
483            AF_PUTSTATIC_INIT  = 5,
484            AF_LIMIT           = 6;
485    // Enumerate the different field kinds using Wrapper,
486    // with an extra case added for checked references.
487    private static int
488            FT_LAST_WRAPPER    = Wrapper.values().length-1,
489            FT_UNCHECKED_REF   = Wrapper.OBJECT.ordinal(),
490            FT_CHECKED_REF     = FT_LAST_WRAPPER+1,
491            FT_LIMIT           = FT_LAST_WRAPPER+2;
492    private static int afIndex(byte formOp, boolean isVolatile, int ftypeKind) {
493        return ((formOp * FT_LIMIT * 2)
494                + (isVolatile ? FT_LIMIT : 0)
495                + ftypeKind);
496    }
497    private static final LambdaForm[] ACCESSOR_FORMS
498            = new LambdaForm[afIndex(AF_LIMIT, false, 0)];
499    private static int ftypeKind(Class<?> ftype) {
500        if (ftype.isPrimitive())
501            return Wrapper.forPrimitiveType(ftype).ordinal();
502        else if (VerifyType.isNullReferenceConversion(Object.class, ftype))
503            return FT_UNCHECKED_REF;
504        else
505            return FT_CHECKED_REF;
506    }

508    /**
509     * Create a LF which can access the given field.
510     * Cache and share this structure among all fields with
511     * the same basicType and refKind.
512     */
513    private static LambdaForm preparedFieldLambdaForm(MemberName m) {
514        Class<?> ftype = m.getFieldType();
515        boolean isVolatile = m.isVolatile();
516        byte formOp;
517        switch (m.getReferenceKind()) {
518        case REF_getField:      formOp = AF_GETFIELD;    break;
519        case REF_putField:      formOp = AF_PUTFIELD;    break;
520        case REF_getStatic:     formOp = AF_GETSTATIC;   break;
521        case REF_putStatic:     formOp = AF_PUTSTATIC;   break;
522        default:  throw new InternalError(m.toString());
523        }
524        if (shouldBeInitialized(m)) {
525            // precompute the barrier-free version:
526            preparedFieldLambdaForm(formOp, isVolatile, ftype);
527            assert((AF_GETSTATIC_INIT - AF_GETSTATIC) ==
528                   (AF_PUTSTATIC_INIT - AF_PUTSTATIC));
529            formOp += (AF_GETSTATIC_INIT - AF_GETSTATIC);
530        }
531        LambdaForm lform = preparedFieldLambdaForm(formOp, isVolatile, ftype);
532        maybeCompile(lform, m);
533        assert(lform.methodType().dropParameterTypes(0, 1)
534                .equals(m.getInvocationType().basicType()))
535                : Arrays.asList(m, m.getInvocationType().basicType(), lform, lform.methodType());
536        return lform;
537    }
538    private static LambdaForm preparedFieldLambdaForm(byte formOp, boolean isVolatile, Class<?> ftype) {
539        int afIndex = afIndex(formOp, isVolatile, ftypeKind(ftype));
540        LambdaForm lform = ACCESSOR_FORMS[afIndex];
541        if (lform != null)  return lform;
542        lform = makePreparedFieldLambdaForm(formOp, isVolatile, ftypeKind(ftype));
543        ACCESSOR_FORMS[afIndex] = lform;  // don't bother with a CAS
544        return lform;
545    }
546
547    private static LambdaForm makePreparedFieldLambdaForm(byte formOp, boolean isVolatile, int ftypeKind) {
548        boolean isGetter  = (formOp & 1) == (AF_GETFIELD & 1);
549        boolean isStatic  = (formOp >= AF_GETSTATIC);
550        boolean needsInit = (formOp >= AF_GETSTATIC_INIT);
551        boolean needsCast = (ftypeKind == FT_CHECKED_REF);
552        Wrapper fw = (needsCast ? Wrapper.OBJECT : Wrapper.values()[ftypeKind]);
553        Class<?> ft = fw.primitiveType();
554        assert(ftypeKind(needsCast ? String.class : ft) == ftypeKind);
555        String tname  = fw.primitiveSimpleName();
556        String ctname = Character.toUpperCase(tname.charAt(0)) + tname.substring(1);
557        if (isVolatile)  ctname += "Volatile";
558        String getOrPut = (isGetter ? "get" : "put");
559        String linkerName = (getOrPut + ctname);  // getObject, putIntVolatile, etc.
560        MethodType linkerType;
561        if (isGetter)
562            linkerType = MethodType.methodType(ft, Object.class, long.class);
563        else
564            linkerType = MethodType.methodType(void.class, Object.class, long.class, ft);
565        MemberName linker = new MemberName(Unsafe.class, linkerName, linkerType, REF_invokeVirtual);
566        try {
567            linker = IMPL_NAMES.resolveOrFail(REF_invokeVirtual, linker, null, NoSuchMethodException.class);
568        } catch (ReflectiveOperationException ex) {
569            throw newInternalError(ex);
570        }
571
572        // What is the external type of the lambda form?
573        MethodType mtype;
574        if (isGetter)
575            mtype = MethodType.methodType(ft);
576        else
577            mtype = MethodType.methodType(void.class, ft);
578        mtype = mtype.basicType();  // erase short to int, etc.
579        if (!isStatic)
580            mtype = mtype.insertParameterTypes(0, Object.class);
581        final int DMH_THIS  = 0;
582        final int ARG_BASE  = 1;
583        final int ARG_LIMIT = ARG_BASE + mtype.parameterCount();
584        // if this is for non-static access, the base pointer is stored at this index:
585        final int OBJ_BASE  = isStatic ? -1 : ARG_BASE;
586        // if this is for write access, the value to be written is stored at this index:
587        final int SET_VALUE  = isGetter ? -1 : ARG_LIMIT - 1;
588        int nameCursor = ARG_LIMIT;
589        final int F_HOLDER  = (isStatic ? nameCursor++ : -1);  // static base if any
590        final int F_OFFSET  = nameCursor++;  // Either static offset or field offset.
591        final int OBJ_CHECK = (OBJ_BASE >= 0 ? nameCursor++ : -1);
592        final int INIT_BAR  = (needsInit ? nameCursor++ : -1);
593        final int PRE_CAST  = (needsCast && !isGetter ? nameCursor++ : -1);
594        final int LINKER_CALL = nameCursor++;
595        final int POST_CAST = (needsCast && isGetter ? nameCursor++ : -1);
596        final int RESULT    = nameCursor-1;  // either the call or the cast
597        Name[] names = arguments(nameCursor - ARG_LIMIT, mtype.invokerType());
598        if (needsInit)
599            names[INIT_BAR] = new Name(Lazy.NF_ensureInitialized, names[DMH_THIS]);
600        if (needsCast && !isGetter)
601            names[PRE_CAST] = new Name(Lazy.NF_checkCast, names[DMH_THIS], names[SET_VALUE]);
602        Object[] outArgs = new Object[1 + linkerType.parameterCount()];
603        assert(outArgs.length == (isGetter ? 3 : 4));
604        outArgs[0] = UNSAFE;
605        if (isStatic) {
606            outArgs[1] = names[F_HOLDER]  = new Name(Lazy.NF_staticBase, names[DMH_THIS]);
607            outArgs[2] = names[F_OFFSET]  = new Name(Lazy.NF_staticOffset, names[DMH_THIS]);
608        } else {
609            outArgs[1] = names[OBJ_CHECK] = new Name(Lazy.NF_checkBase, names[OBJ_BASE]);
610            outArgs[2] = names[F_OFFSET]  = new Name(Lazy.NF_fieldOffset, names[DMH_THIS]);
611        }
612        if (!isGetter) {
613            outArgs[3] = (needsCast ? names[PRE_CAST] : names[SET_VALUE]);
614        }
615        for (Object a : outArgs)  assert(a != null);
616        names[LINKER_CALL] = new Name(linker, outArgs);
617        if (needsCast && isGetter)
618            names[POST_CAST] = new Name(Lazy.NF_checkCast, names[DMH_THIS], names[LINKER_CALL]);
619        for (Name n : names)  assert(n != null);
620        String fieldOrStatic = (isStatic ? "Static" : "Field");
621        String lambdaName = (linkerName + fieldOrStatic);  // significant only for debugging
622        if (needsCast)  lambdaName += "Cast";
623        if (needsInit)  lambdaName += "Init";
624        return new LambdaForm(lambdaName, ARG_LIMIT, names, RESULT);
625    }

627    /**
628     * Pre-initialized NamedFunctions for bootstrapping purposes.
629     * Factored in an inner class to delay initialization until first usage.
630     */
631    private static class Lazy {
632        static final NamedFunction
633                NF_internalMemberName,
634                NF_internalMemberNameEnsureInit,
635                NF_ensureInitialized,
636                NF_fieldOffset,
637                NF_checkBase,
638                NF_staticBase,
639                NF_staticOffset,
640                NF_checkCast,
641                NF_allocateInstance,
642                NF_constructorMethod;
643        static {
644            try {
645                NamedFunction nfs[] = {
646                        NF_internalMemberName = new NamedFunction(DirectMethodHandle.class
647                                .getDeclaredMethod("internalMemberName", Object.class)),
648                        NF_internalMemberNameEnsureInit = new NamedFunction(DirectMethodHandle.class
649                                .getDeclaredMethod("internalMemberNameEnsureInit", Object.class)),
650                        NF_ensureInitialized = new NamedFunction(DirectMethodHandle.class
651                                .getDeclaredMethod("ensureInitialized", Object.class)),
652                        NF_fieldOffset = new NamedFunction(DirectMethodHandle.class
653                                .getDeclaredMethod("fieldOffset", Object.class)),
654                        NF_checkBase = new NamedFunction(DirectMethodHandle.class
655                                .getDeclaredMethod("checkBase", Object.class)),
656                        NF_staticBase = new NamedFunction(DirectMethodHandle.class
657                                .getDeclaredMethod("staticBase", Object.class)),
658                        NF_staticOffset = new NamedFunction(DirectMethodHandle.class
659                                .getDeclaredMethod("staticOffset", Object.class)),
660                        NF_checkCast = new NamedFunction(DirectMethodHandle.class
661                                .getDeclaredMethod("checkCast", Object.class, Object.class)),
662                        NF_allocateInstance = new NamedFunction(DirectMethodHandle.class
663                                .getDeclaredMethod("allocateInstance", Object.class)),
664                        NF_constructorMethod = new NamedFunction(DirectMethodHandle.class
665                                .getDeclaredMethod("constructorMethod", Object.class))
666                };
667                for (NamedFunction nf : nfs) {
668                    // Each nf must be statically invocable or we get tied up in our bootstraps.
669                    assert(InvokerBytecodeGenerator.isStaticallyInvocable(nf.member)) : nf;
670                    nf.resolve();
671                }
672            } catch (ReflectiveOperationException ex) {
673                throw newInternalError(ex);
674            }
675        }
676    }
