056    static MethodHandle makeArrayElementAccessor(Class<?> arrayClass, boolean isSetter) {
057        if (!arrayClass.isArray())
058            throw newIllegalArgumentException("not an array: "+arrayClass);
059        MethodHandle accessor = ArrayAccessor.getAccessor(arrayClass, isSetter);
060        MethodType srcType = accessor.type().erase();
061        MethodType lambdaType = srcType.invokerType();
062        Name[] names = arguments(1, lambdaType);
063        Name[] args  = Arrays.copyOfRange(names, 1, 1 + srcType.parameterCount());
064        names[names.length - 1] = new Name(accessor.asType(srcType), (Object[]) args);
065        LambdaForm form = new LambdaForm("getElement", lambdaType.parameterCount(), names);
066        MethodHandle mh = SimpleMethodHandle.make(srcType, form);
067        if (ArrayAccessor.needCast(arrayClass)) {
068            mh = mh.bindTo(arrayClass);
069        }
070        mh = mh.asType(ArrayAccessor.correctType(arrayClass, isSetter));
071        return mh;
072    }

118        static MethodType type(Class<?> arrayClass, boolean isSetter) {
119            Class<?> elemClass = arrayClass.getComponentType();
120            Class<?> arrayArgClass = arrayClass;
121            if (!elemClass.isPrimitive()) {
122                arrayArgClass = Object[].class;
123                if (USE_WEAKLY_TYPED_ARRAY_ACCESSORS)
124                    arrayArgClass = Object.class;
125            }

127                return !isSetter ?
128                    MethodType.methodType(elemClass,  arrayArgClass, int.class) :
129                    MethodType.methodType(void.class, arrayArgClass, int.class, elemClass);
130            } else {
131                Class<?> classArgClass = Class.class;
132                if (USE_WEAKLY_TYPED_ARRAY_ACCESSORS)
133                    classArgClass = Object.class;
134                return !isSetter ?
135                    MethodType.methodType(Object.class, classArgClass, arrayArgClass, int.class) :
136                    MethodType.methodType(void.class,   classArgClass, arrayArgClass, int.class, Object.class);
137            }
138        }

156    /**
157     * Create a JVM-level adapter method handle to conform the given method
158     * handle to the similar newType, using only pairwise argument conversions.
159     * For each argument, convert incoming argument to the exact type needed.
160     * The argument conversions allowed are casting, boxing and unboxing,
161     * integral widening or narrowing, and floating point widening or narrowing.
162     * @param srcType required call type
163     * @param target original method handle
164     * @param level which strength of conversion is allowed
165     * @return an adapter to the original handle with the desired new type,
166     *          or the original target if the types are already identical
167     *          or null if the adaptation cannot be made
168     */
169    static MethodHandle makePairwiseConvert(MethodHandle target, MethodType srcType, int level) {
170        assert(level >= 0 && level <= 2);
171        MethodType dstType = target.type();
172        assert(dstType.parameterCount() == target.type().parameterCount());
173        if (srcType == dstType)
174            return target;
175
176        // Calculate extra arguments (temporaries) required in the names array.
177        // FIXME: Use an ArrayList<Name>.  Some arguments require more than one conversion step.
178        final int INARG_COUNT = srcType.parameterCount();
179        int conversions = 0;
180        boolean[] needConv = new boolean[1+INARG_COUNT];
181        for (int i = 0; i <= INARG_COUNT; i++) {
182            Class<?> src = (i == INARG_COUNT) ? dstType.returnType() : srcType.parameterType(i);
183            Class<?> dst = (i == INARG_COUNT) ? srcType.returnType() : dstType.parameterType(i);
184            if (!VerifyType.isNullConversion(src, dst) ||
185                level <= 1 && dst.isInterface() && !dst.isAssignableFrom(src)) {
186                needConv[i] = true;
187                conversions++;
188            }
189        }
190        boolean retConv = needConv[INARG_COUNT];
191
192        final int IN_MH         = 0;
193        final int INARG_BASE    = 1;
194        final int INARG_LIMIT   = INARG_BASE + INARG_COUNT;
195        final int NAME_LIMIT    = INARG_LIMIT + conversions + 1;
196        final int RETURN_CONV   = (!retConv ? -1         : NAME_LIMIT - 1);
197        final int OUT_CALL      = (!retConv ? NAME_LIMIT : RETURN_CONV) - 1;
198
199        // Now build a LambdaForm.
200        MethodType lambdaType = srcType.basicType().invokerType();
201        Name[] names = arguments(NAME_LIMIT - INARG_LIMIT, lambdaType);
202
203        // Collect the arguments to the outgoing call, maybe with conversions:
204        final int OUTARG_BASE = 0;  // target MH is Name.function, name Name.arguments[0]
205        Object[] outArgs = new Object[OUTARG_BASE + INARG_COUNT];
206
207        int nameCursor = INARG_LIMIT;
208        for (int i = 0; i < INARG_COUNT; i++) {
209            Class<?> src = srcType.parameterType(i);
210            Class<?> dst = dstType.parameterType(i);
211
212            if (!needConv[i]) {
213                // do nothing: difference is trivial
214                outArgs[OUTARG_BASE + i] = names[INARG_BASE + i];
215                continue;
216            }
217
218            // Tricky case analysis follows.
219            MethodHandle fn = null;
220            if (src.isPrimitive()) {
221                if (dst.isPrimitive()) {
222                    fn = ValueConversions.convertPrimitive(src, dst);
223                } else {
224                    Wrapper w = Wrapper.forPrimitiveType(src);
225                    MethodHandle boxMethod = ValueConversions.box(w);
226                    if (dst == w.wrapperType())
227                        fn = boxMethod;
228                    else
229                        fn = boxMethod.asType(MethodType.methodType(dst, src));
230                }
231            } else {
232                if (dst.isPrimitive()) {
233                    // Caller has boxed a primitive.  Unbox it for the target.
234                    Wrapper w = Wrapper.forPrimitiveType(dst);
235                    if (level == 0 || VerifyType.isNullConversion(src, w.wrapperType())) {
236                        fn = ValueConversions.unbox(dst);
237                    } else if (src == Object.class || !Wrapper.isWrapperType(src)) {
238                        // Examples:  Object->int, Number->int, Comparable->int; Byte->int, Character->int
239                        // must include additional conversions
240                        // src must be examined at runtime, to detect Byte, Character, etc.
241                        MethodHandle unboxMethod = (level == 1
242                                                    ? ValueConversions.unbox(dst)
243                                                    : ValueConversions.unboxCast(dst));
244                        fn = unboxMethod;
245                    } else {
246                        // Example: Byte->int
247                        // Do this by reformulating the problem to Byte->byte.
248                        Class<?> srcPrim = Wrapper.forWrapperType(src).primitiveType();
249                        MethodHandle unbox = ValueConversions.unbox(srcPrim);
250                        // Compose the two conversions.  FIXME:  should make two Names for this job
251                        fn = unbox.asType(MethodType.methodType(dst, src));
252                    }
253                } else {
254                    // Simple reference conversion.
255                    // Note:  Do not check for a class hierarchy relation
256                    // between src and dst.  In all cases a 'null' argument
257                    // will pass the cast conversion.
258                    fn = ValueConversions.cast(dst);
259                }
260            }
261            Name conv = new Name(fn, names[INARG_BASE + i]);
262            assert(names[nameCursor] == null);
263            names[nameCursor++] = conv;
264            assert(outArgs[OUTARG_BASE + i] == null);
265            outArgs[OUTARG_BASE + i] = conv;
266        }
267
268        // Build argument array for the call.
269        assert(nameCursor == OUT_CALL);
270        names[OUT_CALL] = new Name(target, outArgs);
271
272        if (RETURN_CONV < 0) {
273            assert(OUT_CALL == names.length-1);
274        } else {
275            Class<?> needReturn = srcType.returnType();
276            Class<?> haveReturn = dstType.returnType();
277            MethodHandle fn;
278            Object[] arg = { names[OUT_CALL] };
279            if (haveReturn == void.class) {
280                // synthesize a zero value for the given void
281                Object zero = Wrapper.forBasicType(needReturn).zero();
282                fn = MethodHandles.constant(needReturn, zero);
283                arg = new Object[0];  // don't pass names[OUT_CALL] to conversion
284            } else {
285                MethodHandle identity = MethodHandles.identity(needReturn);
286                MethodType needConversion = identity.type().changeParameterType(0, haveReturn);
287                fn = makePairwiseConvert(identity, needConversion, level);
288            }
289            assert(names[RETURN_CONV] == null);
290            names[RETURN_CONV] = new Name(fn, arg);
291            assert(RETURN_CONV == names.length-1);
292        }
293
294        LambdaForm form = new LambdaForm("convert", lambdaType.parameterCount(), names);
295        return SimpleMethodHandle.make(srcType, form);
296    }

306    static MethodHandle makeVarargsCollector(MethodHandle target, Class<?> arrayType) {
307        MethodType type = target.type();
308        int last = type.parameterCount() - 1;
309        if (type.parameterType(last) != arrayType)
310            target = target.asType(type.changeParameterType(last, arrayType));
311        target = target.asFixedArity();  // make sure this attribute is turned off
312        return new AsVarargsCollector(target, target.type(), arrayType);
313    }

315    static class AsVarargsCollector extends MethodHandle {
316        private final MethodHandle target;
317        private final Class<?> arrayType;
318        private MethodHandle cache;
319
320        AsVarargsCollector(MethodHandle target, MethodType type, Class<?> arrayType) {
321            super(type, reinvokerForm(type));
322            this.target = target;
323            this.arrayType = arrayType;
324            this.cache = target.asCollector(arrayType, 0);
325        }
326
327        @Override MethodHandle reinvokerTarget() { return target; }
328
329        @Override
330        public boolean isVarargsCollector() {
331            return true;
332        }
333
334        @Override
335        public MethodHandle asFixedArity() {
336            return target;
337        }
338
339        @Override
340        public MethodHandle asType(MethodType newType) {
341            MethodType type = this.type();
342            int collectArg = type.parameterCount() - 1;
343            int newArity = newType.parameterCount();
344            if (newArity == collectArg+1 &&
345                type.parameterType(collectArg).isAssignableFrom(newType.parameterType(collectArg))) {
346                // if arity and trailing parameter are compatible, do normal thing
347                return asFixedArity().asType(newType);
348            }
349            // check cache
350            if (cache.type().parameterCount() == newArity)
351                return cache.asType(newType);
352            // build and cache a collector
353            int arrayLength = newArity - collectArg;
354            MethodHandle collector;
355            try {
356                collector = asFixedArity().asCollector(arrayType, arrayLength);
357                assert(collector.type().parameterCount() == newArity) : "newArity="+newArity+" but collector="+collector;
358            } catch (IllegalArgumentException ex) {
359                throw new WrongMethodTypeException("cannot build collector", ex);
360            }
361            cache = collector;
362            return collector.asType(newType);
363        }
364
365        @Override
366        MethodHandle setVarargs(MemberName member) {
367            if (member.isVarargs())  return this;
368            return asFixedArity();
369        }
370
371        @Override
372        MethodHandle viewAsType(MethodType newType) {
373            MethodHandle mh = super.viewAsType(newType);
374            // put back the varargs bit:
375            MethodType type = mh.type();
376            int arity = type.parameterCount();
377            return mh.asVarargsCollector(type.parameterType(arity-1));
378        }

381        MemberName internalMemberName() {
382            return asFixedArity().internalMemberName();
383        }

385

388            return asFixedArity().bindArgument(pos, basicType, value);
389        }
390
391        @Override

393            return asFixedArity().bindReceiver(receiver);
394        }

407    /** Factory method:  Spread selected argument. */
408    static MethodHandle makeSpreadArguments(MethodHandle target,
409                                            Class<?> spreadArgType, int spreadArgPos, int spreadArgCount) {
410        MethodType targetType = target.type();
411
412        for (int i = 0; i < spreadArgCount; i++) {
413            Class<?> arg = VerifyType.spreadArgElementType(spreadArgType, i);
414            if (arg == null)  arg = Object.class;
415            targetType = targetType.changeParameterType(spreadArgPos + i, arg);
416        }
417        target = target.asType(targetType);
418
419        MethodType srcType = targetType
420                .replaceParameterTypes(spreadArgPos, spreadArgPos + spreadArgCount, spreadArgType);
421        // Now build a LambdaForm.
422        MethodType lambdaType = srcType.invokerType();
423        Name[] names = arguments(spreadArgCount + 2, lambdaType);
424        int nameCursor = lambdaType.parameterCount();
425        int[] indexes = new int[targetType.parameterCount()];
426
427        for (int i = 0, argIndex = 1; i < targetType.parameterCount() + 1; i++, argIndex++) {
428            Class<?> src = lambdaType.parameterType(i);
429            if (i == spreadArgPos) {
430                // Spread the array.
431                MethodHandle aload = MethodHandles.arrayElementGetter(spreadArgType);
432                Name array = names[argIndex];
433                names[nameCursor++] = new Name(Lazy.NF_checkSpreadArgument, array, spreadArgCount);
434                for (int j = 0; j < spreadArgCount; i++, j++) {
435                    indexes[i] = nameCursor;
436                    names[nameCursor++] = new Name(aload, array, j);
437                }
438            } else if (i < indexes.length) {
439                indexes[i] = argIndex;
440            }
441        }
442        assert(nameCursor == names.length-1);  // leave room for the final call
443
444        // Build argument array for the call.
445        Name[] targetArgs = new Name[targetType.parameterCount()];
446        for (int i = 0; i < targetType.parameterCount(); i++) {
447            int idx = indexes[i];
448            targetArgs[i] = names[idx];
449        }
450        names[names.length - 1] = new Name(target, (Object[]) targetArgs);
451
452        LambdaForm form = new LambdaForm("spread", lambdaType.parameterCount(), names);
453        return SimpleMethodHandle.make(srcType, form);
454    }

456    static void checkSpreadArgument(Object av, int n) {
457        if (av == null) {
458            if (n == 0)  return;
459        } else if (av instanceof Object[]) {
460            int len = ((Object[])av).length;
461            if (len == n)  return;
462        } else {
463            int len = java.lang.reflect.Array.getLength(av);
464            if (len == n)  return;
465        }
466        // fall through to error:
467        throw newIllegalArgumentException("array is not of length "+n);
468    }

487    /** Factory method:  Collect or filter selected argument(s). */
488    static MethodHandle makeCollectArguments(MethodHandle target,
489                MethodHandle collector, int collectArgPos, boolean retainOriginalArgs) {
490        MethodType targetType = target.type();          // (a..., c, [b...])=>r
491        MethodType collectorType = collector.type();    // (b...)=>c
492        int collectArgCount = collectorType.parameterCount();
493        Class<?> collectValType = collectorType.returnType();
494        int collectValCount = (collectValType == void.class ? 0 : 1);
495        MethodType srcType = targetType                 // (a..., [b...])=>r
496                .dropParameterTypes(collectArgPos, collectArgPos+collectValCount);
497        if (!retainOriginalArgs) {                      // (a..., b...)=>r
498            srcType = srcType.insertParameterTypes(collectArgPos, collectorType.parameterList());
499        }
500        // in  arglist: [0: ...keep1 | cpos: collect...  | cpos+cacount: keep2... ]
501        // out arglist: [0: ...keep1 | cpos: collectVal? | cpos+cvcount: keep2... ]
502        // out(retain): [0: ...keep1 | cpos: cV? coll... | cpos+cvc+cac: keep2... ]
503
504        // Now build a LambdaForm.
505        MethodType lambdaType = srcType.invokerType();
506        Name[] names = arguments(2, lambdaType);
507        final int collectNamePos = names.length - 2;
508        final int targetNamePos  = names.length - 1;
509
510        Name[] collectorArgs = Arrays.copyOfRange(names, 1 + collectArgPos, 1 + collectArgPos + collectArgCount);
511        names[collectNamePos] = new Name(collector, (Object[]) collectorArgs);
512
513        // Build argument array for the target.
514        // Incoming LF args to copy are: [ (mh) headArgs collectArgs tailArgs ].
515        // Output argument array is [ headArgs (collectVal)? (collectArgs)? tailArgs ].
516        Name[] targetArgs = new Name[targetType.parameterCount()];
517        int inputArgPos  = 1;  // incoming LF args to copy to target
518        int targetArgPos = 0;  // fill pointer for targetArgs
519        int chunk = collectArgPos;  // |headArgs|
520        System.arraycopy(names, inputArgPos, targetArgs, targetArgPos, chunk);
521        inputArgPos  += chunk;
522        targetArgPos += chunk;
523        if (collectValType != void.class) {
524            targetArgs[targetArgPos++] = names[collectNamePos];
525        }
526        chunk = collectArgCount;
527        if (retainOriginalArgs) {
528            System.arraycopy(names, inputArgPos, targetArgs, targetArgPos, chunk);
529            targetArgPos += chunk;   // optionally pass on the collected chunk
530        }
531        inputArgPos += chunk;
532        chunk = targetArgs.length - targetArgPos;  // all the rest
533        System.arraycopy(names, inputArgPos, targetArgs, targetArgPos, chunk);
534        assert(inputArgPos + chunk == collectNamePos);  // use of rest of input args also
535        names[targetNamePos] = new Name(target, (Object[]) targetArgs);
536
537        LambdaForm form = new LambdaForm("collect", lambdaType.parameterCount(), names);
538        return SimpleMethodHandle.make(srcType, form);
539    }

559    static
560    MethodHandle makeGuardWithTest(MethodHandle test,
561                                   MethodHandle target,
562                                   MethodHandle fallback) {
563        MethodType basicType = target.type().basicType();
564        MethodHandle invokeBasic = MethodHandles.basicInvoker(basicType);
565        int arity = basicType.parameterCount();
566        int extraNames = 3;
567        MethodType lambdaType = basicType.invokerType();
568        Name[] names = arguments(extraNames, lambdaType);
569
570        Object[] testArgs   = Arrays.copyOfRange(names, 1, 1 + arity, Object[].class);
571        Object[] targetArgs = Arrays.copyOfRange(names, 0, 1 + arity, Object[].class);
572
573        // call test
574        names[arity + 1] = new Name(test, testArgs);
575
576        // call selectAlternative
577        Object[] selectArgs = { names[arity + 1], target, fallback };
578        names[arity + 2] = new Name(MethodHandleImpl.selectAlternative(), selectArgs);
579        targetArgs[0] = names[arity + 2];
580
581        // call target or fallback
582        names[arity + 3] = new Name(new NamedFunction(invokeBasic), targetArgs);
583
584        LambdaForm form = new LambdaForm("guard", lambdaType.parameterCount(), names);
585        return SimpleMethodHandle.make(target.type(), form);
586    }

688        static MethodHandle[] makeInvokes() {
689            ArrayList<MethodHandle> invokes = new ArrayList<>();
690            MethodHandles.Lookup lookup = IMPL_LOOKUP;
691            for (;;) {
692                int nargs = invokes.size();
693                String name = "invoke_L"+nargs;
694                MethodHandle invoke = null;
695                try {
696                    invoke = lookup.findVirtual(GuardWithCatch.class, name, MethodType.genericMethodType(nargs));
697                } catch (ReflectiveOperationException ex) {
698                }
699                if (invoke == null)  break;
700                invokes.add(invoke);
701            }
702            assert(invokes.size() == 9);  // current number of methods
703            return invokes.toArray(new MethodHandle[0]);
704        };

719    static
720    MethodHandle makeGuardWithCatch(MethodHandle target,
721                                    Class<? extends Throwable> exType,
722                                    MethodHandle catcher) {
723        MethodType type = target.type();
724        MethodType ctype = catcher.type();
725        int nargs = type.parameterCount();
726        if (nargs < GuardWithCatch.INVOKES.length) {
727            MethodType gtype = type.generic();
728            MethodType gcatchType = gtype.insertParameterTypes(0, Throwable.class);
729            // Note: convertArguments(...2) avoids interface casts present in convertArguments(...0)
730            MethodHandle gtarget = makePairwiseConvert(target, gtype, 2);
731            MethodHandle gcatcher = makePairwiseConvert(catcher, gcatchType, 2);
732            GuardWithCatch gguard = new GuardWithCatch(gtarget, exType, gcatcher);
733            if (gtarget == null || gcatcher == null)  throw new InternalError();
734            MethodHandle ginvoker = GuardWithCatch.INVOKES[nargs].bindReceiver(gguard);
735            return makePairwiseConvert(ginvoker, type, 2);
736        } else {
737            MethodHandle gtarget = makeSpreadArguments(target, Object[].class, 0, nargs);
738            catcher = catcher.asType(ctype.changeParameterType(0, Throwable.class));
739            MethodHandle gcatcher = makeSpreadArguments(catcher, Object[].class, 1, nargs);
740            GuardWithCatch gguard = new GuardWithCatch(gtarget, exType, gcatcher);
741            if (gtarget == null || gcatcher == null)  throw new InternalError();
742            MethodHandle ginvoker = GuardWithCatch.VARARGS_INVOKE.bindReceiver(gguard);
743            MethodHandle gcollect = makeCollectArguments(ginvoker, ValueConversions.varargsArray(nargs), 0, false);
744            return makePairwiseConvert(gcollect, type, 2);
745        }
746    }

800    // Put the whole mess into its own nested class.
801    // That way we can lazily load the code and set up the constants.
802    private static class BindCaller {
803        static
804        MethodHandle bindCaller(MethodHandle mh, Class<?> hostClass) {
805            // Do not use this function to inject calls into system classes.
806            if (hostClass == null
807                ||    (hostClass.isArray() ||
808                       hostClass.isPrimitive() ||
809                       hostClass.getName().startsWith("java.") ||
810                       hostClass.getName().startsWith("sun."))) {
811                throw new InternalError();  // does not happen, and should not anyway
812            }
813            // For simplicity, convert mh to a varargs-like method.
814            MethodHandle vamh = prepareForInvoker(mh);
815            // Cache the result of makeInjectedInvoker once per argument class.
816            MethodHandle bccInvoker = CV_makeInjectedInvoker.get(hostClass);
817            return restoreToType(bccInvoker.bindTo(vamh), mh.type());
818        }
819
820        private static final Unsafe UNSAFE = Unsafe.getUnsafe();
821
822        private static MethodHandle makeInjectedInvoker(Class<?> hostClass) {
823            Class<?> bcc = UNSAFE.defineAnonymousClass(hostClass, T_BYTES, null);
824            if (hostClass.getClassLoader() != bcc.getClassLoader())
825                throw new InternalError(hostClass.getName()+" (CL)");
826            try {
827                if (hostClass.getProtectionDomain() != bcc.getProtectionDomain())
828                    throw new InternalError(hostClass.getName()+" (PD)");
829            } catch (SecurityException ex) {
830                // Self-check was blocked by security manager.  This is OK.
831                // In fact the whole try body could be turned into an assertion.
832            }
833            try {
834                MethodHandle init = IMPL_LOOKUP.findStatic(bcc, "init", MethodType.methodType(void.class));
835                init.invokeExact();  // force initialization of the class
836            } catch (Throwable ex) {
837                throw uncaughtException(ex);
838            }
839            MethodHandle bccInvoker;
840            try {
841                MethodType invokerMT = MethodType.methodType(Object.class, MethodHandle.class, Object[].class);
842                bccInvoker = IMPL_LOOKUP.findStatic(bcc, "invoke_V", invokerMT);
843            } catch (ReflectiveOperationException ex) {
844                throw uncaughtException(ex);
845            }
846            // Test the invoker, to ensure that it really injects into the right place.
847            try {
848                MethodHandle vamh = prepareForInvoker(MH_checkCallerClass);
849                Object ok = bccInvoker.invokeExact(vamh, new Object[]{hostClass, bcc});
850            } catch (Throwable ex) {
851                throw newInternalError(ex);
852            }
853            return bccInvoker;
854        }
855        private static ClassValue<MethodHandle> CV_makeInjectedInvoker = new ClassValue<MethodHandle>() {
856            @Override protected MethodHandle computeValue(Class<?> hostClass) {
857                return makeInjectedInvoker(hostClass);
858            }
859        };
860
861        // Adapt mh so that it can be called directly from an injected invoker:
862        private static MethodHandle prepareForInvoker(MethodHandle mh) {
863            mh = mh.asFixedArity();
864            MethodType mt = mh.type();
865            int arity = mt.parameterCount();
866            MethodHandle vamh = mh.asType(mt.generic());
867            vamh.internalForm().compileToBytecode();  // eliminate LFI stack frames
868            vamh = vamh.asSpreader(Object[].class, arity);
869            vamh.internalForm().compileToBytecode();  // eliminate LFI stack frames
870            return vamh;
871        }
872
873        // Undo the adapter effect of prepareForInvoker:
874        private static MethodHandle restoreToType(MethodHandle vamh, MethodType type) {
875            return vamh.asCollector(Object[].class, type.parameterCount()).asType(type);
876        }

878        private static final MethodHandle MH_checkCallerClass;
879        static {
880            final Class<?> THIS_CLASS = BindCaller.class;
881            assert(checkCallerClass(THIS_CLASS, THIS_CLASS));
882            try {
883                MH_checkCallerClass = IMPL_LOOKUP
884                    .findStatic(THIS_CLASS, "checkCallerClass",
885                                MethodType.methodType(boolean.class, Class.class, Class.class));
886                assert((boolean) MH_checkCallerClass.invokeExact(THIS_CLASS, THIS_CLASS));
887            } catch (Throwable ex) {
888                throw newInternalError(ex);
889            }
890        }

892        @CallerSensitive
893        private static boolean checkCallerClass(Class<?> expected, Class<?> expected2) {
894            // This method is called via MH_checkCallerClass and so it's
895            // correct to ask for the immediate caller here.
896            Class<?> actual = Reflection.getCallerClass();
897            if (actual != expected && actual != expected2)
898                throw new InternalError("found "+actual.getName()+", expected "+expected.getName()
899                                        +(expected == expected2 ? "" : ", or else "+expected2.getName()));
900            return true;
901        }

903        private static final byte[] T_BYTES;
904        static {
905            final Object[] values = {null};
906            AccessController.doPrivileged(new PrivilegedAction<Void>() {
907                    public Void run() {
908                        try {
909                            Class<T> tClass = T.class;
910                            String tName = tClass.getName();
911                            String tResource = tName.substring(tName.lastIndexOf('.')+1)+".class";
912                            java.net.URLConnection uconn = tClass.getResource(tResource).openConnection();
913                            int len = uconn.getContentLength();
914                            byte[] bytes = new byte[len];
915                            try (java.io.InputStream str = uconn.getInputStream()) {
916                                int nr = str.read(bytes);
917                                if (nr != len)  throw new java.io.IOException(tResource);
918                            }
919                            values[0] = bytes;
920                        } catch (java.io.IOException ex) {
921                            throw newInternalError(ex);
922                        }
923                        return null;
924                    }
925                });
926            T_BYTES = (byte[]) values[0];
927        }
