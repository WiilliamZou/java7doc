192    static boolean refKindIsValid(int refKind) {
193        return (refKind > REF_NONE && refKind < REF_LIMIT);
194    }
195    static boolean refKindIsField(byte refKind) {
196        assert(refKindIsValid(refKind));
197        return (refKind <= REF_putStatic);
198    }
199    static boolean refKindIsGetter(byte refKind) {
200        assert(refKindIsValid(refKind));
201        return (refKind <= REF_getStatic);
202    }
203    static boolean refKindIsSetter(byte refKind) {
204        return refKindIsField(refKind) && !refKindIsGetter(refKind);
205    }
206    static boolean refKindIsMethod(byte refKind) {
207        return !refKindIsField(refKind) && (refKind != REF_newInvokeSpecial);
208    }
209    static boolean refKindHasReceiver(byte refKind) {
210        assert(refKindIsValid(refKind));
211        return (refKind & 1) != 0;
212    }
213    static boolean refKindIsStatic(byte refKind) {
214        return !refKindHasReceiver(refKind) && (refKind != REF_newInvokeSpecial);
215    }
216    static boolean refKindDoesDispatch(byte refKind) {
217        assert(refKindIsValid(refKind));
218        return (refKind == REF_invokeVirtual ||
219                refKind == REF_invokeInterface);
220    }
221    static {
222        final int HR_MASK = ((1 << REF_getField) |
223                             (1 << REF_putField) |
224                             (1 << REF_invokeVirtual) |
225                             (1 << REF_invokeSpecial) |
226                             (1 << REF_invokeInterface)
227                            );
228        for (byte refKind = REF_NONE+1; refKind < REF_LIMIT; refKind++) {
229            assert(refKindHasReceiver(refKind) == (((1<<refKind) & HR_MASK) != 0)) : refKind;
230        }
231    }
232    static String refKindName(byte refKind) {
233        assert(refKindIsValid(refKind));
234        return REFERENCE_KIND_NAME[refKind];
235    }

249    private static native int getNamedCon(int which, Object[] name);
250    static boolean verifyConstants() {
251        Object[] box = { null };
252        for (int i = 0; ; i++) {
253            box[0] = null;
254            int vmval = getNamedCon(i, box);
255            if (box[0] == null)  break;
256            String name = (String) box[0];
257            try {
258                Field con = Constants.class.getDeclaredField(name);
259                int jval = con.getInt(null);
260                if (jval == vmval)  continue;
261                String err = (name+": JVM has "+vmval+" while Java has "+jval);
262                if (name.equals("CONV_OP_LIMIT")) {
263                    System.err.println("warning: "+err);
264                    continue;
265                }
266                throw new InternalError(err);
267            } catch (NoSuchFieldException | IllegalAccessException ex) {
268                String err = (name+": JVM has "+vmval+" which Java does not define");
269                // ignore exotic ops the JVM cares about; we just wont issue them
270                //System.err.println("warning: "+err);
271                continue;
272            }
273        }
274        return true;
275    }

283    /**
284     * The JVM is linking an invokedynamic instruction.  Create a reified call site for it.
285     */
286    static MemberName linkCallSite(Object callerObj,
287                                   Object bootstrapMethodObj,
288                                   Object nameObj, Object typeObj,
289                                   Object staticArguments,
290                                   Object[] appendixResult) {
291        MethodHandle bootstrapMethod = (MethodHandle)bootstrapMethodObj;
292        Class<?> caller = (Class<?>)callerObj;
293        String name = nameObj.toString().intern();
294        MethodType type = (MethodType)typeObj;
295        appendixResult[0] = CallSite.makeSite(bootstrapMethod,
296                                              name,
297                                              type,
298                                              staticArguments,
299                                              caller);
300        return Invokers.linkToCallSiteMethod(type);
301    }

310    /**
311     * The JVM wants to link a call site that requires a dynamic type check.
312     * Name is a type-checking invoker, invokeExact or invoke.
313     * Return a JVM method (MemberName) to handle the invoking.
314     * The method assumes the following arguments on the stack:
315     * 0: the method handle being invoked
316     * 1-N: the arguments to the method handle invocation
317     * N+1: an implicitly added type argument (the given MethodType)
318     */
319    static MemberName linkMethod(Class<?> callerClass, int refKind,
320                                 Class<?> defc, String name, Object type,
321                                 Object[] appendixResult) {
322        if (!TRACE_METHOD_LINKAGE)
323            return linkMethodImpl(callerClass, refKind, defc, name, type, appendixResult);
324        return linkMethodTracing(callerClass, refKind, defc, name, type, appendixResult);
325    }
326    static MemberName linkMethodImpl(Class<?> callerClass, int refKind,
327                                     Class<?> defc, String name, Object type,
328                                     Object[] appendixResult) {
329        try {
330            if (defc == MethodHandle.class && refKind == REF_invokeVirtual) {
331                switch (name) {
332                case "invoke":
333                    return Invokers.genericInvokerMethod(fixMethodType(callerClass, type), appendixResult);
334                case "invokeExact":
335                    return Invokers.exactInvokerMethod(fixMethodType(callerClass, type), appendixResult);
336                }
337            }
338        } catch (Throwable ex) {
339            if (ex instanceof LinkageError)
340                throw (LinkageError) ex;
341            else
342                throw new LinkageError(ex.getMessage(), ex);
343        }
344        throw new LinkageError("no such method "+defc.getName()+"."+name+type);
345    }
346    private static MethodType fixMethodType(Class<?> callerClass, Object type) {
347        if (type instanceof MethodType)
348            return (MethodType) type;
349        else
350            return MethodType.fromMethodDescriptorString((String)type, callerClass.getClassLoader());
351    }
352    // Tracing logic:
353    static MemberName linkMethodTracing(Class<?> callerClass, int refKind,
354                                        Class<?> defc, String name, Object type,
355                                        Object[] appendixResult) {
356        System.out.println("linkMethod "+defc.getName()+"."+
357                           name+type+"/"+Integer.toHexString(refKind));
358        try {
359            MemberName res = linkMethodImpl(callerClass, refKind, defc, name, type, appendixResult);
360            System.out.println("linkMethod => "+res+" + "+appendixResult[0]);
361            return res;
362        } catch (Throwable ex) {
363            System.out.println("linkMethod => throw "+ex);
364            throw ex;
365        }
366    }

369    /**
370     * The JVM is resolving a CONSTANT_MethodHandle CP entry.  And it wants our help.
371     * It will make an up-call to this method.  (Do not change the name or signature.)
372     * The type argument is a Class for field requests and a MethodType for non-fields.
373     * <p>
374     * Recent versions of the JVM may also pass a resolved MemberName for the type.
375     * In that case, the name is ignored and may be null.
376     */
377    static MethodHandle linkMethodHandleConstant(Class<?> callerClass, int refKind,
378                                                 Class<?> defc, String name, Object type) {
379        try {
380            Lookup lookup = IMPL_LOOKUP.in(callerClass);
381            assert(refKindIsValid(refKind));
382            return lookup.linkMethodHandleConstant((byte) refKind, defc, name, type);
383        } catch (ReflectiveOperationException ex) {
384            Error err = new IncompatibleClassChangeError();
385            err.initCause(ex);
386            throw err;
387        }
388    }

390    /**
391     * Is this method a caller-sensitive method?
392     * I.e., does it call Reflection.getCallerClass or a similer method
393     * to ask about the identity of its caller?
394     */
395    static boolean isCallerSensitive(MemberName mem) {
396        if (!mem.isMethod())  return false;  // only methods are caller sensitive
397
398        // when the VM support is available, call mem.isCallerSensitive() instead
399        return isCallerSensitiveMethod(mem.getDeclaringClass(), mem.getName()) ||
400                  canBeCalledVirtual(mem);
401    }

404    // to validate the hand-maintained list
405    private static boolean isCallerSensitiveMethod(Class<?> defc, String method) {
406        switch (method) {
407        case "doPrivileged":
408        case "doPrivilegedWithCombiner":
409            return defc == java.security.AccessController.class;
410        case "checkMemberAccess":
411            return defc == java.lang.SecurityManager.class;
412        case "getUnsafe":
413            return defc == sun.misc.Unsafe.class;
414        case "lookup":
415            return defc == java.lang.invoke.MethodHandles.class;
416        case "invoke":
417            return defc == java.lang.reflect.Method.class;
418        case "get":
419        case "getBoolean":
420        case "getByte":
421        case "getChar":
422        case "getShort":
423        case "getInt":
424        case "getLong":
425        case "getFloat":
426        case "getDouble":
427        case "set":
428        case "setBoolean":
429        case "setByte":
430        case "setChar":
431        case "setShort":
432        case "setInt":
433        case "setLong":
434        case "setFloat":
435        case "setDouble":
436            return defc == java.lang.reflect.Field.class;
437        case "newInstance":
438            if (defc == java.lang.reflect.Constructor.class)  return true;
439            if (defc == java.lang.Class.class)  return true;
440            break;
441        case "getFields":
442            return defc == java.lang.Class.class ||
443                   defc == javax.sql.rowset.serial.SerialJavaObject.class;
444        case "forName":
445        case "getClassLoader":
446        case "getClasses":
447        case "getMethods":
448        case "getConstructors":
449        case "getDeclaredClasses":
450        case "getDeclaredFields":
451        case "getDeclaredMethods":
452        case "getDeclaredConstructors":
453        case "getField":
454        case "getMethod":
455        case "getConstructor":
456        case "getDeclaredField":
457        case "getDeclaredMethod":
458        case "getDeclaredConstructor":
459        case "getDeclaringClass":
460        case "getEnclosingClass":
461        case "getEnclosingMethod":
462        case "getEnclosingConstructor":
463            return defc == java.lang.Class.class;
464        case "getConnection":
465        case "getDriver":
466        case "getDrivers":
467        case "deregisterDriver":
468            return defc == java.sql.DriverManager.class;
469        case "newUpdater":
470            if (defc == java.util.concurrent.atomic.AtomicIntegerFieldUpdater.class)  return true;
471            if (defc == java.util.concurrent.atomic.AtomicLongFieldUpdater.class)  return true;
472            if (defc == java.util.concurrent.atomic.AtomicReferenceFieldUpdater.class)  return true;
473            break;
474        case "getContextClassLoader":
475            return defc == java.lang.Thread.class;
476        case "getPackage":
477        case "getPackages":
478            return defc == java.lang.Package.class;
479        case "getParent":
480        case "getSystemClassLoader":
481            return defc == java.lang.ClassLoader.class;
482        case "load":
483        case "loadLibrary":
484            if (defc == java.lang.Runtime.class)  return true;
485            if (defc == java.lang.System.class)  return true;
486            break;
487        case "getCallerClass":
488            if (defc == sun.reflect.Reflection.class)  return true;
489            if (defc == java.lang.System.class)  return true;
490            break;
491        case "getCallerClassLoader":
492            return defc == java.lang.ClassLoader.class;
493        case "registerAsParallelCapable":
494            return defc == java.lang.ClassLoader.class;
495        case "getInvocationHandler":
496        case "getProxyClass":
497        case "newProxyInstance":
498            return defc == java.lang.reflect.Proxy.class;
499        case "asInterfaceInstance":
500            return defc == java.lang.invoke.MethodHandleProxies.class;
501        case "getBundle":
502        case "clearCache":
503            return defc == java.util.ResourceBundle.class;
504        case "getType":
505            return defc == java.io.ObjectStreamField.class;
506        case "forClass":
507            return defc == java.io.ObjectStreamClass.class;
508        case "getLogger":
509            return defc == java.util.logging.Logger.class;
510        case "getAnonymousLogger":
511            return defc == java.util.logging.Logger.class;
512        }
513        return false;
514    }

517    private static boolean canBeCalledVirtual(MemberName mem) {
518        assert(mem.isInvocable());
519        Class<?> defc = mem.getDeclaringClass();
520        switch (mem.getName()) {
521        case "checkMemberAccess":
522            return canBeCalledVirtual(mem, java.lang.SecurityManager.class);
523        case "getContextClassLoader":
524            return canBeCalledVirtual(mem, java.lang.Thread.class);
525        }
526        return false;
527    }
528
529    static boolean canBeCalledVirtual(MemberName symbolicRef, Class<?> definingClass) {
530        Class<?> symbolicRefClass = symbolicRef.getDeclaringClass();
531        if (symbolicRefClass == definingClass)  return true;
532        if (symbolicRef.isStatic() || symbolicRef.isPrivate())  return false;
533        return (definingClass.isAssignableFrom(symbolicRefClass) ||  // Msym overrides Mdef
534                symbolicRefClass.isInterface());                     // Mdef implements Msym
535    }
