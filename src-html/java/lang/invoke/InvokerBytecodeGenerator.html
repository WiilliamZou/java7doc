086    private InvokerBytecodeGenerator(LambdaForm lambdaForm, int localsMapSize,
087                                     String className, String invokerName, MethodType invokerType) {
088        if (invokerName.contains(".")) {
089            int p = invokerName.indexOf(".");
090            className = invokerName.substring(0, p);
091            invokerName = invokerName.substring(p+1);
092        }
093        if (DUMP_CLASS_FILES) {
094            className = makeDumpableClassName(className);
095        }
096        this.className  = superName + "$" + className;
097        this.sourceFile = "LambdaForm$" + className;
098        this.lambdaForm = lambdaForm;
099        this.invokerName = invokerName;
100        this.invokerType = invokerType;
101        this.localsMap = new int[localsMapSize];
102    }

104    private InvokerBytecodeGenerator(String className, String invokerName, MethodType invokerType) {
105        this(null, invokerType.parameterCount(),
106             className, invokerName, invokerType);
107        // Create an array to map name indexes to locals indexes.
108        for (int i = 0; i < localsMap.length; i++) {
109            localsMap[i] = invokerType.parameterSlotCount() - invokerType.parameterSlotDepth(i);
110        }
111    }
112
113    private InvokerBytecodeGenerator(String className, LambdaForm form, MethodType invokerType) {
114        this(form, form.names.length,
115             className, form.debugName, invokerType);
116        // Create an array to map name indexes to locals indexes.
117        Name[] names = form.names;
118        for (int i = 0, index = 0; i < localsMap.length; i++) {
119            localsMap[i] = index;
120            index += Wrapper.forBasicType(names[i].type).stackSlots();
121        }
122    }

173    private static String makeDumpableClassName(String className) {
174        Integer ctr;
175        synchronized (DUMP_CLASS_FILES_COUNTERS) {
176            ctr = DUMP_CLASS_FILES_COUNTERS.get(className);
177            if (ctr == null)  ctr = 0;
178            DUMP_CLASS_FILES_COUNTERS.put(className, ctr+1);
179        }
180        String sfx = ctr.toString();
181        while (sfx.length() < 3)
182            sfx = "0"+sfx;
183        className += sfx;
184        return className;
185    }

187    class CpPatch {
188        final int index;
189        final String placeholder;
190        final Object value;
191        CpPatch(int index, String placeholder, Object value) {
192            this.index = index;
193            this.placeholder = placeholder;
194            this.value = value;
195        }
196        public String toString() {
197            return "CpPatch/index="+index+",placeholder="+placeholder+",value="+value;
198        }
199    }

201    Map<Object, CpPatch> cpPatches = new HashMap<>();

203    int cph = 0;  // for counting constant placeholders
204
205    String constantPlaceholder(Object arg) {
206        String cpPlaceholder = "CONSTANT_PLACEHOLDER_" + cph++;
207        if (DUMP_CLASS_FILES) cpPlaceholder += " <<" + arg.toString() + ">>";  // debugging aid
208        if (cpPatches.containsKey(cpPlaceholder)) {
209            throw new InternalError("observed CP placeholder twice: " + cpPlaceholder);
210        }
211        // insert placeholder in CP and remember the patch
212        int index = cw.newConst((Object) cpPlaceholder);  // TODO check if aready in the constant pool
213        cpPatches.put(cpPlaceholder, new CpPatch(index, cpPlaceholder, arg));
214        return cpPlaceholder;
215    }
216
217    Object[] cpPatches(byte[] classFile) {
218        int size = getConstantPoolSize(classFile);
219        Object[] res = new Object[size];
220        for (CpPatch p : cpPatches.values()) {
221            if (p.index >= size)
222                throw new InternalError("in cpool["+size+"]: "+p+"\n"+Arrays.toString(Arrays.copyOf(classFile, 20)));
223            res[p.index] = p.value;
224        }
225        return res;
226    }

228    /**
229     * Extract the number of constant pool entries from a given class file.
230     *
231     * @param classFile the bytes of the class file in question.
232     * @return the number of entries in the constant pool.
233     */
234    private static int getConstantPoolSize(byte[] classFile) {
235        // The first few bytes:
236        // u4 magic;
237        // u2 minor_version;
238        // u2 major_version;
239        // u2 constant_pool_count;
240        return ((classFile[8] & 0xFF) << 8) | (classFile[9] & 0xFF);
241    }

289     * Set up class file generation.
290     */
291    private void classFilePrologue() {
292        final int NOT_ACC_PUBLIC = 0;  // not ACC_PUBLIC
293        cw = new ClassWriter(ClassWriter.COMPUTE_MAXS + ClassWriter.COMPUTE_FRAMES);
294        cw.visit(Opcodes.V1_6, NOT_ACC_PUBLIC + Opcodes.ACC_FINAL + Opcodes.ACC_SUPER, className, null, superName, null);
295        cw.visitSource(sourceFile, null);
296
297        String invokerDesc = invokerType.toMethodDescriptorString();
298        mv = cw.visitMethod(Opcodes.ACC_STATIC, invokerName, invokerDesc, null, null);
299    }

311     */
312    private void emitConst(Object con) {
313        if (con == null) {
314            mv.visitInsn(Opcodes.ACONST_NULL);
315            return;
316        }
317        if (con instanceof Integer) {
318            emitIconstInsn((int) con);
319            return;
320        }
321        if (con instanceof Long) {
322            long x = (long) con;
323            if (x == (short) x) {
324                emitIconstInsn((int) x);
325                mv.visitInsn(Opcodes.I2L);
326                return;
327            }
328        }
329        if (con instanceof Float) {
330            float x = (float) con;
331            if (x == (short) x) {
332                emitIconstInsn((int) x);
333                mv.visitInsn(Opcodes.I2F);
334                return;
335            }
336        }
337        if (con instanceof Double) {
338            double x = (double) con;
339            if (x == (short) x) {
340                emitIconstInsn((int) x);
341                mv.visitInsn(Opcodes.I2D);
342                return;
343            }
344        }
345        if (con instanceof Boolean) {
346            emitIconstInsn((boolean) con ? 1 : 0);
347            return;
348        }
349        // fall through:
350        mv.visitLdcInsn(con);
351    }
352
353    private void emitIconstInsn(int i) {
354        int opcode;
355        switch (i) {
356        case 0:  opcode = Opcodes.ICONST_0;  break;
357        case 1:  opcode = Opcodes.ICONST_1;  break;
358        case 2:  opcode = Opcodes.ICONST_2;  break;
359        case 3:  opcode = Opcodes.ICONST_3;  break;
360        case 4:  opcode = Opcodes.ICONST_4;  break;
361        case 5:  opcode = Opcodes.ICONST_5;  break;
362        default:
363            if (i == (byte) i) {
364                mv.visitIntInsn(Opcodes.BIPUSH, i & 0xFF);
365            } else if (i == (short) i) {
366                mv.visitIntInsn(Opcodes.SIPUSH, (char) i);
367            } else {
368                mv.visitLdcInsn(i);
369            }
370            return;
371        }
372        mv.visitInsn(opcode);
373    }

375    /*
376     * NOTE: These load/store methods use the localsMap to find the correct index!
377     */
378    private void emitLoadInsn(char type, int index) {
379        int opcode;
380        switch (type) {
381        case 'I':  opcode = Opcodes.ILOAD;  break;
382        case 'J':  opcode = Opcodes.LLOAD;  break;
383        case 'F':  opcode = Opcodes.FLOAD;  break;
384        case 'D':  opcode = Opcodes.DLOAD;  break;
385        case 'L':  opcode = Opcodes.ALOAD;  break;
386        default:
387            throw new InternalError("unknown type: " + type);
388        }
389        mv.visitVarInsn(opcode, localsMap[index]);
390    }
391    private void emitAloadInsn(int index) {
392        emitLoadInsn('L', index);
393    }

395    private void emitStoreInsn(char type, int index) {
396        int opcode;
397        switch (type) {
398        case 'I':  opcode = Opcodes.ISTORE;  break;
399        case 'J':  opcode = Opcodes.LSTORE;  break;
400        case 'F':  opcode = Opcodes.FSTORE;  break;
401        case 'D':  opcode = Opcodes.DSTORE;  break;
402        case 'L':  opcode = Opcodes.ASTORE;  break;
403        default:
404            throw new InternalError("unknown type: " + type);
405        }
406        mv.visitVarInsn(opcode, localsMap[index]);
407    }
408    private void emitAstoreInsn(int index) {
409        emitStoreInsn('L', index);
410    }

417    private void emitBoxing(Class<?> type) {

433        String name  = wrapper.primitiveSimpleName() + "Value";

435        mv.visitTypeInsn(Opcodes.CHECKCAST, owner);
436        mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, owner, name, desc);
437    }

439    /**
440     * Emit an implicit conversion.
441     *
442     * @param ptype type of value present on stack
443     * @param pclass type of value required on stack
444     */
445    private void emitImplicitConversion(char ptype, Class<?> pclass) {
446        switch (ptype) {
447        case 'L':
448            if (VerifyType.isNullConversion(Object.class, pclass))
449                return;
450            if (isStaticallyNameable(pclass)) {
451                mv.visitTypeInsn(Opcodes.CHECKCAST, getInternalName(pclass));
452            } else {
453                mv.visitLdcInsn(constantPlaceholder(pclass));
454                mv.visitTypeInsn(Opcodes.CHECKCAST, CLS);
455                mv.visitInsn(Opcodes.SWAP);
456                mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, CLS, "cast", LL_SIG);
457                if (pclass.isArray())
458                    mv.visitTypeInsn(Opcodes.CHECKCAST, OBJARY);
459            }
460            return;
461        case 'I':
462            if (!VerifyType.isNullConversion(int.class, pclass))
463                emitPrimCast(ptype, Wrapper.basicTypeChar(pclass));
464            return;
465        case 'J':
466            assert(pclass == long.class);
467            return;
468        case 'F':
469            assert(pclass == float.class);
470            return;
471        case 'D':
472            assert(pclass == double.class);
473            return;
474        }
475        throw new InternalError("bad implicit conversion: tc="+ptype+": "+pclass);
476    }

478    /**
479     * Emits an actual return instruction conforming to the given return type.
480     */
481    private void emitReturnInsn(Class<?> type) {
482        int opcode;
483        switch (Wrapper.basicTypeChar(type)) {
484        case 'I':  opcode = Opcodes.IRETURN;  break;
485        case 'J':  opcode = Opcodes.LRETURN;  break;
486        case 'F':  opcode = Opcodes.FRETURN;  break;
487        case 'D':  opcode = Opcodes.DRETURN;  break;
488        case 'L':  opcode = Opcodes.ARETURN;  break;
489        case 'V':  opcode = Opcodes.RETURN;   break;
490        default:
491            throw new InternalError("unknown return type: " + type);
492        }
493        mv.visitInsn(opcode);
494    }

513    /**
514     * Generate an invoker method for the passed {@link LambdaForm}.
515     */
516    private byte[] generateCustomizedCodeBytes() {
517        classFilePrologue();
518
519        // Suppress this method in backtraces displayed to the user.
520        mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
521
522        // Mark this method as a compiled LambdaForm
523        mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Compiled;", true);
524
525        // Force inlining of this invoker method.
526        mv.visitAnnotation("Ljava/lang/invoke/ForceInline;", true);
527
528        // iterate over the form's names, generating bytecode instructions for each
529        // start iterating at the first name following the arguments
530        for (int i = lambdaForm.arity; i < lambdaForm.names.length; i++) {
531            Name name = lambdaForm.names[i];
532            MemberName member = name.function.member();
533
534            if (isSelectAlternative(member)) {
535                // selectAlternative idiom
536                // FIXME: make sure this idiom is really present!
537                emitSelectAlternative(name, lambdaForm.names[i + 1]);
538                i++;  // skip MH.invokeBasic of the selectAlternative result
539            } else if (isStaticallyInvocable(member)) {
540                emitStaticInvoke(member, name);
541            } else {
542                emitInvoke(name);
543            }
544
545            // store the result from evaluating to the target name in a local if required
546            // (if this is the last value, i.e., the one that is going to be returned,
547            // avoid store/load/return and just return)
548            if (i == lambdaForm.names.length - 1 && i == lambdaForm.result) {
549                // return value - do nothing
550            } else if (name.type != 'V') {
551                // non-void: actually assign
552                emitStoreInsn(name.type, name.index());
553            }
554        }
555
556        // return statement
557        emitReturn();
558
559        classFileEpilogue();
560        bogusMethod(lambdaForm);
561
562        final byte[] classFile = cw.toByteArray();
563        maybeDump(className, classFile);
564        return classFile;
565    }

567    /**
568     * Emit an invoke for the given name.
569     *
570     * @param name
571     */
572    void emitInvoke(Name name) {
573        if (true) {
574            // push receiver
575            MethodHandle target = name.function.resolvedHandle;
576            assert(target != null) : name.exprString();
577            mv.visitLdcInsn(constantPlaceholder(target));
578            mv.visitTypeInsn(Opcodes.CHECKCAST, MH);
579        } else {
580            // load receiver
581            emitAloadInsn(0);
582            mv.visitTypeInsn(Opcodes.CHECKCAST, MH);
583            mv.visitFieldInsn(Opcodes.GETFIELD, MH, "form", LF_SIG);
584            mv.visitFieldInsn(Opcodes.GETFIELD, LF, "names", LFN_SIG);
585            // TODO more to come
586        }
587
588        // push arguments
589        for (int i = 0; i < name.arguments.length; i++) {
590            emitPushArgument(name, i);
591        }
592
593        // invocation
594        MethodType type = name.function.methodType();
595        mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", type.basicType().toMethodDescriptorString());
596    }

599        // Sample classes from each package we are willing to bind to statically:
600        java.lang.Object.class,
601        java.util.Arrays.class,
602        sun.misc.Unsafe.class
603        //MethodHandle.class already covered

606    static boolean isStaticallyInvocable(MemberName member) {
607        if (member == null)  return false;
608        if (member.isConstructor())  return false;
609        Class<?> cls = member.getDeclaringClass();
610        if (cls.isArray() || cls.isPrimitive())
611            return false;  // FIXME
612        if (cls.isAnonymousClass() || cls.isLocalClass())
613            return false;  // inner class of some sort
614        if (cls.getClassLoader() != MethodHandle.class.getClassLoader())
615            return false;  // not on BCP
616        MethodType mtype = member.getMethodOrFieldType();
617        if (!isStaticallyNameable(mtype.returnType()))
618            return false;
619        for (Class<?> ptype : mtype.parameterArray())
620            if (!isStaticallyNameable(ptype))
621                return false;
622        if (!member.isPrivate() && VerifyAccess.isSamePackage(MethodHandle.class, cls))
623            return true;   // in java.lang.invoke package
624        if (member.isPublic() && isStaticallyNameable(cls))
625            return true;
626        return false;
627    }
628
629    static boolean isStaticallyNameable(Class<?> cls) {
630        while (cls.isArray())
631            cls = cls.getComponentType();
632        if (cls.isPrimitive())
633            return true;  // int[].class, for example
634        if (cls.getClassLoader() != Object.class.getClassLoader())
635            return false;
636        if (VerifyAccess.isSamePackage(MethodHandle.class, cls))
637            return true;
638        if (!Modifier.isPublic(cls.getModifiers()))
639            return false;
640        for (Class<?> pkgcls : STATICALLY_INVOCABLE_PACKAGES) {
641            if (VerifyAccess.isSamePackage(pkgcls, cls))
642                return true;
643        }
644        return false;
645    }

647    /**
648     * Emit an invoke for the given name, using the MemberName directly.
649     *
650     * @param name
651     */
652    void emitStaticInvoke(MemberName member, Name name) {
653        assert(member.equals(name.function.member()));
654        String cname = getInternalName(member.getDeclaringClass());
655        String mname = member.getName();
656        String mtype;
657        byte refKind = member.getReferenceKind();
658        if (refKind == REF_invokeSpecial) {
659            // in order to pass the verifier, we need to convert this to invokevirtual in all cases
660            assert(member.canBeStaticallyBound()) : member;
661            refKind = REF_invokeVirtual;
662        }
663
664        if (member.getDeclaringClass().isInterface() && refKind == REF_invokeVirtual) {
665            // Methods from Object declared in an interface can be resolved by JVM to invokevirtual kind.
666            // Need to convert it back to invokeinterface to pass verification and make the invocation works as expected.
667            refKind = REF_invokeInterface;
668        }
669
670        // push arguments
671        for (int i = 0; i < name.arguments.length; i++) {
672            emitPushArgument(name, i);
673        }
674
675        // invocation
676        if (member.isMethod()) {
677            mtype = member.getMethodType().toMethodDescriptorString();
678            mv.visitMethodInsn(refKindOpcode(refKind), cname, mname, mtype);
679        } else {
680            mtype = MethodType.toFieldDescriptorString(member.getFieldType());
681            mv.visitFieldInsn(refKindOpcode(refKind), cname, mname, mtype);
682        }
683    }
684    int refKindOpcode(byte refKind) {
685        switch (refKind) {
686        case REF_invokeVirtual:      return Opcodes.INVOKEVIRTUAL;
687        case REF_invokeStatic:       return Opcodes.INVOKESTATIC;
688        case REF_invokeSpecial:      return Opcodes.INVOKESPECIAL;
689        case REF_invokeInterface:    return Opcodes.INVOKEINTERFACE;
690        case REF_getField:           return Opcodes.GETFIELD;
691        case REF_putField:           return Opcodes.PUTFIELD;
692        case REF_getStatic:          return Opcodes.GETSTATIC;
693        case REF_putStatic:          return Opcodes.PUTSTATIC;
694        }
695        throw new InternalError("refKind="+refKind);
696    }

710    /**
711     * Emit bytecode for the selectAlternative idiom.
712     *
713     * The pattern looks like (Cf. MethodHandleImpl.makeGuardWithTest):
714     *
715     *   Lambda(a0:L,a1:I)=>{
716     *     t2:I=foo.test(a1:I);
717     *     t3:L=MethodHandleImpl.selectAlternative(t2:I,(MethodHandle(int)int),(MethodHandle(int)int));
718     *     t4:I=MethodHandle.invokeBasic(t3:L,a1:I);t4:I}
719     *
720     * @param selectAlternativeName
721     * @param invokeBasicName
722     */
723    private void emitSelectAlternative(Name selectAlternativeName, Name invokeBasicName) {
724        MethodType type = selectAlternativeName.function.methodType();
725
726        Name receiver = (Name) invokeBasicName.arguments[0];
727
728        Label L_fallback = new Label();
729        Label L_done     = new Label();
730
731        // load test result
732        emitPushArgument(selectAlternativeName, 0);
733        mv.visitInsn(Opcodes.ICONST_1);
734
735        // if_icmpne L_fallback
736        mv.visitJumpInsn(Opcodes.IF_ICMPNE, L_fallback);
737
738        // invoke selectAlternativeName.arguments[1]
739        MethodHandle target = (MethodHandle) selectAlternativeName.arguments[1];
740        emitPushArgument(selectAlternativeName, 1);  // get 2nd argument of selectAlternative
741        emitAstoreInsn(receiver.index());  // store the MH in the receiver slot
742        emitInvoke(invokeBasicName);
743
744        // goto L_done
745        mv.visitJumpInsn(Opcodes.GOTO, L_done);
746
747        // L_fallback:
748        mv.visitLabel(L_fallback);
749
750        // invoke selectAlternativeName.arguments[2]
751        MethodHandle fallback = (MethodHandle) selectAlternativeName.arguments[2];
752        emitPushArgument(selectAlternativeName, 2);  // get 3rd argument of selectAlternative
753        emitAstoreInsn(receiver.index());  // store the MH in the receiver slot
754        emitInvoke(invokeBasicName);
755
756        // L_done:
757        mv.visitLabel(L_done);
758    }

760    /**
761     *
762     * @param name
763     * @param paramIndex
764     */
765    private void emitPushArgument(Name name, int paramIndex) {
766        Object arg = name.arguments[paramIndex];
767        char ptype = name.function.parameterType(paramIndex);
768        MethodType mtype = name.function.methodType();
769        if (arg instanceof Name) {
770            Name n = (Name) arg;
771            emitLoadInsn(n.type, n.index());
772            emitImplicitConversion(n.type, mtype.parameterType(paramIndex));
773        } else if ((arg == null || arg instanceof String) && ptype == 'L') {
774            emitConst(arg);
775        } else {
776            if (Wrapper.isWrapperType(arg.getClass()) && ptype != 'L') {
777                emitConst(arg);
778            } else {
779                mv.visitLdcInsn(constantPlaceholder(arg));
780                emitImplicitConversion('L', mtype.parameterType(paramIndex));
781            }
782        }
783    }

785    /**
786     * Emits a return statement from a LF invoker. If required, the result type is cast to the correct return type.
787     */
788    private void emitReturn() {
789        // return statement
790        if (lambdaForm.result == -1) {
791            // void
792            mv.visitInsn(Opcodes.RETURN);
793        } else {
794            LambdaForm.Name rn = lambdaForm.names[lambdaForm.result];
795            char rtype = Wrapper.basicTypeChar(invokerType.returnType());
796
797            // put return value on the stack if it is not already there
798            if (lambdaForm.result != lambdaForm.names.length - 1) {
799                emitLoadInsn(rn.type, lambdaForm.result);
800            }
801
802            // potentially generate cast
803            // rtype is the return type of the invoker - generated code must conform to this
804            // rn.type is the type of the result Name in the LF
805            if (rtype != rn.type) {
806                // need cast
807                if (rtype == 'L') {
808                    // possibly cast the primitive to the correct type for boxing
809                    char boxedType = Wrapper.forWrapperType(invokerType.returnType()).basicTypeChar();
810                    if (boxedType != rn.type) {
811                        emitPrimCast(rn.type, boxedType);
812                    }
813                    // cast primitive to reference ("boxing")
814                    emitBoxing(invokerType.returnType());
815                } else {
816                    // to-primitive cast
817                    if (rn.type != 'L') {
818                        // prim-to-prim cast
819                        emitPrimCast(rn.type, rtype);
820                    } else {
821                        // ref-to-prim cast ("unboxing")
822                        throw new InternalError("no ref-to-prim (unboxing) casts supported right now");
823                    }
824                }
825            }
826
827            // generate actual return statement
828            emitReturnInsn(invokerType.returnType());
829        }
830    }

832    /**
833     * Emit a type conversion bytecode casting from "from" to "to".
834     */
835    private void emitPrimCast(char from, char to) {
836        // Here's how.
837        // -   indicates forbidden
838        // <-> indicates implicit
839        //      to ----> boolean  byte     short    char     int      long     float    double
840        // from boolean    <->        -        -        -        -        -        -        -
841        //      byte        -       <->       i2s      i2c      <->      i2l      i2f      i2d
842        //      short       -       i2b       <->      i2c      <->      i2l      i2f      i2d
843        //      char        -       i2b       i2s      <->      <->      i2l      i2f      i2d
844        //      int         -       i2b       i2s      i2c      <->      i2l      i2f      i2d
845        //      long        -     l2i,i2b   l2i,i2s  l2i,i2c    l2i      <->      l2f      l2d
846        //      float       -     f2i,i2b   f2i,i2s  f2i,i2c    f2i      f2l      <->      f2d
847        //      double      -     d2i,i2b   d2i,i2s  d2i,i2c    d2i      d2l      d2f      <->
848        if (from == to) {
849            // no cast required, should be dead code anyway
850            return;
851        }
852        Wrapper wfrom = Wrapper.forBasicType(from);
853        Wrapper wto   = Wrapper.forBasicType(to);
854        if (wfrom.isSubwordOrInt()) {
855            // cast from {byte,short,char,int} to anything
856            emitI2X(to);
857        } else {
858            // cast from {long,float,double} to anything
859            if (wto.isSubwordOrInt()) {
860                // cast to {byte,short,char,int}
861                emitX2I(from);
862                if (wto.bitWidth() < 32) {
863                    // targets other than int require another conversion
864                    emitI2X(to);
865                }
866            } else {
867                // cast to {long,float,double} - this is verbose
868                boolean error = false;
869                switch (from) {
870                case 'J':
871                         if (to == 'F') { mv.visitInsn(Opcodes.L2F); }
872                    else if (to == 'D') { mv.visitInsn(Opcodes.L2D); }
873                    else error = true;
874                    break;
875                case 'F':
876                         if (to == 'J') { mv.visitInsn(Opcodes.F2L); }
877                    else if (to == 'D') { mv.visitInsn(Opcodes.F2D); }
878                    else error = true;
879                    break;
880                case 'D':
881                         if (to == 'J') { mv.visitInsn(Opcodes.D2L); }
882                    else if (to == 'F') { mv.visitInsn(Opcodes.D2F); }
883                    else error = true;
884                    break;
885                default:
886                    error = true;
887                    break;
888                }
889                if (error) {
890                    throw new IllegalStateException("unhandled prim cast: " + from + "2" + to);
891                }
892            }
893        }
894    }

896    private void emitI2X(char type) {
897        switch (type) {
898        case 'B':  mv.visitInsn(Opcodes.I2B);  break;
899        case 'S':  mv.visitInsn(Opcodes.I2S);  break;
900        case 'C':  mv.visitInsn(Opcodes.I2C);  break;
901        case 'I':  /* naught */                break;
902        case 'J':  mv.visitInsn(Opcodes.I2L);  break;
903        case 'F':  mv.visitInsn(Opcodes.I2F);  break;
904        case 'D':  mv.visitInsn(Opcodes.I2D);  break;
905        case 'Z':
906            // For compatibility with ValueConversions and explicitCastArguments:
907            mv.visitInsn(Opcodes.ICONST_1);
908            mv.visitInsn(Opcodes.IAND);
909            break;
910        default:   throw new InternalError("unknown type: " + type);
911        }
912    }

914    private void emitX2I(char type) {
915        switch (type) {
916        case 'J':  mv.visitInsn(Opcodes.L2I);  break;
917        case 'F':  mv.visitInsn(Opcodes.F2I);  break;
918        case 'D':  mv.visitInsn(Opcodes.D2I);  break;
919        default:   throw new InternalError("unknown type: " + type);
920        }
921    }

923    private static String basicTypeCharSignature(String prefix, MethodType type) {
924        StringBuilder buf = new StringBuilder(prefix);
925        for (Class<?> ptype : type.parameterList())
926            buf.append(Wrapper.forBasicType(ptype).basicTypeChar());
927        buf.append('_').append(Wrapper.forBasicType(type.returnType()).basicTypeChar());
928        return buf.toString();
929    }

932     * Generate bytecode for a LambdaForm.vmentry which calls interpretWithArguments.
933     *
934     * @param sig
935     * @return
936     */
937    static MemberName generateLambdaFormInterpreterEntryPoint(String sig) {
938        assert(LambdaForm.isValidSignature(sig));
939        //System.out.println("generateExactInvoker "+sig);
940        // compute method type
941        // first parameter and return type
942        char tret = LambdaForm.signatureReturn(sig);
943        MethodType type = MethodType.methodType(LambdaForm.typeClass(tret), MethodHandle.class);
944        // other parameter types
945        int arity = LambdaForm.signatureArity(sig);
946        for (int i = 1; i < arity; i++) {
947            type = type.appendParameterTypes(LambdaForm.typeClass(sig.charAt(i)));
948        }
949        InvokerBytecodeGenerator g = new InvokerBytecodeGenerator("LFI", "interpret_"+tret, type);
950        return g.loadMethod(g.generateLambdaFormInterpreterEntryPointBytes());
951    }

953    private byte[] generateLambdaFormInterpreterEntryPointBytes() {
954        classFilePrologue();
955
956        // Suppress this method in backtraces displayed to the user.
957        mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
958
959        // Don't inline the interpreter entry.
960        mv.visitAnnotation("Ljava/lang/invoke/DontInline;", true);
961
962        // create parameter array
963        emitIconstInsn(invokerType.parameterCount());
964        mv.visitTypeInsn(Opcodes.ANEWARRAY, "java/lang/Object");
965
966        // fill parameter array
967        for (int i = 0; i < invokerType.parameterCount(); i++) {
968            Class<?> ptype = invokerType.parameterType(i);
969            mv.visitInsn(Opcodes.DUP);
970            emitIconstInsn(i);
971            emitLoadInsn(Wrapper.basicTypeChar(ptype), i);
972            // box if primitive type
973            if (ptype.isPrimitive()) {
974                emitBoxing(ptype);
975            }
976            mv.visitInsn(Opcodes.AASTORE);
977        }
978        // invoke
979        emitAloadInsn(0);
980        mv.visitFieldInsn(Opcodes.GETFIELD, MH, "form", "Ljava/lang/invoke/LambdaForm;");
981        mv.visitInsn(Opcodes.SWAP);  // swap form and array; avoid local variable
982        mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, LF, "interpretWithArguments", "([Ljava/lang/Object;)Ljava/lang/Object;");
983
984        // maybe unbox
985        Class<?> rtype = invokerType.returnType();
986        if (rtype.isPrimitive() && rtype != void.class) {
987            emitUnboxing(Wrapper.asWrapperType(rtype));
988        }
989
990        // return statement
991        emitReturnInsn(rtype);
992
993        classFileEpilogue();
994        bogusMethod(invokerType);
995
996        final byte[] classFile = cw.toByteArray();
997        maybeDump(className, classFile);
998        return classFile;
999    }

1015    static int nfi = 0;
1016
1017    private byte[] generateNamedFunctionInvokerImpl(MethodTypeForm typeForm) {
1018        MethodType dstType = typeForm.erasedType();
1019        classFilePrologue();
1020
1021        // Suppress this method in backtraces displayed to the user.
1022        mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
1023
1024        // Force inlining of this invoker method.
1025        mv.visitAnnotation("Ljava/lang/invoke/ForceInline;", true);
1026
1027        // Load receiver
1028        emitAloadInsn(0);
1029
1030        // Load arguments from array
1031        for (int i = 0; i < dstType.parameterCount(); i++) {
1032            emitAloadInsn(1);
1033            emitIconstInsn(i);
1034            mv.visitInsn(Opcodes.AALOAD);
1035
1036            // Maybe unbox
1037            Class<?> dptype = dstType.parameterType(i);
1038            if (dptype.isPrimitive()) {
1039                Class<?> sptype = dstType.basicType().wrap().parameterType(i);
1040                Wrapper dstWrapper = Wrapper.forBasicType(dptype);
1041                Wrapper srcWrapper = dstWrapper.isSubwordOrInt() ? Wrapper.INT : dstWrapper;  // narrow subword from int
1042                emitUnboxing(srcWrapper.wrapperType());
1043                emitPrimCast(srcWrapper.basicTypeChar(), dstWrapper.basicTypeChar());
1044            }
1045        }
1046
1047        // Invoke
1048        String targetDesc = dstType.basicType().toMethodDescriptorString();
1049        mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", targetDesc);
1050
1051        // Box primitive types
1052        Class<?> rtype = dstType.returnType();
1053        if (rtype != void.class && rtype.isPrimitive()) {
1054            Wrapper srcWrapper = Wrapper.forBasicType(rtype);
1055            Wrapper dstWrapper = srcWrapper.isSubwordOrInt() ? Wrapper.INT : srcWrapper;  // widen subword to int
1056            // boolean casts not allowed
1057            emitPrimCast(srcWrapper.basicTypeChar(), dstWrapper.basicTypeChar());
1058            emitBoxing(dstWrapper.primitiveType());
1059        }
1060
1061        // If the return type is void we return a null reference.
1062        if (rtype == void.class) {
1063            mv.visitInsn(Opcodes.ACONST_NULL);
1064        }
1065        emitReturnInsn(Object.class);  // NOTE: NamedFunction invokers always return a reference value.
1066
1067        classFileEpilogue();
1068        bogusMethod(dstType);
1069
1070        final byte[] classFile = cw.toByteArray();
1071        maybeDump(className, classFile);
1072        return classFile;
1073    }
