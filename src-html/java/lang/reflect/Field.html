040/**
041 * A {@code Field} provides information about, and dynamic access to, a
042 * single field of a class or an interface.  The reflected field may
043 * be a class (static) field or an instance field.
044 *
045 * <p>A {@code Field} permits widening conversions to occur during a get or
046 * set access operation, but throws an {@code IllegalArgumentException} if a
047 * narrowing conversion would occur.
048 *
049 * @see Member
050 * @see java.lang.Class
051 * @see java.lang.Class#getFields()
052 * @see java.lang.Class#getField(String)
053 * @see java.lang.Class#getDeclaredFields()
054 * @see java.lang.Class#getDeclaredField(String)
055 *
056 * @author Kenneth Russell
057 * @author Nakul Saraiya
058 */
059public final
060class Field extends AccessibleObject implements Member {
061
062    private Class<?>            clazz;
063    private int                 slot;
064    // This is guaranteed to be interned by the VM in the 1.4
065    // reflection implementation
066    private String              name;
067    private Class<?>            type;
068    private int                 modifiers;
069    // Generics and annotations support
070    private transient String    signature;
071    // generic info repository; lazily initialized
072    private transient FieldRepository genericInfo;
073    private byte[]              annotations;
074    // Cached field accessor created without override
075    private FieldAccessor fieldAccessor;
076    // Cached field accessor created with override
077    private FieldAccessor overrideFieldAccessor;
078    // For sharing of FieldAccessors. This branching structure is
079    // currently only two levels deep (i.e., one root Field and
080    // potentially many Field objects pointing to it.)
081    private Field               root;
082
083    // Generics infrastructure
084
085    private String getGenericSignature() {return signature;}
086
087    // Accessor for factory
088    private GenericsFactory getFactory() {
089        Class<?> c = getDeclaringClass();
090        // create scope and factory
091        return CoreReflectionFactory.make(c, ClassScope.make(c));
092    }
093
094    // Accessor for generic info repository
095    private FieldRepository getGenericInfo() {
096        // lazily initialize repository if necessary
097        if (genericInfo == null) {
098            // create and cache generic info repository
099            genericInfo = FieldRepository.make(getGenericSignature(),
100                                               getFactory());
101        }
102        return genericInfo; //return cached repository
103    }

106    /**
107     * Package-private constructor used by ReflectAccess to enable
108     * instantiation of these objects in Java code from the java.lang
109     * package via sun.reflect.LangReflectAccess.
110     */
111    Field(Class<?> declaringClass,
112          String name,
113          Class<?> type,
114          int modifiers,
115          int slot,
116          String signature,
117          byte[] annotations)
118    {
119        this.clazz = declaringClass;
120        this.name = name;
121        this.type = type;
122        this.modifiers = modifiers;
123        this.slot = slot;
124        this.signature = signature;
125        this.annotations = annotations;
126    }
127
128    /**
129     * Package-private routine (exposed to java.lang.Class via
130     * ReflectAccess) which returns a copy of this Field. The copy's
131     * "root" field points to this Field.
132     */
133    Field copy() {
134        // This routine enables sharing of FieldAccessor objects
135        // among Field objects which refer to the same underlying
136        // method in the VM. (All of this contortion is only necessary
137        // because of the "accessibility" bit in AccessibleObject,
138        // which implicitly requires that new java.lang.reflect
139        // objects be fabricated for each reflective call on Class
140        // objects.)
141        Field res = new Field(clazz, name, type, modifiers, slot, signature, annotations);
142        res.root = this;
143        // Might as well eagerly propagate this if already present
144        res.fieldAccessor = fieldAccessor;
145        res.overrideFieldAccessor = overrideFieldAccessor;
146        return res;
147    }

211    /**
212     * Returns a {@code Type} object that represents the declared type for
213     * the field represented by this {@code Field} object.
214     *
215     * <p>If the {@code Type} is a parameterized type, the
216     * {@code Type} object returned must accurately reflect the
217     * actual type parameters used in the source code.
218     *
219     * <p>If the type of the underlying field is a type variable or a
220     * parameterized type, it is created. Otherwise, it is resolved.
221     *
222     * @return a {@code Type} object that represents the declared type for
223     *     the field represented by this {@code Field} object
224     * @throws GenericSignatureFormatError if the generic field
225     *     signature does not conform to the format specified in
226     *     <cite>The Java&trade; Virtual Machine Specification</cite>
227     * @throws TypeNotPresentException if the generic type
228     *     signature of the underlying field refers to a non-existent
229     *     type declaration
230     * @throws MalformedParameterizedTypeException if the generic
231     *     signature of the underlying field refers to a parameterized type
232     *     that cannot be instantiated for any reason
233     * @since 1.5
234     */
235    public Type getGenericType() {
236        if (getGenericSignature() != null)
237            return getGenericInfo().getGenericType();
238        else
239            return getType();
240    }

243    /**
244     * Compares this {@code Field} against the specified object.  Returns
245     * true if the objects are the same.  Two {@code Field} objects are the same if
246     * they were declared by the same class and have the same name
247     * and type.
248     */
249    public boolean equals(Object obj) {
250        if (obj != null && obj instanceof Field) {
251            Field other = (Field)obj;
252            return (getDeclaringClass() == other.getDeclaringClass())
253                && (getName() == other.getName())
254                && (getType() == other.getType());
255        }
256        return false;
257    }

259    /**
260     * Returns a hashcode for this {@code Field}.  This is computed as the
261     * exclusive-or of the hashcodes for the underlying field's
262     * declaring class name and its name.
263     */
264    public int hashCode() {
265        return getDeclaringClass().getName().hashCode() ^ getName().hashCode();
266    }

268    /**
269     * Returns a string describing this {@code Field}.  The format is
270     * the access modifiers for the field, if any, followed
271     * by the field type, followed by a space, followed by
272     * the fully-qualified name of the class declaring the field,
273     * followed by a period, followed by the name of the field.
274     * For example:
275     * <pre>
276     *    public static final int java.lang.Thread.MIN_PRIORITY
277     *    private int java.io.FileDescriptor.fd
278     * </pre>
279     *
280     * <p>The modifiers are placed in canonical order as specified by
281     * "The Java Language Specification".  This is {@code public},
282     * {@code protected} or {@code private} first, and then other
283     * modifiers in the following order: {@code static}, {@code final},
284     * {@code transient}, {@code volatile}.
285     */
286    public String toString() {
287        int mod = getModifiers();
288        return (((mod == 0) ? "" : (Modifier.toString(mod) + " "))
289            + getTypeName(getType()) + " "
290            + getTypeName(getDeclaringClass()) + "."
291            + getName());
292    }

294    /**
295     * Returns a string describing this {@code Field}, including
296     * its generic type.  The format is the access modifiers for the
297     * field, if any, followed by the generic field type, followed by
298     * a space, followed by the fully-qualified name of the class
299     * declaring the field, followed by a period, followed by the name
300     * of the field.
301     *
302     * <p>The modifiers are placed in canonical order as specified by
303     * "The Java Language Specification".  This is {@code public},
304     * {@code protected} or {@code private} first, and then other
305     * modifiers in the following order: {@code static}, {@code final},
306     * {@code transient}, {@code volatile}.
307     *
308     * @return a string describing this {@code Field}, including
309     * its generic type
310     *
311     * @since 1.5
312     */
313    public String toGenericString() {
314        int mod = getModifiers();
315        Type fieldType = getGenericType();
316        return (((mod == 0) ? "" : (Modifier.toString(mod) + " "))
317            +  ((fieldType instanceof Class) ?
318                getTypeName((Class)fieldType): fieldType.toString())+ " "
319            + getTypeName(getDeclaringClass()) + "."
320            + getName());
321    }

323    /**
324     * Returns the value of the field represented by this {@code Field}, on
325     * the specified object. The value is automatically wrapped in an
326     * object if it has a primitive type.
327     *
328     * <p>The underlying field's value is obtained as follows:
329     *
330     * <p>If the underlying field is a static field, the {@code obj} argument
331     * is ignored; it may be null.
332     *
333     * <p>Otherwise, the underlying field is an instance field.  If the
334     * specified {@code obj} argument is null, the method throws a
335     * {@code NullPointerException}. If the specified object is not an
336     * instance of the class or interface declaring the underlying
337     * field, the method throws an {@code IllegalArgumentException}.
338     *
339     * <p>If this {@code Field} object is enforcing Java language access control, and
340     * the underlying field is inaccessible, the method throws an
341     * {@code IllegalAccessException}.
342     * If the underlying field is static, the class that declared the
343     * field is initialized if it has not already been initialized.
344     *
345     * <p>Otherwise, the value is retrieved from the underlying instance
346     * or static field.  If the field has a primitive type, the value
347     * is wrapped in an object before being returned, otherwise it is
348     * returned as is.
349     *
350     * <p>If the field is hidden in the type of {@code obj},
351     * the field's value is obtained according to the preceding rules.
352     *
353     * @param obj object from which the represented field's value is
354     * to be extracted
355     * @return the value of the represented field in object
356     * {@code obj}; primitive values are wrapped in an appropriate
357     * object before being returned
358     *
359     * @exception IllegalAccessException    if this {@code Field} object
360     *              is enforcing Java language access control and the underlying
361     *              field is inaccessible.
362     * @exception IllegalArgumentException  if the specified object is not an
363     *              instance of the class or interface declaring the underlying
364     *              field (or a subclass or implementor thereof).
365     * @exception NullPointerException      if the specified object is null
366     *              and the field is an instance field.
367     * @exception ExceptionInInitializerError if the initialization provoked
368     *              by this method fails.
369     */
370    @CallerSensitive
371    public Object get(Object obj)
372        throws IllegalArgumentException, IllegalAccessException
373    {
374        if (!override) {
375            if (!Reflection.quickCheckMemberAccess(clazz, modifiers)) {
376                checkAccess(Reflection.getCallerClass(), clazz, obj, modifiers);
377            }
378        }
379        return getFieldAccessor(obj).get(obj);
380    }

666    /**
667     * Sets the field represented by this {@code Field} object on the
668     * specified object argument to the specified new value. The new
669     * value is automatically unwrapped if the underlying field has a
670     * primitive type.
671     *
672     * <p>The operation proceeds as follows:
673     *
674     * <p>If the underlying field is static, the {@code obj} argument is
675     * ignored; it may be null.
676     *
677     * <p>Otherwise the underlying field is an instance field.  If the
678     * specified object argument is null, the method throws a
679     * {@code NullPointerException}.  If the specified object argument is not
680     * an instance of the class or interface declaring the underlying
681     * field, the method throws an {@code IllegalArgumentException}.
682     *
683     * <p>If this {@code Field} object is enforcing Java language access control, and
684     * the underlying field is inaccessible, the method throws an
685     * {@code IllegalAccessException}.
686     *
687     * <p>If the underlying field is final, the method throws an
688     * {@code IllegalAccessException} unless {@code setAccessible(true)}
689     * has succeeded for this {@code Field} object
690     * and the field is non-static. Setting a final field in this way
691     * is meaningful only during deserialization or reconstruction of
692     * instances of classes with blank final fields, before they are
693     * made available for access by other parts of a program. Use in
694     * any other context may have unpredictable effects, including cases
695     * in which other parts of a program continue to use the original
696     * value of this field.
697     *
698     * <p>If the underlying field is of a primitive type, an unwrapping
699     * conversion is attempted to convert the new value to a value of
700     * a primitive type.  If this attempt fails, the method throws an
701     * {@code IllegalArgumentException}.
702     *
703     * <p>If, after possible unwrapping, the new value cannot be
704     * converted to the type of the underlying field by an identity or
705     * widening conversion, the method throws an

1032    // security check is done before calling this method
1033    private FieldAccessor getFieldAccessor(Object obj)
1034        throws IllegalAccessException
1035    {
1036        boolean ov = override;
1037        FieldAccessor a = (ov) ? overrideFieldAccessor : fieldAccessor;
1038        return (a != null) ? a : acquireFieldAccessor(ov);
1039    }

1041    // NOTE that there is no synchronization used here. It is correct
1042    // (though not efficient) to generate more than one FieldAccessor
1043    // for a given Field. However, avoiding synchronization will
1044    // probably make the implementation more scalable.
1045    private FieldAccessor acquireFieldAccessor(boolean overrideFinalCheck) {
1046        // First check to see if one has been created yet, and take it
1047        // if so
1048        FieldAccessor tmp = null;
1049        if (root != null) tmp = root.getFieldAccessor(overrideFinalCheck);
1050        if (tmp != null) {
1051            if (overrideFinalCheck)
1052                overrideFieldAccessor = tmp;
1053            else
1054                fieldAccessor = tmp;
1055        } else {
1056            // Otherwise fabricate one and propagate it up to the root
1057            tmp = reflectionFactory.newFieldAccessor(this, overrideFinalCheck);
1058            setFieldAccessor(tmp, overrideFinalCheck);
1059        }
1060
1061        return tmp;
1062    }
1063
1064    // Returns FieldAccessor for this Field object, not looking up
1065    // the chain to the root
1066    private FieldAccessor getFieldAccessor(boolean overrideFinalCheck) {
1067        return (overrideFinalCheck)? overrideFieldAccessor : fieldAccessor;
1068    }
1069
1070    // Sets the FieldAccessor for this Field object and
1071    // (recursively) its root
1072    private void setFieldAccessor(FieldAccessor accessor, boolean overrideFinalCheck) {
1073        if (overrideFinalCheck)
1074            overrideFieldAccessor = accessor;
1075        else
1076            fieldAccessor = accessor;
1077        // Propagate up
1078        if (root != null) {
1079            root.setFieldAccessor(accessor, overrideFinalCheck);
1080        }
1081    }

1083    /*
1084     * Utility routine to paper over array type names
1085     */
1086    static String getTypeName(Class<?> type) {
1087        if (type.isArray()) {
1088            try {
1089                Class<?> cl = type;
1090                int dimensions = 0;
1091                while (cl.isArray()) {
1092                    dimensions++;
1093                    cl = cl.getComponentType();
1094                }
1095                StringBuffer sb = new StringBuffer();
1096                sb.append(cl.getName());
1097                for (int i = 0; i < dimensions; i++) {
1098                    sb.append("[]");
1099                }
1100                return sb.toString();
1101            } catch (Throwable e) { /*FALLTHRU*/ }
1102        }
1103        return type.getName();
1104    }
