026package java.lang.reflect;
027
028
029/**
030 * ParameterizedType represents a parameterized type such as
031 * Collection&lt;String&gt;.
032 *
033 * <p>A parameterized type is created the first time it is needed by a
034 * reflective method, as specified in this package. When a
035 * parameterized type p is created, the generic type declaration that
036 * p instantiates is resolved, and all type arguments of p are created
037 * recursively. See {@link java.lang.reflect.TypeVariable
038 * TypeVariable} for details on the creation process for type
039 * variables. Repeated creation of a parameterized type has no effect.
040 *
041 * <p>Instances of classes that implement this interface must implement
042 * an equals() method that equates any two instances that share the
043 * same generic type declaration and have equal type parameters.
044 *
045 * @since 1.5
046 */
047public interface ParameterizedType extends Type {
048    /**
049     * Returns an array of {@code Type} objects representing the actual type
050     * arguments to this type.
051     *
052     * <p>Note that in some cases, the returned array be empty. This can occur
053     * if this type represents a non-parameterized type nested within
054     * a parameterized type.
055     *
056     * @return an array of {@code Type} objects representing the actual type
057     *     arguments to this type
058     * @throws TypeNotPresentException if any of the
059     *     actual type arguments refers to a non-existent type declaration
060     * @throws MalformedParameterizedTypeException if any of the
061     *     actual type parameters refer to a parameterized type that cannot
062     *     be instantiated for any reason
063     * @since 1.5
064     */
065    Type[] getActualTypeArguments();
066
067    /**
068     * Returns the {@code Type} object representing the class or interface
069     * that declared this type.
070     *
071     * @return the {@code Type} object representing the class or interface
072     *     that declared this type
073     * @since 1.5
074     */
075    Type getRawType();
076
077    /**
078     * Returns a {@code Type} object representing the type that this type
079     * is a member of.  For example, if this type is {@code O<T>.I<S>},
080     * return a representation of {@code O<T>}.
081     *
082     * <p>If this type is a top-level type, {@code null} is returned.
083     *
084     * @return a {@code Type} object representing the type that
085     *     this type is a member of. If this type is a top-level type,
086     *     {@code null} is returned
087     * @throws TypeNotPresentException if the owner type
088     *     refers to a non-existent type declaration
089     * @throws MalformedParameterizedTypeException if the owner type
090     *     refers to a parameterized type that cannot be instantiated
091     *     for any reason
092     * @since 1.5
093     */
094    Type getOwnerType();
095}
