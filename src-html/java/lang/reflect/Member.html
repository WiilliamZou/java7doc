026package java.lang.reflect;
027
028/**
029 * Member is an interface that reflects identifying information about
030 * a single member (a field or a method) or a constructor.
031 *
032 * @see java.lang.Class
033 * @see Field
034 * @see Method
035 * @see Constructor
036 *
037 * @author Nakul Saraiya
038 */
039public
040interface Member {
041
042    /**
043     * Identifies the set of all public members of a class or interface,
044     * including inherited members.
045     * @see java.lang.SecurityManager#checkMemberAccess
046     */
047    public static final int PUBLIC = 0;
048
049    /**
050     * Identifies the set of declared members of a class or interface.
051     * Inherited members are not included.
052     * @see java.lang.SecurityManager#checkMemberAccess
053     */
054    public static final int DECLARED = 1;
055
056    /**
057     * Returns the Class object representing the class or interface
058     * that declares the member or constructor represented by this Member.
059     *
060     * @return an object representing the declaring class of the
061     * underlying member
062     */
063    public Class<?> getDeclaringClass();
064
065    /**
066     * Returns the simple name of the underlying member or constructor
067     * represented by this Member.
068     *
069     * @return the simple name of the underlying member
070     */
071    public String getName();
072
073    /**
074     * Returns the Java language modifiers for the member or
075     * constructor represented by this Member, as an integer.  The
076     * Modifier class should be used to decode the modifiers in
077     * the integer.
078     *
079     * @return the Java language modifiers for the underlying member
080     * @see Modifier
081     */
082    public int getModifiers();
083
084    /**
085     * Returns {@code true} if this member was introduced by
086     * the compiler; returns {@code false} otherwise.
087     *
088     * @return true if and only if this member was introduced by
089     * the compiler.
090     * @since 1.5
091     */
092    public boolean isSynthetic();
093}
