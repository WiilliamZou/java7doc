073/**
074 * Instances of the class {@code Class} represent classes and
075 * interfaces in a running Java application.  An enum is a kind of
076 * class and an annotation is a kind of interface.  Every array also
077 * belongs to a class that is reflected as a {@code Class} object
078 * that is shared by all arrays with the same element type and number
079 * of dimensions.  The primitive Java types ({@code boolean},
080 * {@code byte}, {@code char}, {@code short},
081 * {@code int}, {@code long}, {@code float}, and
082 * {@code double}), and the keyword {@code void} are also
083 * represented as {@code Class} objects.
084 *
085 * <p> {@code Class} has no public constructor. Instead {@code Class}
086 * objects are constructed automatically by the Java Virtual Machine as classes
087 * are loaded and by calls to the {@code defineClass} method in the class
088 * loader.
089 *
090 * <p> The following example uses a {@code Class} object to print the
091 * class name of an object:
092 *
093 * <p> <blockquote><pre>
094 *     void printClassName(Object obj) {
095 *         System.out.println("The class of " + obj +
096 *                            " is " + obj.getClass().getName());
097 *     }
098 * </pre></blockquote>
099 *
100 * <p> It is also possible to get the {@code Class} object for a named
101 * type (or for void) using a class literal.  See Section 15.8.2 of
102 * <cite>The Java&trade; Language Specification</cite>.
103 * For example:
104 *
105 * <p> <blockquote>
106 *     {@code System.out.println("The name of class Foo is: "+Foo.class.getName());}
107 * </blockquote>
108 *
109 * @param <T> the type of the class modeled by this {@code Class}
110 * object.  For example, the type of {@code String.class} is {@code
111 * Class<String>}.  Use {@code Class<?>} if the class being modeled is
112 * unknown.
113 *
114 * @author  unascribed
115 * @see     java.lang.ClassLoader#defineClass(byte[], int, int)
116 * @since   JDK1.0
117 */
118public final

199    /**
200     * Returns the {@code Class} object associated with the class or
201     * interface with the given string name, using the given class loader.
202     * Given the fully qualified name for a class or interface (in the same
203     * format returned by {@code getName}) this method attempts to
204     * locate, load, and link the class or interface.  The specified class
205     * loader is used to load the class or interface.  If the parameter
206     * {@code loader} is null, the class is loaded through the bootstrap
207     * class loader.  The class is initialized only if the
208     * {@code initialize} parameter is {@code true} and if it has
209     * not been initialized earlier.
210     *
211     * <p> If {@code name} denotes a primitive type or void, an attempt
212     * will be made to locate a user-defined class in the unnamed package whose
213     * name is {@code name}. Therefore, this method cannot be used to
214     * obtain any of the {@code Class} objects representing primitive
215     * types or void.
216     *
217     * <p> If {@code name} denotes an array class, the component type of
218     * the array class is loaded but not initialized.
219     *
220     * <p> For example, in an instance method the expression:
221     *
222     * <blockquote>
223     *  {@code Class.forName("Foo")}
224     * </blockquote>
225     *
226     * is equivalent to:
227     *
228     * <blockquote>
229     *  {@code Class.forName("Foo", true, this.getClass().getClassLoader())}
230     * </blockquote>
231     *
232     * Note that this method throws errors related to loading, linking or
233     * initializing as specified in Sections 12.2, 12.3 and 12.4 of <em>The
234     * Java Language Specification</em>.
235     * Note that this method does not check whether the requested class
236     * is accessible to its caller.
237     *
238     * <p> If the {@code loader} is {@code null}, and a security
239     * manager is present, and the caller's class loader is not null, then this
240     * method calls the security manager's {@code checkPermission} method
241     * with a {@code RuntimePermission("getClassLoader")} permission to
242     * ensure it's ok to access the bootstrap class loader.
243     *
244     * @param name       fully qualified name of the desired class
245     * @param initialize whether the class must be initialized
246     * @param loader     class loader from which the class must be loaded
247     * @return           class object representing the desired class
248     *
249     * @exception LinkageError if the linkage fails
250     * @exception ExceptionInInitializerError if the initialization provoked
251     *            by this method fails
252     * @exception ClassNotFoundException if the class cannot be located by
253     *            the specified class loader
254     *
255     * @see       java.lang.Class#forName(String)
256     * @see       java.lang.ClassLoader
257     * @since     1.2
258     */
259    @CallerSensitive
260    public static Class<?> forName(String name, boolean initialize,
261                                   ClassLoader loader)
262        throws ClassNotFoundException
263    {
264        Class<?> caller = null;
265        SecurityManager sm = System.getSecurityManager();
266        if (sm != null) {
267            // Reflective call to get caller class is only needed if a security manager
268            // is present.  Avoid the overhead of making this call otherwise.
269            caller = Reflection.getCallerClass();
270            if (loader == null) {
271                ClassLoader ccl = ClassLoader.getClassLoader(caller);
272                if (ccl != null) {
273                    sm.checkPermission(
274                        SecurityConstants.GET_CLASSLOADER_PERMISSION);
275                }
276            }
277        }
278        return forName0(name, initialize, loader, caller);
279    }

287    /**
288     * Creates a new instance of the class represented by this {@code Class}
289     * object.  The class is instantiated as if by a {@code new}
290     * expression with an empty argument list.  The class is initialized if it
291     * has not already been initialized.
292     *
293     * <p>Note that this method propagates any exception thrown by the
294     * nullary constructor, including a checked exception.  Use of
295     * this method effectively bypasses the compile-time exception
296     * checking that would otherwise be performed by the compiler.
297     * The {@link
298     * java.lang.reflect.Constructor#newInstance(java.lang.Object...)
299     * Constructor.newInstance} method avoids this problem by wrapping
300     * any exception thrown by the constructor in a (checked) {@link
301     * java.lang.reflect.InvocationTargetException}.
302     *
303     * @return     a newly allocated instance of the class represented by this
304     *             object.
305     * @exception  IllegalAccessException  if the class or its nullary
306     *               constructor is not accessible.
307     * @exception  InstantiationException
308     *               if this {@code Class} represents an abstract class,
309     *               an interface, an array class, a primitive type, or void;
310     *               or if the class has no nullary constructor;
311     *               or if the instantiation fails for some other reason.
312     * @exception  ExceptionInInitializerError if the initialization
313     *               provoked by this method fails.
314     * @exception  SecurityException
315     *             If a security manager, <i>s</i>, is present and any of the
316     *             following conditions is met:
317     *
318     *             <ul>
319     *
320     *             <li> invocation of
321     *             {@link SecurityManager#checkMemberAccess
322     *             s.checkMemberAccess(this, Member.PUBLIC)} denies
323     *             creation of new instances of this class
324     *
325     *             <li> the caller's class loader is not the same as or an
326     *             ancestor of the class loader for the current class and
327     *             invocation of {@link SecurityManager#checkPackageAccess
328     *             s.checkPackageAccess()} denies access to the package
329     *             of this class
330     *
331     *             </ul>
332     *
333     */
334    @CallerSensitive
335    public T newInstance()
336        throws InstantiationException, IllegalAccessException
337    {
338        if (System.getSecurityManager() != null) {
339            checkMemberAccess(Member.PUBLIC, Reflection.getCallerClass(), false);
340        }
341
342        // NOTE: the following code may not be strictly correct under
343        // the current Java memory model.
344
345        // Constructor lookup
346        if (cachedConstructor == null) {
347            if (this == Class.class) {
348                throw new IllegalAccessException(
349                    "Can not call newInstance() on the Class for java.lang.Class"
350                );
351            }
352            try {
353                Class<?>[] empty = {};
354                final Constructor<T> c = getConstructor0(empty, Member.DECLARED);
355                // Disable accessibility checks on the constructor
356                // since we have to do the security check here anyway
357                // (the stack depth is wrong for the Constructor's
358                // security check to work)
359                java.security.AccessController.doPrivileged(
360                    new java.security.PrivilegedAction<Void>() {
361                        public Void run() {
362                                c.setAccessible(true);
363                                return null;
364                            }
365                        });
366                cachedConstructor = c;
367            } catch (NoSuchMethodException e) {
368                throw new InstantiationException(getName());
369            }
370        }
371        Constructor<T> tmpConstructor = cachedConstructor;
372        // Security check (same as in java.lang.reflect.Constructor)
373        int modifiers = tmpConstructor.getModifiers();
374        if (!Reflection.quickCheckMemberAccess(this, modifiers)) {
375            Class<?> caller = Reflection.getCallerClass();
376            if (newInstanceCallerCache != caller) {
377                Reflection.ensureMemberAccess(caller, this, null, modifiers);
378                newInstanceCallerCache = caller;
379            }
380        }
381        // Run constructor
382        try {
383            return tmpConstructor.newInstance((Object[])null);
384        } catch (InvocationTargetException e) {
385            Unsafe.getUnsafe().throwException(e.getTargetException());
386            // Not reached
387            return null;
388        }
389    }

894     * If this {@code Class} object represents a local or anonymous
895     * class within a method, returns a {@link
896     * java.lang.reflect.Method Method} object representing the
897     * immediately enclosing method of the underlying class. Returns
898     * {@code null} otherwise.
899     *
900     * In particular, this method returns {@code null} if the underlying
901     * class is a local or anonymous class immediately enclosed by a type
902     * declaration, instance initializer or static initializer.
903     *
904     * @return the immediately enclosing method of the underlying class, if
905     *     that class is a local or anonymous class; otherwise {@code null}.
906     * @since 1.5
907     */
908    @CallerSensitive
909    public Method getEnclosingMethod() {
910        EnclosingMethodInfo enclosingInfo = getEnclosingMethodInfo();
911
912        if (enclosingInfo == null)
913            return null;
914        else {
915            if (!enclosingInfo.isMethod())
916                return null;
917
918            MethodRepository typeInfo = MethodRepository.make(enclosingInfo.getDescriptor(),
919                                                              getFactory());
920            Class<?>   returnType       = toClass(typeInfo.getReturnType());
921            Type []    parameterTypes   = typeInfo.getParameterTypes();
922            Class<?>[] parameterClasses = new Class<?>[parameterTypes.length];
923
924            // Convert Types to Classes; returned types *should*
925            // be class objects since the methodDescriptor's used
926            // don't have generics information
927            for(int i = 0; i < parameterClasses.length; i++)
928                parameterClasses[i] = toClass(parameterTypes[i]);
929
930            // Perform access check
931            Class<?> enclosingCandidate = enclosingInfo.getEnclosingClass();
932            // be very careful not to change the stack depth of this
933            // checkMemberAccess call for security reasons
934            // see java.lang.SecurityManager.checkMemberAccess
935            //
936            // Note that we need to do this on the enclosing class
937            enclosingCandidate.checkMemberAccess(Member.DECLARED,
938                                                 Reflection.getCallerClass(), true);
939            /*
940             * Loop over all declared methods; match method name,
941             * number of and type of parameters, *and* return
942             * type.  Matching return type is also necessary
943             * because of covariant returns, etc.
944             */
945            for(Method m: enclosingCandidate.getDeclaredMethods()) {
946                if (m.getName().equals(enclosingInfo.getName()) ) {
947                    Class<?>[] candidateParamClasses = m.getParameterTypes();
948                    if (candidateParamClasses.length == parameterClasses.length) {
949                        boolean matches = true;
950                        for(int i = 0; i < candidateParamClasses.length; i++) {
951                            if (!candidateParamClasses[i].equals(parameterClasses[i])) {
952                                matches = false;
953                                break;
954                            }
955                        }
956
957                        if (matches) { // finally, check return type
958                            if (m.getReturnType().equals(returnType) )
959                                return m;
960                        }
961                    }
962                }
963            }
964
965            throw new InternalError("Enclosing method not found");
966        }
967    }

1166    /**
1167     * Returns the simple name of the underlying class as given in the
1168     * source code. Returns an empty string if the underlying class is
1169     * anonymous.
1170     *
1171     * <p>The simple name of an array is the simple name of the
1172     * component type with "[]" appended.  In particular the simple
1173     * name of an array whose component type is anonymous is "[]".
1174     *
1175     * @return the simple name of the underlying class
1176     * @since 1.5
1177     */
1178    public String getSimpleName() {
1179        if (isArray())
1180            return getComponentType().getSimpleName()+"[]";
1181
1182        String simpleName = getSimpleBinaryName();
1183        if (simpleName == null) { // top level class
1184            simpleName = getName();
1185            return simpleName.substring(simpleName.lastIndexOf(".")+1); // strip the package name
1186        }
1187        // According to JLS3 "Binary Compatibility" (13.1) the binary
1188        // name of non-package classes (not top level) is the binary
1189        // name of the immediately enclosing class followed by a '$' followed by:
1190        // (for nested and inner classes): the simple name.
1191        // (for local classes): 1 or more digits followed by the simple name.
1192        // (for anonymous classes): 1 or more digits.
1193
1194        // Since getSimpleBinaryName() will strip the binary name of
1195        // the immediatly enclosing class, we are now looking at a
1196        // string that matches the regular expression "\$[0-9]*"
1197        // followed by a simple name (considering the simple of an
1198        // anonymous class to be the empty string).
1199
1200        // Remove leading "\$[0-9]*" from the name
1201        int length = simpleName.length();
1202        if (length < 1 || simpleName.charAt(0) != '$')
1203            throw new InternalError("Malformed class name");
1204        int index = 1;
1205        while (index < length && isAsciiDigit(simpleName.charAt(index)))
1206            index++;
1207        // Eventually, this is the empty string iff this is an anonymous class
1208        return simpleName.substring(index);
1209    }

3264    private transient Map<Class<? extends Annotation>, Annotation> declaredAnnotations;
3265    // Value of classRedefinedCount when we last cleared the cached annotations and declaredAnnotations fields

3268    // Clears cached values that might possibly have been obsoleted by
3269    // a class redefinition.
3270    private void clearAnnotationCachesOnClassRedefinition() {
3271        if (lastAnnotationsRedefinedCount != classRedefinedCount) {
3272            annotations = declaredAnnotations = null;
3273            lastAnnotationsRedefinedCount = classRedefinedCount;
3274        }
3275    }

3277    private synchronized void initAnnotationsIfNecessary() {
3278        clearAnnotationCachesOnClassRedefinition();
3279        if (annotations != null)
3280            return;
3281        declaredAnnotations = AnnotationParser.parseAnnotations(
3282            getRawAnnotations(), getConstantPool(), this);
3283        Class<?> superClass = getSuperclass();
3284        if (superClass == null) {
3285            annotations = declaredAnnotations;
3286        } else {
3287            annotations = new HashMap<>();
3288            superClass.initAnnotationsIfNecessary();
3289            for (Map.Entry<Class<? extends Annotation>, Annotation> e : superClass.annotations.entrySet()) {
3290                Class<? extends Annotation> annotationClass = e.getKey();
3291                if (AnnotationType.getInstance(annotationClass).isInherited())
3292                    annotations.put(annotationClass, e.getValue());
3293            }
3294            annotations.putAll(declaredAnnotations);
3295        }
3296    }
