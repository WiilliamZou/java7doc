026package java.lang.management;
027
028/**
029 * Types of {@link MemoryPoolMXBean memory pools}.
030 *
031 * @author  Mandy Chung
032 * @since   1.5
033 */
034public enum MemoryType {
035
036    /**
037     * Heap memory type.
038     * <p>
039     * The Java virtual machine has a <i>heap</i>
040     * that is the runtime data area from which
041     * memory for all class instances and arrays are allocated.
042     */
043    HEAP("Heap memory"),
044
045    /**
046     * Non-heap memory type.
047     * <p>
048     * The Java virtual machine manages memory other than the heap
049     * (referred as <i>non-heap memory</i>).  The non-heap memory includes
050     * the <i>method area</i> and memory required for the internal
051     * processing or optimization for the Java virtual machine.
052     * It stores per-class structures such as a runtime
053     * constant pool, field and method data, and the code for
054     * methods and constructors.
055     */
056    NON_HEAP("Non-heap memory");
057
058    private final String description;
059
060    private MemoryType(String s) {
061        this.description = s;
062    }
063
064    /**
065     * Returns the string representation of this <tt>MemoryType</tt>.
066     * @return the string representation of this <tt>MemoryType</tt>.
067     */
068    public String toString() {
069        return description;
070    }
071
072    private static final long serialVersionUID = 6992337162326171013L;
073}
