026package java.lang;
027
028/**
029 * Thrown to indicate that a method has been passed an illegal or
030 * inappropriate argument.
031 *
032 * @author  unascribed
033 * @see     java.lang.Thread#setPriority(int)
034 * @since   JDK1.0
035 */
036public
037class IllegalArgumentException extends RuntimeException {
038    /**
039     * Constructs an <code>IllegalArgumentException</code> with no
040     * detail message.
041     */
042    public IllegalArgumentException() {
043        super();
044    }
045
046    /**
047     * Constructs an <code>IllegalArgumentException</code> with the
048     * specified detail message.
049     *
050     * @param   s   the detail message.
051     */
052    public IllegalArgumentException(String s) {
053        super(s);
054    }
055
056    /**
057     * Constructs a new exception with the specified detail message and
058     * cause.
059     *
060     * <p>Note that the detail message associated with <code>cause</code> is
061     * <i>not</i> automatically incorporated in this exception's detail
062     * message.
063     *
064     * @param  message the detail message (which is saved for later retrieval
065     *         by the {@link Throwable#getMessage()} method).
066     * @param  cause the cause (which is saved for later retrieval by the
067     *         {@link Throwable#getCause()} method).  (A <tt>null</tt> value
068     *         is permitted, and indicates that the cause is nonexistent or
069     *         unknown.)
070     * @since 1.5
071     */
072    public IllegalArgumentException(String message, Throwable cause) {
073        super(message, cause);
074    }
075
076    /**
077     * Constructs a new exception with the specified cause and a detail
078     * message of <tt>(cause==null ? null : cause.toString())</tt> (which
079     * typically contains the class and detail message of <tt>cause</tt>).
080     * This constructor is useful for exceptions that are little more than
081     * wrappers for other throwables (for example, {@link
082     * java.security.PrivilegedActionException}).
083     *
084     * @param  cause the cause (which is saved for later retrieval by the
085     *         {@link Throwable#getCause()} method).  (A <tt>null</tt> value is
086     *         permitted, and indicates that the cause is nonexistent or
087     *         unknown.)
088     * @since  1.5
089     */
090    public IllegalArgumentException(Throwable cause) {
091        super(cause);
092    }
093
094    private static final long serialVersionUID = -5365630128856068164L;
095}
