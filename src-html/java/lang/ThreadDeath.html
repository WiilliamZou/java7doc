026package java.lang;
027
028/**
029 * An instance of {@code ThreadDeath} is thrown in the victim thread
030 * when the (deprecated) {@link Thread#stop()} method is invoked.
031 *
032 * <p>An application should catch instances of this class only if it
033 * must clean up after being terminated asynchronously.  If
034 * {@code ThreadDeath} is caught by a method, it is important that it
035 * be rethrown so that the thread actually dies.
036 *
037 * <p>The {@linkplain ThreadGroup#uncaughtException top-level error
038 * handler} does not print out a message if {@code ThreadDeath} is
039 * never caught.
040 *
041 * <p>The class {@code ThreadDeath} is specifically a subclass of
042 * {@code Error} rather than {@code Exception}, even though it is a
043 * "normal occurrence", because many applications catch all
044 * occurrences of {@code Exception} and then discard the exception.
045 *
046 * @since   JDK1.0
047 */
048
049public class ThreadDeath extends Error {
050    private static final long serialVersionUID = -4417128565033088268L;
051}
