026package java.util.zip;
027
028/*
029 * This class defines the constants that are used by the classes
030 * which manipulate Zip64 files.
031 */
032
033class ZipConstants64 {
034
035    /*
036     * ZIP64 constants
037     */
038    static final long ZIP64_ENDSIG = 0x06064b50L;  // "PK\006\006"
039    static final long ZIP64_LOCSIG = 0x07064b50L;  // "PK\006\007"
040    static final int  ZIP64_ENDHDR = 56;           // ZIP64 end header size
041    static final int  ZIP64_LOCHDR = 20;           // ZIP64 end loc header size
042    static final int  ZIP64_EXTHDR = 24;           // EXT header size
043    static final int  ZIP64_EXTID  = 0x0001;       // Extra field Zip64 header ID
044
045    static final int  ZIP64_MAGICCOUNT = 0xFFFF;
046    static final long ZIP64_MAGICVAL = 0xFFFFFFFFL;
047
048    /*
049     * Zip64 End of central directory (END) header field offsets
050     */
051    static final int  ZIP64_ENDLEN = 4;       // size of zip64 end of central dir
052    static final int  ZIP64_ENDVEM = 12;      // version made by
053    static final int  ZIP64_ENDVER = 14;      // version needed to extract
054    static final int  ZIP64_ENDNMD = 16;      // number of this disk
055    static final int  ZIP64_ENDDSK = 20;      // disk number of start
056    static final int  ZIP64_ENDTOD = 24;      // total number of entries on this disk
057    static final int  ZIP64_ENDTOT = 32;      // total number of entries
058    static final int  ZIP64_ENDSIZ = 40;      // central directory size in bytes
059    static final int  ZIP64_ENDOFF = 48;      // offset of first CEN header
060    static final int  ZIP64_ENDEXT = 56;      // zip64 extensible data sector
061
062    /*
063     * Zip64 End of central directory locator field offsets
064     */
065    static final int  ZIP64_LOCDSK = 4;       // disk number start
066    static final int  ZIP64_LOCOFF = 8;       // offset of zip64 end
067    static final int  ZIP64_LOCTOT = 16;      // total number of disks
068
069    /*
070     * Zip64 Extra local (EXT) header field offsets
071     */
072    static final int  ZIP64_EXTCRC = 4;       // uncompressed file crc-32 value
073    static final int  ZIP64_EXTSIZ = 8;       // compressed size, 8-byte
074    static final int  ZIP64_EXTLEN = 16;      // uncompressed size, 8-byte
075
076    /*
077     * Language encoding flag EFS
078     */
079    static final int EFS = 0x800;       // If this bit is set the filename and
080                                        // comment fields for this file must be
081                                        // encoded using UTF-8.
082
083    private ZipConstants64() {}
084}
