026package java.util.zip;
027
028import java.nio.ByteBuffer;
029import sun.nio.ch.DirectBuffer;
030
031/**
032 * A class that can be used to compute the Adler-32 checksum of a data
033 * stream. An Adler-32 checksum is almost as reliable as a CRC-32 but
034 * can be computed much faster.
035 *
036 * @see         Checksum
037 * @author      David Connelly
038 */
039public
040class Adler32 implements Checksum {
041    private int adler = 1;
042
043    /**
044     * Creates a new Adler32 object.
045     */
046    public Adler32() {
047    }
048
049    /**
050     * Updates the checksum with the specified byte (the low eight
051     * bits of the argument b).
052     *
053     * @param b the byte to update the checksum with
054     */
055    public void update(int b) {
056        adler = update(adler, b);
057    }
058
059    /**
060     * Updates the checksum with the specified array of bytes.
061     */
062    public void update(byte[] b, int off, int len) {
063        if (b == null) {
064            throw new NullPointerException();
065        }
066        if (off < 0 || len < 0 || off > b.length - len) {
067            throw new ArrayIndexOutOfBoundsException();
068        }
069        adler = updateBytes(adler, b, off, len);
070    }
071
072    /**
073     * Updates the checksum with the specified array of bytes.
074     *
075     * @param b the byte array to update the checksum with
076     */
077    public void update(byte[] b) {
078        adler = updateBytes(adler, b, 0, b.length);
079    }
080
081    /**
082     * Updates the checksum with the bytes from the specified buffer.
083     *
084     * The checksum is updated using
085     * buffer.{@link java.nio.Buffer#remaining() remaining()}
086     * bytes starting at
087     * buffer.{@link java.nio.Buffer#position() position()}
088     * Upon return, the buffer's position will be updated to its
089     * limit; its limit will not have been changed.
090     *
091     * @param buffer the ByteBuffer to update the checksum with
092     */
093    private void update(ByteBuffer buffer) {
094        int pos = buffer.position();
095        int limit = buffer.limit();
096        assert (pos <= limit);
097        int rem = limit - pos;
098        if (rem <= 0)
099            return;
100        if (buffer instanceof DirectBuffer) {
101            adler = updateByteBuffer(adler, ((DirectBuffer)buffer).address(), pos, rem);
102        } else if (buffer.hasArray()) {
103            adler = updateBytes(adler, buffer.array(), pos + buffer.arrayOffset(), rem);
104        } else {
105            byte[] b = new byte[rem];
106            buffer.get(b);
107            adler = updateBytes(adler, b, 0, b.length);
108        }
109        buffer.position(limit);
110    }
111
112    /**
113     * Resets the checksum to initial value.
114     */
115    public void reset() {
116        adler = 1;
117    }
118
119    /**
120     * Returns the checksum value.
121     */
122    public long getValue() {
123        return (long)adler & 0xffffffffL;
124    }
125
126    // Set up JavaUtilZipAccess in SharedSecrets
127    static {
128       sun.misc.SharedSecrets.setJavaUtilZipAccess(new sun.misc.JavaUtilZipAccess() {
129           public void update(Adler32 adler32, ByteBuffer buf) {
130               adler32.update(buf);
131           }
132        });
133    }
134
135    private native static int update(int adler, int b);
136    private native static int updateBytes(int adler, byte[] b, int off,
137                                          int len);
138    private native static int updateByteBuffer(int adler, long addr,
139                                               int off, int len);
140}
