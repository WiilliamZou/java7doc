164    /**
165     * Opens a new <code>ZipFile</code> to read from the specified
166     * <code>File</code> object in the specified mode.  The mode argument
167     * must be either <tt>OPEN_READ</tt> or <tt>OPEN_READ | OPEN_DELETE</tt>.
168     *
169     * <p>First, if there is a security manager, its <code>checkRead</code>
170     * method is called with the <code>name</code> argument as its argument to
171     * ensure the read is allowed.
172     *
173     * @param file the ZIP file to be opened for reading
174     * @param mode the mode in which the file is to be opened
175     * @param charset
176     *        the {@linkplain java.nio.charset.Charset charset} to
177     *        be used to decode the ZIP entry name and comment that are not
178     *        encoded by using UTF-8 encoding (indicated by entry's general
179     *        purpose flag).
180     *
181     * @throws ZipException if a ZIP format error has occurred
182     * @throws IOException if an I/O error has occurred
183     *
184     * @throws SecurityException
185     *         if a security manager exists and its <code>checkRead</code>
186     *         method doesn't allow read access to the file,or its
187     *         <code>checkDelete</code> method doesn't allow deleting the
188     *         file when the <tt>OPEN_DELETE</tt> flag is set
189     *
190     * @throws IllegalArgumentException if the <tt>mode</tt> argument is invalid
191     *
192     * @see SecurityManager#checkRead(java.lang.String)
193     *
194     * @since 1.7
195     */
196    public ZipFile(File file, int mode, Charset charset) throws IOException
197    {
198        if (((mode & OPEN_READ) == 0) ||
199            ((mode & ~(OPEN_READ | OPEN_DELETE)) != 0)) {
200            throw new IllegalArgumentException("Illegal mode: 0x"+
201                                               Integer.toHexString(mode));
202        }
203        String name = file.getPath();
204        SecurityManager sm = System.getSecurityManager();
205        if (sm != null) {
206            sm.checkRead(name);
207            if ((mode & OPEN_DELETE) != 0) {
208                sm.checkDelete(name);
209            }
210        }
211        if (charset == null)
212            throw new NullPointerException("charset is null");
213        this.zc = ZipCoder.get(charset);
214        long t0 = System.nanoTime();
215        jzfile = open(name, mode, file.lastModified(), usemmap);
216        sun.misc.PerfCounter.getZipFileOpenTime().addElapsedTimeFrom(t0);
217        sun.misc.PerfCounter.getZipFileCount().increment();
218        this.name = name;
219        this.total = getTotal(jzfile);
220        this.locsig = startsWithLOC(jzfile);
221    }

291    /**
292     * Returns the zip file entry for the specified name, or null
293     * if not found.
294     *
295     * @param name the name of the entry
296     * @return the zip file entry, or null if not found
297     * @throws IllegalStateException if the zip file has been closed
298     */
299    public ZipEntry getEntry(String name) {
300        if (name == null) {
301            throw new NullPointerException("name");
302        }
303        long jzentry = 0;
304        synchronized (this) {
305            ensureOpen();
306            jzentry = getEntry(jzfile, zc.getBytes(name), true);
307            if (jzentry != 0) {
308                ZipEntry ze = getZipEntry(name, jzentry);
309                freeEntry(jzfile, jzentry);
310                return ze;
311            }
312        }
313        return null;
314    }

322    // the outstanding inputstreams that need to be closed,

324    private final Map<InputStream, Inflater> streams = new WeakHashMap<>();

326    /**
327     * Returns an input stream for reading the contents of the specified
328     * zip file entry.
329     *
330     * <p> Closing this ZIP file will, in turn, close all input
331     * streams that have been returned by invocations of this method.
332     *
333     * @param entry the zip file entry
334     * @return the input stream for reading the contents of the specified
335     * zip file entry.
336     * @throws ZipException if a ZIP format error has occurred
337     * @throws IOException if an I/O error has occurred
338     * @throws IllegalStateException if the zip file has been closed
339     */
340    public InputStream getInputStream(ZipEntry entry) throws IOException {
341        if (entry == null) {
342            throw new NullPointerException("entry");
343        }
344        long jzentry = 0;
345        ZipFileInputStream in = null;
346        synchronized (this) {
347            ensureOpen();
348            if (!zc.isUTF8() && (entry.flag & EFS) != 0) {
349                jzentry = getEntry(jzfile, zc.getBytesUTF8(entry.name), false);
350            } else {
351                jzentry = getEntry(jzfile, zc.getBytes(entry.name), false);
352            }
353            if (jzentry == 0) {
354                return null;
355            }
356            in = new ZipFileInputStream(jzentry);
357
358            switch (getEntryMethod(jzentry)) {
359            case STORED:
360                synchronized (streams) {
361                    streams.put(in, null);
362                }
363                return in;
364            case DEFLATED:
365                // MORE: Compute good size for inflater stream:
366                long size = getEntrySize(jzentry) + 2; // Inflater likes a bit of slack
367                if (size > 65536) size = 8192;
368                if (size <= 0) size = 4096;
369                Inflater inf = getInflater();
370                InputStream is =
371                    new ZipFileInflaterInputStream(in, inf, (int)size);
372                synchronized (streams) {
373                    streams.put(is, inf);
374                }
375                return is;
376            default:
377                throw new ZipException("invalid compression method");
378            }
379        }
380    }

382    private class ZipFileInflaterInputStream extends InflaterInputStream {
383        private volatile boolean closeRequested = false;
384        private boolean eof = false;
385        private final ZipFileInputStream zfin;
386
387        ZipFileInflaterInputStream(ZipFileInputStream zfin, Inflater inf,
388                int size) {
389            super(zfin, inf, size);
390            this.zfin = zfin;
391        }
392
393        public void close() throws IOException {
394            if (closeRequested)
395                return;
396            closeRequested = true;
397
398            super.close();
399            Inflater inf;
400            synchronized (streams) {
401                inf = streams.remove(this);
402            }
403            if (inf != null) {
404                releaseInflater(inf);
405            }
406        }
407
408        // Override fill() method to provide an extra "dummy" byte
409        // at the end of the input stream. This is required when
410        // using the "nowrap" Inflater option.
411        protected void fill() throws IOException {
412            if (eof) {
413                throw new EOFException("Unexpected end of ZLIB input stream");
414            }
415            len = in.read(buf, 0, buf.length);
416            if (len == -1) {
417                buf[0] = 0;
418                len = 1;
419                eof = true;
420            }
421            inf.setInput(buf, 0, len);
422        }
423
424        public int available() throws IOException {
425            if (closeRequested)
426                return 0;
427            long avail = zfin.size() - inf.getBytesWritten();
428            return (avail > (long) Integer.MAX_VALUE ?
429                    Integer.MAX_VALUE : (int) avail);
430        }
431
432        protected void finalize() throws Throwable {
433            close();
434        }
435    }

437    /*
438     * Gets an inflater from the list of available inflaters or allocates
439     * a new one.
440     */
441    private Inflater getInflater() {
442        Inflater inf;
443        synchronized (inflaterCache) {
444            while (null != (inf = inflaterCache.poll())) {
445                if (false == inf.ended()) {
446                    return inf;
447                }
448            }
449        }
450        return new Inflater(true);
451    }
452
453    /*
454     * Releases the specified inflater to the list of available inflaters.
455     */
456    private void releaseInflater(Inflater inf) {
457        if (false == inf.ended()) {
458            inf.reset();
459            synchronized (inflaterCache) {
460                inflaterCache.add(inf);
461            }
462        }
463    }

476    /**
477     * Returns an enumeration of the ZIP file entries.
478     * @return an enumeration of the ZIP file entries
479     * @throws IllegalStateException if the zip file has been closed
480     */
481    public Enumeration<? extends ZipEntry> entries() {
482        ensureOpen();
483        return new Enumeration<ZipEntry>() {
484                private int i = 0;
485                public boolean hasMoreElements() {
486                    synchronized (ZipFile.this) {
487                        ensureOpen();
488                        return i < total;
489                    }
490                }
491                public ZipEntry nextElement() throws NoSuchElementException {
492                    synchronized (ZipFile.this) {
493                        ensureOpen();
494                        if (i >= total) {
495                            throw new NoSuchElementException();
496                        }
497                        long jzentry = getNextEntry(jzfile, i++);
498                        if (jzentry == 0) {
499                            String message;
500                            if (closeRequested) {
501                                message = "ZipFile concurrently closed";
502                            } else {
503                                message = getZipMessage(ZipFile.this.jzfile);
504                            }
505                            throw new ZipError("jzentry == 0" +
506                                               ",\n jzfile = " + ZipFile.this.jzfile +
507                                               ",\n total = " + ZipFile.this.total +
508                                               ",\n name = " + ZipFile.this.name +
509                                               ",\n i = " + i +
510                                               ",\n message = " + message
511                                );
512                        }
513                        ZipEntry ze = getZipEntry(null, jzentry);
514                        freeEntry(jzfile, jzentry);
515                        return ze;
516                    }
517                }
518            };
519    }

521    private ZipEntry getZipEntry(String name, long jzentry) {
522        ZipEntry e = new ZipEntry();
523        e.flag = getEntryFlag(jzentry);  // get the flag first
524        if (name != null) {
525            e.name = name;
526        } else {
527            byte[] bname = getEntryBytes(jzentry, JZENTRY_NAME);
528            if (!zc.isUTF8() && (e.flag & EFS) != 0) {
529                e.name = zc.toStringUTF8(bname, bname.length);
530            } else {
531                e.name = zc.toString(bname, bname.length);
532            }
533        }
534        e.time = getEntryTime(jzentry);
535        e.crc = getEntryCrc(jzentry);
536        e.size = getEntrySize(jzentry);
537        e. csize = getEntryCSize(jzentry);
538        e.method = getEntryMethod(jzentry);
539        e.extra = getEntryBytes(jzentry, JZENTRY_EXTRA);
540        byte[] bcomm = getEntryBytes(jzentry, JZENTRY_COMMENT);
541        if (bcomm == null) {
542            e.comment = null;
543        } else {
544            if (!zc.isUTF8() && (e.flag & EFS) != 0) {
545                e.comment = zc.toStringUTF8(bcomm, bcomm.length);
546            } else {
547                e.comment = zc.toString(bcomm, bcomm.length);
548            }
549        }
550        return e;
551    }

565    /**
566     * Closes the ZIP file.
567     * <p> Closing this ZIP file will close all of the input streams
568     * previously returned by invocations of the {@link #getInputStream
569     * getInputStream} method.
570     *
571     * @throws IOException if an I/O error has occurred
572     */
573    public void close() throws IOException {
574        if (closeRequested)
575            return;
576        closeRequested = true;
577
578        synchronized (this) {
579            // Close streams, release their inflaters
580            synchronized (streams) {
581                if (false == streams.isEmpty()) {
582                    Map<InputStream, Inflater> copy = new HashMap<>(streams);
583                    streams.clear();
584                    for (Map.Entry<InputStream, Inflater> e : copy.entrySet()) {
585                        e.getKey().close();
586                        Inflater inf = e.getValue();
587                        if (inf != null) {
588                            inf.end();
589                        }
590                    }
591                }
592            }
593
594            // Release cached inflaters
595            Inflater inf;
596            synchronized (inflaterCache) {
597                while (null != (inf = inflaterCache.poll())) {
598                    inf.end();
599                }
600            }
601
602            if (jzfile != 0) {
603                // Close the zip file
604                long zf = this.jzfile;
605                jzfile = 0;
606
607                close(zf);
608            }
609        }
610    }

648    /*
649     * Inner class implementing the input stream used to read a
650     * (possibly compressed) zip file entry.
651     */
652   private class ZipFileInputStream extends InputStream {
653        private volatile boolean closeRequested = false;
654        protected long jzentry; // address of jzentry data
655        private   long pos;     // current position within entry data
656        protected long rem;     // number of remaining bytes within entry
657        protected long size;    // uncompressed size of this entry
658
659        ZipFileInputStream(long jzentry) {
660            pos = 0;
661            rem = getEntryCSize(jzentry);
662            size = getEntrySize(jzentry);
663            this.jzentry = jzentry;
664        }
665
666        public int read(byte b[], int off, int len) throws IOException {
667            if (rem == 0) {
668                return -1;
669            }
670            if (len <= 0) {
671                return 0;
672            }
673            if (len > rem) {
674                len = (int) rem;
675            }
676            synchronized (ZipFile.this) {
677                ensureOpenOrZipException();
678
679                len = ZipFile.read(ZipFile.this.jzfile, jzentry, pos, b,
680                                   off, len);
681            }
682            if (len > 0) {
683                pos += len;
684                rem -= len;
685            }
686            if (rem == 0) {
687                close();
688            }
689            return len;
690        }
691
692        public int read() throws IOException {
693            byte[] b = new byte[1];
694            if (read(b, 0, 1) == 1) {
695                return b[0] & 0xff;
696            } else {
697                return -1;
698            }
699        }
700
701        public long skip(long n) {
702            if (n > rem)
703                n = rem;
704            pos += n;
705            rem -= n;
706            if (rem == 0) {
707                close();
708            }
709            return n;
710        }
711
712        public int available() {
713            return rem > Integer.MAX_VALUE ? Integer.MAX_VALUE : (int) rem;
714        }
715
716        public long size() {
717            return size;
718        }
719
720        public void close() {
721            if (closeRequested)
722                return;
723            closeRequested = true;
724
725            rem = 0;
726            synchronized (ZipFile.this) {
727                if (jzentry != 0 && ZipFile.this.jzfile != 0) {
728                    freeEntry(ZipFile.this.jzfile, jzentry);
729                    jzentry = 0;
730                }
731            }
732            synchronized (streams) {
733                streams.remove(this);
734            }
735        }
736
737        protected void finalize() {
738            close();
739        }
740    }
