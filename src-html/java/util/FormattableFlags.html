026package java.util;
027
028/**
029 * FomattableFlags are passed to the {@link Formattable#formatTo
030 * Formattable.formatTo()} method and modify the output format for {@linkplain
031 * Formattable Formattables}.  Implementations of {@link Formattable} are
032 * responsible for interpreting and validating any flags.
033 *
034 * @since  1.5
035 */
036public class FormattableFlags {
037
038    // Explicit instantiation of this class is prohibited.
039    private FormattableFlags() {}
040
041    /**
042     * Left-justifies the output.  Spaces (<tt>'&#92;u0020'</tt>) will be added
043     * at the end of the converted value as required to fill the minimum width
044     * of the field.  If this flag is not set then the output will be
045     * right-justified.
046     *
047     * <p> This flag corresponds to <tt>'-'</tt> (<tt>'&#92;u002d'</tt>) in
048     * the format specifier.
049     */
050    public static final int LEFT_JUSTIFY = 1<<0; // '-'
051
052    /**
053     * Converts the output to upper case according to the rules of the
054     * {@linkplain java.util.Locale locale} given during creation of the
055     * <tt>formatter</tt> argument of the {@link Formattable#formatTo
056     * formatTo()} method.  The output should be equivalent the following
057     * invocation of {@link String#toUpperCase(java.util.Locale)}
058     *
059     * <pre>
060     *     out.toUpperCase() </pre>
061     *
062     * <p> This flag corresponds to <tt>'S'</tt> (<tt>'&#92;u0053'</tt>) in
063     * the format specifier.
064     */
065    public static final int UPPERCASE = 1<<1;    // 'S'
066
067    /**
068     * Requires the output to use an alternate form.  The definition of the
069     * form is specified by the <tt>Formattable</tt>.
070     *
071     * <p> This flag corresponds to <tt>'#'</tt> (<tt>'&#92;u0023'</tt>) in
072     * the format specifier.
073     */
074    public static final int ALTERNATE = 1<<2;    // '#'
075}
