026package java.util;
027
028/**
029 * Unchecked exception thrown when a conversion and flag are incompatible.
030 *
031 * <p> Unless otherwise specified, passing a <tt>null</tt> argument to any
032 * method or constructor in this class will cause a {@link
033 * NullPointerException} to be thrown.
034 *
035 * @since 1.5
036 */
037public class FormatFlagsConversionMismatchException
038    extends IllegalFormatException
039{
040    private static final long serialVersionUID = 19120414L;
041
042    private String f;
043
044    private char c;
045
046    /**
047     * Constructs an instance of this class with the specified flag
048     * and conversion.
049     *
050     * @param  f
051     *         The flag
052     *
053     * @param  c
054     *         The conversion
055     */
056    public FormatFlagsConversionMismatchException(String f, char c) {
057        if (f == null)
058            throw new NullPointerException();
059        this.f = f;
060        this.c = c;
061    }
062
063    /**
064     * Returns the incompatible flag.
065     *
066     * @return  The flag
067     */
068     public String getFlags() {
069        return f;
070    }
071
072    /**
073     * Returns the incompatible conversion.
074     *
075     * @return  The conversion
076     */
077    public char getConversion() {
078        return c;
079    }
080
081    public String getMessage() {
082        return "Conversion = " + c + ", Flags = " + f;
083    }
084}
