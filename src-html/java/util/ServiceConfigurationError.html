026package java.util;
027
028
029/**
030 * Error thrown when something goes wrong while loading a service provider.
031 *
032 * <p> This error will be thrown in the following situations:
033 *
034 * <ul>
035 *
036 *   <li> The format of a provider-configuration file violates the <a
037 *   href="ServiceLoader.html#format">specification</a>; </li>
038 *
039 *   <li> An {@link java.io.IOException IOException} occurs while reading a
040 *   provider-configuration file; </li>
041 *
042 *   <li> A concrete provider class named in a provider-configuration file
043 *   cannot be found; </li>
044 *
045 *   <li> A concrete provider class is not a subclass of the service class;
046 *   </li>
047 *
048 *   <li> A concrete provider class cannot be instantiated; or
049 *
050 *   <li> Some other kind of error occurs. </li>
051 *
052 * </ul>
053 *
054 *
055 * @author Mark Reinhold
056 * @since 1.6
057 */
058
059public class ServiceConfigurationError
060    extends Error
061{
062
063    private static final long serialVersionUID = 74132770414881L;
064
065    /**
066     * Constructs a new instance with the specified message.
067     *
068     * @param  msg  The message, or <tt>null</tt> if there is no message
069     *
070     */
071    public ServiceConfigurationError(String msg) {
072        super(msg);
073    }
074
075    /**
076     * Constructs a new instance with the specified message and cause.
077     *
078     * @param  msg  The message, or <tt>null</tt> if there is no message
079     *
080     * @param  cause  The cause, or <tt>null</tt> if the cause is nonexistent
081     *                or unknown
082     */
083    public ServiceConfigurationError(String msg, Throwable cause) {
084        super(msg, cause);
085    }
086
087}
