026package java.util.prefs;
027
028import java.io.NotSerializableException;
029
030/**
031 * Thrown to indicate that an operation could not complete because
032 * the input did not conform to the appropriate XML document type
033 * for a collection of preferences, as per the {@link Preferences}
034 * specification.
035 *
036 * @author  Josh Bloch
037 * @see     Preferences
038 * @since   1.4
039 */
040public class InvalidPreferencesFormatException extends Exception {
041    /**
042     * Constructs an InvalidPreferencesFormatException with the specified
043     * cause.
044     *
045     * @param  cause the cause (which is saved for later retrieval by the
046     *         {@link Throwable#getCause()} method).
047     */
048    public InvalidPreferencesFormatException(Throwable cause) {
049        super(cause);
050    }
051
052   /**
053    * Constructs an InvalidPreferencesFormatException with the specified
054    * detail message.
055    *
056    * @param   message   the detail message. The detail message is saved for
057    *          later retrieval by the {@link Throwable#getMessage()} method.
058    */
059    public InvalidPreferencesFormatException(String message) {
060        super(message);
061    }
062
063    /**
064     * Constructs an InvalidPreferencesFormatException with the specified
065     * detail message and cause.
066     *
067     * @param  message   the detail message. The detail message is saved for
068     *         later retrieval by the {@link Throwable#getMessage()} method.
069     * @param  cause the cause (which is saved for later retrieval by the
070     *         {@link Throwable#getCause()} method).
071     */
072    public InvalidPreferencesFormatException(String message, Throwable cause) {
073        super(message, cause);
074    }
075
076    private static final long serialVersionUID = -791715184232119669L;
077}
