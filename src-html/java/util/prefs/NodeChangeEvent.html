026package java.util.prefs;
027
028import java.io.NotSerializableException;
029
030/**
031 * An event emitted by a <tt>Preferences</tt> node to indicate that
032 * a child of that node has been added or removed.<p>
033 *
034 * Note, that although NodeChangeEvent inherits Serializable interface from
035 * java.util.EventObject, it is not intended to be Serializable. Appropriate
036 * serialization methods are implemented to throw NotSerializableException.
037 *
038 * @author  Josh Bloch
039 * @see     Preferences
040 * @see     NodeChangeListener
041 * @see     PreferenceChangeEvent
042 * @since   1.4
043 * @serial  exclude
044 */
045
046public class NodeChangeEvent extends java.util.EventObject {
047    /**
048     * The node that was added or removed.
049     *
050     * @serial
051     */
052    private Preferences child;
053
054    /**
055     * Constructs a new <code>NodeChangeEvent</code> instance.
056     *
057     * @param parent  The parent of the node that was added or removed.
058     * @param child   The node that was added or removed.
059     */
060    public NodeChangeEvent(Preferences parent, Preferences child) {
061        super(parent);
062        this.child = child;
063    }
064
065    /**
066     * Returns the parent of the node that was added or removed.
067     *
068     * @return  The parent Preferences node whose child was added or removed
069     */
070    public Preferences getParent() {
071        return (Preferences) getSource();
072    }
073
074    /**
075     * Returns the node that was added or removed.
076     *
077     * @return  The node that was added or removed.
078     */
079    public Preferences getChild() {
080        return child;
081    }
082
083    /**
084     * Throws NotSerializableException, since NodeChangeEvent objects are not
085     * intended to be serializable.
086     */
087     private void writeObject(java.io.ObjectOutputStream out)
088                                               throws NotSerializableException {
089         throw new NotSerializableException("Not serializable.");
090     }
091
092    /**
093     * Throws NotSerializableException, since NodeChangeEvent objects are not
094     * intended to be serializable.
095     */
096     private void readObject(java.io.ObjectInputStream in)
097                                               throws NotSerializableException {
098         throw new NotSerializableException("Not serializable.");
099     }
100
101    // Defined so that this class isn't flagged as a potential problem when
102    // searches for missing serialVersionUID fields are done.
103    private static final long serialVersionUID = 8068949086596572957L;
104}
