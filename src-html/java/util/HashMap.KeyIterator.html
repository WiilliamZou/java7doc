026package java.util;
027import java.io.*;

029/**
030 * Hash table based implementation of the <tt>Map</tt> interface.  This
031 * implementation provides all of the optional map operations, and permits
032 * <tt>null</tt> values and the <tt>null</tt> key.  (The <tt>HashMap</tt>
033 * class is roughly equivalent to <tt>Hashtable</tt>, except that it is
034 * unsynchronized and permits nulls.)  This class makes no guarantees as to
035 * the order of the map; in particular, it does not guarantee that the order
036 * will remain constant over time.
037 *
038 * <p>This implementation provides constant-time performance for the basic
039 * operations (<tt>get</tt> and <tt>put</tt>), assuming the hash function
040 * disperses the elements properly among the buckets.  Iteration over
041 * collection views requires time proportional to the "capacity" of the
042 * <tt>HashMap</tt> instance (the number of buckets) plus its size (the number
043 * of key-value mappings).  Thus, it's very important not to set the initial
044 * capacity too high (or the load factor too low) if iteration performance is
045 * important.
046 *
047 * <p>An instance of <tt>HashMap</tt> has two parameters that affect its
048 * performance: <i>initial capacity</i> and <i>load factor</i>.  The
049 * <i>capacity</i> is the number of buckets in the hash table, and the initial
050 * capacity is simply the capacity at the time the hash table is created.  The
051 * <i>load factor</i> is a measure of how full the hash table is allowed to
052 * get before its capacity is automatically increased.  When the number of
053 * entries in the hash table exceeds the product of the load factor and the
054 * current capacity, the hash table is <i>rehashed</i> (that is, internal data
055 * structures are rebuilt) so that the hash table has approximately twice the
056 * number of buckets.
057 *
058 * <p>As a general rule, the default load factor (.75) offers a good tradeoff
059 * between time and space costs.  Higher values decrease the space overhead
060 * but increase the lookup cost (reflected in most of the operations of the
061 * <tt>HashMap</tt> class, including <tt>get</tt> and <tt>put</tt>).  The
062 * expected number of entries in the map and its load factor should be taken
063 * into account when setting its initial capacity, so as to minimize the
064 * number of rehash operations.  If the initial capacity is greater
065 * than the maximum number of entries divided by the load factor, no
066 * rehash operations will ever occur.
067 *
068 * <p>If many mappings are to be stored in a <tt>HashMap</tt> instance,
069 * creating it with a sufficiently large capacity will allow the mappings to
070 * be stored more efficiently than letting it perform automatic rehashing as
071 * needed to grow the table.
072 *
073 * <p><strong>Note that this implementation is not synchronized.</strong>
074 * If multiple threads access a hash map concurrently, and at least one of
075 * the threads modifies the map structurally, it <i>must</i> be
076 * synchronized externally.  (A structural modification is any operation
077 * that adds or deletes one or more mappings; merely changing the value
078 * associated with a key that an instance already contains is not a
079 * structural modification.)  This is typically accomplished by
080 * synchronizing on some object that naturally encapsulates the map.
081 *
082 * If no such object exists, the map should be "wrapped" using the
083 * {@link Collections#synchronizedMap Collections.synchronizedMap}
084 * method.  This is best done at creation time, to prevent accidental
085 * unsynchronized access to the map:<pre>
086 *   Map m = Collections.synchronizedMap(new HashMap(...));</pre>
087 *
088 * <p>The iterators returned by all of this class's "collection view methods"
089 * are <i>fail-fast</i>: if the map is structurally modified at any time after
090 * the iterator is created, in any way except through the iterator's own
091 * <tt>remove</tt> method, the iterator will throw a
092 * {@link ConcurrentModificationException}.  Thus, in the face of concurrent
093 * modification, the iterator fails quickly and cleanly, rather than risking
094 * arbitrary, non-deterministic behavior at an undetermined time in the
095 * future.
096 *
097 * <p>Note that the fail-fast behavior of an iterator cannot be guaranteed
098 * as it is, generally speaking, impossible to make any hard guarantees in the
099 * presence of unsynchronized concurrent modification.  Fail-fast iterators
100 * throw <tt>ConcurrentModificationException</tt> on a best-effort basis.
101 * Therefore, it would be wrong to write a program that depended on this
102 * exception for its correctness: <i>the fail-fast behavior of iterators
103 * should be used only to detect bugs.</i>
104 *
105 * <p>This class is a member of the
106 * <a href="{@docRoot}/../technotes/guides/collections/index.html">
107 * Java Collections Framework</a>.
108 *
109 * @param <K> the type of keys maintained by this map
110 * @param <V> the type of mapped values
111 *
112 * @author  Doug Lea
113 * @author  Josh Bloch
114 * @author  Arthur van Hoff
115 * @author  Neal Gafter
116 * @see     Object#hashCode()
117 * @see     Collection
118 * @see     Map
119 * @see     TreeMap
120 * @see     Hashtable
121 * @since   1.2
122 */
123
124public class HashMap<K,V>
125    extends AbstractMap<K,V>
126    implements Map<K,V>, Cloneable, Serializable
127{

130     * The default initial capacity - MUST be a power of two.
131     */
132    static final int DEFAULT_INITIAL_CAPACITY = 1 << 4; // aka 16
133
134    /**
135     * The maximum capacity, used if a higher value is implicitly specified
136     * by either of the constructors with arguments.
137     * MUST be a power of two <= 1<<30.
138     */
139    static final int MAXIMUM_CAPACITY = 1 << 30;

179     * the HashMap or otherwise modify its internal structure (e.g.,

185    /**
186     * The default threshold of map capacity above which alternative hashing is
187     * used for String keys. Alternative hashing reduces the incidence of
188     * collisions due to weak hash code calculation for String keys.
189     * <p/>
190     * This value may be overridden by defining the system property
191     * {@code jdk.map.althashing.threshold}. A property value of {@code 1}
192     * forces alternative hashing to be used at all times whereas
193     * {@code -1} value ensures that alternative hashing is never used.
194     */
195    static final int ALTERNATIVE_HASHING_THRESHOLD_DEFAULT = Integer.MAX_VALUE;
196
197    /**
198     * holds values which can't be initialized until after VM is booted.
199     */
200    private static class Holder {
201
202        /**
203         * Table capacity above which to switch to use alternative hashing.
204         */
205        static final int ALTERNATIVE_HASHING_THRESHOLD;
206
207        static {
208            String altThreshold = java.security.AccessController.doPrivileged(
209                new sun.security.action.GetPropertyAction(
210                    "jdk.map.althashing.threshold"));
211
212            int threshold;
213            try {
214                threshold = (null != altThreshold)
215                        ? Integer.parseInt(altThreshold)
216                        : ALTERNATIVE_HASHING_THRESHOLD_DEFAULT;
217
218                // disable alternative hashing if -1
219                if (threshold == -1) {
220                    threshold = Integer.MAX_VALUE;
221                }
222
223                if (threshold < 0) {
224                    throw new IllegalArgumentException("value must be positive integer.");
225                }
226            } catch(IllegalArgumentException failed) {
227                throw new Error("Illegal value for 'jdk.map.althashing.threshold'", failed);
228            }
229
230            ALTERNATIVE_HASHING_THRESHOLD = threshold;
231        }
232    }

234    /**
235     * A randomizing value associated with this instance that is applied to
236     * hash code of keys to make hash collisions harder to find. If 0 then
237     * alternative hashing is disabled.
238     */
239    transient int hashSeed = 0;

241    /**
242     * Constructs an empty <tt>HashMap</tt> with the specified initial
243     * capacity and load factor.
244     *
245     * @param  initialCapacity the initial capacity
246     * @param  loadFactor      the load factor
247     * @throws IllegalArgumentException if the initial capacity is negative
248     *         or the load factor is nonpositive
249     */
250    public HashMap(int initialCapacity, float loadFactor) {
251        if (initialCapacity < 0)
252            throw new IllegalArgumentException("Illegal initial capacity: " +
253                                               initialCapacity);
254        if (initialCapacity > MAXIMUM_CAPACITY)
255            initialCapacity = MAXIMUM_CAPACITY;
256        if (loadFactor <= 0 || Float.isNaN(loadFactor))
257            throw new IllegalArgumentException("Illegal load factor: " +
258                                               loadFactor);
259
260        this.loadFactor = loadFactor;
261        threshold = initialCapacity;
262        init();
263    }

265    /**
266     * Constructs an empty <tt>HashMap</tt> with the specified initial
267     * capacity and the default load factor (0.75).
268     *
269     * @param  initialCapacity the initial capacity.
270     * @throws IllegalArgumentException if the initial capacity is negative.
271     */
272    public HashMap(int initialCapacity) {
273        this(initialCapacity, DEFAULT_LOAD_FACTOR);
274    }

276    /**
277     * Constructs an empty <tt>HashMap</tt> with the default initial capacity
278     * (16) and the default load factor (0.75).
279     */
280    public HashMap() {
281        this(DEFAULT_INITIAL_CAPACITY, DEFAULT_LOAD_FACTOR);
282    }

284    /**
285     * Constructs a new <tt>HashMap</tt> with the same mappings as the
286     * specified <tt>Map</tt>.  The <tt>HashMap</tt> is created with
287     * default load factor (0.75) and an initial capacity sufficient to
288     * hold the mappings in the specified <tt>Map</tt>.
289     *
290     * @param   m the map whose mappings are to be placed in this map
291     * @throws  NullPointerException if the specified map is null
292     */
293    public HashMap(Map<? extends K, ? extends V> m) {
294        this(Math.max((int) (m.size() / DEFAULT_LOAD_FACTOR) + 1,
295                      DEFAULT_INITIAL_CAPACITY), DEFAULT_LOAD_FACTOR);
296        inflateTable(threshold);
297
298        putAllForCreate(m);
299    }

301    private static int roundUpToPowerOf2(int number) {
302        // assert number >= 0 : "number must be non-negative";
303        return number >= MAXIMUM_CAPACITY
304                ? MAXIMUM_CAPACITY
305                : (number > 1) ? Integer.highestOneBit((number - 1) << 1) : 1;
306    }
307
308    /**
309     * Inflates the table.
310     */
311    private void inflateTable(int toSize) {
312        // Find a power of 2 >= toSize
313        int capacity = roundUpToPowerOf2(toSize);
314
315        threshold = (int) Math.min(capacity * loadFactor, MAXIMUM_CAPACITY + 1);
316        table = new Entry[capacity];
317        initHashSeedAsNeeded(capacity);
318    }

332    /**
333     * Initialize the hashing mask value. We defer initialization until we
334     * really need it.
335     */
336    final boolean initHashSeedAsNeeded(int capacity) {
337        boolean currentAltHashing = hashSeed != 0;
338        boolean useAltHashing = sun.misc.VM.isBooted() &&
339                (capacity >= Holder.ALTERNATIVE_HASHING_THRESHOLD);
340        boolean switching = currentAltHashing ^ useAltHashing;
341        if (switching) {
342            hashSeed = useAltHashing
343                ? sun.misc.Hashing.randomHashSeed(this)
344                : 0;
345        }
346        return switching;
347    }
348
349    /**
350     * Retrieve object hash code and applies a supplemental hash function to the
351     * result hash, which defends against poor quality hash functions.  This is
352     * critical because HashMap uses power-of-two length hash tables, that
353     * otherwise encounter collisions for hashCodes that do not differ
354     * in lower bits. Note: Null keys always map to hash 0, thus index 0.
355     */
356    final int hash(Object k) {
357        int h = hashSeed;
358        if (0 != h && k instanceof String) {
359            return sun.misc.Hashing.stringHash32((String) k);
360        }
361
362        h ^= k.hashCode();
363
364        // This function ensures that hashCodes that differ only by
365        // constant multiples at each bit position have a bounded
366        // number of collisions (approximately 8 at default load factor).
367        h ^= (h >>> 20) ^ (h >>> 12);
368        return h ^ (h >>> 7) ^ (h >>> 4);
369    }
370
371    /**
372     * Returns index for hash code h.
373     */
374    static int indexFor(int h, int length) {
375        // assert Integer.bitCount(length) == 1 : "length must be a non-zero power of 2";
376        return h & (length-1);
377    }

397    /**
398     * Returns the value to which the specified key is mapped,
399     * or {@code null} if this map contains no mapping for the key.
400     *
401     * <p>More formally, if this map contains a mapping from a key
402     * {@code k} to a value {@code v} such that {@code (key==null ? k==null :
403     * key.equals(k))}, then this method returns {@code v}; otherwise
404     * it returns {@code null}.  (There can be at most one such mapping.)
405     *
406     * <p>A return value of {@code null} does not <i>necessarily</i>
407     * indicate that the map contains no mapping for the key; it's also
408     * possible that the map explicitly maps the key to {@code null}.
409     * The {@link #containsKey containsKey} operation may be used to
410     * distinguish these two cases.
411     *
412     * @see #put(Object, Object)
413     */
414    public V get(Object key) {
415        if (key == null)
416            return getForNullKey();
417        Entry<K,V> entry = getEntry(key);
418
419        return null == entry ? null : entry.getValue();
420    }
421
422    /**
423     * Offloaded version of get() to look up null keys.  Null keys map
424     * to index 0.  This null case is split out into separate methods
425     * for the sake of performance in the two most commonly used
426     * operations (get and put), but incorporated with conditionals in
427     * others.
428     */
429    private V getForNullKey() {
430        if (size == 0) {
431            return null;
432        }
433        for (Entry<K,V> e = table[0]; e != null; e = e.next) {
434            if (e.key == null)
435                return e.value;
436        }
437        return null;
438    }

440    /**
441     * Returns <tt>true</tt> if this map contains a mapping for the
442     * specified key.
443     *
444     * @param   key   The key whose presence in this map is to be tested
445     * @return <tt>true</tt> if this map contains a mapping for the specified
446     * key.
447     */
448    public boolean containsKey(Object key) {
449        return getEntry(key) != null;
450    }

452    /**
453     * Returns the entry associated with the specified key in the
454     * HashMap.  Returns null if the HashMap contains no mapping
455     * for the key.
456     */
457    final Entry<K,V> getEntry(Object key) {
458        if (size == 0) {
459            return null;
460        }
461
462        int hash = (key == null) ? 0 : hash(key);
463        for (Entry<K,V> e = table[indexFor(hash, table.length)];
464             e != null;
465             e = e.next) {
466            Object k;
467            if (e.hash == hash &&
468                ((k = e.key) == key || (key != null && key.equals(k))))
469                return e;
470        }
471        return null;
472    }

474    /**
475     * Associates the specified value with the specified key in this map.
476     * If the map previously contained a mapping for the key, the old
477     * value is replaced.
478     *
479     * @param key key with which the specified value is to be associated
480     * @param value value to be associated with the specified key
481     * @return the previous value associated with <tt>key</tt>, or
482     *         <tt>null</tt> if there was no mapping for <tt>key</tt>.
483     *         (A <tt>null</tt> return can also indicate that the map
484     *         previously associated <tt>null</tt> with <tt>key</tt>.)
485     */
486    public V put(K key, V value) {
487        if (table == EMPTY_TABLE) {
488            inflateTable(threshold);
489        }
490        if (key == null)
491            return putForNullKey(value);
492        int hash = hash(key);
493        int i = indexFor(hash, table.length);
494        for (Entry<K,V> e = table[i]; e != null; e = e.next) {
495            Object k;
496            if (e.hash == hash && ((k = e.key) == key || key.equals(k))) {
497                V oldValue = e.value;
498                e.value = value;
499                e.recordAccess(this);
500                return oldValue;
501            }
502        }
503
504        modCount++;
505        addEntry(hash, key, value, i);
506        return null;
507    }
508
509    /**
510     * Offloaded version of put for null keys
511     */
512    private V putForNullKey(V value) {
513        for (Entry<K,V> e = table[0]; e != null; e = e.next) {
514            if (e.key == null) {
515                V oldValue = e.value;
516                e.value = value;
517                e.recordAccess(this);
518                return oldValue;
519            }
520        }
521        modCount++;
522        addEntry(0, null, value, 0);
523        return null;
524    }

526    /**
527     * This method is used instead of put by constructors and
528     * pseudoconstructors (clone, readObject).  It does not resize the table,
529     * check for comodification, etc.  It calls createEntry rather than
530     * addEntry.
531     */
532    private void putForCreate(K key, V value) {
533        int hash = null == key ? 0 : hash(key);
534        int i = indexFor(hash, table.length);
535
536        /**
537         * Look for preexisting entry for key.  This will never happen for
538         * clone or deserialize.  It will only happen for construction if the
539         * input Map is a sorted map whose ordering is inconsistent w/ equals.
540         */
541        for (Entry<K,V> e = table[i]; e != null; e = e.next) {
542            Object k;
543            if (e.hash == hash &&
544                ((k = e.key) == key || (key != null && key.equals(k)))) {
545                e.value = value;
546                return;
547            }
548        }
549
550        createEntry(hash, key, value, i);
551    }
552
553    private void putAllForCreate(Map<? extends K, ? extends V> m) {
554        for (Map.Entry<? extends K, ? extends V> e : m.entrySet())
555            putForCreate(e.getKey(), e.getValue());
556    }

558    /**
559     * Rehashes the contents of this map into a new array with a
560     * larger capacity.  This method is called automatically when the
561     * number of keys in this map reaches its threshold.
562     *
563     * If current capacity is MAXIMUM_CAPACITY, this method does not
564     * resize the map, but sets threshold to Integer.MAX_VALUE.
565     * This has the effect of preventing future calls.
566     *
567     * @param newCapacity the new capacity, MUST be a power of two;
568     *        must be greater than current capacity unless current
569     *        capacity is MAXIMUM_CAPACITY (in which case value
570     *        is irrelevant).
571     */
572    void resize(int newCapacity) {
573        Entry[] oldTable = table;
574        int oldCapacity = oldTable.length;
575        if (oldCapacity == MAXIMUM_CAPACITY) {
576            threshold = Integer.MAX_VALUE;
577            return;
578        }
579
580        Entry[] newTable = new Entry[newCapacity];
581        transfer(newTable, initHashSeedAsNeeded(newCapacity));
582        table = newTable;
583        threshold = (int)Math.min(newCapacity * loadFactor, MAXIMUM_CAPACITY + 1);
584    }
585
586    /**
587     * Transfers all entries from current table to newTable.
588     */
589    void transfer(Entry[] newTable, boolean rehash) {
590        int newCapacity = newTable.length;
591        for (Entry<K,V> e : table) {
592            while(null != e) {
593                Entry<K,V> next = e.next;
594                if (rehash) {
595                    e.hash = null == e.key ? 0 : hash(e.key);
596                }
597                int i = indexFor(e.hash, newCapacity);
598                e.next = newTable[i];
599                newTable[i] = e;
600                e = next;
601            }
602        }
603    }

605    /**
606     * Copies all of the mappings from the specified map to this map.
607     * These mappings will replace any mappings that this map had for
608     * any of the keys currently in the specified map.
609     *
610     * @param m mappings to be stored in this map
611     * @throws NullPointerException if the specified map is null
612     */
613    public void putAll(Map<? extends K, ? extends V> m) {
614        int numKeysToBeAdded = m.size();
615        if (numKeysToBeAdded == 0)
616            return;
617
618        if (table == EMPTY_TABLE) {
619            inflateTable((int) Math.max(numKeysToBeAdded * loadFactor, threshold));
620        }
621
622        /*
623         * Expand the map if the map if the number of mappings to be added
624         * is greater than or equal to threshold.  This is conservative; the
625         * obvious condition is (m.size() + size) >= threshold, but this
626         * condition could result in a map with twice the appropriate capacity,
627         * if the keys to be added overlap with the keys already in this map.
628         * By using the conservative calculation, we subject ourself
629         * to at most one extra resize.
630         */
631        if (numKeysToBeAdded > threshold) {
632            int targetCapacity = (int)(numKeysToBeAdded / loadFactor + 1);
633            if (targetCapacity > MAXIMUM_CAPACITY)
634                targetCapacity = MAXIMUM_CAPACITY;
635            int newCapacity = table.length;
636            while (newCapacity < targetCapacity)
637                newCapacity <<= 1;
638            if (newCapacity > table.length)
639                resize(newCapacity);
640        }
641
642        for (Map.Entry<? extends K, ? extends V> e : m.entrySet())
643            put(e.getKey(), e.getValue());
644    }

646    /**
647     * Removes the mapping for the specified key from this map if present.
648     *
649     * @param  key key whose mapping is to be removed from the map
650     * @return the previous value associated with <tt>key</tt>, or
651     *         <tt>null</tt> if there was no mapping for <tt>key</tt>.
652     *         (A <tt>null</tt> return can also indicate that the map
653     *         previously associated <tt>null</tt> with <tt>key</tt>.)
654     */
655    public V remove(Object key) {
656        Entry<K,V> e = removeEntryForKey(key);
657        return (e == null ? null : e.value);
658    }
659
660    /**
661     * Removes and returns the entry associated with the specified key
662     * in the HashMap.  Returns null if the HashMap contains no mapping
663     * for this key.
664     */
665    final Entry<K,V> removeEntryForKey(Object key) {
666        if (size == 0) {
667            return null;
668        }
669        int hash = (key == null) ? 0 : hash(key);
670        int i = indexFor(hash, table.length);
671        Entry<K,V> prev = table[i];
672        Entry<K,V> e = prev;
673
674        while (e != null) {
675            Entry<K,V> next = e.next;
676            Object k;
677            if (e.hash == hash &&
678                ((k = e.key) == key || (key != null && key.equals(k)))) {
679                modCount++;
680                size--;
681                if (prev == e)
682                    table[i] = next;
683                else
684                    prev.next = next;
685                e.recordRemoval(this);
686                return e;
687            }
688            prev = e;
689            e = next;
690        }
691
692        return e;
693    }

695    /**
696     * Special version of remove for EntrySet using {@code Map.Entry.equals()}
697     * for matching.
698     */
699    final Entry<K,V> removeMapping(Object o) {
700        if (size == 0 || !(o instanceof Map.Entry))
701            return null;
702
703        Map.Entry<K,V> entry = (Map.Entry<K,V>) o;
704        Object key = entry.getKey();
705        int hash = (key == null) ? 0 : hash(key);
706        int i = indexFor(hash, table.length);
707        Entry<K,V> prev = table[i];
708        Entry<K,V> e = prev;
709
710        while (e != null) {
711            Entry<K,V> next = e.next;
712            if (e.hash == hash && e.equals(entry)) {
713                modCount++;
714                size--;
715                if (prev == e)
716                    table[i] = next;
717                else
718                    prev.next = next;
719                e.recordRemoval(this);
720                return e;
721            }
722            prev = e;
723            e = next;
724        }
725
726        return e;
727    }

730     * Removes all of the mappings from this map.
731     * The map will be empty after this call returns.
732     */
733    public void clear() {
734        modCount++;
735        Arrays.fill(table, null);
736        size = 0;

739    /**
740     * Returns <tt>true</tt> if this map maps one or more keys to the
741     * specified value.
742     *
743     * @param value value whose presence in this map is to be tested
744     * @return <tt>true</tt> if this map maps one or more keys to the
745     *         specified value
746     */
747    public boolean containsValue(Object value) {
748        if (value == null)
749            return containsNullValue();
750
751        Entry[] tab = table;
752        for (int i = 0; i < tab.length ; i++)
753            for (Entry e = tab[i] ; e != null ; e = e.next)
754                if (value.equals(e.value))
755                    return true;
756        return false;
757    }

760     * Special-case code for containsValue with null argument
761     */
762    private boolean containsNullValue() {
763        Entry[] tab = table;
764        for (int i = 0; i < tab.length ; i++)
765            for (Entry e = tab[i] ; e != null ; e = e.next)
766                if (e.value == null)
767                    return true;
768        return false;
769    }

771    /**
772     * Returns a shallow copy of this <tt>HashMap</tt> instance: the keys and
773     * values themselves are not cloned.
774     *
775     * @return a shallow copy of this map
776     */
777    public Object clone() {
778        HashMap<K,V> result = null;
779        try {
780            result = (HashMap<K,V>)super.clone();
781        } catch (CloneNotSupportedException e) {
782            // assert false;
783        }
784        if (result.table != EMPTY_TABLE) {
785            result.inflateTable(Math.min(
786                (int) Math.min(
787                    size * Math.min(1 / loadFactor, 4.0f),
788                    // we have limits...
789                    HashMap.MAXIMUM_CAPACITY),
790               table.length));
791        }
792        result.entrySet = null;
793        result.modCount = 0;
794        result.size = 0;
795        result.init();
796        result.putAllForCreate(this);
797
798        return result;
799    }

836            Object k2 = e.getKey();

870    /**
871     * Adds a new entry with the specified key, value and hash code to
872     * the specified bucket.  It is the responsibility of this
873     * method to resize the table if appropriate.
874     *
875     * Subclass overrides this to alter the behavior of put method.
876     */
877    void addEntry(int hash, K key, V value, int bucketIndex) {
878        if ((size >= threshold) && (null != table[bucketIndex])) {
879            resize(2 * table.length);
880            hash = (null != key) ? hash(key) : 0;
881            bucketIndex = indexFor(hash, table.length);
882        }
883
884        createEntry(hash, key, value, bucketIndex);
885    }

887    /**
888     * Like addEntry except that this version is used when creating entries
889     * as part of Map construction or "pseudo-construction" (cloning,
890     * deserialization).  This version needn't worry about resizing the table.
891     *
892     * Subclass overrides this to alter the behavior of HashMap(Map),
893     * clone, and readObject.
894     */
895    void createEntry(int hash, K key, V value, int bucketIndex) {
896        Entry<K,V> e = table[bucketIndex];
897        table[bucketIndex] = new Entry<>(hash, key, value, e);
898        size++;
899    }

901    private abstract class HashIterator<E> implements Iterator<E> {
902        Entry<K,V> next;        // next entry to return
903        int expectedModCount;   // For fast-fail
904        int index;              // current slot
905        Entry<K,V> current;     // current entry
906
907        HashIterator() {
908            expectedModCount = modCount;
909            if (size > 0) { // advance to first entry
910                Entry[] t = table;
911                while (index < t.length && (next = t[index++]) == null)
912                    ;
913            }
914        }
915
916        public final boolean hasNext() {
917            return next != null;
918        }
919
920        final Entry<K,V> nextEntry() {
921            if (modCount != expectedModCount)
922                throw new ConcurrentModificationException();
923            Entry<K,V> e = next;
924            if (e == null)
925                throw new NoSuchElementException();
926
927            if ((next = e.next) == null) {
928                Entry[] t = table;
929                while (index < t.length && (next = t[index++]) == null)
930                    ;
931            }
932            current = e;
933            return e;
934        }
935
936        public void remove() {
937            if (current == null)
938                throw new IllegalStateException();
939            if (modCount != expectedModCount)
940                throw new ConcurrentModificationException();
941            Object k = current.key;
942            current = null;
943            HashMap.this.removeEntryForKey(k);
944            expectedModCount = modCount;
945        }
946    }

1098    /**
1099     * Save the state of the <tt>HashMap</tt> instance to a stream (i.e.,
1100     * serialize it).
1101     *
1102     * @serialData The <i>capacity</i> of the HashMap (the length of the
1103     *             bucket array) is emitted (int), followed by the
1104     *             <i>size</i> (an int, the number of key-value
1105     *             mappings), followed by the key (Object) and value (Object)
1106     *             for each key-value mapping.  The key-value mappings are
1107     *             emitted in no particular order.
1108     */
1109    private void writeObject(java.io.ObjectOutputStream s)
1110        throws IOException
1111    {
1112        // Write out the threshold, loadfactor, and any hidden stuff
1113        s.defaultWriteObject();
1114
1115        // Write out number of buckets
1116        if (table==EMPTY_TABLE) {
1117            s.writeInt(roundUpToPowerOf2(threshold));
1118        } else {
1119           s.writeInt(table.length);
1120        }
1121
1122        // Write out size (number of Mappings)
1123        s.writeInt(size);
1124
1125        // Write out keys and values (alternating)
1126        if (size > 0) {
1127            for(Map.Entry<K,V> e : entrySet0()) {
1128                s.writeObject(e.getKey());
1129                s.writeObject(e.getValue());
1130            }
1131        }
1132    }

1136    /**
1137     * Reconstitute the {@code HashMap} instance from a stream (i.e.,
1138     * deserialize it).
1139     */
1140    private void readObject(java.io.ObjectInputStream s)
1141         throws IOException, ClassNotFoundException
1142    {
1143        // Read in the threshold (ignored), loadfactor, and any hidden stuff
1144        s.defaultReadObject();
1145        if (loadFactor <= 0 || Float.isNaN(loadFactor)) {
1146            throw new InvalidObjectException("Illegal load factor: " +
1147                                               loadFactor);
1148        }
1149
1150        // set other fields that need values
1151        table = (Entry<K,V>[]) EMPTY_TABLE;
1152
1153        // Read in number of buckets
1154        s.readInt(); // ignored.
1155
1156        // Read number of mappings
1157        int mappings = s.readInt();
1158        if (mappings < 0)
1159            throw new InvalidObjectException("Illegal mappings count: " +
1160                                               mappings);
1161
1162        // capacity chosen by number of mappings and desired load (if >= 0.25)
1163        int capacity = (int) Math.min(
1164                    mappings * Math.min(1 / loadFactor, 4.0f),
1165                    // we have limits...
1166                    HashMap.MAXIMUM_CAPACITY);
1167
1168        // allocate the bucket array;
1169        if (mappings > 0) {
1170            inflateTable(capacity);
1171        } else {
1172            threshold = capacity;
1173        }
1174
1175        init();  // Give subclass a chance to do its thing.
1176
1177        // Read the keys and values, and put the mappings in the HashMap
1178        for (int i = 0; i < mappings; i++) {
1179            K key = (K) s.readObject();
1180            V value = (V) s.readObject();
1181            putForCreate(key, value);
1182        }
1183    }
