026package java.util;
027
028import java.io.NotSerializableException;
029import java.io.IOException;
030
031/**
032 * Thrown to indicate that an operation could not complete because
033 * the input did not conform to the appropriate XML document type
034 * for a collection of properties, as per the {@link Properties}
035 * specification.<p>
036 *
037 * Note, that although InvalidPropertiesFormatException inherits Serializable
038 * interface from Exception, it is not intended to be Serializable. Appropriate
039 * serialization methods are implemented to throw NotSerializableException.
040 *
041 * @see     Properties
042 * @since   1.5
043 * @serial exclude
044 */
045
046public class InvalidPropertiesFormatException extends IOException {
047    /**
048     * Constructs an InvalidPropertiesFormatException with the specified
049     * cause.
050     *
051     * @param  cause the cause (which is saved for later retrieval by the
052     *         {@link Throwable#getCause()} method).
053     */
054    public InvalidPropertiesFormatException(Throwable cause) {
055        super(cause==null ? null : cause.toString());
056        this.initCause(cause);
057    }
058
059   /**
060    * Constructs an InvalidPropertiesFormatException with the specified
061    * detail message.
062    *
063    * @param   message   the detail message. The detail message is saved for
064    *          later retrieval by the {@link Throwable#getMessage()} method.
065    */
066    public InvalidPropertiesFormatException(String message) {
067        super(message);
068    }
069
070    /**
071     * Throws NotSerializableException, since InvalidPropertiesFormatException
072     * objects are not intended to be serializable.
073     */
074    private void writeObject(java.io.ObjectOutputStream out)
075        throws NotSerializableException
076    {
077        throw new NotSerializableException("Not serializable.");
078    }
079
080    /**
081     * Throws NotSerializableException, since InvalidPropertiesFormatException
082     * objects are not intended to be serializable.
083     */
084    private void readObject(java.io.ObjectInputStream in)
085        throws NotSerializableException
086    {
087        throw new NotSerializableException("Not serializable.");
088    }
089
090}
