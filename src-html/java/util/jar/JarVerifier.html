106    /**
107     * This method scans to see which entry we're parsing and
108     * keeps various state information depending on what type of
109     * file is being parsed.
110     */
111    public void beginEntry(JarEntry je, ManifestEntryVerifier mev)
112        throws IOException
113    {
114        if (je == null)
115            return;
116
117        if (debug != null) {
118            debug.println("beginEntry "+je.getName());
119        }
120
121        String name = je.getName();
122
123        /*
124         * Assumptions:
125         * 1. The manifest should be the first entry in the META-INF directory.
126         * 2. The .SF/.DSA/.EC files follow the manifest, before any normal entries
127         * 3. Any of the following will throw a SecurityException:
128         *    a. digest mismatch between a manifest section and
129         *       the SF section.
130         *    b. digest mismatch between the actual jar entry and the manifest
131         */
132
133        if (parsingMeta) {
134            String uname = name.toUpperCase(Locale.ENGLISH);
135            if ((uname.startsWith("META-INF/") ||
136                 uname.startsWith("/META-INF/"))) {
137
138                if (je.isDirectory()) {
139                    mev.setEntry(null, je);
140                    return;
141                }
142
143                if (uname.equals(JarFile.MANIFEST_NAME) ||
144                        uname.equals(JarIndex.INDEX_NAME)) {
145                    return;
146                }
147
148                if (SignatureFileVerifier.isBlockOrSF(uname)) {
149                    /* We parse only DSA, RSA or EC PKCS7 blocks. */
150                    parsingBlockOrSF = true;
151                    baos.reset();
152                    mev.setEntry(null, je);
153                    return;
154                }
155
156                // If a META-INF entry is not MF or block or SF, they should
157                // be normal entries. According to 2 above, no more block or
158                // SF will appear. Let's doneWithMeta.
159            }
160        }
161
162        if (parsingMeta) {
163            doneWithMeta();
164        }
165
166        if (je.isDirectory()) {
167            mev.setEntry(null, je);
168            return;
169        }
170
171        // be liberal in what you accept. If the name starts with ./, remove
172        // it as we internally canonicalize it with out the ./.
173        if (name.startsWith("./"))
174            name = name.substring(2);
175
176        // be liberal in what you accept. If the name starts with /, remove
177        // it as we internally canonicalize it with out the /.
178        if (name.startsWith("/"))
179            name = name.substring(1);
180
181        // only set the jev object for entries that have a signature
182        // (either verified or not)
183        if (sigFileSigners.get(name) != null ||
184                verifiedSigners.get(name) != null) {
185            mev.setEntry(name, je);
186            return;
187        }
188
189        // don't compute the digest for this entry
190        mev.setEntry(null, je);
191
192        return;
193    }

195    /**
196     * update a single byte.
197     */
198
199    public void update(int b, ManifestEntryVerifier mev)
200        throws IOException
201    {
202        if (b != -1) {
203            if (parsingBlockOrSF) {
204                baos.write(b);
205            } else {
206                mev.update((byte)b);
207            }
208        } else {
209            processEntry(mev);
210        }
211    }

213    /**
214     * update an array of bytes.
215     */
216
217    public void update(int n, byte[] b, int off, int len,
218                       ManifestEntryVerifier mev)
219        throws IOException
220    {
221        if (n != -1) {
222            if (parsingBlockOrSF) {
223                baos.write(b, off, n);
224            } else {
225                mev.update(b, off, n);
226            }
227        } else {
228            processEntry(mev);
229        }
230    }

232    /**
233     * called when we reach the end of entry in one of the read() methods.
234     */
235    private void processEntry(ManifestEntryVerifier mev)
236        throws IOException
237    {
238        if (!parsingBlockOrSF) {
239            JarEntry je = mev.getEntry();
240            if ((je != null) && (je.signers == null)) {
241                je.signers = mev.verify(verifiedSigners, sigFileSigners);
242                je.certs = mapSignersToCertArray(je.signers);
243            }
244        } else {
245
246            try {
247                parsingBlockOrSF = false;
248
249                if (debug != null) {
250                    debug.println("processEntry: processing block");
251                }
252
253                String uname = mev.getEntry().getName()
254                                             .toUpperCase(Locale.ENGLISH);
255
256                if (uname.endsWith(".SF")) {
257                    String key = uname.substring(0, uname.length()-3);
258                    byte bytes[] = baos.toByteArray();
259                    // add to sigFileData in case future blocks need it
260                    sigFileData.put(key, bytes);
261                    // check pending blocks, we can now process
262                    // anyone waiting for this .SF file
263                    Iterator it = pendingBlocks.iterator();
264                    while (it.hasNext()) {
265                        SignatureFileVerifier sfv =
266                            (SignatureFileVerifier) it.next();
267                        if (sfv.needSignatureFile(key)) {
268                            if (debug != null) {
269                                debug.println(
270                                 "processEntry: processing pending block");
271                            }
272
273                            sfv.setSignatureFile(bytes);
274                            sfv.process(sigFileSigners, manifestDigests);
275                        }
276                    }
277                    return;
278                }
279
280                // now we are parsing a signature block file
281
282                String key = uname.substring(0, uname.lastIndexOf("."));
283
284                if (signerCache == null)
285                    signerCache = new ArrayList();
286
287                if (manDig == null) {
288                    synchronized(manifestRawBytes) {
289                        if (manDig == null) {
290                            manDig = new ManifestDigester(manifestRawBytes);
291                            manifestRawBytes = null;
292                        }
293                    }
294                }
295
296                SignatureFileVerifier sfv =
297                  new SignatureFileVerifier(signerCache,
298                                            manDig, uname, baos.toByteArray());
299
300                if (sfv.needSignatureFileBytes()) {
301                    // see if we have already parsed an external .SF file
302                    byte[] bytes = (byte[]) sigFileData.get(key);
303
304                    if (bytes == null) {
305                        // put this block on queue for later processing
306                        // since we don't have the .SF bytes yet
307                        // (uname, block);
308                        if (debug != null) {
309                            debug.println("adding pending block");
310                        }
311                        pendingBlocks.add(sfv);
312                        return;
313                    } else {
314                        sfv.setSignatureFile(bytes);
315                    }
316                }
317                sfv.process(sigFileSigners, manifestDigests);
318
319            } catch (IOException ioe) {
320                // e.g. sun.security.pkcs.ParsingException
321                if (debug != null) debug.println("processEntry caught: "+ioe);
322                // ignore and treat as unsigned
323            } catch (SignatureException se) {
324                if (debug != null) debug.println("processEntry caught: "+se);
325                // ignore and treat as unsigned
326            } catch (NoSuchAlgorithmException nsae) {
327                if (debug != null) debug.println("processEntry caught: "+nsae);
328                // ignore and treat as unsigned
329            } catch (CertificateException ce) {
330                if (debug != null) debug.println("processEntry caught: "+ce);
331                // ignore and treat as unsigned
332            }
333        }
334    }

361    public CodeSigner[] getCodeSigners(JarFile jar, JarEntry entry)
362    {
363        String name = entry.getName();
364        if (eagerValidation && sigFileSigners.get(name) != null) {
365            /*
366             * Force a read of the entry data to generate the
367             * verification hash.
368             */
369            try {
370                InputStream s = jar.getInputStream(entry);
371                byte[] buffer = new byte[1024];
372                int n = buffer.length;
373                while (n != -1) {
374                    n = s.read(buffer, 0, buffer.length);
375                }
376                s.close();
377            } catch (IOException e) {
378            }
379        }
380        return getCodeSigners(name);
381    }

383    /*
384     * Convert an array of signers into an array of concatenated certificate
385     * arrays.
386     */
387    private static java.security.cert.Certificate[] mapSignersToCertArray(
388        CodeSigner[] signers) {
389
390        if (signers != null) {
391            ArrayList certChains = new ArrayList();
392            for (int i = 0; i < signers.length; i++) {
393                certChains.addAll(
394                    signers[i].getSignerCertPath().getCertificates());
395            }
396
397            // Convert into a Certificate[]
398            return (java.security.cert.Certificate[])
399                certChains.toArray(
400                    new java.security.cert.Certificate[certChains.size()]);
401        }
402        return null;
403    }

405    /**
406     * returns true if there no files to verify.
407     * should only be called after all the META-INF entries
408     * have been processed.
409     */
410    boolean nothingToVerify()
411    {
412        return (anyToVerify == false);

415    /**
416     * called to let us know we have processed all the
417     * META-INF entries, and if we re-read one of them, don't
418     * re-process it. Also gets rid of any data structures
419     * we needed when parsing META-INF entries.
420     */
421    void doneWithMeta()
422    {
423        parsingMeta = false;
424        anyToVerify = !sigFileSigners.isEmpty();
425        baos = null;
426        sigFileData = null;
427        pendingBlocks = null;
428        signerCache = null;
429        manDig = null;
430        // MANIFEST.MF is always treated as signed and verified,
431        // move its signers from sigFileSigners to verifiedSigners.
432        if (sigFileSigners.containsKey(JarFile.MANIFEST_NAME)) {
433            verifiedSigners.put(JarFile.MANIFEST_NAME,
434                    sigFileSigners.remove(JarFile.MANIFEST_NAME));
435        }
436    }

438    static class VerifierStream extends java.io.InputStream {
439
440        private InputStream is;
441        private JarVerifier jv;
442        private ManifestEntryVerifier mev;
443        private long numLeft;
444
445        VerifierStream(Manifest man,
446                       JarEntry je,
447                       InputStream is,
448                       JarVerifier jv) throws IOException
449        {
450            this.is = is;
451            this.jv = jv;
452            this.mev = new ManifestEntryVerifier(man);
453            this.jv.beginEntry(je, mev);
454            this.numLeft = je.getSize();
455            if (this.numLeft == 0)
456                this.jv.update(-1, this.mev);
457        }
458
459        public int read() throws IOException
460        {
461            if (numLeft > 0) {
462                int b = is.read();
463                jv.update(b, mev);
464                numLeft--;
465                if (numLeft == 0)
466                    jv.update(-1, mev);
467                return b;
468            } else {
469                return -1;
470            }
471        }
472
473        public int read(byte b[], int off, int len) throws IOException {
474            if ((numLeft > 0) && (numLeft < len)) {
475                len = (int)numLeft;
476            }
477
478            if (numLeft > 0) {
479                int n = is.read(b, off, len);
480                jv.update(n, b, off, len, mev);
481                numLeft -= n;
482                if (numLeft == 0)
483                    jv.update(-1, b, off, len, mev);
484                return n;
485            } else {
486                return -1;
487            }
488        }
489
490        public void close()
491            throws IOException
492        {
493            if (is != null)
494                is.close();
495            is = null;
496            mev = null;
497            jv = null;
498        }
499
500        public int available() throws IOException {
501            return is.available();
502        }
503
504    }

508    private Map urlToCodeSourceMap = new HashMap();
509    private Map signerToCodeSource = new HashMap();
510    private URL lastURL;

513    /*
514     * Create a unique mapping from codeSigner cache entries to CodeSource.
515     * In theory, multiple URLs origins could map to a single locally cached
516     * and shared JAR file although in practice there will be a single URL in use.
517     */
518    private synchronized CodeSource mapSignersToCodeSource(URL url, CodeSigner[] signers) {
519        Map map;
520        if (url == lastURL) {
521            map = lastURLMap;
522        } else {
523            map = (Map) urlToCodeSourceMap.get(url);
524            if (map == null) {
525                map = new HashMap();
526                urlToCodeSourceMap.put(url, map);
527            }
528            lastURLMap = map;
529            lastURL = url;
530        }
531        CodeSource cs = (CodeSource) map.get(signers);
532        if (cs == null) {
533            cs = new VerifierCodeSource(csdomain, url, signers);
534            signerToCodeSource.put(signers, cs);
535        }
536        return cs;
537    }
538
539    private CodeSource[] mapSignersToCodeSources(URL url, List signers, boolean unsigned) {
540        List sources = new ArrayList();
541
542        for (int i = 0; i < signers.size(); i++) {
543            sources.add(mapSignersToCodeSource(url, (CodeSigner[]) signers.get(i)));
544        }
545        if (unsigned) {
546            sources.add(mapSignersToCodeSource(url, null));
547        }
548        return (CodeSource[]) sources.toArray(new CodeSource[sources.size()]);
549    }
550    private CodeSigner[] emptySigner = new CodeSigner[0];
551
552    /*
553     * Match CodeSource to a CodeSigner[] in the signer cache.
554     */
555    private CodeSigner[] findMatchingSigners(CodeSource cs) {
556        if (cs instanceof VerifierCodeSource) {
557            VerifierCodeSource vcs = (VerifierCodeSource) cs;
558            if (vcs.isSameDomain(csdomain)) {
559                return ((VerifierCodeSource) cs).getPrivateSigners();
560            }
561        }
562
563        /*
564         * In practice signers should always be optimized above
565         * but this handles a CodeSource of any type, just in case.
566         */
567        CodeSource[] sources = mapSignersToCodeSources(cs.getLocation(), getJarCodeSigners(), true);
568        List sourceList = new ArrayList();
569        for (int i = 0; i < sources.length; i++) {
570            sourceList.add(sources[i]);
571        }
572        int j = sourceList.indexOf(cs);
573        if (j != -1) {
574            CodeSigner[] match;
575            match = ((VerifierCodeSource) sourceList.get(j)).getPrivateSigners();
576            if (match == null) {
577                match = emptySigner;
578            }
579            return match;
580        }
581        return null;
582    }

584    /*
585     * Instances of this class hold uncopied references to internal
586     * signing data that can be compared by object reference identity.
587     */
588    private static class VerifierCodeSource extends CodeSource {
589
590        URL vlocation;
591        CodeSigner[] vsigners;
592        java.security.cert.Certificate[] vcerts;
593        Object csdomain;
594
595        VerifierCodeSource(Object csdomain, URL location, CodeSigner[] signers) {
596            super(location, signers);
597            this.csdomain = csdomain;
598            vlocation = location;
599            vsigners = signers; // from signerCache
600        }
601
602        VerifierCodeSource(Object csdomain, URL location, java.security.cert.Certificate[] certs) {
603            super(location, certs);
604            this.csdomain = csdomain;
605            vlocation = location;
606            vcerts = certs; // from signerCache
607        }
608
609        /*
610         * All VerifierCodeSource instances are constructed based on
611         * singleton signerCache or signerCacheCert entries for each unique signer.
612         * No CodeSigner<->Certificate[] conversion is required.
613         * We use these assumptions to optimize equality comparisons.
614         */
615        public boolean equals(Object obj) {
616            if (obj == this) {
617                return true;
618            }
619            if (obj instanceof VerifierCodeSource) {
620                VerifierCodeSource that = (VerifierCodeSource) obj;
621
622                /*
623                 * Only compare against other per-signer singletons constructed
624                 * on behalf of the same JarFile instance. Otherwise, compare
625                 * things the slower way.
626                 */
627                if (isSameDomain(that.csdomain)) {
628                    if (that.vsigners != this.vsigners
629                            || that.vcerts != this.vcerts) {
630                        return false;
631                    }
632                    if (that.vlocation != null) {
633                        return that.vlocation.equals(this.vlocation);
634                    } else if (this.vlocation != null) {
635                        return this.vlocation.equals(that.vlocation);
636                    } else { // both null
637                        return true;
638                    }
639                }
640            }
641            return super.equals(obj);
642        }
643
644        boolean isSameDomain(Object csdomain) {
645            return this.csdomain == csdomain;
646        }

648        private CodeSigner[] getPrivateSigners() {
649            return vsigners;
650        }
651
652        private java.security.cert.Certificate[] getPrivateCertificates() {
653            return vcerts;
654        }

658    private synchronized Map signerMap() {
659        if (signerMap == null) {
660            /*
661             * Snapshot signer state so it doesn't change on us. We care
662             * only about the asserted signatures. Verification of
663             * signature validity happens via the JarEntry apis.
664             */
665            signerMap = new HashMap(verifiedSigners.size() + sigFileSigners.size());
666            signerMap.putAll(verifiedSigners);
667            signerMap.putAll(sigFileSigners);
668        }
669        return signerMap;
670    }

672    public synchronized Enumeration<String> entryNames(JarFile jar, final CodeSource[] cs) {
673        final Map map = signerMap();
674        final Iterator itor = map.entrySet().iterator();
675        boolean matchUnsigned = false;
676
677        /*
678         * Grab a single copy of the CodeSigner arrays. Check
679         * to see if we can optimize CodeSigner equality test.
680         */
681        List req = new ArrayList(cs.length);
682        for (int i = 0; i < cs.length; i++) {
683            CodeSigner[] match = findMatchingSigners(cs[i]);
684            if (match != null) {
685                if (match.length > 0) {
686                    req.add(match);
687                } else {
688                    matchUnsigned = true;
689                }
690            } else {
691                matchUnsigned = true;
692            }
693        }
694
695        final List signersReq = req;
696        final Enumeration enum2 = (matchUnsigned) ? unsignedEntryNames(jar) : emptyEnumeration;
697
698        return new Enumeration<String>() {
699
700            String name;
701
702            public boolean hasMoreElements() {
703                if (name != null) {
704                    return true;
705                }
706
707                while (itor.hasNext()) {
708                    Map.Entry e = (Map.Entry) itor.next();
709                    if (signersReq.contains((CodeSigner[]) e.getValue())) {
710                        name = (String) e.getKey();
711                        return true;
712                    }
713                }
714                while (enum2.hasMoreElements()) {
715                    name = (String) enum2.nextElement();
716                    return true;
717                }
718                return false;
719            }
720
721            public String nextElement() {
722                if (hasMoreElements()) {
723                    String value = name;
724                    name = null;
725                    return value;
726                }
727                throw new NoSuchElementException();
728            }
729        };
730    }

732    /*
733     * Like entries() but screens out internal JAR mechanism entries
734     * and includes signed entries with no ZIP data.
735     */
736    public Enumeration<JarEntry> entries2(final JarFile jar, Enumeration e) {
737        final Map map = new HashMap();
738        map.putAll(signerMap());
739        final Enumeration enum_ = e;
740        return new Enumeration<JarEntry>() {
741
742            Enumeration signers = null;
743            JarEntry entry;
744
745            public boolean hasMoreElements() {
746                if (entry != null) {
747                    return true;
748                }
749                while (enum_.hasMoreElements()) {
750                    ZipEntry ze = (ZipEntry) enum_.nextElement();
751                    if (JarVerifier.isSigningRelated(ze.getName())) {
752                        continue;
753                    }
754                    entry = jar.newEntry(ze);
755                    return true;
756                }
757                if (signers == null) {
758                    signers = Collections.enumeration(map.keySet());
759                }
760                while (signers.hasMoreElements()) {
761                    String name = (String) signers.nextElement();
762                    entry = jar.newEntry(new ZipEntry(name));
763                    return true;
764                }
765
766                // Any map entries left?
767                return false;
768            }
769
770            public JarEntry nextElement() {
771                if (hasMoreElements()) {
772                    JarEntry je = entry;
773                    map.remove(je.getName());
774                    entry = null;
775                    return je;
776                }
777                throw new NoSuchElementException();
778            }
779        };
780    }

797    private Enumeration<String> unsignedEntryNames(JarFile jar) {
798        final Map map = signerMap();
799        final Enumeration entries = jar.entries();
800        return new Enumeration<String>() {
801
802            String name;
803
804            /*
805             * Grab entries from ZIP directory but screen out
806             * metadata.
807             */
808            public boolean hasMoreElements() {
809                if (name != null) {
810                    return true;
811                }
812                while (entries.hasMoreElements()) {
813                    String value;
814                    ZipEntry e = (ZipEntry) entries.nextElement();
815                    value = e.getName();
816                    if (e.isDirectory() || isSigningRelated(value)) {
817                        continue;
818                    }
819                    if (map.get(value) == null) {
820                        name = value;
821                        return true;
822                    }
823                }
824                return false;
825            }
826
827            public String nextElement() {
828                if (hasMoreElements()) {
829                    String value = name;
830                    name = null;
831                    return value;
832                }
833                throw new NoSuchElementException();
834            }
835        };
836    }
