050/**
051 * Represents a currency. Currencies are identified by their ISO 4217 currency
052 * codes. Visit the <a href="http://www.iso.org/iso/en/prods-services/popstds/currencycodes.html">
053 * ISO web site</a> for more information, including a table of
054 * currency codes.
055 * <p>
056 * The class is designed so that there's never more than one
057 * <code>Currency</code> instance for any given currency. Therefore, there's
058 * no public constructor. You obtain a <code>Currency</code> instance using
059 * the <code>getInstance</code> methods.
060 * <p>
061 * Users can supersede the Java runtime currency data by creating a properties
062 * file named <code>&lt;JAVA_HOME&gt;/lib/currency.properties</code>.  The contents
063 * of the properties file are key/value pairs of the ISO 3166 country codes
064 * and the ISO 4217 currency data respectively.  The value part consists of
065 * three ISO 4217 values of a currency, i.e., an alphabetic code, a numeric
066 * code, and a minor unit.  Those three ISO 4217 values are separated by commas.
067 * The lines which start with '#'s are considered comment lines.  For example,
068 * <p>
069 * <code>
070 * #Sample currency properties<br>
071 * JP=JPZ,999,0
072 * </code>
073 * <p>
074 * will supersede the currency data for Japan.
075 *
076 * @since 1.4
077 */
078public final class Currency implements Serializable {

108    // Class data: currency data obtained from currency.data file.
109    // Purpose:
110    // - determine valid country codes
111    // - determine valid currency codes
112    // - map country codes to currency codes
113    // - obtain default fraction digits for currency codes
114    //
115    // sc = special case; dfd = default fraction digits
116    // Simple countries are those where the country code is a prefix of the
117    // currency code, and there are no known plans to change the currency.
118    //
119    // table formats:
120    // - mainTable:
121    //   - maps country code to 32-bit int
122    //   - 26*26 entries, corresponding to [A-Z]*[A-Z]
123    //   - \u007F -> not valid country
124    //   - bits 18-31: unused
125    //   - bits 8-17: numeric code (0 to 1023)
126    //   - bit 7: 1 - special case, bits 0-4 indicate which one
127    //            0 - simple country, bits 0-4 indicate final char of currency code
128    //   - bits 5-6: fraction digits for simple countries, 0 for special cases
129    //   - bits 0-4: final char for currency code for simple country, or ID of special case
130    // - special case IDs:
131    //   - 0: country has no currency
132    //   - other: index into sc* arrays + 1
133    // - scCutOverTimes: cut-over time in millis as returned by
134    //   System.currentTimeMillis for special case countries that are changing
135    //   currencies; Long.MAX_VALUE for countries that are not changing currencies
136    // - scOldCurrencies: old currencies for special case countries
137    // - scNewCurrencies: new currencies for special case countries that are
138    //   changing currencies; null for others
139    // - scOldCurrenciesDFD: default fraction digits for old currencies
140    // - scNewCurrenciesDFD: default fraction digits for new currencies, 0 for
141    //   countries that are not changing currencies
142    // - otherCurrencies: concatenation of all currency codes that are not the
143    //   main currency of a simple country, separated by "-"
144    // - otherCurrenciesDFD: decimal format digits for currencies in otherCurrencies, same order
145
146    static int formatVersion;
147    static int dataVersion;
148    static int[] mainTable;
149    static long[] scCutOverTimes;
150    static String[] scOldCurrencies;
151    static String[] scNewCurrencies;
152    static int[] scOldCurrenciesDFD;
153    static int[] scNewCurrenciesDFD;
154    static int[] scOldCurrenciesNumericCode;
155    static int[] scNewCurrenciesNumericCode;
156    static String otherCurrencies;
157    static int[] otherCurrenciesDFD;
158    static int[] otherCurrenciesNumericCode;

160    // handy constants - must match definitions in GenerateCurrencyData
161    // magic number
162    private static final int MAGIC_NUMBER = 0x43757244;
163    // number of characters from A to Z
164    private static final int A_TO_Z = ('Z' - 'A') + 1;
165    // entry for invalid country codes
166    private static final int INVALID_COUNTRY_ENTRY = 0x007F;
167    // entry for countries without currency
168    private static final int COUNTRY_WITHOUT_CURRENCY_ENTRY = 0x0080;
169    // mask for simple case country entries
170    private static final int SIMPLE_CASE_COUNTRY_MASK = 0x0000;
171    // mask for simple case country entry final character
172    private static final int SIMPLE_CASE_COUNTRY_FINAL_CHAR_MASK = 0x001F;
173    // mask for simple case country entry default currency digits
174    private static final int SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_MASK = 0x0060;
175    // shift count for simple case country entry default currency digits
176    private static final int SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_SHIFT = 5;
177    // mask for special case country entries
178    private static final int SPECIAL_CASE_COUNTRY_MASK = 0x0080;
179    // mask for special case country index
180    private static final int SPECIAL_CASE_COUNTRY_INDEX_MASK = 0x001F;
181    // delta from entry index component in main table to index into special case tables
182    private static final int SPECIAL_CASE_COUNTRY_INDEX_DELTA = 1;
183    // mask for distinguishing simple and special case countries
184    private static final int COUNTRY_TYPE_MASK = SIMPLE_CASE_COUNTRY_MASK | SPECIAL_CASE_COUNTRY_MASK;
185    // mask for the numeric code of the currency
186    private static final int NUMERIC_CODE_MASK = 0x0003FF00;
187    // shift count for the numeric code of the currency
188    private static final int NUMERIC_CODE_SHIFT = 8;
189
190    // Currency data format version
191    private static final int VALID_FORMAT_VERSION = 1;
192
193    static {
194        AccessController.doPrivileged(new PrivilegedAction<Object>() {
195            public Object run() {
196                String homeDir = System.getProperty("java.home");
197                try {
198                    String dataFile = homeDir + File.separator +
199                            "lib" + File.separator + "currency.data";
200                    DataInputStream dis = new DataInputStream(
201                        new BufferedInputStream(
202                        new FileInputStream(dataFile)));
203                    if (dis.readInt() != MAGIC_NUMBER) {
204                        throw new InternalError("Currency data is possibly corrupted");
205                    }
206                    formatVersion = dis.readInt();
207                    if (formatVersion != VALID_FORMAT_VERSION) {
208                        throw new InternalError("Currency data format is incorrect");
209                    }
210                    dataVersion = dis.readInt();
211                    mainTable = readIntArray(dis, A_TO_Z * A_TO_Z);
212                    int scCount = dis.readInt();
213                    scCutOverTimes = readLongArray(dis, scCount);
214                    scOldCurrencies = readStringArray(dis, scCount);
215                    scNewCurrencies = readStringArray(dis, scCount);
216                    scOldCurrenciesDFD = readIntArray(dis, scCount);
217                    scNewCurrenciesDFD = readIntArray(dis, scCount);
218                    scOldCurrenciesNumericCode = readIntArray(dis, scCount);
219                    scNewCurrenciesNumericCode = readIntArray(dis, scCount);
220                    int ocCount = dis.readInt();
221                    otherCurrencies = dis.readUTF();
222                    otherCurrenciesDFD = readIntArray(dis, ocCount);
223                    otherCurrenciesNumericCode = readIntArray(dis, ocCount);
224                    dis.close();
225                } catch (IOException e) {
226                    InternalError ie = new InternalError();
227                    ie.initCause(e);
228                    throw ie;
229                }
230
231                // look for the properties file for overrides
232                try {
233                    File propFile = new File(homeDir + File.separator +
234                                             "lib" + File.separator +
235                                             "currency.properties");
236                    if (propFile.exists()) {
237                        Properties props = new Properties();
238                        try (FileReader fr = new FileReader(propFile)) {
239                            props.load(fr);
240                        }
241                        Set<String> keys = props.stringPropertyNames();
242                        Pattern propertiesPattern =
243                            Pattern.compile("([A-Z]{3})\\s*,\\s*(\\d{3})\\s*,\\s*([0-3])");
244                        for (String key : keys) {
245                           replaceCurrencyData(propertiesPattern,
246                               key.toUpperCase(Locale.ROOT),
247                               props.getProperty(key).toUpperCase(Locale.ROOT));
248                        }
249                    }
250                } catch (IOException e) {
251                    info("currency.properties is ignored because of an IOException", e);
252                }
253                return null;
254            }
255        });
256    }

276    /**
277     * Returns the <code>Currency</code> instance for the given currency code.
278     *
279     * @param currencyCode the ISO 4217 code of the currency
280     * @return the <code>Currency</code> instance for the given currency code
281     * @exception NullPointerException if <code>currencyCode</code> is null
282     * @exception IllegalArgumentException if <code>currencyCode</code> is not
283     * a supported ISO 4217 code.
284     */

286        return getInstance(currencyCode, Integer.MIN_VALUE, 0);
287    }

289    private static Currency getInstance(String currencyCode, int defaultFractionDigits,
290        int numericCode) {
291        // Try to look up the currency code in the instances table.
292        // This does the null pointer check as a side effect.
293        // Also, if there already is an entry, the currencyCode must be valid.
294        Currency instance = instances.get(currencyCode);
295        if (instance != null) {
296            return instance;
297        }
298
299        if (defaultFractionDigits == Integer.MIN_VALUE) {
300            // Currency code not internally generated, need to verify first
301            // A currency code must have 3 characters and exist in the main table
302            // or in the list of other currencies.
303            if (currencyCode.length() != 3) {
304                throw new IllegalArgumentException();
305            }
306            char char1 = currencyCode.charAt(0);
307            char char2 = currencyCode.charAt(1);
308            int tableEntry = getMainTableEntry(char1, char2);
309            if ((tableEntry & COUNTRY_TYPE_MASK) == SIMPLE_CASE_COUNTRY_MASK
310                    && tableEntry != INVALID_COUNTRY_ENTRY
311                    && currencyCode.charAt(2) - 'A' == (tableEntry & SIMPLE_CASE_COUNTRY_FINAL_CHAR_MASK)) {
312                defaultFractionDigits = (tableEntry & SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_MASK) >> SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_SHIFT;
313                numericCode = (tableEntry & NUMERIC_CODE_MASK) >> NUMERIC_CODE_SHIFT;
314            } else {
315                // Check for '-' separately so we don't get false hits in the table.
316                if (currencyCode.charAt(2) == '-') {
317                    throw new IllegalArgumentException();
318                }
319                int index = otherCurrencies.indexOf(currencyCode);
320                if (index == -1) {
321                    throw new IllegalArgumentException();
322                }
323                defaultFractionDigits = otherCurrenciesDFD[index / 4];
324                numericCode = otherCurrenciesNumericCode[index / 4];
325            }
326        }
327
328        Currency currencyVal =
329            new Currency(currencyCode, defaultFractionDigits, numericCode);
330        instance = instances.putIfAbsent(currencyCode, currencyVal);
331        return (instance != null ? instance : currencyVal);
332    }

334    /**
335     * Returns the <code>Currency</code> instance for the country of the
336     * given locale. The language and variant components of the locale
337     * are ignored. The result may vary over time, as countries change their
338     * currencies. For example, for the original member countries of the
339     * European Monetary Union, the method returns the old national currencies
340     * until December 31, 2001, and the Euro from January 1, 2002, local time
341     * of the respective countries.
342     * <p>
343     * The method returns <code>null</code> for territories that don't
344     * have a currency, such as Antarctica.
345     *
346     * @param locale the locale for whose country a <code>Currency</code>
347     * instance is needed
348     * @return the <code>Currency</code> instance for the country of the given
349     * locale, or null
350     * @exception NullPointerException if <code>locale</code> or its country
351     * code is null
352     * @exception IllegalArgumentException if the country of the given locale
353     * is not a supported ISO 3166 country code.
354     */
355    public static Currency getInstance(Locale locale) {
356        String country = locale.getCountry();
357        if (country == null) {
358            throw new NullPointerException();
359        }
360
361        if (country.length() != 2) {
362            throw new IllegalArgumentException();
363        }
364
365        char char1 = country.charAt(0);
366        char char2 = country.charAt(1);
367        int tableEntry = getMainTableEntry(char1, char2);
368        if ((tableEntry & COUNTRY_TYPE_MASK) == SIMPLE_CASE_COUNTRY_MASK
369                    && tableEntry != INVALID_COUNTRY_ENTRY) {
370            char finalChar = (char) ((tableEntry & SIMPLE_CASE_COUNTRY_FINAL_CHAR_MASK) + 'A');
371            int defaultFractionDigits = (tableEntry & SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_MASK) >> SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_SHIFT;
372            int numericCode = (tableEntry & NUMERIC_CODE_MASK) >> NUMERIC_CODE_SHIFT;
373            StringBuffer sb = new StringBuffer(country);
374            sb.append(finalChar);
375            return getInstance(sb.toString(), defaultFractionDigits, numericCode);
376        } else {
377            // special cases
378            if (tableEntry == INVALID_COUNTRY_ENTRY) {
379                throw new IllegalArgumentException();
380            }
381            if (tableEntry == COUNTRY_WITHOUT_CURRENCY_ENTRY) {
382                return null;
383            } else {
384                int index = (tableEntry & SPECIAL_CASE_COUNTRY_INDEX_MASK) - SPECIAL_CASE_COUNTRY_INDEX_DELTA;
385                if (scCutOverTimes[index] == Long.MAX_VALUE || System.currentTimeMillis() < scCutOverTimes[index]) {
386                    return getInstance(scOldCurrencies[index], scOldCurrenciesDFD[index],
387                        scOldCurrenciesNumericCode[index]);
388                } else {
389                    return getInstance(scNewCurrencies[index], scNewCurrenciesDFD[index],
390                        scNewCurrenciesNumericCode[index]);
391                }
392            }
393        }
394    }

396    /**
397     * Gets the set of available currencies.  The returned set of currencies
398     * contains all of the available currencies, which may include currencies
399     * that represent obsolete ISO 4217 codes.  The set can be modified
400     * without affecting the available currencies in the runtime.
401     *
402     * @return the set of available currencies.  If there is no currency
403     *    available in the runtime, the returned set is empty.
404     * @since 1.7
405     */
406    public static Set<Currency> getAvailableCurrencies() {
407        synchronized(Currency.class) {
408            if (available == null) {
409                available = new HashSet<Currency>(256);
410
411                // Add simple currencies first
412                for (char c1 = 'A'; c1 <= 'Z'; c1 ++) {
413                    for (char c2 = 'A'; c2 <= 'Z'; c2 ++) {
414                        int tableEntry = getMainTableEntry(c1, c2);
415                        if ((tableEntry & COUNTRY_TYPE_MASK) == SIMPLE_CASE_COUNTRY_MASK
416                             && tableEntry != INVALID_COUNTRY_ENTRY) {
417                            char finalChar = (char) ((tableEntry & SIMPLE_CASE_COUNTRY_FINAL_CHAR_MASK) + 'A');
418                            int defaultFractionDigits = (tableEntry & SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_MASK) >> SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_SHIFT;
419                            int numericCode = (tableEntry & NUMERIC_CODE_MASK) >> NUMERIC_CODE_SHIFT;
420                            StringBuilder sb = new StringBuilder();
421                            sb.append(c1);
422                            sb.append(c2);
423                            sb.append(finalChar);
424                            available.add(getInstance(sb.toString(), defaultFractionDigits, numericCode));
425                        }
426                    }
427                }
428
429                // Now add other currencies
430                StringTokenizer st = new StringTokenizer(otherCurrencies, "-");
431                while (st.hasMoreElements()) {
432                    available.add(getInstance((String)st.nextElement()));
433                }
434            }
435        }
436
437        @SuppressWarnings("unchecked")
438        Set<Currency> result = (Set<Currency>) available.clone();
439        return result;
440    }

463    /**
464     * Gets the symbol of this currency for the specified locale.
465     * For example, for the US Dollar, the symbol is "$" if the specified
466     * locale is the US, while for other locales it may be "US$". If no
467     * symbol can be determined, the ISO 4217 currency code is returned.
468     *
469     * @param locale the locale for which a display name for this currency is
470     * needed
471     * @return the symbol of this currency for the specified locale
472     * @exception NullPointerException if <code>locale</code> is null
473     */
474    public String getSymbol(Locale locale) {
475        try {
476            // Check whether a provider can provide an implementation that's closer
477            // to the requested locale than what the Java runtime itself can provide.
478            LocaleServiceProviderPool pool =
479                LocaleServiceProviderPool.getPool(CurrencyNameProvider.class);
480
481            if (pool.hasProviders()) {
482                // Assuming that all the country locales include necessary currency
483                // symbols in the Java runtime's resources,  so there is no need to
484                // examine whether Java runtime's currency resource bundle is missing
485                // names.  Therefore, no resource bundle is provided for calling this
486                // method.
487                String symbol = pool.getLocalizedObject(
488                                    CurrencyNameGetter.INSTANCE,
489                                    locale, (OpenListResourceBundle)null,
490                                    currencyCode, SYMBOL);
491                if (symbol != null) {
492                    return symbol;
493                }
494            }
495
496            ResourceBundle bundle = LocaleData.getCurrencyNames(locale);
497            return bundle.getString(currencyCode);
498        } catch (MissingResourceException e) {
499            // use currency code as symbol of last resort
500            return currencyCode;
501        }
502    }

539    /**
540     * Gets the name that is suitable for displaying this currency for
541     * the specified locale.  If there is no suitable display name found
542     * for the specified locale, the ISO 4217 currency code is returned.
543     *
544     * @param locale the locale for which a display name for this currency is
545     * needed
546     * @return the display name of this currency for the specified locale
547     * @exception NullPointerException if <code>locale</code> is null
548     * @since 1.7
549     */
550    public String getDisplayName(Locale locale) {
551        try {
552            OpenListResourceBundle bundle = LocaleData.getCurrencyNames(locale);
553            String result = null;
554            String bundleKey = currencyCode.toLowerCase(Locale.ROOT);
555
556            // Check whether a provider can provide an implementation that's closer
557            // to the requested locale than what the Java runtime itself can provide.
558            LocaleServiceProviderPool pool =
559                LocaleServiceProviderPool.getPool(CurrencyNameProvider.class);
560            if (pool.hasProviders()) {
561                result = pool.getLocalizedObject(
562                                    CurrencyNameGetter.INSTANCE,
563                                    locale, bundleKey, bundle, currencyCode, DISPLAYNAME);
564            }
565
566            if (result == null) {
567                result = bundle.getString(bundleKey);
568            }
569
570            if (result != null) {
571                return result;
572            }
573        } catch (MissingResourceException e) {
574            // fall through
575        }
576
577        // use currency code as symbol of last resort
578        return currencyCode;
579    }

597    /**
598     * Gets the main table entry for the country whose country code consists
599     * of char1 and char2.
600     */
601    private static int getMainTableEntry(char char1, char char2) {
602        if (char1 < 'A' || char1 > 'Z' || char2 < 'A' || char2 > 'Z') {
603            throw new IllegalArgumentException();
604        }
605        return mainTable[(char1 - 'A') * A_TO_Z + (char2 - 'A')];
606    }
607
608    /**
609     * Sets the main table entry for the country whose country code consists
610     * of char1 and char2.
611     */
612    private static void setMainTableEntry(char char1, char char2, int entry) {
613        if (char1 < 'A' || char1 > 'Z' || char2 < 'A' || char2 > 'Z') {
614            throw new IllegalArgumentException();
615        }
616        mainTable[(char1 - 'A') * A_TO_Z + (char2 - 'A')] = entry;
617    }

619    /**
620     * Obtains a localized currency names from a CurrencyNameProvider
621     * implementation.
622     */
623    private static class CurrencyNameGetter
624        implements LocaleServiceProviderPool.LocalizedObjectGetter<CurrencyNameProvider,
625                                                                   String> {
626        private static final CurrencyNameGetter INSTANCE = new CurrencyNameGetter();
627
628        public String getObject(CurrencyNameProvider currencyNameProvider,
629                                Locale locale,
630                                String key,
631                                Object... params) {
632            assert params.length == 1;
633            int type = (Integer)params[0];
634
635            switch(type) {
636            case SYMBOL:
637                return currencyNameProvider.getSymbol(key, locale);
638            case DISPLAYNAME:
639                return currencyNameProvider.getDisplayName(key, locale);
640            default:
641                assert false; // shouldn't happen
642            }
643
644            return null;
645        }
646    }

648    private static int[] readIntArray(DataInputStream dis, int count) throws IOException {
649        int[] ret = new int[count];
650        for (int i = 0; i < count; i++) {
651            ret[i] = dis.readInt();
652        }
653
654        return ret;
655    }

657    private static long[] readLongArray(DataInputStream dis, int count) throws IOException {
658        long[] ret = new long[count];
659        for (int i = 0; i < count; i++) {
660            ret[i] = dis.readLong();
661        }
662
663        return ret;
664    }

675    /**
676     * Replaces currency data found in the currencydata.properties file
677     *
678     * @param pattern regex pattern for the properties
679     * @param ctry country code
680     * @param data currency data.  This is a comma separated string that
681     *    consists of "three-letter alphabet code", "three-digit numeric code",
682     *    and "one-digit (0,1,2, or 3) default fraction digit".
683     *    For example, "JPZ,392,0".
684     * @throws
685     */
686    private static void replaceCurrencyData(Pattern pattern, String ctry, String curdata) {
687
688        if (ctry.length() != 2) {
689            // ignore invalid country code
690            String message = new StringBuilder()
691                .append("The entry in currency.properties for ")
692                .append(ctry).append(" is ignored because of the invalid country code.")
693                .toString();
694            info(message, null);
695            return;
696        }
697
698        Matcher m = pattern.matcher(curdata);
699        if (!m.find()) {
700            // format is not recognized.  ignore the data
701            String message = new StringBuilder()
702                .append("The entry in currency.properties for ")
703                .append(ctry)
704                .append(" is ignored because the value format is not recognized.")
705                .toString();
706            info(message, null);
707            return;
708        }
709
710        String code = m.group(1);
711        int numeric = Integer.parseInt(m.group(2));
712        int fraction = Integer.parseInt(m.group(3));
713        int entry = numeric << NUMERIC_CODE_SHIFT;
714
715        int index;
716        for (index = 0; index < scOldCurrencies.length; index++) {
717            if (scOldCurrencies[index].equals(code)) {
718                break;
719            }
720        }
721
722        if (index == scOldCurrencies.length) {
723            // simple case
724            entry |= (fraction << SIMPLE_CASE_COUNTRY_DEFAULT_DIGITS_SHIFT) |
725                     (code.charAt(2) - 'A');
726        } else {
727            // special case
728            entry |= SPECIAL_CASE_COUNTRY_MASK |
729                     (index + SPECIAL_CASE_COUNTRY_INDEX_DELTA);
730        }
731        setMainTableEntry(ctry.charAt(0), ctry.charAt(1), entry);
732    }
