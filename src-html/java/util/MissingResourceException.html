041package java.util;
042
043/**
044 * Signals that a resource is missing.
045 * @see java.lang.Exception
046 * @see ResourceBundle
047 * @author      Mark Davis
048 * @since       JDK1.1
049 */
050public
051class MissingResourceException extends RuntimeException {
052
053    /**
054     * Constructs a MissingResourceException with the specified information.
055     * A detail message is a String that describes this particular exception.
056     * @param s the detail message
057     * @param className the name of the resource class
058     * @param key the key for the missing resource.
059     */
060    public MissingResourceException(String s, String className, String key) {
061        super(s);
062        this.className = className;
063        this.key = key;
064    }
065
066    /**
067     * Constructs a <code>MissingResourceException</code> with
068     * <code>message</code>, <code>className</code>, <code>key</code>,
069     * and <code>cause</code>. This constructor is package private for
070     * use by <code>ResourceBundle.getBundle</code>.
071     *
072     * @param message
073     *        the detail message
074     * @param className
075     *        the name of the resource class
076     * @param key
077     *        the key for the missing resource.
078     * @param cause
079     *        the cause (which is saved for later retrieval by the
080     *        {@link Throwable.getCause()} method). (A null value is
081     *        permitted, and indicates that the cause is nonexistent
082     *        or unknown.)
083     */
084    MissingResourceException(String message, String className, String key, Throwable cause) {
085        super(message, cause);
086        this.className = className;
087        this.key = key;
088    }
089
090    /**
091     * Gets parameter passed by constructor.
092     *
093     * @return the name of the resource class
094     */
095    public String getClassName() {
096        return className;
097    }
098
099    /**
100     * Gets parameter passed by constructor.
101     *
102     * @return the key for the missing resource
103     */
104    public String getKey() {
105        return key;
106    }
107
108    //============ privates ============
109
110    // serialization compatibility with JDK1.1
111    private static final long serialVersionUID = -4876345176062000401L;
112
113    /**
114     * The class name of the resource bundle requested by the user.
115     * @serial
116     */
117    private String className;
118
119    /**
120     * The name of the specific resource requested by the user.
121     * @serial
122     */
123    private String key;
124}
