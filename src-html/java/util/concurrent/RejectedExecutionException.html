036package java.util.concurrent;
037
038/**
039 * Exception thrown by an {@link Executor} when a task cannot be
040 * accepted for execution.
041 *
042 * @since 1.5
043 * @author Doug Lea
044 */
045public class RejectedExecutionException extends RuntimeException {
046    private static final long serialVersionUID = -375805702767069545L;
047
048    /**
049     * Constructs a <tt>RejectedExecutionException</tt> with no detail message.
050     * The cause is not initialized, and may subsequently be
051     * initialized by a call to {@link #initCause(Throwable) initCause}.
052     */
053    public RejectedExecutionException() { }
054
055    /**
056     * Constructs a <tt>RejectedExecutionException</tt> with the
057     * specified detail message. The cause is not initialized, and may
058     * subsequently be initialized by a call to {@link
059     * #initCause(Throwable) initCause}.
060     *
061     * @param message the detail message
062     */
063    public RejectedExecutionException(String message) {
064        super(message);
065    }
066
067    /**
068     * Constructs a <tt>RejectedExecutionException</tt> with the
069     * specified detail message and cause.
070     *
071     * @param  message the detail message
072     * @param  cause the cause (which is saved for later retrieval by the
073     *         {@link #getCause()} method)
074     */
075    public RejectedExecutionException(String message, Throwable cause) {
076        super(message, cause);
077    }
078
079    /**
080     * Constructs a <tt>RejectedExecutionException</tt> with the
081     * specified cause.  The detail message is set to: <pre> (cause ==
082     * null ? null : cause.toString())</pre> (which typically contains
083     * the class and detail message of <tt>cause</tt>).
084     *
085     * @param  cause the cause (which is saved for later retrieval by the
086     *         {@link #getCause()} method)
087     */
088    public RejectedExecutionException(Throwable cause) {
089        super(cause);
090    }
091}
