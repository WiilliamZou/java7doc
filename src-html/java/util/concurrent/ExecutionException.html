036package java.util.concurrent;
037
038/**
039 * Exception thrown when attempting to retrieve the result of a task
040 * that aborted by throwing an exception. This exception can be
041 * inspected using the {@link #getCause()} method.
042 *
043 * @see Future
044 * @since 1.5
045 * @author Doug Lea
046 */
047public class ExecutionException extends Exception {
048    private static final long serialVersionUID = 7830266012832686185L;
049
050    /**
051     * Constructs an <tt>ExecutionException</tt> with no detail message.
052     * The cause is not initialized, and may subsequently be
053     * initialized by a call to {@link #initCause(Throwable) initCause}.
054     */
055    protected ExecutionException() { }
056
057    /**
058     * Constructs an <tt>ExecutionException</tt> with the specified detail
059     * message. The cause is not initialized, and may subsequently be
060     * initialized by a call to {@link #initCause(Throwable) initCause}.
061     *
062     * @param message the detail message
063     */
064    protected ExecutionException(String message) {
065        super(message);
066    }
067
068    /**
069     * Constructs an <tt>ExecutionException</tt> with the specified detail
070     * message and cause.
071     *
072     * @param  message the detail message
073     * @param  cause the cause (which is saved for later retrieval by the
074     *         {@link #getCause()} method)
075     */
076    public ExecutionException(String message, Throwable cause) {
077        super(message, cause);
078    }
079
080    /**
081     * Constructs an <tt>ExecutionException</tt> with the specified cause.
082     * The detail message is set to:
083     * <pre>
084     *  (cause == null ? null : cause.toString())</pre>
085     * (which typically contains the class and detail message of
086     * <tt>cause</tt>).
087     *
088     * @param  cause the cause (which is saved for later retrieval by the
089     *         {@link #getCause()} method)
090     */
091    public ExecutionException(Throwable cause) {
092        super(cause);
093    }
094}
