051/**
052 * The Introspector class provides a standard way for tools to learn about
053 * the properties, events, and methods supported by a target Java Bean.
054 * <p>
055 * For each of those three kinds of information, the Introspector will
056 * separately analyze the bean's class and superclasses looking for
057 * either explicit or implicit information and use that information to
058 * build a BeanInfo object that comprehensively describes the target bean.
059 * <p>
060 * For each class "Foo", explicit information may be available if there exists
061 * a corresponding "FooBeanInfo" class that provides a non-null value when
062 * queried for the information.   We first look for the BeanInfo class by
063 * taking the full package-qualified name of the target bean class and
064 * appending "BeanInfo" to form a new class name.  If this fails, then
065 * we take the final classname component of this name, and look for that
066 * class in each of the packages specified in the BeanInfo package search
067 * path.
068 * <p>
069 * Thus for a class such as "sun.xyz.OurButton" we would first look for a
070 * BeanInfo class called "sun.xyz.OurButtonBeanInfo" and if that failed we'd
071 * look in each package in the BeanInfo search path for an OurButtonBeanInfo
072 * class.  With the default search path, this would mean looking for
073 * "sun.beans.infos.OurButtonBeanInfo".
074 * <p>
075 * If a class provides explicit BeanInfo about itself then we add that to
076 * the BeanInfo information we obtained from analyzing any derived classes,
077 * but we regard the explicit information as being definitive for the current
078 * class and its base classes, and do not proceed any further up the superclass
079 * chain.
080 * <p>
081 * If we don't find explicit BeanInfo on a class, we use low-level
082 * reflection to study the methods of the class and apply standard design
083 * patterns to identify property accessors, event sources, or public
084 * methods.  We then proceed to analyze the class's superclass and add
085 * in the information from it (and possibly on up the superclass chain).
086 * <p>
087 * For more information about introspection and design patterns, please
088 * consult the
089 *  <a href="http://java.sun.com/products/javabeans/docs/index.html">JavaBeans&trade; specification</a>.
090 */

137    /**
138     * Introspect on a Java Bean and learn about all its properties, exposed
139     * methods, and events.
140     * <p>
141     * If the BeanInfo class for a Java Bean has been previously Introspected
142     * then the BeanInfo class is retrieved from the BeanInfo cache.
143     *
144     * @param beanClass  The bean class to be analyzed.
145     * @return  A BeanInfo object describing the target bean.
146     * @exception IntrospectionException if an exception occurs during
147     *              introspection.
148     * @see #flushCaches
149     * @see #flushFromCaches
150     */
151    public static BeanInfo getBeanInfo(Class<?> beanClass)
152        throws IntrospectionException
153    {
154        if (!ReflectUtil.isPackageAccessible(beanClass)) {
155            return (new Introspector(beanClass, null, USE_ALL_BEANINFO)).getBeanInfo();
156        }
157        ThreadGroupContext context = ThreadGroupContext.getContext();
158        BeanInfo beanInfo;
159        synchronized (declaredMethodCache) {
160            beanInfo = context.getBeanInfo(beanClass);
161        }
162        if (beanInfo == null) {
163            beanInfo = new Introspector(beanClass, null, USE_ALL_BEANINFO).getBeanInfo();
164            synchronized (declaredMethodCache) {
165                context.putBeanInfo(beanClass, beanInfo);
166            }
167        }
168        return beanInfo;
169    }

217    /**
218     * Introspect on a Java Bean and learn about all its properties,
219     * exposed methods and events, below a given {@code stopClass} point
220     * subject to some control {@code flags}.
221     * <dl>
222     *  <dt>USE_ALL_BEANINFO</dt>
223     *  <dd>Any BeanInfo that can be discovered will be used.</dd>
224     *  <dt>IGNORE_IMMEDIATE_BEANINFO</dt>
225     *  <dd>Any BeanInfo associated with the specified {@code beanClass} will be ignored.</dd>
226     *  <dt>IGNORE_ALL_BEANINFO</dt>
227     *  <dd>Any BeanInfo associated with the specified {@code beanClass}
228     *      or any of its parent classes will be ignored.</dd>
229     * </dl>
230     * Any methods/properties/events in the {@code stopClass}
231     * or in its parent classes will be ignored in the analysis.
232     * <p>
233     * If the BeanInfo class for a Java Bean has been
234     * previously introspected based on the same arguments then
235     * the BeanInfo class is retrieved from the BeanInfo cache.
236     *
237     * @param beanClass  the bean class to be analyzed
238     * @param stopClass  the parent class at which to stop the analysis
239     * @param flags      flags to control the introspection
240     * @return a BeanInfo object describing the target bean
241     * @exception IntrospectionException if an exception occurs during introspection
242     *
243     * @since 1.7
244     */
245    public static BeanInfo getBeanInfo(Class<?> beanClass, Class<?> stopClass,
246                                        int flags) throws IntrospectionException {
247        BeanInfo bi;
248        if (stopClass == null && flags == USE_ALL_BEANINFO) {
249            // Same parameters to take advantage of caching.
250            bi = getBeanInfo(beanClass);
251        } else {
252            bi = (new Introspector(beanClass, stopClass, flags)).getBeanInfo();
253        }
254        return bi;
255
256        // Old behaviour: Make an independent copy of the BeanInfo.
257        //return new GenericBeanInfo(bi);
258    }

261    /**
262     * Utility method to take a string and convert it to normal Java variable
263     * name capitalization.  This normally means converting the first
264     * character from upper case to lower case, but in the (unusual) special
265     * case when there is more than one character and both the first and
266     * second characters are upper case, we leave it alone.
267     * <p>
268     * Thus "FooBah" becomes "fooBah" and "X" becomes "x", but "URL" stays
269     * as "URL".
270     *
271     * @param  name The string to be decapitalized.
272     * @return  The decapitalized version of the string.
273     */
274    public static String decapitalize(String name) {
275        if (name == null || name.length() == 0) {
276            return name;
277        }
278        if (name.length() > 1 && Character.isUpperCase(name.charAt(1)) &&
279                        Character.isUpperCase(name.charAt(0))){
280            return name;
281        }
282        char chars[] = name.toCharArray();
283        chars[0] = Character.toLowerCase(chars[0]);
284        return new String(chars);
285    }

340    /**
341     * Flush the Introspector's internal cached information for a given class.
342     * This method is not normally required.  It is normally only needed
343     * by advanced tools that update existing "Class" objects in-place
344     * and need to make the Introspector re-analyze an existing Class object.
345     *
346     * Note that only the direct state associated with the target Class
347     * object is flushed.  We do not flush state for other Class objects
348     * with the same name, nor do we flush state for any related Class
349     * objects (such as subclasses), even though their state may include
350     * information indirectly obtained from the target Class object.
351     *
352     * @param clz  Class object to be flushed.
353     * @throws NullPointerException If the Class object is null.
354     */
355    public static void flushFromCaches(Class<?> clz) {
356        if (clz == null) {
357            throw new NullPointerException();
358        }
359        synchronized (declaredMethodCache) {
360            ThreadGroupContext.getContext().removeBeanInfo(clz);
361            declaredMethodCache.put(clz, null);
362        }
363    }

369    private Introspector(Class beanClass, Class stopClass, int flags)
370                                            throws IntrospectionException {
371        this.beanClass = beanClass;
372
373        // Check stopClass is a superClass of startClass.
374        if (stopClass != null) {
375            boolean isSuper = false;
376            for (Class c = beanClass.getSuperclass(); c != null; c = c.getSuperclass()) {
377                if (c == stopClass) {
378                    isSuper = true;
379                }
380            }
381            if (!isSuper) {
382                throw new IntrospectionException(stopClass.getName() + " not superclass of " +
383                                        beanClass.getName());
384            }
385        }
386
387        if (flags == USE_ALL_BEANINFO) {
388            explicitBeanInfo = findExplicitBeanInfo(beanClass);
389        }
390
391        Class superClass = beanClass.getSuperclass();
392        if (superClass != stopClass) {
393            int newFlags = flags;
394            if (newFlags == IGNORE_IMMEDIATE_BEANINFO) {
395                newFlags = USE_ALL_BEANINFO;
396            }
397            superBeanInfo = getBeanInfo(superClass, stopClass, newFlags);
398        }
399        if (explicitBeanInfo != null) {
400            additionalBeanInfo = explicitBeanInfo.getAdditionalBeanInfo();
401        }
402        if (additionalBeanInfo == null) {
403            additionalBeanInfo = new BeanInfo[0];
404        }
405    }

407    /**
408     * Constructs a GenericBeanInfo class from the state of the Introspector
409     */
410    private BeanInfo getBeanInfo() throws IntrospectionException {
411
412        // the evaluation order here is import, as we evaluate the
413        // event sets and locate PropertyChangeListeners before we
414        // look for properties.
415        BeanDescriptor bd = getTargetBeanDescriptor();
416        MethodDescriptor mds[] = getTargetMethodInfo();
417        EventSetDescriptor esds[] = getTargetEventInfo();
418        PropertyDescriptor pds[] = getTargetPropertyInfo();
419
420        int defaultEvent = getTargetDefaultEventIndex();
421        int defaultProperty = getTargetDefaultPropertyIndex();
422
423        return new GenericBeanInfo(bd, esds, defaultEvent, pds,
424                        defaultProperty, mds, explicitBeanInfo);
425
426    }

441    /**
442     * @return An array of PropertyDescriptors describing the editable
443     * properties supported by the target bean.
444     */
445
446    private PropertyDescriptor[] getTargetPropertyInfo() {
447
448        // Check if the bean has its own BeanInfo that will provide
449        // explicit information.
450        PropertyDescriptor[] explicitProperties = null;
451        if (explicitBeanInfo != null) {
452            explicitProperties = getPropertyDescriptors(this.explicitBeanInfo);
453        }
454
455        if (explicitProperties == null && superBeanInfo != null) {
456            // We have no explicit BeanInfo properties.  Check with our parent.
457            addPropertyDescriptors(getPropertyDescriptors(this.superBeanInfo));
458        }
459
460        for (int i = 0; i < additionalBeanInfo.length; i++) {
461            addPropertyDescriptors(additionalBeanInfo[i].getPropertyDescriptors());
462        }
463
464        if (explicitProperties != null) {
465            // Add the explicit BeanInfo data to our results.
466            addPropertyDescriptors(explicitProperties);
467
468        } else {
469
470            // Apply some reflection to the current class.
471
472            // First get an array of all the public methods at this level
473            Method methodList[] = getPublicDeclaredMethods(beanClass);
474
475            // Now analyze each method.
476            for (int i = 0; i < methodList.length; i++) {
477                Method method = methodList[i];
478                if (method == null) {
479                    continue;
480                }
481                // skip static methods.
482                int mods = method.getModifiers();
483                if (Modifier.isStatic(mods)) {
484                    continue;
485                }
486                String name = method.getName();
487                Class argTypes[] = method.getParameterTypes();
488                Class resultType = method.getReturnType();
489                int argCount = argTypes.length;
490                PropertyDescriptor pd = null;
491
492                if (name.length() <= 3 && !name.startsWith(IS_PREFIX)) {
493                    // Optimization. Don't bother with invalid propertyNames.
494                    continue;
495                }
496
497                try {
498
499                    if (argCount == 0) {
500                        if (name.startsWith(GET_PREFIX)) {
501                            // Simple getter
502                            pd = new PropertyDescriptor(this.beanClass, name.substring(3), method, null);
503                        } else if (resultType == boolean.class && name.startsWith(IS_PREFIX)) {
504                            // Boolean getter
505                            pd = new PropertyDescriptor(this.beanClass, name.substring(2), method, null);
506                        }
507                    } else if (argCount == 1) {
508                        if (int.class.equals(argTypes[0]) && name.startsWith(GET_PREFIX)) {
509                            pd = new IndexedPropertyDescriptor(this.beanClass, name.substring(3), null, null, method, null);
510                        } else if (void.class.equals(resultType) && name.startsWith(SET_PREFIX)) {
511                            // Simple setter
512                            pd = new PropertyDescriptor(this.beanClass, name.substring(3), null, method);
513                            if (throwsException(method, PropertyVetoException.class)) {
514                                pd.setConstrained(true);
515                            }
516                        }
517                    } else if (argCount == 2) {
518                            if (void.class.equals(resultType) && int.class.equals(argTypes[0]) && name.startsWith(SET_PREFIX)) {
519                            pd = new IndexedPropertyDescriptor(this.beanClass, name.substring(3), null, null, null, method);
520                            if (throwsException(method, PropertyVetoException.class)) {
521                                pd.setConstrained(true);
522                            }
523                        }
524                    }
525                } catch (IntrospectionException ex) {
526                    // This happens if a PropertyDescriptor or IndexedPropertyDescriptor
527                    // constructor fins that the method violates details of the deisgn
528                    // pattern, e.g. by having an empty name, or a getter returning
529                    // void , or whatever.
530                    pd = null;
531                }
532
533                if (pd != null) {
534                    // If this class or one of its base classes is a PropertyChange
535                    // source, then we assume that any properties we discover are "bound".
536                    if (propertyChangeSource) {
537                        pd.setBound(true);
538                    }
539                    addPropertyDescriptor(pd);
540                }
541            }
542        }
543        processPropertyDescriptors();
544
545        // Allocate and populate the result array.
546        PropertyDescriptor result[] = new PropertyDescriptor[properties.size()];
547        result = (PropertyDescriptor[])properties.values().toArray(result);
548
549        // Set the default index.
550        if (defaultPropertyName != null) {
551            for (int i = 0; i < result.length; i++) {
552                if (defaultPropertyName.equals(result[i].getName())) {
553                    defaultPropertyIndex = i;
554                }
555            }
556        }
557
558        return result;
559    }

563    /**
564     * Adds the property descriptor to the list store.
565     */
566    private void addPropertyDescriptor(PropertyDescriptor pd) {
567        String propName = pd.getName();
568        List list = (List)pdStore.get(propName);
569        if (list == null) {
570            list = new ArrayList();
571            pdStore.put(propName, list);
572        }
573        if (this.beanClass != pd.getClass0()) {
574            // replace existing property descriptor
575            // only if we have types to resolve
576            // in the context of this.beanClass
577            Method read = pd.getReadMethod();
578            Method write = pd.getWriteMethod();
579            boolean cls = true;
580            if (read != null) cls = cls && read.getGenericReturnType() instanceof Class;
581            if (write != null) cls = cls && write.getGenericParameterTypes()[0] instanceof Class;
582            if (pd instanceof IndexedPropertyDescriptor) {
583                IndexedPropertyDescriptor ipd = (IndexedPropertyDescriptor) pd;
584                Method readI = ipd.getIndexedReadMethod();
585                Method writeI = ipd.getIndexedWriteMethod();
586                if (readI != null) cls = cls && readI.getGenericReturnType() instanceof Class;
587                if (writeI != null) cls = cls && writeI.getGenericParameterTypes()[1] instanceof Class;
588                if (!cls) {
589                    pd = new IndexedPropertyDescriptor(ipd);
590                    pd.updateGenericsFor(this.beanClass);
591                }
592            }
593            else if (!cls) {
594                pd = new PropertyDescriptor(pd);
595                pd.updateGenericsFor(this.beanClass);
596            }
597        }
598        list.add(pd);
599    }

601    private void addPropertyDescriptors(PropertyDescriptor[] descriptors) {
602        if (descriptors != null) {
603            for (PropertyDescriptor descriptor : descriptors) {
604                addPropertyDescriptor(descriptor);
605            }
606        }
607    }

609    private PropertyDescriptor[] getPropertyDescriptors(BeanInfo info) {
610        PropertyDescriptor[] descriptors = info.getPropertyDescriptors();
611        int index = info.getDefaultPropertyIndex();
612        if ((0 <= index) && (index < descriptors.length)) {
613            this.defaultPropertyName = descriptors[index].getName();
614        }
615        return descriptors;
616    }

618    /**
619     * Populates the property descriptor table by merging the
620     * lists of Property descriptors.
621     */
622    private void processPropertyDescriptors() {
623        if (properties == null) {
624            properties = new TreeMap();
625        }
626
627        List list;
628
629        PropertyDescriptor pd, gpd, spd;
630        IndexedPropertyDescriptor ipd, igpd, ispd;
631
632        Iterator it = pdStore.values().iterator();
633        while (it.hasNext()) {
634            pd = null; gpd = null; spd = null;
635            ipd = null; igpd = null; ispd = null;
636
637            list = (List)it.next();
638
639            // First pass. Find the latest getter method. Merge properties
640            // of previous getter methods.
641            for (int i = 0; i < list.size(); i++) {
642                pd = (PropertyDescriptor)list.get(i);
643                if (pd instanceof IndexedPropertyDescriptor) {
644                    ipd = (IndexedPropertyDescriptor)pd;
645                    if (ipd.getIndexedReadMethod() != null) {
646                        if (igpd != null) {
647                            igpd = new IndexedPropertyDescriptor(igpd, ipd);
648                        } else {
649                            igpd = ipd;
650                        }
651                    }
652                } else {
653                    if (pd.getReadMethod() != null) {
654                        if (gpd != null) {
655                            // Don't replace the existing read
656                            // method if it starts with "is"
657                            Method method = gpd.getReadMethod();
658                            if (!method.getName().startsWith(IS_PREFIX)) {
659                                gpd = new PropertyDescriptor(gpd, pd);
660                            }
661                        } else {
662                            gpd = pd;
663                        }
664                    }
665                }
666            }
667
668            // Second pass. Find the latest setter method which
669            // has the same type as the getter method.
670            for (int i = 0; i < list.size(); i++) {
671                pd = (PropertyDescriptor)list.get(i);
672                if (pd instanceof IndexedPropertyDescriptor) {
673                    ipd = (IndexedPropertyDescriptor)pd;
674                    if (ipd.getIndexedWriteMethod() != null) {
675                        if (igpd != null) {
676                            if (igpd.getIndexedPropertyType()
677                                == ipd.getIndexedPropertyType()) {
678                                if (ispd != null) {
679                                    ispd = new IndexedPropertyDescriptor(ispd, ipd);
680                                } else {
681                                    ispd = ipd;
682                                }
683                            }
684                        } else {
685                            if (ispd != null) {
686                                ispd = new IndexedPropertyDescriptor(ispd, ipd);
687                            } else {
688                                ispd = ipd;
689                            }
690                        }
691                    }
692                } else {
693                    if (pd.getWriteMethod() != null) {
694                        if (gpd != null) {
695                            if (gpd.getPropertyType() == pd.getPropertyType()) {
696                                if (spd != null) {
697                                    spd = new PropertyDescriptor(spd, pd);
698                                } else {
699                                    spd = pd;
700                                }
701                            }
702                        } else {
703                            if (spd != null) {
704                                spd = new PropertyDescriptor(spd, pd);
705                            } else {
706                                spd = pd;
707                            }
708                        }
709                    }
710                }
711            }
712
713            // At this stage we should have either PDs or IPDs for the
714            // representative getters and setters. The order at which the
715            // property descriptors are determined represent the
716            // precedence of the property ordering.
717            pd = null; ipd = null;
718
719            if (igpd != null && ispd != null) {
720                // Complete indexed properties set
721                // Merge any classic property descriptors
722                if (gpd != null) {
723                    PropertyDescriptor tpd = mergePropertyDescriptor(igpd, gpd);
724                    if (tpd instanceof IndexedPropertyDescriptor) {
725                        igpd = (IndexedPropertyDescriptor)tpd;
726                    }
727                }
728                if (spd != null) {
729                    PropertyDescriptor tpd = mergePropertyDescriptor(ispd, spd);
730                    if (tpd instanceof IndexedPropertyDescriptor) {
731                        ispd = (IndexedPropertyDescriptor)tpd;
732                    }
733                }
734                if (igpd == ispd) {
735                    pd = igpd;
736                } else {
737                    pd = mergePropertyDescriptor(igpd, ispd);
738                }
739            } else if (gpd != null && spd != null) {
740                // Complete simple properties set
741                if (gpd == spd) {
742                    pd = gpd;
743                } else {
744                    pd = mergePropertyDescriptor(gpd, spd);
745                }
746            } else if (ispd != null) {
747                // indexed setter
748                pd = ispd;
749                // Merge any classic property descriptors
750                if (spd != null) {
751                    pd = mergePropertyDescriptor(ispd, spd);
752                }
753                if (gpd != null) {
754                    pd = mergePropertyDescriptor(ispd, gpd);
755                }
756            } else if (igpd != null) {
757                // indexed getter
758                pd = igpd;
759                // Merge any classic property descriptors
760                if (gpd != null) {
761                    pd = mergePropertyDescriptor(igpd, gpd);
762                }
763                if (spd != null) {
764                    pd = mergePropertyDescriptor(igpd, spd);
765                }
766            } else if (spd != null) {
767                // simple setter
768                pd = spd;
769            } else if (gpd != null) {
770                // simple getter
771                pd = gpd;
772            }
773
774            // Very special case to ensure that an IndexedPropertyDescriptor
775            // doesn't contain less information than the enclosed
776            // PropertyDescriptor. If it does, then recreate as a
777            // PropertyDescriptor. See 4168833
778            if (pd instanceof IndexedPropertyDescriptor) {
779                ipd = (IndexedPropertyDescriptor)pd;
780                if (ipd.getIndexedReadMethod() == null && ipd.getIndexedWriteMethod() == null) {
781                    pd = new PropertyDescriptor(ipd);
782                }
783            }
784
785            // Find the first property descriptor
786            // which does not have getter and setter methods.
787            // See regression bug 4984912.
788            if ( (pd == null) && (list.size() > 0) ) {
789                pd = (PropertyDescriptor) list.get(0);
790            }
791
792            if (pd != null) {
793                properties.put(pd.getName(), pd);
794            }
795        }
796    }

798    /**
799     * Adds the property descriptor to the indexedproperty descriptor only if the
800     * types are the same.
801     *
802     * The most specific property descriptor will take precedence.
803     */
804    private PropertyDescriptor mergePropertyDescriptor(IndexedPropertyDescriptor ipd,
805                                                       PropertyDescriptor pd) {
806        PropertyDescriptor result = null;
807
808        Class propType = pd.getPropertyType();
809        Class ipropType = ipd.getIndexedPropertyType();
810
811        if (propType.isArray() && propType.getComponentType() == ipropType) {
812            if (pd.getClass0().isAssignableFrom(ipd.getClass0())) {
813                result = new IndexedPropertyDescriptor(pd, ipd);
814            } else {
815                result = new IndexedPropertyDescriptor(ipd, pd);
816            }
817        } else {
818            // Cannot merge the pd because of type mismatch
819            // Return the most specific pd
820            if (pd.getClass0().isAssignableFrom(ipd.getClass0())) {
821                result = ipd;
822            } else {
823                result = pd;
824                // Try to add methods which may have been lost in the type change
825                // See 4168833
826                Method write = result.getWriteMethod();
827                Method read = result.getReadMethod();
828
829                if (read == null && write != null) {
830                    read = findMethod(result.getClass0(),
831                                      GET_PREFIX + NameGenerator.capitalize(result.getName()), 0);
832                    if (read != null) {
833                        try {
834                            result.setReadMethod(read);
835                        } catch (IntrospectionException ex) {
836                            // no consequences for failure.
837                        }
838                    }
839                }
840                if (write == null && read != null) {
841                    write = findMethod(result.getClass0(),
842                                       SET_PREFIX + NameGenerator.capitalize(result.getName()), 1,
843                                       new Class[] { FeatureDescriptor.getReturnType(result.getClass0(), read) });
844                    if (write != null) {
845                        try {
846                            result.setWriteMethod(write);
847                        } catch (IntrospectionException ex) {
848                            // no consequences for failure.
849                        }
850                    }
851                }
852            }
853        }
854        return result;
855    }

877    /**
878     * @return An array of EventSetDescriptors describing the kinds of
879     * events fired by the target bean.
880     */
881    private EventSetDescriptor[] getTargetEventInfo() throws IntrospectionException {
882        if (events == null) {
883            events = new HashMap();
884        }
885
886        // Check if the bean has its own BeanInfo that will provide
887        // explicit information.
888        EventSetDescriptor[] explicitEvents = null;
889        if (explicitBeanInfo != null) {
890            explicitEvents = explicitBeanInfo.getEventSetDescriptors();
891            int ix = explicitBeanInfo.getDefaultEventIndex();
892            if (ix >= 0 && ix < explicitEvents.length) {
893                defaultEventName = explicitEvents[ix].getName();
894            }
895        }
896
897        if (explicitEvents == null && superBeanInfo != null) {
898            // We have no explicit BeanInfo events.  Check with our parent.
899            EventSetDescriptor supers[] = superBeanInfo.getEventSetDescriptors();
900            for (int i = 0 ; i < supers.length; i++) {
901                addEvent(supers[i]);
902            }
903            int ix = superBeanInfo.getDefaultEventIndex();
904            if (ix >= 0 && ix < supers.length) {
905                defaultEventName = supers[ix].getName();
906            }
907        }
908
909        for (int i = 0; i < additionalBeanInfo.length; i++) {
910            EventSetDescriptor additional[] = additionalBeanInfo[i].getEventSetDescriptors();
911            if (additional != null) {
912                for (int j = 0 ; j < additional.length; j++) {
913                    addEvent(additional[j]);
914                }
915            }
916        }
917
918        if (explicitEvents != null) {
919            // Add the explicit explicitBeanInfo data to our results.
920            for (int i = 0 ; i < explicitEvents.length; i++) {
921                addEvent(explicitEvents[i]);
922            }
923
924        } else {
925
926            // Apply some reflection to the current class.
927
928            // Get an array of all the public beans methods at this level
929            Method methodList[] = getPublicDeclaredMethods(beanClass);
930
931            // Find all suitable "add", "remove" and "get" Listener methods
932            // The name of the listener type is the key for these hashtables
933            // i.e, ActionListener
934            Map adds = null;
935            Map removes = null;
936            Map gets = null;
937
938            for (int i = 0; i < methodList.length; i++) {
939                Method method = methodList[i];
940                if (method == null) {
941                    continue;
942                }
943                // skip static methods.
944                int mods = method.getModifiers();
945                if (Modifier.isStatic(mods)) {
946                    continue;
947                }
948                String name = method.getName();
949                // Optimization avoid getParameterTypes
950                if (!name.startsWith(ADD_PREFIX) && !name.startsWith(REMOVE_PREFIX)
951                    && !name.startsWith(GET_PREFIX)) {
952                    continue;
953                }
954
955                if (name.startsWith(ADD_PREFIX)) {
956                    Class<?> returnType = method.getReturnType();
957                    if (returnType == void.class) {
958                        Type[] parameterTypes = method.getGenericParameterTypes();
959                        if (parameterTypes.length == 1) {
960                            Class<?> type = TypeResolver.erase(TypeResolver.resolveInClass(beanClass, parameterTypes[0]));
961                            if (Introspector.isSubclass(type, eventListenerType)) {
962                                String listenerName = name.substring(3);
963                                if (listenerName.length() > 0 &&
964                                    type.getName().endsWith(listenerName)) {
965                                    if (adds == null) {
966                                        adds = new HashMap();
967                                    }
968                                    adds.put(listenerName, method);
969                                }
970                            }
971                        }
972                    }
973                }
974                else if (name.startsWith(REMOVE_PREFIX)) {
975                    Class<?> returnType = method.getReturnType();
976                    if (returnType == void.class) {
977                        Type[] parameterTypes = method.getGenericParameterTypes();
978                        if (parameterTypes.length == 1) {
979                            Class<?> type = TypeResolver.erase(TypeResolver.resolveInClass(beanClass, parameterTypes[0]));
980                            if (Introspector.isSubclass(type, eventListenerType)) {
981                                String listenerName = name.substring(6);
982                                if (listenerName.length() > 0 &&
983                                    type.getName().endsWith(listenerName)) {
984                                    if (removes == null) {
985                                        removes = new HashMap();
986                                    }
987                                    removes.put(listenerName, method);
988                                }
989                            }
990                        }
991                    }
992                }
993                else if (name.startsWith(GET_PREFIX)) {
994                    Class<?>[] parameterTypes = method.getParameterTypes();
995                    if (parameterTypes.length == 0) {
996                        Class<?> returnType = FeatureDescriptor.getReturnType(beanClass, method);
997                        if (returnType.isArray()) {
998                            Class<?> type = returnType.getComponentType();
999                            if (Introspector.isSubclass(type, eventListenerType)) {
1000                                String listenerName  = name.substring(3, name.length() - 1);
1001                                if (listenerName.length() > 0 &&
1002                                    type.getName().endsWith(listenerName)) {
1003                                    if (gets == null) {
1004                                        gets = new HashMap();
1005                                    }
1006                                    gets.put(listenerName, method);
1007                                }
1008                            }
1009                        }
1010                    }
1011                }
1012            }
1013
1014            if (adds != null && removes != null) {
1015                // Now look for matching addFooListener+removeFooListener pairs.
1016                // Bonus if there is a matching getFooListeners method as well.
1017                Iterator keys = adds.keySet().iterator();
1018                while (keys.hasNext()) {
1019                    String listenerName = (String) keys.next();
1020                    // Skip any "add" which doesn't have a matching "remove" or
1021                    // a listener name that doesn't end with Listener
1022                    if (removes.get(listenerName) == null || !listenerName.endsWith("Listener")) {
1023                        continue;
1024                    }
1025                    String eventName = decapitalize(listenerName.substring(0, listenerName.length()-8));
1026                    Method addMethod = (Method)adds.get(listenerName);
1027                    Method removeMethod = (Method)removes.get(listenerName);
1028                    Method getMethod = null;
1029                    if (gets != null) {
1030                        getMethod = (Method)gets.get(listenerName);
1031                    }
1032                    Class argType = FeatureDescriptor.getParameterTypes(beanClass, addMethod)[0];
1033
1034                    // generate a list of Method objects for each of the target methods:
1035                    Method allMethods[] = getPublicDeclaredMethods(argType);
1036                    List validMethods = new ArrayList(allMethods.length);
1037                    for (int i = 0; i < allMethods.length; i++) {
1038                        if (allMethods[i] == null) {
1039                            continue;
1040                        }
1041
1042                        if (isEventHandler(allMethods[i])) {
1043                            validMethods.add(allMethods[i]);
1044                        }
1045                    }
1046                    Method[] methods = (Method[])validMethods.toArray(new Method[validMethods.size()]);
1047
1048                    EventSetDescriptor esd = new EventSetDescriptor(eventName, argType,
1049                                                                    methods, addMethod,
1050                                                                    removeMethod,
1051                                                                    getMethod);
1052
1053                    // If the adder method throws the TooManyListenersException then it
1054                    // is a Unicast event source.
1055                    if (throwsException(addMethod,
1056                                        java.util.TooManyListenersException.class)) {
1057                        esd.setUnicast(true);
1058                    }
1059                    addEvent(esd);
1060                }
1061            } // if (adds != null ...
1062        }
1063        EventSetDescriptor[] result;
1064        if (events.size() == 0) {
1065            result = EMPTY_EVENTSETDESCRIPTORS;
1066        } else {
1067            // Allocate and populate the result array.
1068            result = new EventSetDescriptor[events.size()];
1069            result = (EventSetDescriptor[])events.values().toArray(result);
1070
1071            // Set the default index.
1072            if (defaultEventName != null) {
1073                for (int i = 0; i < result.length; i++) {
1074                    if (defaultEventName.equals(result[i].getName())) {
1075                        defaultEventIndex = i;
1076                    }
1077                }
1078            }
1079        }
1080        return result;
1081    }

1083    private void addEvent(EventSetDescriptor esd) {
1084        String key = esd.getName();
1085        if (esd.getName().equals("propertyChange")) {
1086            propertyChangeSource = true;
1087        }
1088        EventSetDescriptor old = (EventSetDescriptor)events.get(key);
1089        if (old == null) {
1090            events.put(key, esd);
1091            return;
1092        }
1093        EventSetDescriptor composite = new EventSetDescriptor(old, esd);
1094        events.put(key, composite);
1095    }

1097    /**
1098     * @return An array of MethodDescriptors describing the private
1099     * methods supported by the target bean.
1100     */
1101    private MethodDescriptor[] getTargetMethodInfo() {
1102        if (methods == null) {
1103            methods = new HashMap(100);
1104        }
1105
1106        // Check if the bean has its own BeanInfo that will provide
1107        // explicit information.
1108        MethodDescriptor[] explicitMethods = null;
1109        if (explicitBeanInfo != null) {
1110            explicitMethods = explicitBeanInfo.getMethodDescriptors();
1111        }
1112
1113        if (explicitMethods == null && superBeanInfo != null) {
1114            // We have no explicit BeanInfo methods.  Check with our parent.
1115            MethodDescriptor supers[] = superBeanInfo.getMethodDescriptors();
1116            for (int i = 0 ; i < supers.length; i++) {
1117                addMethod(supers[i]);
1118            }
1119        }
1120
1121        for (int i = 0; i < additionalBeanInfo.length; i++) {
1122            MethodDescriptor additional[] = additionalBeanInfo[i].getMethodDescriptors();
1123            if (additional != null) {
1124                for (int j = 0 ; j < additional.length; j++) {
1125                    addMethod(additional[j]);
1126                }
1127            }
1128        }
1129
1130        if (explicitMethods != null) {
1131            // Add the explicit explicitBeanInfo data to our results.
1132            for (int i = 0 ; i < explicitMethods.length; i++) {
1133                addMethod(explicitMethods[i]);
1134            }
1135
1136        } else {
1137
1138            // Apply some reflection to the current class.
1139
1140            // First get an array of all the beans methods at this level
1141            Method methodList[] = getPublicDeclaredMethods(beanClass);
1142
1143            // Now analyze each method.
1144            for (int i = 0; i < methodList.length; i++) {
1145                Method method = methodList[i];
1146                if (method == null) {
1147                    continue;
1148                }
1149                MethodDescriptor md = new MethodDescriptor(method);
1150                addMethod(md);
1151            }
1152        }
1153
1154        // Allocate and populate the result array.
1155        MethodDescriptor result[] = new MethodDescriptor[methods.size()];
1156        result = (MethodDescriptor[])methods.values().toArray(result);
1157
1158        return result;
1159    }

1161    private void addMethod(MethodDescriptor md) {
1162        // We have to be careful here to distinguish method by both name
1163        // and argument lists.
1164        // This method gets called a *lot, so we try to be efficient.
1165        String name = md.getName();
1166
1167        MethodDescriptor old = (MethodDescriptor)methods.get(name);
1168        if (old == null) {
1169            // This is the common case.
1170            methods.put(name, md);
1171            return;
1172        }
1173
1174        // We have a collision on method names.  This is rare.
1175
1176        // Check if old and md have the same type.
1177        String[] p1 = md.getParamNames();
1178        String[] p2 = old.getParamNames();
1179
1180        boolean match = false;
1181        if (p1.length == p2.length) {
1182            match = true;
1183            for (int i = 0; i < p1.length; i++) {
1184                if (p1[i] != p2[i]) {
1185                    match = false;
1186                    break;
1187                }
1188            }
1189        }
1190        if (match) {
1191            MethodDescriptor composite = new MethodDescriptor(old, md);
1192            methods.put(name, composite);
1193            return;
1194        }
1195
1196        // We have a collision on method names with different type signatures.
1197        // This is very rare.
1198
1199        String longKey = makeQualifiedMethodName(name, p1);
1200        old = (MethodDescriptor)methods.get(longKey);
1201        if (old == null) {
1202            methods.put(longKey, md);
1203            return;
1204        }
1205        MethodDescriptor composite = new MethodDescriptor(old, md);
1206        methods.put(longKey, composite);
1207    }

1213        StringBuffer sb = new StringBuffer(name);

1216            sb.append(':');
1217            sb.append(params[i]);

1219        return sb.toString();

1259        // We assume that a method is an event handler if it has a single
1260        // argument, whose type inherit from java.util.Event.
1261        Type argTypes[] = m.getGenericParameterTypes();
1262        if (argTypes.length != 1) {

1268    /*
1269     * Internal method to return *public* methods within a class.
1270     */
1271    private static Method[] getPublicDeclaredMethods(Class clz) {
1272        // Looking up Class.getDeclaredMethods is relatively expensive,
1273        // so we cache the results.
1274        if (!ReflectUtil.isPackageAccessible(clz)) {
1275            return new Method[0];
1276        }
1277        synchronized (declaredMethodCache) {
1278            Method[] result = declaredMethodCache.get(clz);
1279            if (result == null) {
1280                result = clz.getMethods();
1281                for (int i = 0; i < result.length; i++) {
1282                    Method method = result[i];
1283                    if (!method.getDeclaringClass().equals(clz)) {
1284                        result[i] = null;
1285                    }
1286                }
1287                declaredMethodCache.put(clz, result);
1288            }
1289            return result;
1290        }
1291    }

1297    /**
1298     * Internal support for finding a target methodName with a given
1299     * parameter list on a given class.
1300     */
1301    private static Method internalFindMethod(Class start, String methodName,
1302                                                 int argCount, Class args[]) {
1303        // For overriden methods we need to find the most derived version.
1304        // So we start with the given class and walk up the superclass chain.
1305
1306        Method method = null;
1307
1308        for (Class cl = start; cl != null; cl = cl.getSuperclass()) {
1309            Method methods[] = getPublicDeclaredMethods(cl);
1310            for (int i = 0; i < methods.length; i++) {
1311                method = methods[i];
1312                if (method == null) {
1313                    continue;
1314                }
1315
1316                // make sure method signature matches.
1317                if (method.getName().equals(methodName)) {
1318                    Type[] params = method.getGenericParameterTypes();
1319                    if (params.length == argCount) {
1320                        if (args != null) {
1321                            boolean different = false;
1322                            if (argCount > 0) {
1323                                for (int j = 0; j < argCount; j++) {
1324                                    if (TypeResolver.erase(TypeResolver.resolveInClass(start, params[j])) != args[j]) {
1325                                        different = true;
1326                                        continue;
1327                                    }
1328                                }
1329                                if (different) {
1330                                    continue;
1331                                }
1332                            }
1333                        }
1334                        return method;
1335                    }
1336                }
1337            }
1338        }
1339        method = null;
1340
1341        // Now check any inherited interfaces.  This is necessary both when
1342        // the argument class is itself an interface, and when the argument
1343        // class is an abstract class.
1344        Class ifcs[] = start.getInterfaces();
1345        for (int i = 0 ; i < ifcs.length; i++) {
1346            // Note: The original implementation had both methods calling
1347            // the 3 arg method. This is preserved but perhaps it should
1348            // pass the args array instead of null.
1349            method = internalFindMethod(ifcs[i], methodName, argCount, null);
1350            if (method != null) {
1351                break;
1352            }
1353        }
1354        return method;
1355    }

1364    /**
1365     * Find a target methodName with specific parameter list on a given class.
1366     * <p>
1367     * Used in the contructors of the EventSetDescriptor,
1368     * PropertyDescriptor and the IndexedPropertyDescriptor.
1369     * <p>
1370     * @param cls The Class object on which to retrieve the method.
1371     * @param methodName Name of the method.
1372     * @param argCount Number of arguments for the desired method.
1373     * @param args Array of argument types for the method.
1374     * @return the method or null if not found
1375     */
1376    static Method findMethod(Class cls, String methodName, int argCount,
1377                             Class args[]) {
1378        if (methodName == null) {
1379            return null;
1380        }
1381        return internalFindMethod(cls, methodName, argCount, args);
1382    }
1383
1384    /**
1385     * Return true if class a is either equivalent to class b, or
1386     * if class a is a subclass of class b, i.e. if a either "extends"
1387     * or "implements" b.
1388     * Note tht either or both "Class" objects may represent interfaces.
1389     */
1390    static  boolean isSubclass(Class a, Class b) {
1391        // We rely on the fact that for any given java class or
1392        // primtitive type there is a unqiue Class object, so
1393        // we can use object equivalence in the comparisons.
1394        if (a == b) {
1395            return true;
1396        }
1397        if (a == null || b == null) {
1398            return false;
1399        }
1400        for (Class x = a; x != null; x = x.getSuperclass()) {
1401            if (x == b) {
1402                return true;
1403            }
1404            if (b.isInterface()) {
1405                Class interfaces[] = x.getInterfaces();
1406                for (int i = 0; i < interfaces.length; i++) {
1407                    if (isSubclass(interfaces[i], b)) {
1408                        return true;
1409                    }
1410                }
1411            }
1412        }
1413        return false;
1414    }
1415
1416    /**
1417     * Return true iff the given method throws the given exception.
1418     */
1419    private boolean throwsException(Method method, Class exception) {
1420        Class exs[] = method.getExceptionTypes();
1421        for (int i = 0; i < exs.length; i++) {
1422            if (exs[i] == exception) {
1423                return true;
1424            }
1425        }
1426        return false;
1427    }
1428
1429    /**
1430     * Try to create an instance of a named class.
1431     * First try the classloader of "sibling", then try the system
1432     * classloader then the class loader of the current Thread.
1433     */
1434    static Object instantiate(Class sibling, String className)
1435                 throws InstantiationException, IllegalAccessException,
1436                                                ClassNotFoundException {
1437        // First check with sibling's classloader (if any).
1438        ClassLoader cl = sibling.getClassLoader();
1439        Class cls = ClassFinder.findClass(className, cl);
1440        return cls.newInstance();
1441    }

1455
1456    private BeanDescriptor beanDescriptor;
1457    private EventSetDescriptor[] events;
1458    private int defaultEvent;
1459    private PropertyDescriptor[] properties;
1460    private int defaultProperty;
1461    private MethodDescriptor[] methods;
1462    private Reference<BeanInfo> targetBeanInfoRef;
1463
1464    public GenericBeanInfo(BeanDescriptor beanDescriptor,
1465                EventSetDescriptor[] events, int defaultEvent,
1466                PropertyDescriptor[] properties, int defaultProperty,
1467                MethodDescriptor[] methods, BeanInfo targetBeanInfo) {
1468        this.beanDescriptor = beanDescriptor;
1469        this.events = events;
1470        this.defaultEvent = defaultEvent;
1471        this.properties = properties;
1472        this.defaultProperty = defaultProperty;
1473        this.methods = methods;
1474        this.targetBeanInfoRef = (targetBeanInfo != null)
1475                ? new SoftReference<>(targetBeanInfo)
1476                : null;
1477    }
1478
1479    /**
1480     * Package-private dup constructor
1481     * This must isolate the new object from any changes to the old object.
1482     */
1483    GenericBeanInfo(GenericBeanInfo old) {
1484
1485        beanDescriptor = new BeanDescriptor(old.beanDescriptor);
1486        if (old.events != null) {
1487            int len = old.events.length;
1488            events = new EventSetDescriptor[len];
1489            for (int i = 0; i < len; i++) {
1490                events[i] = new EventSetDescriptor(old.events[i]);
1491            }
1492        }
1493        defaultEvent = old.defaultEvent;
1494        if (old.properties != null) {
1495            int len = old.properties.length;
1496            properties = new PropertyDescriptor[len];
1497            for (int i = 0; i < len; i++) {
1498                PropertyDescriptor oldp = old.properties[i];
1499                if (oldp instanceof IndexedPropertyDescriptor) {
1500                    properties[i] = new IndexedPropertyDescriptor(
1501                                        (IndexedPropertyDescriptor) oldp);
1502                } else {
1503                    properties[i] = new PropertyDescriptor(oldp);
1504                }
1505            }
1506        }
1507        defaultProperty = old.defaultProperty;
1508        if (old.methods != null) {
1509            int len = old.methods.length;
1510            methods = new MethodDescriptor[len];
1511            for (int i = 0; i < len; i++) {
1512                methods[i] = new MethodDescriptor(old.methods[i]);
1513            }
1514        }
1515        this.targetBeanInfoRef = old.targetBeanInfoRef;
1516    }

1519        return properties;
1520    }

1522    public int getDefaultPropertyIndex() {
1523        return defaultProperty;
1524    }

1528    }

1532    }

1536    }

1539        return beanDescriptor;
1540    }

1542    public java.awt.Image getIcon(int iconKind) {
1543        BeanInfo targetBeanInfo = getTargetBeanInfo();
1544        if (targetBeanInfo != null) {
1545            return targetBeanInfo.getIcon(iconKind);
1546        }
1547        return super.getIcon(iconKind);
1548    }
1549
1550    private BeanInfo getTargetBeanInfo() {
1551        if (this.targetBeanInfoRef == null) {
1552            return null;
1553        }
1554        BeanInfo targetBeanInfo = this.targetBeanInfoRef.get();
1555        if (targetBeanInfo == null) {
1556            targetBeanInfo = ThreadGroupContext.getContext().getBeanInfoFinder()
1557                    .find(this.beanDescriptor.getBeanClass());
1558            if (targetBeanInfo != null) {
1559                this.targetBeanInfoRef = new SoftReference<>(targetBeanInfo);
1560            }
1561        }
1562        return targetBeanInfo;
1563    }
