026package java.beans;
027
028/**
029 * Under some circumstances a bean may be run on servers where a GUI
030 * is not available.  This interface can be used to query a bean to
031 * determine whether it absolutely needs a gui, and to advise the
032 * bean whether a GUI is available.
033 * <p>
034 * This interface is for expert developers, and is not needed
035 * for normal simple beans.  To avoid confusing end-users we
036 * avoid using getXXX setXXX design patterns for these methods.
037 */
038
039public interface Visibility {
040
041    /**
042     * Determines whether this bean needs a GUI.
043     *
044     * @return True if the bean absolutely needs a GUI available in
045     *          order to get its work done.
046     */
047    boolean needsGui();
048
049    /**
050     * This method instructs the bean that it should not use the Gui.
051     */
052    void dontUseGui();
053
054    /**
055     * This method instructs the bean that it is OK to use the Gui.
056     */
057    void okToUseGui();
058
059    /**
060     * Determines whether this bean is avoiding using a GUI.
061     *
062     * @return true if the bean is currently avoiding use of the Gui.
063     *   e.g. due to a call on dontUseGui().
064     */
065    boolean avoidingGui();
066
067}
