026package java.beans;
027
028/**
029 * A customizer class provides a complete custom GUI for customizing
030 * a target Java Bean.
031 * <P>
032 * Each customizer should inherit from the java.awt.Component class so
033 * it can be instantiated inside an AWT dialog or panel.
034 * <P>
035 * Each customizer should have a null constructor.
036 */
037
038public interface Customizer {
039
040    /**
041     * Set the object to be customized.  This method should be called only
042     * once, before the Customizer has been added to any parent AWT container.
043     * @param bean  The object to be customized.
044     */
045    void setObject(Object bean);
046
047    /**
048     * Register a listener for the PropertyChange event.  The customizer
049     * should fire a PropertyChange event whenever it changes the target
050     * bean in a way that might require the displayed properties to be
051     * refreshed.
052     *
053     * @param listener  An object to be invoked when a PropertyChange
054     *          event is fired.
055     */
056     void addPropertyChangeListener(PropertyChangeListener listener);
057
058    /**
059     * Remove a listener for the PropertyChange event.
060     *
061     * @param listener  The PropertyChange listener to be removed.
062     */
063    void removePropertyChangeListener(PropertyChangeListener listener);
064
065}
