026package java.beans;
027
028/**
029 * <p>
030 * This interface is intended to be implemented by, or delegated from, instances
031 * of java.beans.beancontext.BeanContext, in order to propagate to its nested hierarchy
032 * of java.beans.beancontext.BeanContextChild instances, the current "designTime" property.
033 * <p>
034 * The JavaBeans&trade; specification defines the notion of design time as is a
035 * mode in which JavaBeans instances should function during their composition
036 * and customization in a interactive design, composition or construction tool,
037 * as opposed to runtime when the JavaBean is part of an applet, application,
038 * or other live Java executable abstraction.
039 *
040 * @author Laurence P. G. Cable
041 * @since 1.2
042 *
043 * @see java.beans.beancontext.BeanContext
044 * @see java.beans.beancontext.BeanContextChild
045 * @see java.beans.beancontext.BeanContextMembershipListener
046 * @see java.beans.PropertyChangeEvent
047 */
048
049public interface DesignMode {
050
051    /**
052     * The standard value of the propertyName as fired from a BeanContext or
053     * other source of PropertyChangeEvents.
054     */
055
056    static String PROPERTYNAME = "designTime";
057
058    /**
059     * Sets the "value" of the "designTime" property.
060     * <p>
061     * If the implementing object is an instance of java.beans.beancontext.BeanContext,
062     * or a subinterface thereof, then that BeanContext should fire a
063     * PropertyChangeEvent, to its registered BeanContextMembershipListeners, with
064     * parameters:
065     * <ul>
066     *    <li><code>propertyName</code> - <code>java.beans.DesignMode.PROPERTYNAME</code>
067     *    <li><code>oldValue</code> - previous value of "designTime"
068     *    <li><code>newValue</code> - current value of "designTime"
069     * </ul>
070     * Note it is illegal for a BeanContextChild to invoke this method
071     * associated with a BeanContext that it is nested within.
072     *
073     * @param designTime  the current "value" of the "designTime" property
074     * @see java.beans.beancontext.BeanContext
075     * @see java.beans.beancontext.BeanContextMembershipListener
076     * @see java.beans.PropertyChangeEvent
077     */
078
079    void setDesignTime(boolean designTime);
080
081    /**
082     * A value of true denotes that JavaBeans should behave in design time
083     * mode, a value of false denotes runtime behavior.
084     *
085     * @return the current "value" of the "designTime" property.
086     */
087
088    boolean isDesignTime();
089}
