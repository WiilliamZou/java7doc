026package java.beans.beancontext;
027
028import java.util.EventObject;
029
030import java.beans.beancontext.BeanContext;
031
032/**
033 * <p>
034 * <code>BeanContextEvent</code> is the abstract root event class
035 * for all events emitted
036 * from, and pertaining to the semantics of, a <code>BeanContext</code>.
037 * This class introduces a mechanism to allow the propagation of
038 * <code>BeanContextEvent</code> subclasses through a hierarchy of
039 * <code>BeanContext</code>s. The <code>setPropagatedFrom()</code>
040 * and <code>getPropagatedFrom()</code> methods allow a
041 * <code>BeanContext</code> to identify itself as the source
042 * of a propagated event.
043 * </p>
044 *
045 * @author      Laurence P. G. Cable
046 * @since       1.2
047 * @see         java.beans.beancontext.BeanContext
048 */
049
050public abstract class BeanContextEvent extends EventObject {
051    private static final long serialVersionUID = 7267998073569045052L;
052
053    /**
054     * Contruct a BeanContextEvent
055     *
056     * @param bc        The BeanContext source
057     */
058    protected BeanContextEvent(BeanContext bc) {
059        super(bc);
060    }
061
062    /**
063     * Gets the <code>BeanContext</code> associated with this event.
064     * @return the <code>BeanContext</code> associated with this event.
065     */
066    public BeanContext getBeanContext() { return (BeanContext)getSource(); }
067
068    /**
069     * Sets the <code>BeanContext</code> from which this event was propagated.
070     * @param bc the <code>BeanContext</code> from which this event
071     * was propagated
072     */
073    public synchronized void setPropagatedFrom(BeanContext bc) {
074        propagatedFrom = bc;
075    }
076
077    /**
078     * Gets the <code>BeanContext</code> from which this event was propagated.
079     * @return the <code>BeanContext</code> from which this
080     * event was propagated
081     */
082    public synchronized BeanContext getPropagatedFrom() {
083        return propagatedFrom;
084    }
085
086    /**
087     * Reports whether or not this event is
088     * propagated from some other <code>BeanContext</code>.
089     * @return <code>true</code> if propagated, <code>false</code>
090     * if not
091     */
092    public synchronized boolean isPropagated() {
093        return propagatedFrom != null;
094    }
095
096    /*
097     * fields
098     */
099
100    /**
101     * The <code>BeanContext</code> from which this event was propagated
102     */
103    protected BeanContext propagatedFrom;
104}
