101    /**
102     * Instantiate a bean.
103     * <p>
104     * The bean is created based on a name relative to a class-loader.
105     * This name should be a dot-separated name such as "a.b.c".
106     * <p>
107     * In Beans 1.0 the given name can indicate either a serialized object
108     * or a class.  Other mechanisms may be added in the future.  In
109     * beans 1.0 we first try to treat the beanName as a serialized object
110     * name then as a class name.
111     * <p>
112     * When using the beanName as a serialized object name we convert the
113     * given beanName to a resource pathname and add a trailing ".ser" suffix.
114     * We then try to load a serialized object from that resource.
115     * <p>
116     * For example, given a beanName of "x.y", Beans.instantiate would first
117     * try to read a serialized object from the resource "x/y.ser" and if
118     * that failed it would try to load the class "x.y" and create an
119     * instance of that class.
120     * <p>
121     * If the bean is a subtype of java.applet.Applet, then it is given
122     * some special initialization.  First, it is supplied with a default
123     * AppletStub and AppletContext.  Second, if it was instantiated from
124     * a classname the applet's "init" method is called.  (If the bean was
125     * deserialized this step is skipped.)
126     * <p>
127     * Note that for beans which are applets, it is the caller's responsiblity
128     * to call "start" on the applet.  For correct behaviour, this should be done
129     * after the applet has been added into a visible AWT container.
130     * <p>
131     * Note that applets created via beans.instantiate run in a slightly
132     * different environment than applets running inside browsers.  In
133     * particular, bean applets have no access to "parameters", so they may
134     * wish to provide property get/set methods to set parameter values.  We
135     * advise bean-applet developers to test their bean-applets against both
136     * the JDK appletviewer (for a reference browser environment) and the
137     * BDK BeanBox (for a reference bean container).
138     *
139     * @param     cls         the class-loader from which we should create
140     *                        the bean.  If this is null, then the system
141     *                        class-loader is used.
142     * @param     beanName    the name of the bean within the class-loader.
143     *                        For example "sun.beanbox.foobah"
144     * @param     beanContext The BeanContext in which to nest the new bean
145     * @param     initializer The AppletInitializer for the new bean
146     *
147     * @exception ClassNotFoundException if the class of a serialized
148     *              object could not be found.
149     * @exception IOException if an I/O error occurs.
150     */
151
152    public static Object instantiate(ClassLoader cls, String beanName, BeanContext beanContext, AppletInitializer initializer)
153                        throws IOException, ClassNotFoundException {
154
155        InputStream ins;
156        ObjectInputStream oins = null;
157        Object result = null;
158        boolean serialized = false;
159        IOException serex = null;
160
161        // If the given classloader is null, we check if an
162        // system classloader is available and (if so)
163        // use that instead.
164        // Note that calls on the system class loader will
165        // look in the bootstrap class loader first.
166        if (cls == null) {
167            try {
168                cls = ClassLoader.getSystemClassLoader();
169            } catch (SecurityException ex) {
170                // We're not allowed to access the system class loader.
171                // Drop through.
172            }
173        }
174
175        // Try to find a serialized object with this name
176        final String serName = beanName.replace('.','/').concat(".ser");
177        if (cls == null)
178            ins =  ClassLoader.getSystemResourceAsStream(serName);
179        else
180            ins =  cls.getResourceAsStream(serName);
181        if (ins != null) {
182            try {
183                if (cls == null) {
184                    oins = new ObjectInputStream(ins);
185                } else {
186                    oins = new ObjectInputStreamWithLoader(ins, cls);
187                }
188                result = oins.readObject();
189                serialized = true;
190                oins.close();
191            } catch (IOException ex) {
192                ins.close();
193                // Drop through and try opening the class.  But remember
194                // the exception in case we can't find the class either.
195                serex = ex;
196            } catch (ClassNotFoundException ex) {
197                ins.close();
198                throw ex;
199            }
200        }
201
202        if (result == null) {
203            // No serialized object, try just instantiating the class
204            Class cl;
205
206            try {
207                cl = ClassFinder.findClass(beanName, cls);
208            } catch (ClassNotFoundException ex) {
209                // There is no appropriate class.  If we earlier tried to
210                // deserialize an object and got an IO exception, throw that,
211                // otherwise rethrow the ClassNotFoundException.
212                if (serex != null) {
213                    throw serex;
214                }
215                throw ex;
216            }
217
218            if (!Modifier.isPublic(cl.getModifiers())) {
219                throw new ClassNotFoundException("" + cl + " : no public access");
220            }
221
222            /*
223             * Try to instantiate the class.
224             */
225
226            try {
227                result = cl.newInstance();
228            } catch (Exception ex) {
229                // We have to remap the exception to one in our signature.
230                // But we pass extra information in the detail message.
231                throw new ClassNotFoundException("" + cl + " : " + ex, ex);
232            }
233        }
234
235        if (result != null) {
236
237            // Ok, if the result is an applet initialize it.
238
239            AppletStub stub = null;
240
241            if (result instanceof Applet) {
242                Applet  applet      = (Applet) result;
243                boolean needDummies = initializer == null;
244
245                if (needDummies) {
246
247                    // Figure our the codebase and docbase URLs.  We do this
248                    // by locating the URL for a known resource, and then
249                    // massaging the URL.
250
251                    // First find the "resource name" corresponding to the bean
252                    // itself.  So a serialzied bean "a.b.c" would imply a
253                    // resource name of "a/b/c.ser" and a classname of "x.y"
254                    // would imply a resource name of "x/y.class".
255
256                    final String resourceName;
257
258                    if (serialized) {
259                        // Serialized bean
260                        resourceName = beanName.replace('.','/').concat(".ser");
261                    } else {
262                        // Regular class
263                        resourceName = beanName.replace('.','/').concat(".class");
264                    }
265
266                    URL objectUrl = null;
267                    URL codeBase  = null;
268                    URL docBase   = null;
269
270                    // Now get the URL correponding to the resource name.
271                    if (cls == null) {
272                        objectUrl = ClassLoader.getSystemResource(resourceName);
273                    } else
274                        objectUrl = cls.getResource(resourceName);
275
276                    // If we found a URL, we try to locate the docbase by taking
277                    // of the final path name component, and the code base by taking
278                    // of the complete resourceName.
279                    // So if we had a resourceName of "a/b/c.class" and we got an
280                    // objectURL of "file://bert/classes/a/b/c.class" then we would
281                    // want to set the codebase to "file://bert/classes/" and the
282                    // docbase to "file://bert/classes/a/b/"
283
284                    if (objectUrl != null) {
285                        String s = objectUrl.toExternalForm();
286
287                        if (s.endsWith(resourceName)) {
288                            int ix   = s.length() - resourceName.length();
289                            codeBase = new URL(s.substring(0,ix));
290                            docBase  = codeBase;
291
292                            ix = s.lastIndexOf('/');
293
294                            if (ix >= 0) {
295                                docBase = new URL(s.substring(0,ix+1));
296                            }
297                        }
298                    }
299
300                    // Setup a default context and stub.
301                    BeansAppletContext context = new BeansAppletContext(applet);
302
303                    stub = (AppletStub)new BeansAppletStub(applet, context, codeBase, docBase);
304                    applet.setStub(stub);
305                } else {
306                    initializer.initialize(applet, beanContext);
307                }
308
309                // now, if there is a BeanContext, add the bean, if applicable.
310
311                if (beanContext != null) {
312                    beanContext.add(result);
313                }
314
315                // If it was deserialized then it was already init-ed.
316                // Otherwise we need to initialize it.
317
318                if (!serialized) {
319                    // We need to set a reasonable initial size, as many
320                    // applets are unhappy if they are started without
321                    // having been explicitly sized.
322                    applet.setSize(100,100);
323                    applet.init();
324                }
325
326                if (needDummies) {
327                  ((BeansAppletStub)stub).active = true;
328                } else initializer.activate(applet);
329
330            } else if (beanContext != null) beanContext.add(result);
331        }
332
333        return result;
334    }

339     * type view of that source object.
340     * <p>
341     * The result may be the same object or a different object.  If
342     * the requested target view isn't available then the given

356    /**
357     * Check if a bean can be viewed as a given target type.
358     * The result will be true if the Beans.getInstanceof method
359     * can be used on the given bean to obtain an object that
360     * represents the specified targetType type view.
361     *
362     * @param bean  Bean from which we want to obtain a view.
363     * @param targetType  The type of view we'd like to get.
364     * @return "true" if the given bean supports the given targetType.
365     *
366     */

368        return Introspector.isSubclass(bean.getClass(), targetType);

454/**
455 * This subclass of ObjectInputStream delegates loading of classes to
456 * an existing ClassLoader.
457 */
458
459class ObjectInputStreamWithLoader extends ObjectInputStream
460{
461    private ClassLoader loader;
462
463    /**
464     * Loader must be non-null;
465     */
466
467    public ObjectInputStreamWithLoader(InputStream in, ClassLoader loader)
468            throws IOException, StreamCorruptedException {
469
470        super(in);
471        if (loader == null) {
472            throw new IllegalArgumentException("Illegal null argument to ObjectInputStreamWithLoader");
473        }
474        this.loader = loader;
475    }
476
477    /**
478     * Use the given ClassLoader rather than using the system class
479     */
480    protected Class resolveClass(ObjectStreamClass classDesc)
481        throws IOException, ClassNotFoundException {
482
483        String cname = classDesc.getName();
484        return ClassFinder.resolveClass(cname, this.loader);
485    }
486}

488/**
489 * Package private support class.  This provides a default AppletContext
490 * for beans which are applets.
491 */
492
493class BeansAppletContext implements AppletContext {
494    Applet target;
495    Hashtable imageCache = new Hashtable();
496
497    BeansAppletContext(Applet target) {
498        this.target = target;
499    }
500
501    public AudioClip getAudioClip(URL url) {
502        // We don't currently support audio clips in the Beans.instantiate
503        // applet context, unless by some luck there exists a URL content
504        // class that can generate an AudioClip from the audio URL.
505        try {
506            return (AudioClip) url.getContent();
507        } catch (Exception ex) {
508            return null;
509        }
510    }
511
512    public synchronized Image getImage(URL url) {
513        Object o = imageCache.get(url);
514        if (o != null) {
515            return (Image)o;
516        }
517        try {
518            o = url.getContent();
519            if (o == null) {
520                return null;
521            }
522            if (o instanceof Image) {
523                imageCache.put(url, o);
524                return (Image) o;
525            }
526            // Otherwise it must be an ImageProducer.
527            Image img = target.createImage((java.awt.image.ImageProducer)o);
528            imageCache.put(url, img);
529            return img;
530
531        } catch (Exception ex) {
532            return null;
533        }
534    }
535
536    public Applet getApplet(String name) {
537        return null;
538    }
539
540    public Enumeration getApplets() {
541        Vector applets = new Vector();
542        applets.addElement(target);
543        return applets.elements();
544    }
545
546    public void showDocument(URL url) {
547        // We do nothing.
548    }
549
550    public void showDocument(URL url, String target) {
551        // We do nothing.
552    }
553
554    public void showStatus(String status) {
555        // We do nothing.
556    }
557
558    public void setStream(String key, InputStream stream)throws IOException{
559        // We do nothing.
560    }
561
562    public InputStream getStream(String key){
563        // We do nothing.
564        return null;
565    }
566
567    public Iterator getStreamKeys(){
568        // We do nothing.
569        return null;
570    }
571}
