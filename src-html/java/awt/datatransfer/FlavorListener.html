025package java.awt.datatransfer;
026
027import java.util.EventListener;
028
029
030/**
031 * Defines an object which listens for {@link FlavorEvent}s.
032 *
033 * @author Alexander Gerasimov
034 * @since 1.5
035 */
036public interface FlavorListener extends EventListener {
037    /**
038     * Invoked when the target {@link Clipboard} of the listener
039     * has changed its available {@link DataFlavor}s.
040     * <p>
041     * Some notifications may be redundant &#151; they are not
042     * caused by a change of the set of DataFlavors available
043     * on the clipboard.
044     * For example, if the clipboard subsystem supposes that
045     * the system clipboard's contents has been changed but it
046     * can't ascertain whether its DataFlavors have been changed
047     * because of some exceptional condition when accessing the
048     * clipboard, the notification is sent to ensure from omitting
049     * a significant notification. Ordinarily, those redundant
050     * notifications should be occasional.
051     *
052     * @param e  a <code>FlavorEvent</code> object
053     */
054    void flavorsChanged(FlavorEvent e);
055}
