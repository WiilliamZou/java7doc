026package java.awt.datatransfer;
027
028import java.util.List;
029
030
031/**
032 * A FlavorMap which relaxes the traditional 1-to-1 restriction of a Map. A
033 * flavor is permitted to map to any number of natives, and likewise a native
034 * is permitted to map to any number of flavors. FlavorTables need not be
035 * symmetric, but typically are.
036 *
037 * @author David Mendenhall
038 *
039 * @since 1.4
040 */
041public interface FlavorTable extends FlavorMap {
042
043    /**
044     * Returns a <code>List</code> of <code>String</code> natives to which the
045     * specified <code>DataFlavor</code> corresponds. The <code>List</code>
046     * will be sorted from best native to worst. That is, the first native will
047     * best reflect data in the specified flavor to the underlying native
048     * platform. The returned <code>List</code> is a modifiable copy of this
049     * <code>FlavorTable</code>'s internal data. Client code is free to modify
050     * the <code>List</code> without affecting this object.
051     *
052     * @param flav the <code>DataFlavor</code> whose corresponding natives
053     *        should be returned. If <code>null</code> is specified, all
054     *        natives currently known to this <code>FlavorTable</code> are
055     *        returned in a non-deterministic order.
056     * @return a <code>java.util.List</code> of <code>java.lang.String</code>
057     *         objects which are platform-specific representations of platform-
058     *         specific data formats
059     */
060    List<String> getNativesForFlavor(DataFlavor flav);
061
062    /**
063     * Returns a <code>List</code> of <code>DataFlavor</code>s to which the
064     * specified <code>String</code> corresponds. The <code>List</code> will be
065     * sorted from best <code>DataFlavor</code> to worst. That is, the first
066     * <code>DataFlavor</code> will best reflect data in the specified
067     * native to a Java application. The returned <code>List</code> is a
068     * modifiable copy of this <code>FlavorTable</code>'s internal data.
069     * Client code is free to modify the <code>List</code> without affecting
070     * this object.
071     *
072     * @param nat the native whose corresponding <code>DataFlavor</code>s
073     *        should be returned. If <code>null</code> is specified, all
074     *        <code>DataFlavor</code>s currently known to this
075     *        <code>FlavorTable</code> are returned in a non-deterministic
076     *        order.
077     * @return a <code>java.util.List</code> of <code>DataFlavor</code>
078     *         objects into which platform-specific data in the specified,
079     *         platform-specific native can be translated
080     */
081    List<DataFlavor> getFlavorsForNative(String nat);
082}
