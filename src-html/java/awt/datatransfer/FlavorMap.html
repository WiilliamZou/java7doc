026package java.awt.datatransfer;
027
028import java.util.Map;
029
030
031/**
032 * A two-way Map between "natives" (Strings), which correspond to platform-
033 * specfic data formats, and "flavors" (DataFlavors), which corerspond to
034 * platform-independent MIME types. FlavorMaps need not be symmetric, but
035 * typically are.
036 *
037 *
038 * @since 1.2
039 */
040public interface FlavorMap {
041
042    /**
043     * Returns a <code>Map</code> of the specified <code>DataFlavor</code>s to
044     * their corresponding <code>String</code> native. The returned
045     * <code>Map</code> is a modifiable copy of this <code>FlavorMap</code>'s
046     * internal data. Client code is free to modify the <code>Map</code>
047     * without affecting this object.
048     *
049     * @param flavors an array of <code>DataFlavor</code>s which will be the
050     *        key set of the returned <code>Map</code>. If <code>null</code> is
051     *        specified, a mapping of all <code>DataFlavor</code>s currently
052     *        known to this <code>FlavorMap</code> to their corresponding
053     *        <code>String</code> natives will be returned.
054     * @return a <code>java.util.Map</code> of <code>DataFlavor</code>s to
055     *         <code>String</code> natives
056     */
057    Map<DataFlavor,String> getNativesForFlavors(DataFlavor[] flavors);
058
059    /**
060     * Returns a <code>Map</code> of the specified <code>String</code> natives
061     * to their corresponding <code>DataFlavor</code>. The returned
062     * <code>Map</code> is a modifiable copy of this <code>FlavorMap</code>'s
063     * internal data. Client code is free to modify the <code>Map</code>
064     * without affecting this object.
065     *
066     * @param natives an array of <code>String</code>s which will be the
067     *        key set of the returned <code>Map</code>. If <code>null</code> is
068     *        specified, a mapping of all <code>String</code> natives currently
069     *        known to this <code>FlavorMap</code> to their corresponding
070     *        <code>DataFlavor</code>s will be returned.
071     * @return a <code>java.util.Map</code> of <code>String</code> natives to
072     *         <code>DataFlavor</code>s
073     */
074    Map<String,DataFlavor> getFlavorsForNatives(String[] natives);
075}
