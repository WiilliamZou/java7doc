026package java.awt;
027
028import java.io.*;
029
030/**
031 * The <code>GraphicsConfigTemplate</code> class is used to obtain a valid
032 * {@link GraphicsConfiguration}.  A user instantiates one of these
033 * objects and then sets all non-default attributes as desired.  The
034 * {@link GraphicsDevice#getBestConfiguration} method found in the
035 * {@link GraphicsDevice} class is then called with this
036 * <code>GraphicsConfigTemplate</code>.  A valid
037 * <code>GraphicsConfiguration</code> is returned that meets or exceeds
038 * what was requested in the <code>GraphicsConfigTemplate</code>.
039 * @see GraphicsDevice
040 * @see GraphicsConfiguration
041 *
042 * @since       1.2
043 */
044public abstract class GraphicsConfigTemplate implements Serializable {
045    /*
046     * serialVersionUID
047     */
048    private static final long serialVersionUID = -8061369279557787079L;
049
050    /**
051     * This class is an abstract class so only subclasses can be
052     * instantiated.
053     */
054    public GraphicsConfigTemplate() {
055    }
056
057    /**
058     * Value used for "Enum" (Integer) type.  States that this
059     * feature is required for the <code>GraphicsConfiguration</code>
060     * object.  If this feature is not available, do not select the
061     * <code>GraphicsConfiguration</code> object.
062     */
063    public static final int REQUIRED    = 1;
064
065    /**
066     * Value used for "Enum" (Integer) type.  States that this
067     * feature is desired for the <code>GraphicsConfiguration</code>
068     * object.  A selection with this feature is preferred over a
069     * selection that does not include this feature, although both
070     * selections can be considered valid matches.
071     */
072    public static final int PREFERRED   = 2;
073
074    /**
075     * Value used for "Enum" (Integer) type.  States that this
076     * feature is not necessary for the selection of the
077     * <code>GraphicsConfiguration</code> object.  A selection
078     * without this feature is preferred over a selection that
079     * includes this feature since it is not used.
080     */
081    public static final int UNNECESSARY = 3;
082
083    /**
084     * Returns the "best" configuration possible that passes the
085     * criteria defined in the <code>GraphicsConfigTemplate</code>.
086     * @param gc the array of <code>GraphicsConfiguration</code>
087     * objects to choose from.
088     * @return a <code>GraphicsConfiguration</code> object that is
089     * the best configuration possible.
090     * @see GraphicsConfiguration
091     */
092    public abstract GraphicsConfiguration
093      getBestConfiguration(GraphicsConfiguration[] gc);
094
095    /**
096     * Returns a <code>boolean</code> indicating whether or
097     * not the specified <code>GraphicsConfiguration</code> can be
098     * used to create a drawing surface that supports the indicated
099     * features.
100     * @param gc the <code>GraphicsConfiguration</code> object to test
101     * @return <code>true</code> if this
102     * <code>GraphicsConfiguration</code> object can be used to create
103     * surfaces that support the indicated features;
104     * <code>false</code> if the <code>GraphicsConfiguration</code> can
105     * not be used to create a drawing surface usable by this Java(tm)
106     * API.
107     */
108    public abstract boolean
109      isGraphicsConfigSupported(GraphicsConfiguration gc);
110
111}
