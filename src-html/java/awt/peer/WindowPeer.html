026package java.awt.peer;
027
028import java.awt.*;
029
030import java.awt.image.BufferedImage;
031
032/**
033 * The peer interface for {@link Window}.
034 *
035 * The peer interfaces are intended only for use in porting
036 * the AWT. They are not intended for use by application
037 * developers, and developers should not implement peers
038 * nor invoke any of the peer methods directly on the peer
039 * instances.
040 */
041public interface WindowPeer extends ContainerPeer {
042
043    /**
044     * Makes this window the topmost window on the desktop.
045     *
046     * @see Window#toFront()
047     */
048    void toFront();
049
050    /**
051     * Makes this window the bottommost window on the desktop.
052     *
053     * @see Window#toBack()
054     */
055    void toBack();
056
057    /**
058     * Updates the window's always-on-top state.
059     * Sets if the window should always stay
060     * on top of all other windows or not.
061     *
062     * @see Window#getAlwaysOnTop()
063     * @see Window#setAlwaysOnTop(boolean)
064     */
065    void updateAlwaysOnTopState();
066
067    /**
068     * Updates the window's focusable state.
069     *
070     * @see Window#setFocusableWindowState(boolean)
071     */
072    void updateFocusableWindowState();
073
074    /**
075     * Sets if this window is blocked by a modal dialog or not.
076     *
077     * @param blocker the blocking modal dialog
078     * @param blocked {@code true} to block the window, {@code false}
079     *        to unblock it
080     */
081    void setModalBlocked(Dialog blocker, boolean blocked);
082
083    /**
084     * Updates the minimum size on the peer.
085     *
086     * @see Window#setMinimumSize(Dimension)
087     */
088    void updateMinimumSize();
089
090    /**
091     * Updates the icons for the window.
092     *
093     * @see Window#setIconImages(java.util.List)
094     */
095    void updateIconImages();
096
097    /**
098     * Sets the level of opacity for the window.
099     *
100     * @see Window#setOpacity(float)
101     */
102    void setOpacity(float opacity);
103
104    /**
105     * Enables the per-pixel alpha support for the window.
106     *
107     * @see Window#setBackground(Color)
108     */
109    void setOpaque(boolean isOpaque);
110
111    /**
112     * Updates the native part of non-opaque window.
113     *
114     * @see Window#setBackground(Color)
115     */
116    void updateWindow();
117
118    /**
119     * Instructs the peer to update the position of the security warning.
120     */
121    void repositionSecurityWarning();
122}
