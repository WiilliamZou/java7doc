025package java.awt.peer;
026
027import java.awt.Choice;
028
029/**
030 * The peer interface for {@link Choice}.
031 *
032 * The peer interfaces are intended only for use in porting
033 * the AWT. They are not intended for use by application
034 * developers, and developers should not implement peers
035 * nor invoke any of the peer methods directly on the peer
036 * instances.
037 */
038public interface ChoicePeer extends ComponentPeer {
039
040    /**
041     * Adds an item with the string {@code item} to the combo box list
042     * at index {@code index}.
043     *
044     * @param item the label to be added to the list
045     * @param index the index where to add the item
046     *
047     * @see Choice#add(String)
048     */
049    void add(String item, int index);
050
051    /**
052     * Removes the item at index {@code index} from the combo box list.
053     *
054     * @param index the index where to remove the item
055     *
056     * @see Choice#remove(int)
057     */
058    void remove(int index);
059
060    /**
061     * Removes all items from the combo box list.
062     *
063     * @see Choice#removeAll()
064     */
065    void removeAll();
066
067    /**
068     * Selects the item at index {@code index}.
069     *
070     * @param index the index which should be selected
071     *
072     * @see Choice#select(int)
073     */
074    void select(int index);
075
076}
