026package java.awt.peer;
027
028import java.awt.Window;
029import java.awt.Point;
030
031/**
032 * Peer interface for {@link MouseInfo}. This is used to get some additional
033 * information about the mouse.
034 *
035 * The peer interfaces are intended only for use in porting
036 * the AWT. They are not intended for use by application
037 * developers, and developers should not implement peers
038 * nor invoke any of the peer methods directly on the peer
039 * instances.
040 */
041public interface MouseInfoPeer {
042
043    /**
044     * This method does two things: it fills the point fields with
045     * the current coordinates of the mouse cursor and returns the
046     * number of the screen device where the pointer is located.
047     * The number of the screen device is only returned for independent
048     * devices (which are not parts of a virtual screen device).
049     * For virtual screen devices, 0 is returned.
050     * Mouse coordinates are also calculated depending on whether
051     * or not the screen device is virtual. For virtual screen
052     * devices, pointer coordinates are calculated in the virtual
053     * coordinate system. Otherwise, coordinates are calculated in
054     * the coordinate system of the screen device where the pointer
055     * is located.
056     * See java.awt.GraphicsConfiguration documentation for more
057     * details about virtual screen devices.
058     */
059    int fillPointWithCoords(Point point);
060
061    /**
062     * Returns whether or not the window is located under the mouse
063     * pointer. The window is considered to be under the mouse pointer
064     * if it is showing on the screen, and the mouse pointer is above
065     * the part of the window that is not obscured by any other windows.
066     */
067    boolean isWindowUnderMouse(Window w);
068
069}
