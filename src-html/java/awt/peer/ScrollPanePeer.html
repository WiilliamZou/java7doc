025package java.awt.peer;
026
027import java.awt.Adjustable;
028import java.awt.ScrollPane;
029import java.awt.ScrollPaneAdjustable;
030
031/**
032 * The peer interface for {@link ScrollPane}.
033 *
034 * The peer interfaces are intended only for use in porting
035 * the AWT. They are not intended for use by application
036 * developers, and developers should not implement peers
037 * nor invoke any of the peer methods directly on the peer
038 * instances.
039 */
040public interface ScrollPanePeer extends ContainerPeer {
041
042    /**
043     * Returns the height of the horizontal scroll bar.
044     *
045     * @return the height of the horizontal scroll bar
046     *
047     * @see ScrollPane#getHScrollbarHeight()
048     */
049    int getHScrollbarHeight();
050
051    /**
052     * Returns the width of the vertical scroll bar.
053     *
054     * @return the width of the vertical scroll bar
055     *
056     * @see ScrollPane#getVScrollbarWidth()
057     */
058    int getVScrollbarWidth();
059
060    /**
061     * Sets the scroll position of the child.
062     *
063     * @param x the X coordinate of the scroll position
064     * @param y the Y coordinate of the scroll position
065     *
066     * @see ScrollPane#setScrollPosition(int, int)
067     */
068    void setScrollPosition(int x, int y);
069
070    /**
071     * Called when the child component changes its size.
072     *
073     * @param w the new width of the child component
074     * @param h the new height of the child component
075     *
076     * @see ScrollPane#layout()
077     */
078    void childResized(int w, int h);
079
080    /**
081     * Sets the unit increment of one of the scroll pane's adjustables.
082     *
083     * @param adj the scroll pane adjustable object
084     * @param u the unit increment
085     *
086     * @see ScrollPaneAdjustable#setUnitIncrement(int)
087     */
088    void setUnitIncrement(Adjustable adj, int u);
089
090    /**
091     * Sets the value for one of the scroll pane's adjustables.
092     *
093     * @param adj the scroll pane adjustable object
094     * @param v the value to set
095     */
096    void setValue(Adjustable adj, int v);
097}
