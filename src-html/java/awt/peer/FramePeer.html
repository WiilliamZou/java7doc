026package java.awt.peer;
027
028import java.awt.*;
029
030import sun.awt.EmbeddedFrame;
031
032/**
033 * The peer interface for {@link Frame}. This adds a couple of frame specific
034 * methods to the {@link WindowPeer} interface.
035 *
036 * The peer interfaces are intended only for use in porting
037 * the AWT. They are not intended for use by application
038 * developers, and developers should not implement peers
039 * nor invoke any of the peer methods directly on the peer
040 * instances.
041 */
042public interface FramePeer extends WindowPeer {
043
044    /**
045     * Sets the title on the frame.
046     *
047     * @param title the title to set
048     *
049     * @see Frame#setTitle(String)
050     */
051    void setTitle(String title);
052
053    /**
054     * Sets the menu bar for the frame.
055     *
056     * @param mb the menu bar to set
057     *
058     * @see Frame#setMenuBar(MenuBar)
059     */
060    void setMenuBar(MenuBar mb);
061
062    /**
063     * Sets if the frame should be resizable or not.
064     *
065     * @param resizeable {@code true} when the frame should be resizable,
066     *        {@code false} if not
067     *
068     * @see Frame#setResizable(boolean)
069     */
070    void setResizable(boolean resizeable);
071
072    /**
073     * Changes the state of the frame.
074     *
075     * @param state the new state
076     *
077     * @see Frame#setExtendedState(int)
078     */
079    void setState(int state);
080
081    /**
082     * Returns the current state of the frame.
083     *
084     * @return the current state of the frame
085     *
086     * @see Frame#getExtendedState()
087     */
088    int getState();
089
090    /**
091     * Sets the bounds of the frame when it becomes maximized.
092     *
093     * @param bounds the maximized bounds of the frame
094     *
095     * @see Frame#setMaximizedBounds(Rectangle)
096     */
097    void setMaximizedBounds(Rectangle bounds);
098
099    /**
100     * Sets the size and location for embedded frames. (On embedded frames,
101     * setLocation() and setBounds() always set the frame to (0,0) for
102     * backwards compatibility.
103     *
104     * @param x the X location
105     * @param y the Y location
106     * @param width the width of the frame
107     * @param height the height of the frame
108     *
109     * @see EmbeddedFrame#setBoundsPrivate(int, int, int, int)
110     */
111    // TODO: This is only used in EmbeddedFrame, and should probably be moved
112    // into an EmbeddedFramePeer which would extend FramePeer
113    void setBoundsPrivate(int x, int y, int width, int height);
114
115    /**
116     * Returns the size and location for embedded frames. (On embedded frames,
117     * setLocation() and setBounds() always set the frame to (0,0) for
118     * backwards compatibility.
119     *
120     * @return the bounds of an embedded frame
121     *
122     * @see EmbeddedFrame#getBoundsPrivate()
123     */
124    // TODO: This is only used in EmbeddedFrame, and should probably be moved
125    // into an EmbeddedFramePeer which would extend FramePeer
126    Rectangle getBoundsPrivate();
127
128}
