025package java.awt.peer;
026
027import java.awt.TextComponent;
028import java.awt.im.InputMethodRequests;
029
030/**
031 * The peer interface for {@link TextComponent}.
032 *
033 * The peer interfaces are intended only for use in porting
034 * the AWT. They are not intended for use by application
035 * developers, and developers should not implement peers
036 * nor invoke any of the peer methods directly on the peer
037 * instances.
038 */
039public interface TextComponentPeer extends ComponentPeer {
040
041    /**
042     * Sets if the text component should be editable or not.
043     *
044     * @param editable {@code true} for editable text components,
045     *        {@code false} for non-editable text components
046     *
047     * @see TextComponent#setEditable(boolean)
048     */
049    void setEditable(boolean editable);
050
051    /**
052     * Returns the current content of the text component.
053     *
054     * @return the current content of the text component
055     *
056     * @see TextComponent#getText()
057     */
058    String getText();
059
060    /**
061     * Sets the content for the text component.
062     *
063     * @param l the content to set
064     *
065     * @see TextComponent#setText(String)
066     */
067    void setText(String l);
068
069    /**
070     * Returns the start index of the current selection.
071     *
072     * @return the start index of the current selection
073     *
074     * @see TextComponent#getSelectionStart()
075     */
076    int getSelectionStart();
077
078    /**
079     * Returns the end index of the current selection.
080     *
081     * @return the end index of the current selection
082     *
083     * @see TextComponent#getSelectionEnd()
084     */
085    int getSelectionEnd();
086
087    /**
088     * Selects an area of the text component.
089     *
090     * @param selStart the start index of the new selection
091     * @param selEnd the end index of the new selection
092     *
093     * @see TextComponent#select(int, int)
094     */
095    void select(int selStart, int selEnd);
096
097    /**
098     * Sets the caret position of the text component.
099     *
100     * @param pos the caret position to set
101     *
102     * @see TextComponent#setCaretPosition(int)
103     */
104    void setCaretPosition(int pos);
105
106    /**
107     * Returns the current caret position.
108     *
109     * @return the current caret position
110     *
111     * @see TextComponent#getCaretPosition()
112     */
113    int getCaretPosition();
114
115    /**
116     * Returns the input method requests.
117     *
118     * @return the input method requests
119     */
120    InputMethodRequests getInputMethodRequests();
121}
