025package java.awt;
026
027import javax.accessibility.*;
028
029/**
030 * <code>Panel</code> is the simplest container class. A panel
031 * provides space in which an application can attach any other
032 * component, including other panels.
033 * <p>
034 * The default layout manager for a panel is the
035 * <code>FlowLayout</code> layout manager.
036 *
037 * @author      Sami Shaio
038 * @see     java.awt.FlowLayout
039 * @since   JDK1.0
040 */
041public class Panel extends Container implements Accessible {
042    private static final String base = "panel";
043    private static int nameCounter = 0;
044
045    /*
046     * JDK 1.1 serialVersionUID
047     */
048     private static final long serialVersionUID = -2728009084054400034L;
049
050    /**
051     * Creates a new panel using the default layout manager.
052     * The default layout manager for all panels is the
053     * <code>FlowLayout</code> class.
054     */
055    public Panel() {
056        this(new FlowLayout());
057    }
058
059    /**
060     * Creates a new panel with the specified layout manager.
061     * @param layout the layout manager for this panel.
062     * @since JDK1.1
063     */
064    public Panel(LayoutManager layout) {
065        setLayout(layout);
066    }
067
068    /**
069     * Construct a name for this component.  Called by getName() when the
070     * name is null.
071     */
072    String constructComponentName() {
073        synchronized (Panel.class) {
074            return base + nameCounter++;
075        }
076    }
077
078    /**
079     * Creates the Panel's peer.  The peer allows you to modify the
080     * appearance of the panel without changing its functionality.
081     */
082
083    public void addNotify() {
084        synchronized (getTreeLock()) {
085            if (peer == null)
086                peer = getToolkit().createPanel(this);
087            super.addNotify();
088        }
089    }
090
091/////////////////
092// Accessibility support
093////////////////
094
095    /**
096     * Gets the AccessibleContext associated with this Panel.
097     * For panels, the AccessibleContext takes the form of an
098     * AccessibleAWTPanel.
099     * A new AccessibleAWTPanel instance is created if necessary.
100     *
101     * @return an AccessibleAWTPanel that serves as the
102     *         AccessibleContext of this Panel
103     * @since 1.3
104     */
105    public AccessibleContext getAccessibleContext() {
106        if (accessibleContext == null) {
107            accessibleContext = new AccessibleAWTPanel();
108        }
109        return accessibleContext;
110    }
111
112    /**
113     * This class implements accessibility support for the
114     * <code>Panel</code> class.  It provides an implementation of the
115     * Java Accessibility API appropriate to panel user-interface elements.
116     * @since 1.3
117     */
118    protected class AccessibleAWTPanel extends AccessibleAWTContainer {
119
120        private static final long serialVersionUID = -6409552226660031050L;
121
122        /**
123         * Get the role of this object.
124         *
125         * @return an instance of AccessibleRole describing the role of the
126         * object
127         */
128        public AccessibleRole getAccessibleRole() {
129            return AccessibleRole.PANEL;
130        }
131    }
132
133}
