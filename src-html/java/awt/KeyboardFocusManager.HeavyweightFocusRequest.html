064/**
065 * The KeyboardFocusManager is responsible for managing the active and focused
066 * Windows, and the current focus owner. The focus owner is defined as the
067 * Component in an application that will typically receive all KeyEvents
068 * generated by the user. The focused Window is the Window that is, or
069 * contains, the focus owner. Only a Frame or a Dialog can be the active
070 * Window. The native windowing system may denote the active Window or its
071 * children with special decorations, such as a highlighted title bar. The
072 * active Window is always either the focused Window, or the first Frame or
073 * Dialog that is an owner of the focused Window.
074 * <p>
075 * The KeyboardFocusManager is both a centralized location for client code to
076 * query for the focus owner and initiate focus changes, and an event
077 * dispatcher for all FocusEvents, WindowEvents related to focus, and
078 * KeyEvents.
079 * <p>
080 * Some browsers partition applets in different code bases into separate
081 * contexts, and establish walls between these contexts. In such a scenario,
082 * there will be one KeyboardFocusManager per context. Other browsers place all
083 * applets into the same context, implying that there will be only a single,
084 * global KeyboardFocusManager for all applets. This behavior is
085 * implementation-dependent. Consult your browser's documentation for more
086 * information. No matter how many contexts there may be, however, there can
087 * never be more than one focus owner, focused Window, or active Window, per
088 * ClassLoader.
089 * <p>
090 * Please see
091 * <a href="http://java.sun.com/docs/books/tutorial/uiswing/misc/focus.html">
092 * How to Use the Focus Subsystem</a>,
093 * a section in <em>The Java Tutorial</em>, and the
094 * <a href="../../java/awt/doc-files/FocusSpec.html">Focus Specification</a>
095 * for more information.
096 *
097 * @author David Mendenhall
098 *
099 * @see Window
100 * @see Frame
101 * @see Dialog
102 * @see java.awt.event.FocusEvent
103 * @see java.awt.event.WindowEvent
104 * @see java.awt.event.KeyEvent
105 * @since 1.4
106 */
107public abstract class KeyboardFocusManager
108    implements KeyEventDispatcher, KeyEventPostProcessor

231    /**
232     * Sets the current KeyboardFocusManager instance for the calling thread's
233     * context. If null is specified, then the current KeyboardFocusManager
234     * is replaced with a new instance of DefaultKeyboardFocusManager.
235     * <p>
236     * If a SecurityManager is installed, the calling thread must be granted
237     * the AWTPermission "replaceKeyboardFocusManager" in order to replace the
238     * the current KeyboardFocusManager. If this permission is not granted,
239     * this method will throw a SecurityException, and the current
240     * KeyboardFocusManager will be unchanged.
241     *
242     * @param newManager the new KeyboardFocusManager for this thread's context
243     * @see #getCurrentKeyboardFocusManager
244     * @see DefaultKeyboardFocusManager
245     * @throws SecurityException if the calling thread does not have permission
246     *         to replace the current KeyboardFocusManager
247     */
248    public static void setCurrentKeyboardFocusManager(
249        KeyboardFocusManager newManager) throws SecurityException
250    {
251        checkReplaceKFMPermission();
252
253        KeyboardFocusManager oldManager = null;
254
255        synchronized (KeyboardFocusManager.class) {
256            AppContext appcontext = AppContext.getAppContext();
257
258            if (newManager != null) {
259                oldManager = getCurrentKeyboardFocusManager(appcontext);
260
261                appcontext.put(KeyboardFocusManager.class, newManager);
262            } else {
263                oldManager = getCurrentKeyboardFocusManager(appcontext);
264                appcontext.remove(KeyboardFocusManager.class);
265            }
266        }
267
268        if (oldManager != null) {
269            oldManager.firePropertyChange("managingFocus",
270                                          Boolean.TRUE,
271                                          Boolean.FALSE);
272        }
273        if (newManager != null) {
274            newManager.firePropertyChange("managingFocus",
275                                          Boolean.FALSE,
276                                          Boolean.TRUE);
277        }
278    }

425        StringTokenizer tokens = new StringTokenizer(value, ",");
426        while (tokens.hasMoreTokens()) {
427            targetSet.add(AWTKeyStroke.getAWTKeyStroke(tokens.nextToken()));

435     * Initializes a KeyboardFocusManager.
436     */
437    public KeyboardFocusManager() {
438        for (int i = 0; i < TRAVERSAL_KEY_LENGTH; i++) {
439            Set work_set = new HashSet();
440            for (int j = 0; j < defaultFocusTraversalKeyStrokes[i].length; j++) {
441                work_set.add(defaultFocusTraversalKeyStrokes[i][j]);
442            }
443            defaultFocusTraversalKeys[i] = (work_set.isEmpty())
444                ? Collections.EMPTY_SET
445                : Collections.unmodifiableSet(work_set);
446        }
447        initPeer();
448    }

508    /**
509     * Sets the focus owner. The operation will be cancelled if the Component
510     * is not focusable. The focus owner is defined as the Component in an
511     * application that will typically receive all KeyEvents generated by the
512     * user. KeyEvents which map to the focus owner's focus traversal keys will
513     * not be delivered if focus traversal keys are enabled for the focus
514     * owner. In addition, KeyEventDispatchers may retarget or consume
515     * KeyEvents before they reach the focus owner.
516     * <p>
517     * This method does not actually set the focus to the specified Component.
518     * It merely stores the value to be subsequently returned by
519     * <code>getFocusOwner()</code>. Use <code>Component.requestFocus()</code>
520     * or <code>Component.requestFocusInWindow()</code> to change the focus
521     * owner, subject to platform limitations.
522     *
523     * @param focusOwner the focus owner
524     * @see #getFocusOwner
525     * @see #getGlobalFocusOwner
526     * @see Component#requestFocus()
527     * @see Component#requestFocusInWindow()
528     * @see Component#isFocusable
529     * @beaninfo
530     *       bound: true
531     */
532    protected void setGlobalFocusOwner(Component focusOwner) {
533        Component oldFocusOwner = null;
534        boolean shouldFire = false;
535
536        if (focusOwner == null || focusOwner.isFocusable()) {
537            synchronized (KeyboardFocusManager.class) {
538                checkKFMSecurity();
539
540                oldFocusOwner = getFocusOwner();
541
542                try {
543                    fireVetoableChange("focusOwner", oldFocusOwner,
544                                       focusOwner);
545                } catch (PropertyVetoException e) {
546                    // rejected
547                    return;
548                }
549
550                KeyboardFocusManager.focusOwner = focusOwner;
551
552                if (focusOwner != null &&
553                    (getCurrentFocusCycleRoot() == null ||
554                     !focusOwner.isFocusCycleRoot(getCurrentFocusCycleRoot())))
555                {
556                    Container rootAncestor =
557                        focusOwner.getFocusCycleRootAncestor();
558                    if (rootAncestor == null && (focusOwner instanceof Window))
559                    {
560                        rootAncestor = (Container)focusOwner;
561                    }
562                    if (rootAncestor != null) {
563                        setGlobalCurrentFocusCycleRoot(rootAncestor);
564                    }
565                }
566
567                shouldFire = true;
568            }
569        }
570
571        if (shouldFire) {
572            firePropertyChange("focusOwner", oldFocusOwner, focusOwner);
573        }
574    }

973     * Sets the default focus traversal keys for a given traversal operation.
974     * This traversal key <code>Set</code> will be in effect on all
975     * <code>Window</code>s that have no such <code>Set</code> of
976     * their own explicitly defined. This <code>Set</code> will also be
977     * inherited, recursively, by any child <code>Component</code> of

982     * implementation-dependent. Sun recommends that all implementations for a
983     * particular native platform use the same default values. The
984     * recommendations for Windows and Unix are listed below. These
985     * recommendations are used in the Sun AWT implementations.

988     * <tr>
989     *    <th>Identifier</th>

992     * </tr>
993     * <tr>
994     *    <td><code>KeyboardFocusManager.FORWARD_TRAVERSAL_KEYS</code></td>
995     *    <td>Normal forward keyboard traversal</td>
996     *    <td><code>TAB</code> on <code>KEY_PRESSED</code>,
997     *        <code>CTRL-TAB</code> on <code>KEY_PRESSED</code></td>
998     * </tr>
999     * <tr>
1000     *    <td><code>KeyboardFocusManager.BACKWARD_TRAVERSAL_KEYS</code></td>
1001     *    <td>Normal reverse keyboard traversal</td>
1002     *    <td><code>SHIFT-TAB</code> on <code>KEY_PRESSED</code>,
1003     *        <code>CTRL-SHIFT-TAB</code> on <code>KEY_PRESSED</code></td>
1004     * </tr>
1005     * <tr>
1006     *    <td><code>KeyboardFocusManager.UP_CYCLE_TRAVERSAL_KEYS</code></td>
1007     *    <td>Go up one focus traversal cycle</td>
1008     *    <td>none</td>
1009     * </tr>
1010     * <tr>
1011     *    <td><code>KeyboardFocusManager.DOWN_CYCLE_TRAVERSAL_KEYS</code></td>
1012     *    <td>Go down one focus traversal cycle</td>
1013     *    <td>none</td>
1014     * </tr>
1015     * </table>
1016     *
1017     * To disable a traversal key, use an empty <code>Set</code>;
1018     * <code>Collections.EMPTY_SET</code> is recommended.
1019     * <p>
1020     * Using the <code>AWTKeyStroke</code> API, client code can
1021     * specify on which of two
1022     * specific <code>KeyEvent</code>s, <code>KEY_PRESSED</code> or
1023     * <code>KEY_RELEASED</code>, the focus traversal operation will
1024     * occur. Regardless of which <code>KeyEvent</code> is specified,
1025     * however, all <code>KeyEvent</code>s related to the focus
1026     * traversal key, including the associated <code>KEY_TYPED</code>
1027     * event, will be consumed, and will not be dispatched
1028     * to any <code>Component</code>. It is a runtime error to
1029     * specify a <code>KEY_TYPED</code> event as
1030     * mapping to a focus traversal operation, or to map the same event to
1031     * multiple default focus traversal operations.
1032     *
1033     * @param id one of
1034     *        <code>KeyboardFocusManager.FORWARD_TRAVERSAL_KEYS</code>,
1035     *        <code>KeyboardFocusManager.BACKWARD_TRAVERSAL_KEYS</code>,
1036     *        <code>KeyboardFocusManager.UP_CYCLE_TRAVERSAL_KEYS</code>, or
1037     *        <code>KeyboardFocusManager.DOWN_CYCLE_TRAVERSAL_KEYS</code>
1038     * @param keystrokes the Set of <code>AWTKeyStroke</code>s for the
1039     *        specified operation
1040     * @see #getDefaultFocusTraversalKeys
1041     * @see Component#setFocusTraversalKeys
1042     * @see Component#getFocusTraversalKeys
1043     * @throws IllegalArgumentException if id is not one of
1044     *         <code>KeyboardFocusManager.FORWARD_TRAVERSAL_KEYS</code>,
1045     *         <code>KeyboardFocusManager.BACKWARD_TRAVERSAL_KEYS</code>,
1046     *         <code>KeyboardFocusManager.UP_CYCLE_TRAVERSAL_KEYS</code>, or
1047     *         <code>KeyboardFocusManager.DOWN_CYCLE_TRAVERSAL_KEYS</code>,
1048     *         or if keystrokes is <code>null</code>,
1049     *         or if keystrokes contains <code>null</code>,
1050     *         or if any <code>Object</code> in
1051     *         keystrokes is not an <code>AWTKeyStroke</code>,
1052     *         or if any keystroke
1053     *         represents a <code>KEY_TYPED</code> event,
1054     *         or if any keystroke already maps
1055     *         to another default focus traversal operation
1056     * @beaninfo
1057     *       bound: true
1058     */
1059    public void
1060        setDefaultFocusTraversalKeys(int id,
1061                                     Set<? extends AWTKeyStroke> keystrokes)
1062    {
1063        if (id < 0 || id >= TRAVERSAL_KEY_LENGTH) {
1064            throw new IllegalArgumentException("invalid focus traversal key identifier");
1065        }
1066        if (keystrokes == null) {
1067            throw new IllegalArgumentException("cannot set null Set of default focus traversal keys");
1068        }
1069
1070        Set oldKeys;
1071
1072        synchronized (this) {
1073            for (Iterator iter = keystrokes.iterator(); iter.hasNext(); ) {
1074                Object obj = iter.next();
1075
1076                if (obj == null) {
1077                    throw new IllegalArgumentException("cannot set null focus traversal key");
1078                }
1079
1080                // Fix for 6195831:
1081                //According to javadoc this method should throw IAE instead of ClassCastException
1082                if (!(obj instanceof AWTKeyStroke)) {
1083                    throw new IllegalArgumentException("object is expected to be AWTKeyStroke");
1084                }
1085                AWTKeyStroke keystroke = (AWTKeyStroke)obj;
1086
1087                if (keystroke.getKeyChar() != KeyEvent.CHAR_UNDEFINED) {
1088                    throw new IllegalArgumentException("focus traversal keys cannot map to KEY_TYPED events");
1089                }
1090
1091                // Check to see if key already maps to another traversal
1092                // operation
1093                for (int i = 0; i < TRAVERSAL_KEY_LENGTH; i++) {
1094                    if (i == id) {
1095                        continue;
1096                    }
1097
1098                    if (defaultFocusTraversalKeys[i].contains(keystroke)) {
1099                        throw new IllegalArgumentException("focus traversal keys must be unique for a Component");
1100                    }
1101                }
1102            }
1103
1104            oldKeys = defaultFocusTraversalKeys[id];
1105            defaultFocusTraversalKeys[id] =
1106                Collections.unmodifiableSet(new HashSet(keystrokes));
1107        }
1108
1109        firePropertyChange(defaultFocusTraversalKeyPropertyNames[id],
1110                           oldKeys, keystrokes);
1111    }

1788        Component window = component;
1789        while (window != null && !(window instanceof Window)) {
1790            window = window.parent;
1791        }
1792        if (window != null) {

1796    static synchronized void setMostRecentFocusOwner(Window window,
1797                                                     Component component) {
1798        // ATTN: component has a strong reference to window via chain
1799        // of Component.parent fields.  Since WeakHasMap refers to its
1800        // values strongly, we need to break the strong link from the
1801        // value (component) back to its key (window).
1802        WeakReference weakValue = null;
1803        if (component != null) {
1804            weakValue = new WeakReference(component);
1805        }
1806        mostRecentFocusOwners.put(window, weakValue);
1807    }
1808    static void clearMostRecentFocusOwner(Component comp) {
1809        Container window;
1810
1811        if (comp == null) {
1812            return;
1813        }
1814
1815        synchronized (comp.getTreeLock()) {
1816            window = comp.getParent();
1817            while (window != null && !(window instanceof Window)) {
1818                window = window.getParent();
1819            }
1820        }
1821
1822        synchronized (KeyboardFocusManager.class) {
1823            if ((window != null)
1824                && (getMostRecentFocusOwner((Window)window) == comp))
1825            {
1826                setMostRecentFocusOwner((Window)window, null);
1827            }
1828            // Also clear temporary lost component stored in Window
1829            if (window != null) {
1830                Window realWindow = (Window)window;
1831                if (realWindow.getTemporaryLostComponent() == comp) {
1832                    realWindow.setTemporaryLostComponent(null);
1833                }
1834            }
1835        }
1836    }

2558            // The current app context differs from the app context of a focus
2559            // owner (and all pending lightweight requests), so we do nothing
2560            // now and wait for a next event.
2561            return;
2562        }
2563
2564        synchronized(heavyweightRequests) {
2565            if (currentLightweightRequests != null) {
2566                clearingCurrentLightweightRequests = true;
2567                disableRestoreFocus = true;
2568                localLightweightRequests = currentLightweightRequests;
2569                allowSyncFocusRequests = (localLightweightRequests.size() < 2);
2570                currentLightweightRequests = null;
2571            } else {
2572                // do nothing
2573                return;
2574            }
2575        }
2576
2577        Throwable caughtEx = null;
2578        try {
2579            if (localLightweightRequests != null) {
2580                Component lastFocusOwner = null;
2581                Component currentFocusOwner = null;
2582
2583                for (Iterator iter = localLightweightRequests.iterator(); iter.hasNext(); ) {
2584
2585                    currentFocusOwner = manager.getGlobalFocusOwner();
2586                    LightweightFocusRequest lwFocusRequest =
2587                        (LightweightFocusRequest)iter.next();
2588
2589                    /*
2590                     * WARNING: This is based on DKFM's logic solely!
2591                     *
2592                     * We allow to trigger restoreFocus() in the dispatching process
2593                     * only if we have the last request to dispatch. If the last request
2594                     * fails, focus will be restored to either the component of the last
2595                     * previously succedded request, or to to the focus owner that was
2596                     * before this clearing proccess.
2597                     */
2598                    if (!iter.hasNext()) {
2599                        disableRestoreFocus = false;
2600                    }
2601
2602                    FocusEvent currentFocusOwnerEvent = null;
2603                    /*
2604                     * We're not dispatching FOCUS_LOST while the current focus owner is null.
2605                     * But regardless of whether it's null or not, we're clearing ALL the local
2606                     * lw requests.
2607                     */
2608                    if (currentFocusOwner != null) {
2609                        currentFocusOwnerEvent = new CausedFocusEvent(currentFocusOwner,
2610                                       FocusEvent.FOCUS_LOST,
2611                                       lwFocusRequest.temporary,
2612                                       lwFocusRequest.component, lwFocusRequest.cause);
2613                    }
2614                    FocusEvent newFocusOwnerEvent =
2615                        new CausedFocusEvent(lwFocusRequest.component,
2616                                       FocusEvent.FOCUS_GAINED,
2617                                       lwFocusRequest.temporary,
2618                                       currentFocusOwner == null ? lastFocusOwner : currentFocusOwner,
2619                                       lwFocusRequest.cause);
2620
2621                    if (currentFocusOwner != null) {
2622                        ((AWTEvent) currentFocusOwnerEvent).isPosted = true;
2623                        caughtEx = dispatchAndCatchException(caughtEx, currentFocusOwner, currentFocusOwnerEvent);
2624                    }
2625
2626                    ((AWTEvent) newFocusOwnerEvent).isPosted = true;
2627                    caughtEx = dispatchAndCatchException(caughtEx, lwFocusRequest.component, newFocusOwnerEvent);
2628
2629                    if (manager.getGlobalFocusOwner() == lwFocusRequest.component) {
2630                        lastFocusOwner = lwFocusRequest.component;
2631                    }
2632                }
2633            }
2634        } finally {
2635            clearingCurrentLightweightRequests = false;
2636            disableRestoreFocus = false;
2637            localLightweightRequests = null;
2638            allowSyncFocusRequests = true;
2639        }
2640        if (caughtEx instanceof RuntimeException) {
2641            throw (RuntimeException)caughtEx;
2642        } else if (caughtEx instanceof Error) {
2643            throw (Error)caughtEx;
2644        }
2645    }

2670    static FocusEvent retargetFocusGained(FocusEvent fe) {
2671        assert (fe.getID() == FocusEvent.FOCUS_GAINED);
2672
2673        Component currentFocusOwner = getCurrentKeyboardFocusManager().
2674            getGlobalFocusOwner();
2675        Component source = fe.getComponent();
2676        Component opposite = fe.getOppositeComponent();
2677        Component nativeSource = getHeavyweight(source);
2678
2679        synchronized (heavyweightRequests) {
2680            HeavyweightFocusRequest hwFocusRequest = getFirstHWRequest();
2681
2682            if (hwFocusRequest == HeavyweightFocusRequest.CLEAR_GLOBAL_FOCUS_OWNER)
2683            {
2684                return retargetUnexpectedFocusEvent(fe);
2685            }
2686
2687            if (source != null && nativeSource == null && hwFocusRequest != null) {
2688                // if source w/o peer and
2689                // if source is equal to first lightweight
2690                // then we should correct source and nativeSource
2691                if (source == hwFocusRequest.getFirstLightweightRequest().component)
2692                {
2693                    source = hwFocusRequest.heavyweight;
2694                    nativeSource = source; // source is heavuweight itself
2695                }
2696            }
2697            if (hwFocusRequest != null &&
2698                nativeSource == hwFocusRequest.heavyweight)
2699            {
2700                // Focus change as a result of a known call to requestFocus(),
2701                // or known click on a peer focusable heavyweight Component.
2702
2703                heavyweightRequests.removeFirst();
2704
2705                LightweightFocusRequest lwFocusRequest =
2706                    hwFocusRequest.lightweightRequests.removeFirst();
2707
2708                Component newSource = lwFocusRequest.component;
2709                if (currentFocusOwner != null) {
2710                    /*
2711                     * Since we receive FOCUS_GAINED when current focus
2712                     * owner is not null, correcponding FOCUS_LOST is supposed
2713                     * to be lost.  And so,  we keep new focus owner
2714                     * to determine synthetic FOCUS_LOST event which will be
2715                     * generated by KeyboardFocusManager for this FOCUS_GAINED.
2716                     *
2717                     * This code based on knowledge of
2718                     * DefaultKeyboardFocusManager's implementation and might
2719                     * be not applicable for another KeyboardFocusManager.
2720                     */
2721                    newFocusOwner = newSource;
2722                }
2723
2724                boolean temporary = (opposite == null ||
2725                                     isTemporary(newSource, opposite))
2726                        ? false
2727                        : lwFocusRequest.temporary;
2728
2729                if (hwFocusRequest.lightweightRequests.size() > 0) {
2730                    currentLightweightRequests =
2731                        hwFocusRequest.lightweightRequests;
2732                    EventQueue.invokeLater(new Runnable() {
2733                            public void run() {
2734                                processCurrentLightweightRequests();
2735                            }
2736                        });
2737                }
2738
2739                // 'opposite' will be fixed by
2740                // DefaultKeyboardFocusManager.realOppositeComponent
2741                return new CausedFocusEvent(newSource,
2742                                      FocusEvent.FOCUS_GAINED, temporary,
2743                                      opposite, lwFocusRequest.cause);
2744            }
2745
2746            if (currentFocusOwner != null
2747                && currentFocusOwner.getContainingWindow() == source
2748                && (hwFocusRequest == null || source != hwFocusRequest.heavyweight))
2749            {
2750                // Special case for FOCUS_GAINED in top-levels
2751                // If it arrives as the result of activation we should skip it
2752                // This event will not have appropriate request record and
2753                // on arrival there will be already some focus owner set.
2754                return new CausedFocusEvent(currentFocusOwner, FocusEvent.FOCUS_GAINED, false,
2755                                            null, CausedFocusEvent.Cause.ACTIVATION);
2756            }
2757
2758            return retargetUnexpectedFocusEvent(fe);
2759        } // end synchronized(heavyweightRequests)
2760    }

2762    static FocusEvent retargetFocusLost(FocusEvent fe) {
2763        assert (fe.getID() == FocusEvent.FOCUS_LOST);
2764
2765        Component currentFocusOwner = getCurrentKeyboardFocusManager().
2766            getGlobalFocusOwner();
2767        Component opposite = fe.getOppositeComponent();
2768        Component nativeOpposite = getHeavyweight(opposite);
2769
2770        synchronized (heavyweightRequests) {
2771            HeavyweightFocusRequest hwFocusRequest = getFirstHWRequest();
2772
2773            if (hwFocusRequest == HeavyweightFocusRequest.CLEAR_GLOBAL_FOCUS_OWNER)
2774            {
2775                if (currentFocusOwner != null) {
2776                    // Call to KeyboardFocusManager.clearGlobalFocusOwner()
2777                    heavyweightRequests.removeFirst();
2778                    return new CausedFocusEvent(currentFocusOwner,
2779                                                FocusEvent.FOCUS_LOST, false, null,
2780                                                CausedFocusEvent.Cause.CLEAR_GLOBAL_FOCUS_OWNER);
2781                }
2782
2783                // Otherwise, fall through to failure case below
2784
2785            } else if (opposite == null)
2786            {
2787                // Focus leaving application
2788                if (currentFocusOwner != null) {
2789                    return new CausedFocusEvent(currentFocusOwner,
2790                                                FocusEvent.FOCUS_LOST,
2791                                                true, null, CausedFocusEvent.Cause.ACTIVATION);
2792                } else {
2793                    return fe;
2794                }
2795            } else if (hwFocusRequest != null &&
2796                       (nativeOpposite == hwFocusRequest.heavyweight ||
2797                        nativeOpposite == null &&
2798                        opposite == hwFocusRequest.getFirstLightweightRequest().component))
2799            {
2800                if (currentFocusOwner == null) {
2801                    return fe;
2802                }
2803                // Focus change as a result of a known call to requestFocus(),
2804                // or click on a peer focusable heavyweight Component.
2805                // If a focus transfer is made across top-levels, then the
2806                // FOCUS_LOST event is always temporary, and the FOCUS_GAINED
2807                // event is always permanent. Otherwise, the stored temporary
2808                // value is honored.
2809
2810                LightweightFocusRequest lwFocusRequest =
2811                    hwFocusRequest.lightweightRequests.getFirst();
2812
2813                boolean temporary = isTemporary(opposite, currentFocusOwner)
2814                    ? true
2815                    : lwFocusRequest.temporary;
2816
2817                return new CausedFocusEvent(currentFocusOwner, FocusEvent.FOCUS_LOST,
2818                                            temporary, lwFocusRequest.component, lwFocusRequest.cause);
2819            } else if (focusedWindowChanged(opposite, currentFocusOwner)) {
2820                // If top-level changed there might be no focus request in a list
2821                // But we know the opposite, we now it is temporary - dispatch the event.
2822                if (!fe.isTemporary() && currentFocusOwner != null) {
2823                    // Create copy of the event with only difference in temporary parameter.
2824                    fe = new CausedFocusEvent(currentFocusOwner, FocusEvent.FOCUS_LOST,
2825                                              true, opposite, CausedFocusEvent.Cause.ACTIVATION);
2826                }
2827                return fe;
2828            }
2829
2830            return retargetUnexpectedFocusEvent(fe);
2831        }  // end synchronized(heavyweightRequests)
2832    }

2834    static AWTEvent retargetFocusEvent(AWTEvent event) {
2835        if (clearingCurrentLightweightRequests) {
2836            return event;
2837        }
2838
2839        KeyboardFocusManager manager = getCurrentKeyboardFocusManager();
2840        if (focusLog.isLoggable(PlatformLogger.FINER)) {
2841            if (event instanceof FocusEvent || event instanceof WindowEvent) {
2842                focusLog.finer(">>> {0}", String.valueOf(event));
2843            }
2844            if (focusLog.isLoggable(PlatformLogger.FINER) && event instanceof KeyEvent) {
2845                focusLog.finer("    focus owner is {0}",
2846                               String.valueOf(manager.getGlobalFocusOwner()));
2847                focusLog.finer(">>> {0}", String.valueOf(event));
2848            }
2849        }
2850
2851        synchronized(heavyweightRequests) {
2852            /*
2853             * This code handles FOCUS_LOST event which is generated by
2854             * DefaultKeyboardFocusManager for FOCUS_GAINED.
2855             *
2856             * This code based on knowledge of DefaultKeyboardFocusManager's
2857             * implementation and might be not applicable for another
2858             * KeyboardFocusManager.
2859             *
2860             * Fix for 4472032
2861             */
2862            if (newFocusOwner != null &&
2863                event.getID() == FocusEvent.FOCUS_LOST)
2864            {
2865                FocusEvent fe = (FocusEvent)event;
2866
2867                if (manager.getGlobalFocusOwner() == fe.getComponent() &&
2868                    fe.getOppositeComponent() == newFocusOwner)
2869                {
2870                    newFocusOwner = null;
2871                    return event;
2872                }
2873            }
2874        }

2876        processCurrentLightweightRequests();
2877
2878        switch (event.getID()) {
2879            case FocusEvent.FOCUS_GAINED: {
2880                event = retargetFocusGained((FocusEvent)event);
2881                break;
2882            }
2883            case FocusEvent.FOCUS_LOST: {
2884                event = retargetFocusLost((FocusEvent)event);
2885                break;
2886            }
2887            default:
2888                /* do nothing */
2889        }
2890        return event;
2891    }
