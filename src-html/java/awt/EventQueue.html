059/**
060 * <code>EventQueue</code> is a platform-independent class
061 * that queues events, both from the underlying peer classes
062 * and from trusted application classes.
063 * <p>
064 * It encapsulates asynchronous event dispatch machinery which
065 * extracts events from the queue and dispatches them by calling
066 * {@link #dispatchEvent(AWTEvent) dispatchEvent(AWTEvent)} method
067 * on this <code>EventQueue</code> with the event to be dispatched
068 * as an argument.  The particular behavior of this machinery is
069 * implementation-dependent.  The only requirements are that events
070 * which were actually enqueued to this queue (note that events
071 * being posted to the <code>EventQueue</code> can be coalesced)
072 * are dispatched:
073 * <dl>
074 *   <dt> Sequentially.
075 *   <dd> That is, it is not permitted that several events from
076 *        this queue are dispatched simultaneously.
077 *   <dt> In the same order as they are enqueued.
078 *   <dd> That is, if <code>AWTEvent</code>&nbsp;A is enqueued
079 *        to the <code>EventQueue</code> before
080 *        <code>AWTEvent</code>&nbsp;B then event B will not be
081 *        dispatched before event A.
082 * </dl>
083 * <p>
084 * Some browsers partition applets in different code bases into
085 * separate contexts, and establish walls between these contexts.
086 * In such a scenario, there will be one <code>EventQueue</code>
087 * per context. Other browsers place all applets into the same
088 * context, implying that there will be only a single, global
089 * <code>EventQueue</code> for all applets. This behavior is
090 * implementation-dependent.  Consult your browser's documentation
091 * for more information.
092 * <p>
093 * For information on the threading issues of the event dispatch
094 * machinery, see <a href="doc-files/AWTThreadIssues.html#Autoshutdown"
095 * >AWT Threading Issues</a>.
096 *
097 * @author Thomas Ball
098 * @author Fred Ecks
099 * @author David Mendenhall
100 *
101 * @since       1.1
102 */

115     * That is, the EventQueue object is actually implemented as
116     * NUM_PRIORITIES queues and all Events on a particular internal Queue
117     * have identical priority. Events are pulled off the EventQueue starting
118     * with the Queue of highest priority. We progress in decreasing order
119     * across all Queues.
120     */
121    private Queue[] queues = new Queue[NUM_PRIORITIES];

227    public EventQueue() {
228        for (int i = 0; i < NUM_PRIORITIES; i++) {
229            queues[i] = new Queue();
230        }
231        /*
232         * NOTE: if you ever have to start the associated event dispatch
233         * thread at this point, be aware of the following problem:
234         * If this EventQueue instance is created in
235         * SunToolkit.createNewAppContext() the started dispatch thread
236         * may call AppContext.getAppContext() before createNewAppContext()
237         * completes thus causing mess in thread group to appcontext mapping.
238         */
239
240        appContext = AppContext.getAppContext();
241        pushPopLock = (Lock)appContext.get(AppContext.EVENT_QUEUE_LOCK_KEY);
242        pushPopCond = (Condition)appContext.get(AppContext.EVENT_QUEUE_COND_KEY);
243    }

245    /**
246     * Posts a 1.1-style event to the <code>EventQueue</code>.
247     * If there is an existing event on the queue with the same ID
248     * and event source, the source <code>Component</code>'s
249     * <code>coalesceEvents</code> method will be called.
250     *
251     * @param theEvent an instance of <code>java.awt.AWTEvent</code>,
252     *          or a subclass of it
253     * @throws NullPointerException if <code>theEvent</code> is <code>null</code>
254     */
255    public void postEvent(AWTEvent theEvent) {
256        SunToolkit.flushPendingEvents(appContext);
257        postEventPrivate(theEvent);
258    }

260    /**
261     * Posts a 1.1-style event to the <code>EventQueue</code>.
262     * If there is an existing event on the queue with the same ID
263     * and event source, the source <code>Component</code>'s
264     * <code>coalesceEvents</code> method will be called.
265     *
266     * @param theEvent an instance of <code>java.awt.AWTEvent</code>,
267     *          or a subclass of it
268     */
269    private final void postEventPrivate(AWTEvent theEvent) {
270        theEvent.isPosted = true;
271        pushPopLock.lock();
272        try {
273            if (nextQueue != null) {
274                // Forward the event to the top of EventQueue stack
275                nextQueue.postEventPrivate(theEvent);
276                return;
277            }
278            if (dispatchThread == null) {
279                if (theEvent.getSource() == AWTAutoShutdown.getInstance()) {
280                    return;
281                } else {
282                    initDispatchThread();
283                }
284            }
285            postEvent(theEvent, getPriority(theEvent));
286        } finally {
287            pushPopLock.unlock();
288        }
289    }

291    private static int getPriority(AWTEvent theEvent) {
292        if (theEvent instanceof PeerEvent) {
293            PeerEvent peerEvent = (PeerEvent)theEvent;
294            if ((peerEvent.getFlags() & PeerEvent.ULTIMATE_PRIORITY_EVENT) != 0) {
295                return ULTIMATE_PRIORITY;
296            }
297            if ((peerEvent.getFlags() & PeerEvent.PRIORITY_EVENT) != 0) {
298                return HIGH_PRIORITY;
299            }
300            if ((peerEvent.getFlags() & PeerEvent.LOW_PRIORITY_EVENT) != 0) {
301                return LOW_PRIORITY;
302            }
303        }
304        int id = theEvent.getID();
305        if ((id >= PaintEvent.PAINT_FIRST) && (id <= PaintEvent.PAINT_LAST)) {
306            return LOW_PRIORITY;
307        }
308        return NORM_PRIORITY;
309    }

311    /**
312     * Posts the event to the internal Queue of specified priority,
313     * coalescing as appropriate.
314     *
315     * @param theEvent an instance of <code>java.awt.AWTEvent</code>,
316     *          or a subclass of it
317     * @param priority  the desired priority of the event
318     */
319    private void postEvent(AWTEvent theEvent, int priority) {
320        if (coalesceEvent(theEvent, priority)) {
321            return;
322        }
323
324        EventQueueItem newItem = new EventQueueItem(theEvent);
325
326        cacheEQItem(newItem);
327
328        boolean notifyID = (theEvent.getID() == this.waitForID);
329
330        if (queues[priority].head == null) {
331            boolean shouldNotify = noEvents();
332            queues[priority].head = queues[priority].tail = newItem;
333
334            if (shouldNotify) {
335                if (theEvent.getSource() != AWTAutoShutdown.getInstance()) {
336                    AWTAutoShutdown.getInstance().notifyThreadBusy(dispatchThread);
337                }
338                pushPopCond.signalAll();
339            } else if (notifyID) {
340                pushPopCond.signalAll();
341            }
342        } else {
343            // The event was not coalesced or has non-Component source.
344            // Insert it at the end of the appropriate Queue.
345            queues[priority].tail.next = newItem;
346            queues[priority].tail = newItem;
347            if (notifyID) {
348                pushPopCond.signalAll();
349            }
350        }
351    }

353    private boolean coalescePaintEvent(PaintEvent e) {
354        ComponentPeer sourcePeer = ((Component)e.getSource()).peer;
355        if (sourcePeer != null) {
356            sourcePeer.coalescePaintEvent(e);
357        }
358        EventQueueItem[] cache = ((Component)e.getSource()).eventCache;
359        if (cache == null) {
360            return false;
361        }
362        int index = eventToCacheIndex(e);
363
364        if (index != -1 && cache[index] != null) {
365            PaintEvent merged = mergePaintEvents(e, (PaintEvent)cache[index].event);
366            if (merged != null) {
367                cache[index].event = merged;
368                return true;
369            }
370        }
371        return false;
372    }
373
374    private PaintEvent mergePaintEvents(PaintEvent a, PaintEvent b) {
375        Rectangle aRect = a.getUpdateRect();
376        Rectangle bRect = b.getUpdateRect();
377        if (bRect.contains(aRect)) {
378            return b;
379        }
380        if (aRect.contains(bRect)) {
381            return a;
382        }
383        return null;
384    }

386    private boolean coalesceMouseEvent(MouseEvent e) {
387        if (e instanceof SunDropTargetEvent) {
388            // SunDropTargetEvent should not coalesce with MouseEvent
389            return false;
390        }
391        EventQueueItem[] cache = ((Component)e.getSource()).eventCache;
392        if (cache == null) {
393            return false;
394        }
395        int index = eventToCacheIndex(e);
396        if (index != -1 && cache[index] != null) {
397            cache[index].event = e;
398            return true;
399        }
400        return false;
401    }

403    private boolean coalescePeerEvent(PeerEvent e) {
404        EventQueueItem[] cache = ((Component)e.getSource()).eventCache;
405        if (cache == null) {
406            return false;
407        }
408        int index = eventToCacheIndex(e);
409        if (index != -1 && cache[index] != null) {
410            e = e.coalesceEvents((PeerEvent)cache[index].event);
411            if (e != null) {
412                cache[index].event = e;
413                return true;
414            } else {
415                cache[index] = null;
416            }
417        }
418        return false;
419    }

421    /*
422     * Should avoid of calling this method by any means
423     * as it's working time is dependant on EQ length.
424     * In the wors case this method alone can slow down the entire application
425     * 10 times by stalling the Event processing.
426     * Only here by backward compatibility reasons.
427     */
428    private boolean coalesceOtherEvent(AWTEvent e, int priority) {
429        int id = e.getID();
430        Component source = (Component)e.getSource();
431        for (EventQueueItem entry = queues[priority].head;
432            entry != null; entry = entry.next)
433        {
434            // Give Component.coalesceEvents a chance
435            if (entry.event.getSource() == source && entry.event.getID() == id) {
436                AWTEvent coalescedEvent = source.coalesceEvents(
437                    entry.event, e);
438                if (coalescedEvent != null) {
439                    entry.event = coalescedEvent;
440                    return true;
441                }
442            }
443        }
444        return false;
445    }
446
447    private boolean coalesceEvent(AWTEvent e, int priority) {
448        if (!(e.getSource() instanceof Component)) {
449            return false;
450        }
451        if (e instanceof PeerEvent) {
452            return coalescePeerEvent((PeerEvent)e);
453        }
454        // The worst case
455        if (((Component)e.getSource()).isCoalescingEnabled()
456            && coalesceOtherEvent(e, priority))
457        {
458            return true;
459        }
460        if (e instanceof PaintEvent) {
461            return coalescePaintEvent((PaintEvent)e);
462        }
463        if (e instanceof MouseEvent) {
464            return coalesceMouseEvent((MouseEvent)e);
465        }
466        return false;
467    }

469    private void cacheEQItem(EventQueueItem entry) {
470        if(entry.event instanceof SunDropTargetEvent) {
471            // Do not cache SunDropTargetEvent, it should not coalesce
472            return;
473        }
474        int index = eventToCacheIndex(entry.event);
475        if (index != -1 && entry.event.getSource() instanceof Component) {
476            Component source = (Component)entry.event.getSource();
477            if (source.eventCache == null) {
478                source.eventCache = new EventQueueItem[CACHE_LENGTH];
479            }
480            source.eventCache[index] = entry;
481        }
482    }
483
484    private void uncacheEQItem(EventQueueItem entry) {
485        int index = eventToCacheIndex(entry.event);
486        if (index != -1 && entry.event.getSource() instanceof Component) {
487            Component source = (Component)entry.event.getSource();
488            if (source.eventCache == null) {
489                return;
490            }
491            source.eventCache[index] = null;
492        }
493    }

495    private static final int PAINT = 0;
496    private static final int UPDATE = 1;
497    private static final int MOVE = 2;
498    private static final int DRAG = 3;
499    private static final int PEER = 4;
500    private static final int CACHE_LENGTH = 5;

502    private static int eventToCacheIndex(AWTEvent e) {
503        switch(e.getID()) {
504        case PaintEvent.PAINT:
505            return PAINT;
506        case PaintEvent.UPDATE:
507            return UPDATE;
508        case MouseEvent.MOUSE_MOVED:
509            return MOVE;
510        case MouseEvent.MOUSE_DRAGGED:
511            return DRAG;
512        default:
513            return e instanceof PeerEvent ? PEER : -1;
514        }
515    }

517    /**
518     * Returns whether an event is pending on any of the separate
519     * Queues.
520     * @return whether an event is pending on any of the separate Queues
521     */
522    private boolean noEvents() {
523        for (int i = 0; i < NUM_PRIORITIES; i++) {
524            if (queues[i].head != null) {
525                return false;
526            }
527        }
528
529        return true;
530    }

532    /**
533     * Removes an event from the <code>EventQueue</code> and
534     * returns it.  This method will block until an event has
535     * been posted by another thread.
536     * @return the next <code>AWTEvent</code>
537     * @exception InterruptedException
538     *            if any thread has interrupted this thread
539     */
540    public AWTEvent getNextEvent() throws InterruptedException {
541        do {
542            /*
543             * SunToolkit.flushPendingEvents must be called outside
544             * of the synchronized block to avoid deadlock when
545             * event queues are nested with push()/pop().
546             */
547            SunToolkit.flushPendingEvents(appContext);
548            pushPopLock.lock();
549            try {
550                AWTEvent event = getNextEventPrivate();
551                if (event != null) {
552                    return event;
553                }
554                AWTAutoShutdown.getInstance().notifyThreadFree(dispatchThread);
555                pushPopCond.await();
556            } finally {
557                pushPopLock.unlock();
558            }
559        } while(true);
560    }
561
562    /*
563     * Must be called under the lock. Doesn't call flushPendingEvents()
564     */
565    AWTEvent getNextEventPrivate() throws InterruptedException {
566        for (int i = NUM_PRIORITIES - 1; i >= 0; i--) {
567            if (queues[i].head != null) {
568                EventQueueItem entry = queues[i].head;
569                queues[i].head = entry.next;
570                if (entry.next == null) {
571                    queues[i].tail = null;
572                }
573                uncacheEQItem(entry);
574                return entry.event;
575            }
576        }
577        return null;
578    }
579
580    AWTEvent getNextEvent(int id) throws InterruptedException {
581        do {
582            /*
583             * SunToolkit.flushPendingEvents must be called outside
584             * of the synchronized block to avoid deadlock when
585             * event queues are nested with push()/pop().
586             */
587            SunToolkit.flushPendingEvents(appContext);
588            pushPopLock.lock();
589            try {
590                for (int i = 0; i < NUM_PRIORITIES; i++) {
591                    for (EventQueueItem entry = queues[i].head, prev = null;
592                         entry != null; prev = entry, entry = entry.next)
593                    {
594                        if (entry.event.getID() == id) {
595                            if (prev == null) {
596                                queues[i].head = entry.next;
597                            } else {
598                                prev.next = entry.next;
599                            }
600                            if (queues[i].tail == entry) {
601                                queues[i].tail = prev;
602                            }
603                            uncacheEQItem(entry);
604                            return entry.event;
605                        }
606                    }
607                }
608                waitForID = id;
609                pushPopCond.await();
610                waitForID = 0;
611            } finally {
612                pushPopLock.unlock();
613            }
614        } while(true);
615    }

617    /**
618     * Returns the first event on the <code>EventQueue</code>
619     * without removing it.
620     * @return the first event
621     */
622    public AWTEvent peekEvent() {
623        pushPopLock.lock();
624        try {
625            for (int i = NUM_PRIORITIES - 1; i >= 0; i--) {
626                if (queues[i].head != null) {
627                    return queues[i].head.event;
628                }
629            }
630        } finally {
631            pushPopLock.unlock();
632        }
633
634        return null;
635    }

637    /**
638     * Returns the first event with the specified id, if any.
639     * @param id the id of the type of event desired
640     * @return the first event of the specified id or <code>null</code>
641     *    if there is no such event
642     */
643    public AWTEvent peekEvent(int id) {
644        pushPopLock.lock();
645        try {
646            for (int i = NUM_PRIORITIES - 1; i >= 0; i--) {
647                EventQueueItem q = queues[i].head;
648                for (; q != null; q = q.next) {
649                    if (q.event.getID() == id) {
650                        return q.event;
651                    }
652                }
653            }
654        } finally {
655            pushPopLock.unlock();
656        }
657
658        return null;
659    }

664    /**
665     * Dispatches an event. The manner in which the event is
666     * dispatched depends upon the type of the event and the
667     * type of the event's source object:
668     * <p> </p>
669     * <table border=1 summary="Event types, source types, and dispatch methods">
670     * <tr>
671     *     <th>Event Type</th>
672     *     <th>Source Type</th>
673     *     <th>Dispatched To</th>
674     * </tr>
675     * <tr>
676     *     <td>ActiveEvent</td>
677     *     <td>Any</td>
678     *     <td>event.dispatch()</td>
679     * </tr>
680     * <tr>
681     *     <td>Other</td>
682     *     <td>Component</td>
683     *     <td>source.dispatchEvent(AWTEvent)</td>
684     * </tr>
685     * <tr>
686     *     <td>Other</td>
687     *     <td>MenuComponent</td>
688     *     <td>source.dispatchEvent(AWTEvent)</td>
689     * </tr>
690     * <tr>
691     *     <td>Other</td>
692     *     <td>Other</td>
693     *     <td>No action (ignored)</td>
694     * </tr>
695     * </table>
696     * <p> </p>
697     * @param event an instance of <code>java.awt.AWTEvent</code>,
698     *          or a subclass of it
699     * @throws NullPointerException if <code>event</code> is <code>null</code>
700     * @since           1.2
701     */
702    protected void dispatchEvent(final AWTEvent event) {
703        final Object src = event.getSource();
704        final PrivilegedAction<Void> action = new PrivilegedAction<Void>() {
705            public Void run() {
706                dispatchEventImpl(event, src);
707                return null;
708            }
709        };
710
711        final AccessControlContext stack = AccessController.getContext();
712        final AccessControlContext srcAcc = getAccessControlContextFrom(src);
713        final AccessControlContext eventAcc = event.getAccessControlContext();
714        if (srcAcc == null) {
715            javaSecurityAccess.doIntersectionPrivilege(action, stack, eventAcc);
716        } else {
717            javaSecurityAccess.doIntersectionPrivilege(
718                new PrivilegedAction<Void>() {
719                    public Void run() {
720                        javaSecurityAccess.doIntersectionPrivilege(action, eventAcc);
721                        return null;
722                    }
723                }, stack, srcAcc);
724        }
725    }

727    private static AccessControlContext getAccessControlContextFrom(Object src) {
728        return src instanceof Component ?
729            ((Component)src).getAccessControlContext() :
730            src instanceof MenuComponent ?
731                ((MenuComponent)src).getAccessControlContext() :
732                src instanceof TrayIcon ?
733                    ((TrayIcon)src).getAccessControlContext() :
734                    null;
735    }

737    /**
738     * Called from dispatchEvent() under a correct AccessControlContext
739     */
740    private void dispatchEventImpl(final AWTEvent event, final Object src) {
741        event.isPosted = true;
742        if (event instanceof ActiveEvent) {
743            // This could become the sole method of dispatching in time.
744            setCurrentEventAndMostRecentTimeImpl(event);
745            ((ActiveEvent)event).dispatch();
746        } else if (src instanceof Component) {
747            ((Component)src).dispatchEvent(event);

749        } else if (src instanceof MenuComponent) {

751        } else if (src instanceof TrayIcon) {
752            ((TrayIcon)src).dispatchEvent(event);
753        } else if (src instanceof AWTAutoShutdown) {
754            if (noEvents()) {
755                dispatchThread.stopDispatching();
756            }
757        } else {
758            if (getEventLog().isLoggable(PlatformLogger.FINE)) {
759                getEventLog().fine("Unable to dispatch event: " + event);
760            }
761        }

844    /**
845     * Replaces the existing <code>EventQueue</code> with the specified one.
846     * Any pending events are transferred to the new <code>EventQueue</code>
847     * for processing by it.
848     *
849     * @param newEventQueue an <code>EventQueue</code>
850     *          (or subclass thereof) instance to be use
851     * @see      java.awt.EventQueue#pop
852     * @throws NullPointerException if <code>newEventQueue</code> is <code>null</code>
853     * @since           1.2
854     */
855    public void push(EventQueue newEventQueue) {
856        if (getEventLog().isLoggable(PlatformLogger.FINE)) {
857            getEventLog().fine("EventQueue.push(" + newEventQueue + ")");
858        }
859
860        pushPopLock.lock();
861        try {
862            EventQueue topQueue = this;
863            while (topQueue.nextQueue != null) {
864                topQueue = topQueue.nextQueue;
865            }
866
867            if ((topQueue.dispatchThread != null) &&
868                (topQueue.dispatchThread.getEventQueue() == this))
869            {
870                newEventQueue.dispatchThread = topQueue.dispatchThread;
871                topQueue.dispatchThread.setEventQueue(newEventQueue);
872            }
873
874            // Transfer all events forward to new EventQueue.
875            while (topQueue.peekEvent() != null) {
876                try {
877                    // Use getNextEventPrivate() as it doesn't call flushPendingEvents()
878                    newEventQueue.postEventPrivate(topQueue.getNextEventPrivate());
879                } catch (InterruptedException ie) {
880                    if (getEventLog().isLoggable(PlatformLogger.FINE)) {
881                        getEventLog().fine("Interrupted push", ie);
882                    }
883                }
884            }
885
886            // Wake up EDT waiting in getNextEvent(), so it can
887            // pick up a new EventQueue. Post the waking event before
888            // topQueue.nextQueue is assigned, otherwise the event would
889            // go newEventQueue
890            topQueue.postEventPrivate(new InvocationEvent(topQueue, dummyRunnable));
891
892            newEventQueue.previousQueue = topQueue;
893            topQueue.nextQueue = newEventQueue;
894
895            if (appContext.get(AppContext.EVENT_QUEUE_KEY) == topQueue) {
896                appContext.put(AppContext.EVENT_QUEUE_KEY, newEventQueue);
897            }
898
899            pushPopCond.signalAll();
900        } finally {
901            pushPopLock.unlock();
902        }
903    }

905    /**
906     * Stops dispatching events using this <code>EventQueue</code>.
907     * Any pending events are transferred to the previous
908     * <code>EventQueue</code> for processing.
909     * <p>
910     * Warning: To avoid deadlock, do not declare this method
911     * synchronized in a subclass.
912     *
913     * @exception EmptyStackException if no previous push was made
914     *  on this <code>EventQueue</code>
915     * @see      java.awt.EventQueue#push
916     * @since           1.2
917     */
918    protected void pop() throws EmptyStackException {
919        if (getEventLog().isLoggable(PlatformLogger.FINE)) {
920            getEventLog().fine("EventQueue.pop(" + this + ")");
921        }
922
923        pushPopLock.lock();
924        try {
925            EventQueue topQueue = this;
926            while (topQueue.nextQueue != null) {
927                topQueue = topQueue.nextQueue;
928            }
929            EventQueue prevQueue = topQueue.previousQueue;
930            if (prevQueue == null) {
931                throw new EmptyStackException();
932            }
933
934            topQueue.previousQueue = null;
935            prevQueue.nextQueue = null;
936
937            // Transfer all events back to previous EventQueue.
938            while (topQueue.peekEvent() != null) {
939                try {
940                    prevQueue.postEventPrivate(topQueue.getNextEventPrivate());
941                } catch (InterruptedException ie) {
942                    if (getEventLog().isLoggable(PlatformLogger.FINE)) {
943                        getEventLog().fine("Interrupted pop", ie);
944                    }
945                }
946            }
947
948            if ((topQueue.dispatchThread != null) &&
949                (topQueue.dispatchThread.getEventQueue() == this))
950            {
951                prevQueue.dispatchThread = topQueue.dispatchThread;
952                topQueue.dispatchThread.setEventQueue(prevQueue);
953            }
954
955            if (appContext.get(AppContext.EVENT_QUEUE_KEY) == this) {
956                appContext.put(AppContext.EVENT_QUEUE_KEY, prevQueue);
957            }
958
959            // Wake up EDT waiting in getNextEvent(), so it can
960            // pick up a new EventQueue
961            topQueue.postEventPrivate(new InvocationEvent(topQueue, dummyRunnable));
962
963            pushPopCond.signalAll();
964        } finally {
965            pushPopLock.unlock();
966        }
967    }

1070    final boolean detachDispatchThread(EventDispatchThread edt, boolean forceDetach) {
1071        /*
1072         * This synchronized block is to secure that the event dispatch
1073         * thread won't die in the middle of posting a new event to the
1074         * associated event queue. It is important because we notify
1075         * that the event dispatch thread is busy after posting a new event
1076         * to its queue, so the EventQueue.dispatchThread reference must
1077         * be valid at that point.
1078         */
1079        pushPopLock.lock();
1080        try {
1081            if (edt == dispatchThread) {
1082                /*
1083                 * Don't detach the thread if any events are pending. Not
1084                 * sure if it's a possible scenario, though.
1085                 *
1086                 * Fix for 4648733. Check both the associated java event
1087                 * queue and the PostEventQueue.
1088                 */
1089                if (!forceDetach && (peekEvent() != null) || !SunToolkit.isPostEventQueueEmpty()) {
1090                    return false;
1091                }
1092                dispatchThread = null;
1093            }
1094            AWTAutoShutdown.getInstance().notifyThreadFree(edt);
1095            return true;
1096        } finally {
1097            pushPopLock.unlock();
1098        }

1119    /*
1120     * Removes any pending events for the specified source object.
1121     * If removeAllEvents parameter is <code>true</code> then all
1122     * events for the specified source object are removed, if it
1123     * is <code>false</code> then <code>SequencedEvent</code>, <code>SentEvent</code>,
1124     * <code>FocusEvent</code>, <code>WindowEvent</code>, <code>KeyEvent</code>,
1125     * and <code>InputMethodEvent</code> are kept in the queue, but all other
1126     * events are removed.
1127     *
1128     * This method is normally called by the source's
1129     * <code>removeNotify</code> method.
1130     */
1131    final void removeSourceEvents(Object source, boolean removeAllEvents) {
1132        SunToolkit.flushPendingEvents(appContext);
1133        pushPopLock.lock();
1134        try {
1135            for (int i = 0; i < NUM_PRIORITIES; i++) {
1136                EventQueueItem entry = queues[i].head;
1137                EventQueueItem prev = null;
1138                while (entry != null) {
1139                    if ((entry.event.getSource() == source)
1140                        && (removeAllEvents
1141                            || ! (entry.event instanceof SequencedEvent
1142                                  || entry.event instanceof SentEvent
1143                                  || entry.event instanceof FocusEvent
1144                                  || entry.event instanceof WindowEvent
1145                                  || entry.event instanceof KeyEvent
1146                                  || entry.event instanceof InputMethodEvent)))
1147                    {
1148                        if (entry.event instanceof SequencedEvent) {
1149                            ((SequencedEvent)entry.event).dispose();
1150                        }
1151                        if (entry.event instanceof SentEvent) {
1152                            ((SentEvent)entry.event).dispose();
1153                        }
1154                        if (entry.event instanceof InvocationEvent) {
1155                            AWTAccessor.getInvocationEventAccessor()
1156                                    .dispose((InvocationEvent)entry.event);
1157                        }
1158                        if (prev == null) {
1159                            queues[i].head = entry.next;
1160                        } else {
1161                            prev.next = entry.next;
1162                        }
1163                        uncacheEQItem(entry);
1164                    } else {
1165                        prev = entry;
1166                    }
1167                    entry = entry.next;
1168                }
1169                queues[i].tail = prev;
1170            }
1171        } finally {
1172            pushPopLock.unlock();
1173        }
1174    }

1179    private void setCurrentEventAndMostRecentTimeImpl(AWTEvent e) {
1180        pushPopLock.lock();
1181        try {
1182            if (Thread.currentThread() != dispatchThread) {
1183                return;
1184            }
1185
1186            currentEvent = new WeakReference(e);
1187
1188            // This series of 'instanceof' checks should be replaced with a
1189            // polymorphic type (for example, an interface which declares a
1190            // getWhen() method). However, this would require us to make such
1191            // a type public, or to place it in sun.awt. Both of these approaches
1192            // have been frowned upon. So for now, we hack.

1203            } else if (e instanceof ActionEvent) {

1206            } else if (e instanceof InvocationEvent) {
1207                InvocationEvent ie = (InvocationEvent)e;
1208                mostRecentEventTime2 = ie.getWhen();

1211        } finally {

1245     *
1246     * @param runnable  the <code>Runnable</code> whose <code>run</code>
1247     *                  method should be executed
1248     *                  synchronously in the
1249     *                  {@link #isDispatchThread event dispatch thread}
1250     *                  of {@link Toolkit#getSystemEventQueue the system EventQueue}
1251     * @exception       InterruptedException  if any thread has
1252     *                  interrupted this thread
1253     * @exception       InvocationTargetException  if an throwable is thrown
1254     *                  when running <code>runnable</code>
1255     * @see             #invokeLater
1256     * @see             Toolkit#getSystemEventQueue
1257     * @see             #isDispatchThread
1258     * @since           1.2
1259     */

1261        throws InterruptedException, InvocationTargetException

1266    static void invokeAndWait(Object source, Runnable runnable)
1267        throws InterruptedException, InvocationTargetException
1268    {
1269        if (EventQueue.isDispatchThread()) {
1270            throw new Error("Cannot call invokeAndWait from the event dispatcher thread");
1271        }
1272
1273        class AWTInvocationLock {}
1274        Object lock = new AWTInvocationLock();
1275
1276        InvocationEvent event =
1277            new InvocationEvent(source, runnable, lock, true);
1278
1279        synchronized (lock) {
1280            Toolkit.getEventQueue().postEvent(event);
1281            while (!event.isDispatched()) {
1282                lock.wait();
1283            }
1284        }
1285
1286        Throwable eventThrowable = event.getThrowable();
1287        if (eventThrowable != null) {
1288            throw new InvocationTargetException(eventThrowable);
1289        }
1290    }

1293     * Called from PostEventQueue.postEvent to notify that a new event
1294     * appeared. First it proceeds to the EventQueue on the top of the
1295     * stack, then notifies the associated dispatch thread if it exists
1296     * or starts a new one otherwise.

1298    private void wakeup(boolean isShutdown) {
1299        pushPopLock.lock();
1300        try {
1301            if (nextQueue != null) {
1302                // Forward call to the top of EventQueue stack.
1303                nextQueue.wakeup(isShutdown);
1304            } else if (dispatchThread != null) {
1305                pushPopCond.signalAll();
1306            } else if (!isShutdown) {
1307                initDispatchThread();
1308            }
1309        } finally {
1310            pushPopLock.unlock();
1311        }
