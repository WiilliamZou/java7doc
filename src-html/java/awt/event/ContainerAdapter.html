026package java.awt.event;
027
028/**
029 * An abstract adapter class for receiving container events.
030 * The methods in this class are empty. This class exists as
031 * convenience for creating listener objects.
032 * <P>
033 * Extend this class to create a <code>ContainerEvent</code> listener
034 * and override the methods for the events of interest. (If you implement the
035 * <code>ContainerListener</code> interface, you have to define all of
036 * the methods in it. This abstract class defines null methods for them
037 * all, so you can only have to define methods for events you care about.)
038 * <P>
039 * Create a listener object using the extended class and then register it with
040 * a component using the component's <code>addContainerListener</code>
041 * method. When the container's contents change because a component has
042 * been added or removed, the relevant method in the listener object is invoked,
043 * and the <code>ContainerEvent</code> is passed to it.
044 *
045 * @see ContainerEvent
046 * @see ContainerListener
047 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/containerlistener.html">Tutorial: Writing a Container Listener</a>
048 *
049 * @author Amy Fowler
050 * @since 1.1
051 */
052public abstract class ContainerAdapter implements ContainerListener {
053    /**
054     * Invoked when a component has been added to the container.
055     */
056    public void componentAdded(ContainerEvent e) {}
057
058    /**
059     * Invoked when a component has been removed from the container.
060     */
061    public void componentRemoved(ContainerEvent e) {}
062}
