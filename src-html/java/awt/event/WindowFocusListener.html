026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving <code>WindowEvents</code>, including
032 * <code>WINDOW_GAINED_FOCUS</code> and <code>WINDOW_LOST_FOCUS</code> events.
033 * The class that is interested in processing a <code>WindowEvent</code>
034 * either implements this interface (and
035 * all the methods it contains) or extends the abstract
036 * <code>WindowAdapter</code> class (overriding only the methods of interest).
037 * The listener object created from that class is then registered with a
038 * <code>Window</code>
039 * using the <code>Window</code>'s <code>addWindowFocusListener</code> method.
040 * When the <code>Window</code>'s
041 * status changes by virtue of it being opened, closed, activated, deactivated,
042 * iconified, or deiconified, or by focus being transfered into or out of the
043 * <code>Window</code>, the relevant method in the listener object is invoked,
044 * and the <code>WindowEvent</code> is passed to it.
045 *
046 * @author David Mendenhall
047 *
048 * @see WindowAdapter
049 * @see WindowEvent
050 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/windowlistener.html">Tutorial: Writing a Window Listener</a>
051 *
052 * @since 1.4
053 */
054public interface WindowFocusListener extends EventListener {
055
056    /**
057     * Invoked when the Window is set to be the focused Window, which means
058     * that the Window, or one of its subcomponents, will receive keyboard
059     * events.
060     */
061    public void windowGainedFocus(WindowEvent e);
062
063    /**
064     * Invoked when the Window is no longer the focused Window, which means
065     * that keyboard events will no longer be delivered to the Window or any of
066     * its subcomponents.
067     */
068    public void windowLostFocus(WindowEvent e);
069}
