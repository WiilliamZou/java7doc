026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving container events.
032 * The class that is interested in processing a container event
033 * either implements this interface (and all the methods it
034 * contains) or extends the abstract <code>ContainerAdapter</code> class
035 * (overriding only the methods of interest).
036 * The listener object created from that class is then registered with a
037 * component using the component's <code>addContainerListener</code>
038 * method. When the container's contents change because a component
039 * has been added or removed, the relevant method in the listener object
040 * is invoked, and the <code>ContainerEvent</code> is passed to it.
041 * <P>
042 * Container events are provided for notification purposes ONLY;
043 * The AWT will automatically handle add and remove operations
044 * internally so the program works properly regardless of
045 * whether the program registers a <code>ComponentListener</code> or not.
046 *
047 * @see ContainerAdapter
048 * @see ContainerEvent
049 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/containerlistener.html">Tutorial: Writing a Container Listener</a>
050 *
051 * @author Tim Prinzing
052 * @author Amy Fowler
053 * @since 1.1
054 */
055public interface ContainerListener extends EventListener {
056    /**
057     * Invoked when a component has been added to the container.
058     */
059    public void componentAdded(ContainerEvent e);
060
061    /**
062     * Invoked when a component has been removed from the container.
063     */
064    public void componentRemoved(ContainerEvent e);
065
066}
