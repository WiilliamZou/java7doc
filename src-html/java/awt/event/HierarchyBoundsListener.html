026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving ancestor moved and resized events.
032 * The class that is interested in processing these events either implements
033 * this interface (and all the methods it contains) or extends the abstract
034 * <code>HierarchyBoundsAdapter</code> class (overriding only the method of
035 * interest).
036 * The listener object created from that class is then registered with a
037 * Component using the Component's <code>addHierarchyBoundsListener</code>
038 * method. When the hierarchy to which the Component belongs changes by
039 * the resizing or movement of an ancestor, the relevant method in the listener
040 * object is invoked, and the <code>HierarchyEvent</code> is passed to it.
041 * <p>
042 * Hierarchy events are provided for notification purposes ONLY;
043 * The AWT will automatically handle changes to the hierarchy internally so
044 * that GUI layout works properly regardless of whether a
045 * program registers an <code>HierarchyBoundsListener</code> or not.
046 *
047 * @author      David Mendenhall
048 * @see         HierarchyBoundsAdapter
049 * @see         HierarchyEvent
050 * @since       1.3
051 */
052public interface HierarchyBoundsListener extends EventListener {
053    /**
054     * Called when an ancestor of the source is moved.
055     */
056    public void ancestorMoved(HierarchyEvent e);
057
058    /**
059     * Called when an ancestor of the source is resized.
060     */
061    public void ancestorResized(HierarchyEvent e);
062}
