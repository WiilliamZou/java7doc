026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving component events.
032 * The class that is interested in processing a component event
033 * either implements this interface (and all the methods it
034 * contains) or extends the abstract <code>ComponentAdapter</code> class
035 * (overriding only the methods of interest).
036 * The listener object created from that class is then registered with a
037 * component using the component's <code>addComponentListener</code>
038 * method. When the component's size, location, or visibility
039 * changes, the relevant method in the listener object is invoked,
040 * and the <code>ComponentEvent</code> is passed to it.
041 * <P>
042 * Component events are provided for notification purposes ONLY;
043 * The AWT will automatically handle component moves and resizes
044 * internally so that GUI layout works properly regardless of
045 * whether a program registers a <code>ComponentListener</code> or not.
046 *
047 * @see ComponentAdapter
048 * @see ComponentEvent
049 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/componentlistener.html">Tutorial: Writing a Component Listener</a>
050 *
051 * @author Carl Quinn
052 * @since 1.1
053 */
054public interface ComponentListener extends EventListener {
055    /**
056     * Invoked when the component's size changes.
057     */
058    public void componentResized(ComponentEvent e);
059
060    /**
061     * Invoked when the component's position changes.
062     */
063    public void componentMoved(ComponentEvent e);
064
065    /**
066     * Invoked when the component has been made visible.
067     */
068    public void componentShown(ComponentEvent e);
069
070    /**
071     * Invoked when the component has been made invisible.
072     */
073    public void componentHidden(ComponentEvent e);
074}
