026package java.awt.event;
027
028import sun.awt.AWTAccessor;
029
030import java.awt.ActiveEvent;
031import java.awt.AWTEvent;
032
033/**
034 * An event which executes the <code>run()</code> method on a <code>Runnable
035 * </code> when dispatched by the AWT event dispatcher thread. This class can
036 * be used as a reference implementation of <code>ActiveEvent</code> rather
037 * than declaring a new class and defining <code>dispatch()</code>.<p>
038 *
039 * Instances of this class are placed on the <code>EventQueue</code> by calls
040 * to <code>invokeLater</code> and <code>invokeAndWait</code>. Client code
041 * can use this fact to write replacement functions for <code>invokeLater
042 * </code> and <code>invokeAndWait</code> without writing special-case code
043 * in any <code>AWTEventListener</code> objects.
044 * <p>
045 * An unspecified behavior will be caused if the {@code id} parameter
046 * of any particular {@code InvocationEvent} instance is not
047 * in the range from {@code INVOCATION_FIRST} to {@code INVOCATION_LAST}.
048 *
049 * @author      Fred Ecks
050 * @author      David Mendenhall
051 *
052 * @see         java.awt.ActiveEvent
053 * @see         java.awt.EventQueue#invokeLater
054 * @see         java.awt.EventQueue#invokeAndWait
055 * @see         AWTEventListener
056 *
057 * @since       1.2
058 */
059public class InvocationEvent extends AWTEvent implements ActiveEvent {
060
061    static {
062        AWTAccessor.setInvocationEventAccessor(new AWTAccessor.InvocationEventAccessor() {
063            @Override
064            public void dispose(InvocationEvent invocationEvent) {
065                invocationEvent.finishedDispatching(false);
066            }
067            @Override
068            public InvocationEvent createEvent(Object source, Runnable runnable, Runnable listener,
069                                               boolean catchThrowables) {
070                return new InvocationEvent(source, runnable, listener, catchThrowables);
071            }
072        });
073    }
074
075    /**
076     * Marks the first integer id for the range of invocation event ids.
077     */
078    public static final int INVOCATION_FIRST = 1200;
079
080    /**
081     * The default id for all InvocationEvents.
082     */
083    public static final int INVOCATION_DEFAULT = INVOCATION_FIRST;
084
085    /**
086     * Marks the last integer id for the range of invocation event ids.
087     */
088    public static final int INVOCATION_LAST = INVOCATION_DEFAULT;
089
090    /**
091     * The Runnable whose run() method will be called.
092     */
093    protected Runnable runnable;
094
095    /**
096     * The (potentially null) Object whose notifyAll() method will be called
097     * immediately after the Runnable.run() method has returned or thrown an exception.
098     *
099     * @see #isDispatched
100     */
101    protected Object notifier;
102
103    /**
104     * The (potentially null) Runnable whose run() method will be called
105     * immediately after the event was dispatched or disposed.
106     *
107     * @see #isDispatched
108     */
109    private final Runnable listener;
110
111    /**
112     * Indicates whether the <code>run()</code> method of the <code>runnable</code>
113     * was executed or not.
114     *
115     * @see #isDispatched
116     * @since 1.7
117     */
118    private volatile boolean dispatched = false;
119
120    /**
121     * Set to true if dispatch() catches Throwable and stores it in the
122     * exception instance variable. If false, Throwables are propagated up
123     * to the EventDispatchThread's dispatch loop.
124     */
125    protected boolean catchExceptions;
126
127    /**
128     * The (potentially null) Exception thrown during execution of the
129     * Runnable.run() method. This variable will also be null if a particular
130     * instance does not catch exceptions.
131     */
132    private Exception exception = null;
133
134    /**
135     * The (potentially null) Throwable thrown during execution of the
136     * Runnable.run() method. This variable will also be null if a particular
137     * instance does not catch exceptions.
138     */
139    private Throwable throwable = null;

141    /**
142     * The timestamp of when this event occurred.
143     *
144     * @serial
145     * @see #getWhen
146     */
147    private long when;

150     * JDK 1.1 serialVersionUID.

154    /**
155     * Constructs an <code>InvocationEvent</code> with the specified
156     * source which will execute the runnable's <code>run</code>
157     * method when dispatched.
158     * <p>This is a convenience constructor.  An invocation of the form
159     * <tt>InvocationEvent(source, runnable)</tt>
160     * behaves in exactly the same way as the invocation of
161     * <tt>{@link #InvocationEvent(Object, Runnable, Object, boolean) InvocationEvent}(source, runnable, null, false)</tt>.
162     * <p> This method throws an <code>IllegalArgumentException</code>
163     * if <code>source</code> is <code>null</code>.
164     *
165     * @param source    The <code>Object</code> that originated the event
166     * @param runnable  The <code>Runnable</code> whose <code>run</code>
167     *                  method will be executed
168     * @throws IllegalArgumentException if <code>source</code> is null
169     *
170     * @see #getSource()
171     * @see #InvocationEvent(Object, Runnable, Object, boolean)
172     */
173    public InvocationEvent(Object source, Runnable runnable) {
174        this(source, INVOCATION_DEFAULT, runnable, null, null, false);
175    }

177    /**
178     * Constructs an <code>InvocationEvent</code> with the specified
179     * source which will execute the runnable's <code>run</code>
180     * method when dispatched.  If notifier is non-<code>null</code>,
181     * <code>notifyAll()</code> will be called on it
182     * immediately after <code>run</code> has returned or thrown an exception.
183     * <p>An invocation of the form <tt>InvocationEvent(source,
184     * runnable, notifier, catchThrowables)</tt>
185     * behaves in exactly the same way as the invocation of
186     * <tt>{@link #InvocationEvent(Object, int, Runnable, Object, boolean) InvocationEvent}(source, InvocationEvent.INVOCATION_DEFAULT, runnable, notifier, catchThrowables)</tt>.
187     * <p>This method throws an <code>IllegalArgumentException</code>
188     * if <code>source</code> is <code>null</code>.
189     *
190     * @param source            The <code>Object</code> that originated
191     *                          the event
192     * @param runnable          The <code>Runnable</code> whose
193     *                          <code>run</code> method will be
194     *                          executed
195     * @param notifier          The {@code Object} whose <code>notifyAll</code>
196     *                          method will be called after
197     *                          <code>Runnable.run</code> has returned or
198     *                          thrown an exception
199     * @param catchThrowables   Specifies whether <code>dispatch</code>
200     *                          should catch Throwable when executing
201     *                          the <code>Runnable</code>'s <code>run</code>
202     *                          method, or should instead propagate those
203     *                          Throwables to the EventDispatchThread's
204     *                          dispatch loop
205     * @throws IllegalArgumentException if <code>source</code> is null
206     *
207     * @see #getSource()
208     * @see     #InvocationEvent(Object, int, Runnable, Object, boolean)
209     */
210    public InvocationEvent(Object source, Runnable runnable, Object notifier,
211                           boolean catchThrowables) {
212        this(source, INVOCATION_DEFAULT, runnable, notifier, null, catchThrowables);
213    }

215    /**
216     * Constructs an <code>InvocationEvent</code> with the specified
217     * source which will execute the runnable's <code>run</code>
218     * method when dispatched.  If listener is non-<code>null</code>,
219     * <code>listener.run()</code> will be called immediately after
220     * <code>run</code> has returned, thrown an exception or the event
221     * was disposed.
222     * <p>This method throws an <code>IllegalArgumentException</code>
223     * if <code>source</code> is <code>null</code>.
224     *
225     * @param source            The <code>Object</code> that originated
226     *                          the event
227     * @param runnable          The <code>Runnable</code> whose
228     *                          <code>run</code> method will be
229     *                          executed
230     * @param listener          The <code>Runnable</code>Runnable whose
231     *                          <code>run()</code> method will be called
232     *                          after the {@code InvocationEvent}
233     *                          was dispatched or disposed
234     * @param catchThrowables   Specifies whether <code>dispatch</code>
235     *                          should catch Throwable when executing
236     *                          the <code>Runnable</code>'s <code>run</code>
237     *                          method, or should instead propagate those
238     *                          Throwables to the EventDispatchThread's
239     *                          dispatch loop
240     * @throws IllegalArgumentException if <code>source</code> is null
241     */
242    private InvocationEvent(Object source, Runnable runnable, Runnable listener,
243                           boolean catchThrowables)  {
244        this(source, INVOCATION_DEFAULT, runnable, null, listener, catchThrowables);
245    }

247    /**
248     * Constructs an <code>InvocationEvent</code> with the specified
249     * source and ID which will execute the runnable's <code>run</code>
250     * method when dispatched.  If notifier is non-<code>null</code>,
251     * <code>notifyAll</code> will be called on it immediately after
252     * <code>run</code> has returned or thrown an exception.
253     * <p>This method throws an
254     * <code>IllegalArgumentException</code> if <code>source</code>
255     * is <code>null</code>.
256     *
257     * @param source            The <code>Object</code> that originated
258     *                          the event
259     * @param id     An integer indicating the type of event.
260     *                     For information on allowable values, see
261     *                     the class description for {@link InvocationEvent}
262     * @param runnable          The <code>Runnable</code> whose
263     *                          <code>run</code> method will be executed
264     * @param notifier          The <code>Object</code> whose <code>notifyAll</code>
265     *                          method will be called after
266     *                          <code>Runnable.run</code> has returned or
267     *                          thrown an exception
268     * @param catchThrowables   Specifies whether <code>dispatch</code>
269     *                          should catch Throwable when executing the
270     *                          <code>Runnable</code>'s <code>run</code>
271     *                          method, or should instead propagate those
272     *                          Throwables to the EventDispatchThread's
273     *                          dispatch loop
274     * @throws IllegalArgumentException if <code>source</code> is null
275     * @see #getSource()
276     * @see #getID()
277     */
278    protected InvocationEvent(Object source, int id, Runnable runnable,
279                              Object notifier, boolean catchThrowables) {
280        this(source, id, runnable, notifier, null, catchThrowables);
281    }
282
283    private InvocationEvent(Object source, int id, Runnable runnable,
284                            Object notifier, Runnable listener, boolean catchThrowables) {
285        super(source, id);
286        this.runnable = runnable;
287        this.notifier = notifier;
288        this.listener = listener;
289        this.catchExceptions = catchThrowables;
290        this.when = System.currentTimeMillis();
291    }
292    /**
293     * Executes the Runnable's <code>run()</code> method and notifies the
294     * notifier (if any) when <code>run()</code> has returned or thrown an exception.
295     *
296     * @see #isDispatched
297     */
298    public void dispatch() {
299        try {
300            if (catchExceptions) {
301                try {
302                    runnable.run();
303                }
304                catch (Throwable t) {
305                    if (t instanceof Exception) {
306                        exception = (Exception) t;
307                    }
308                    throwable = t;
309                }
310            }
311            else {
312                runnable.run();
313            }
314        } finally {
315            finishedDispatching(true);
316        }
317    }
318
319    /**
320     * Returns any Exception caught while executing the Runnable's <code>run()
321     * </code> method.
322     *
323     * @return  A reference to the Exception if one was thrown; null if no
324     *          Exception was thrown or if this InvocationEvent does not
325     *          catch exceptions
326     */
327    public Exception getException() {
328        return (catchExceptions) ? exception : null;
329    }
330
331    /**
332     * Returns any Throwable caught while executing the Runnable's <code>run()
333     * </code> method.
334     *
335     * @return  A reference to the Throwable if one was thrown; null if no
336     *          Throwable was thrown or if this InvocationEvent does not
337     *          catch Throwables
338     * @since 1.5
339     */
340    public Throwable getThrowable() {
341        return (catchExceptions) ? throwable : null;
342    }
343
344    /**
345     * Returns the timestamp of when this event occurred.
346     *
347     * @return this event's timestamp
348     * @since 1.4
349     */
350    public long getWhen() {
351        return when;
352    }
353
354    /**
355     * Returns {@code true} if the event is dispatched or any exception is
356     * thrown while dispatching, {@code false} otherwise. The method should
357     * be called by a waiting thread that calls the {@code notifier.wait()} method.
358     * Since spurious wakeups are possible (as explained in {@link Object#wait()}),
359     * this method should be used in a waiting loop to ensure that the event
360     * got dispatched:
361     * <pre>
362     *     while (!event.isDispatched()) {
363     *         notifier.wait();
364     *     }
365     * </pre>
366     * If the waiting thread wakes up without dispatching the event,
367     * the {@code isDispatched()} method returns {@code false}, and
368     * the {@code while} loop executes once more, thus, causing
369     * the awakened thread to revert to the waiting mode.
370     * <p>
371     * If the {@code notifier.notifyAll()} happens before the waiting thread
372     * enters the {@code notifier.wait()} method, the {@code while} loop ensures
373     * that the waiting thread will not enter the {@code notifier.wait()} method.
374     * Otherwise, there is no guarantee that the waiting thread will ever be woken
375     * from the wait.
376     *
377     * @return {@code true} if the event has been dispatched, or any exception
378     * has been thrown while dispatching, {@code false} otherwise
379     * @see #dispatch
380     * @see #notifier
381     * @see #catchExceptions
382     * @since 1.7
383     */
384    public boolean isDispatched() {
385        return dispatched;
386    }
387
388    /**
389     * Called when the event was dispatched or disposed
390     * @param dispatched true if the event was dispatched
391     *                   false if the event was disposed
392     */
393    private void finishedDispatching(boolean dispatched) {
394        this.dispatched = dispatched;
395
396        if (notifier != null) {
397            synchronized (notifier) {
398                notifier.notifyAll();
399            }
400        }
401
402        if (listener != null) {
403            listener.run();
404        }
405    }
406
407    /**
408     * Returns a parameter string identifying this event.
409     * This method is useful for event-logging and for debugging.
410     *
411     * @return  A string identifying the event and its attributes
412     */
413    public String paramString() {
414        String typeStr;
415        switch(id) {
416            case INVOCATION_DEFAULT:
417                typeStr = "INVOCATION_DEFAULT";
418                break;
419            default:
420                typeStr = "unknown type";
421        }
422        return typeStr + ",runnable=" + runnable + ",notifier=" + notifier +
423            ",catchExceptions=" + catchExceptions + ",when=" + when;
424    }
