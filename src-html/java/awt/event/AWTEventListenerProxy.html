026package java.awt.event;
027
028import java.util.EventListenerProxy;
029import java.awt.AWTEvent;
030
031/**
032 * A class which extends the {@code EventListenerProxy}
033 * specifically for adding an {@code AWTEventListener}
034 * for a specific event mask.
035 * Instances of this class can be added as {@code AWTEventListener}s
036 * to a {@code Toolkit} object.
037 * <p>
038 * The {@code getAWTEventListeners} method of {@code Toolkit}
039 * can return a mixture of {@code AWTEventListener}
040 * and {@code AWTEventListenerProxy} objects.
041 *
042 * @see java.awt.Toolkit
043 * @see java.util.EventListenerProxy
044 * @since 1.4
045 */
046public class AWTEventListenerProxy
047        extends EventListenerProxy<AWTEventListener>
048        implements AWTEventListener {
049
050    private final long eventMask;
051
052    /**
053     * Constructor which binds the {@code AWTEventListener}
054     * to a specific event mask.
055     *
056     * @param eventMask  the bitmap of event types to receive
057     * @param listener   the listener object
058     */
059    public AWTEventListenerProxy (long eventMask, AWTEventListener listener) {
060        super(listener);
061        this.eventMask = eventMask;
062    }
063
064    /**
065     * Forwards the AWT event to the listener delegate.
066     *
067     * @param event  the AWT event
068     */
069    public void eventDispatched(AWTEvent event) {
070        getListener().eventDispatched(event);
071    }
072
073    /**
074     * Returns the event mask associated with the listener.
075     *
076     * @return the event mask associated with the listener
077     */
078    public long getEventMask() {
079        return this.eventMask;
080    }
081}
