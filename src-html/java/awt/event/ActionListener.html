026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving action events.
032 * The class that is interested in processing an action event
033 * implements this interface, and the object created with that
034 * class is registered with a component, using the component's
035 * <code>addActionListener</code> method. When the action event
036 * occurs, that object's <code>actionPerformed</code> method is
037 * invoked.
038 *
039 * @see ActionEvent
040 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/eventmodel.html">Tutorial: Java 1.1 Event Model</a>
041 *
042 * @author Carl Quinn
043 * @since 1.1
044 */
045public interface ActionListener extends EventListener {
046
047    /**
048     * Invoked when an action occurs.
049     */
050    public void actionPerformed(ActionEvent e);
051
052}
