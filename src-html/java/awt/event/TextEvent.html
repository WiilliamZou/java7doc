026package java.awt.event;
027
028import java.awt.AWTEvent;
029
030/**
031 * A semantic event which indicates that an object's text changed.
032 * This high-level event is generated by an object (such as a TextComponent)
033 * when its text changes. The event is passed to
034 * every <code>TextListener</code> object which registered to receive such
035 * events using the component's <code>addTextListener</code> method.
036 * <P>
037 * The object that implements the <code>TextListener</code> interface gets
038 * this <code>TextEvent</code> when the event occurs. The listener is
039 * spared the details of processing individual mouse movements and key strokes
040 * Instead, it can process a "meaningful" (semantic) event like "text changed".
041 * <p>
042 * An unspecified behavior will be caused if the {@code id} parameter
043 * of any particular {@code TextEvent} instance is not
044 * in the range from {@code TEXT_FIRST} to {@code TEXT_LAST}.
045 *
046 * @author Georges Saab
047 *
048 * @see java.awt.TextComponent
049 * @see TextListener
050 *
051 * @since 1.1
052 */
053
054public class TextEvent extends AWTEvent {
055
056    /**
057     * The first number in the range of ids used for text events.
058     */
059    public static final int TEXT_FIRST  = 900;
060
061    /**
062     * The last number in the range of ids used for text events.
063     */
064    public static final int TEXT_LAST   = 900;
065
066    /**
067     * This event id indicates that object's text changed.
068     */
069    public static final int TEXT_VALUE_CHANGED  = TEXT_FIRST;
070
071    /*
072     * JDK 1.1 serialVersionUID
073     */
074    private static final long serialVersionUID = 6269902291250941179L;
075
076    /**
077     * Constructs a <code>TextEvent</code> object.
078     * <p> This method throws an
079     * <code>IllegalArgumentException</code> if <code>source</code>
080     * is <code>null</code>.
081     *
082     * @param source The (<code>TextComponent</code>) object that
083     *               originated the event
084     * @param id     An integer that identifies the event type.
085     *                     For information on allowable values, see
086     *                     the class description for {@link TextEvent}
087     * @throws IllegalArgumentException if <code>source</code> is null
088     * @see #getSource()
089     * @see #getID()
090     */
091    public TextEvent(Object source, int id) {
092        super(source, id);
093    }
094
095
096    /**
097     * Returns a parameter string identifying this text event.
098     * This method is useful for event-logging and for debugging.
099     *
100     * @return a string identifying the event and its attributes
101     */
102    public String paramString() {
103        String typeStr;
104        switch(id) {
105          case TEXT_VALUE_CHANGED:
106              typeStr = "TEXT_VALUE_CHANGED";
107              break;
108          default:
109              typeStr = "unknown type";
110        }
111        return typeStr;
112    }
113}
