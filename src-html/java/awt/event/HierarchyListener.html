026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving hierarchy changed events.
032 * The class that is interested in processing a hierarchy changed event
033 * should implement this interface.
034 * The listener object created from that class is then registered with a
035 * Component using the Component's <code>addHierarchyListener</code>
036 * method. When the hierarchy to which the Component belongs changes, the
037 * <code>hierarchyChanged</code> method in the listener object is invoked,
038 * and the <code>HierarchyEvent</code> is passed to it.
039 * <p>
040 * Hierarchy events are provided for notification purposes ONLY;
041 * The AWT will automatically handle changes to the hierarchy internally so
042 * that GUI layout, displayability, and visibility work properly regardless
043 * of whether a program registers a <code>HierarchyListener</code> or not.
044 *
045 * @author      David Mendenhall
046 * @see         HierarchyEvent
047 * @since       1.3
048 */
049public interface HierarchyListener extends EventListener {
050    /**
051     * Called when the hierarchy has been changed. To discern the actual
052     * type of change, call <code>HierarchyEvent.getChangeFlags()</code>.
053     *
054     * @see HierarchyEvent#getChangeFlags()
055     */
056    public void hierarchyChanged(HierarchyEvent e);
057}
