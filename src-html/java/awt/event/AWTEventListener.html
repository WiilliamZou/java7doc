026package java.awt.event;
027
028import java.util.EventListener;
029import java.awt.AWTEvent;
030
031/**
032 * The listener interface for receiving notification of events
033 * dispatched to objects that are instances of Component or
034 * MenuComponent or their subclasses.  Unlike the other EventListeners
035 * in this package, AWTEventListeners passively observe events
036 * being dispatched in the AWT, system-wide.  Most applications
037 * should never use this class; applications which might use
038 * AWTEventListeners include event recorders for automated testing,
039 * and facilities such as the Java Accessibility package.
040 * <p>
041 * The class that is interested in monitoring AWT events
042 * implements this interface, and the object created with that
043 * class is registered with the Toolkit, using the Toolkit's
044 * <code>addAWTEventListener</code> method.  When an event is
045 * dispatched anywhere in the AWT, that object's
046 * <code>eventDispatched</code> method is invoked.
047 *
048 * @see java.awt.AWTEvent
049 * @see java.awt.Toolkit#addAWTEventListener
050 * @see java.awt.Toolkit#removeAWTEventListener
051 *
052 * @author Fred Ecks
053 * @since 1.2
054 */
055public interface AWTEventListener extends EventListener {
056
057    /**
058     * Invoked when an event is dispatched in the AWT.
059     */
060    public void eventDispatched(AWTEvent event);
061
062}
