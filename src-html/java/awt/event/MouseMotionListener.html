026package java.awt.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving mouse motion events on a component.
032 * (For clicks and other mouse events, use the <code>MouseListener</code>.)
033 * <P>
034 * The class that is interested in processing a mouse motion event
035 * either implements this interface (and all the methods it
036 * contains) or extends the abstract <code>MouseMotionAdapter</code> class
037 * (overriding only the methods of interest).
038 * <P>
039 * The listener object created from that class is then registered with a
040 * component using the component's <code>addMouseMotionListener</code>
041 * method. A mouse motion event is generated when the mouse is moved
042 * or dragged. (Many such events will be generated). When a mouse motion event
043 * occurs, the relevant method in the listener object is invoked, and
044 * the <code>MouseEvent</code> is passed to it.
045 *
046 * @author Amy Fowler
047 *
048 * @see MouseMotionAdapter
049 * @see MouseEvent
050 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/mousemotionlistener.html">Tutorial: Writing a Mouse Motion Listener</a>
051 *
052 * @since 1.1
053 */
054public interface MouseMotionListener extends EventListener {
055
056    /**
057     * Invoked when a mouse button is pressed on a component and then
058     * dragged.  <code>MOUSE_DRAGGED</code> events will continue to be
059     * delivered to the component where the drag originated until the
060     * mouse button is released (regardless of whether the mouse position
061     * is within the bounds of the component).
062     * <p>
063     * Due to platform-dependent Drag&Drop implementations,
064     * <code>MOUSE_DRAGGED</code> events may not be delivered during a native
065     * Drag&Drop operation.
066     */
067    public void mouseDragged(MouseEvent e);
068
069    /**
070     * Invoked when the mouse cursor has been moved onto a component
071     * but no buttons have been pushed.
072     */
073    public void mouseMoved(MouseEvent e);
074
075}
