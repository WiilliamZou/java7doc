026package java.awt.event;
027
028class NativeLibLoader {
029
030    /**
031     * This is copied from java.awt.Toolkit since we need the library
032     * loaded in sun.awt.image also:
033     *
034     * WARNING: This is a temporary workaround for a problem in the
035     * way the AWT loads native libraries. A number of classes in this
036     * package (sun.awt.image) have a native method, initIDs(),
037     * which initializes
038     * the JNI field and method ids used in the native portion of
039     * their implementation.
040     *
041     * Since the use and storage of these ids is done by the
042     * implementation libraries, the implementation of these method is
043     * provided by the particular AWT implementations (for example,
044     *  "Toolkit"s/Peer), such as Motif, Microsoft Windows, or Tiny. The
045     * problem is that this means that the native libraries must be
046     * loaded by the java.* classes, which do not necessarily know the
047     * names of the libraries to load. A better way of doing this
048     * would be to provide a separate library which defines java.awt.*
049     * initIDs, and exports the relevant symbols out to the
050     * implementation libraries.
051     *
052     * For now, we know it's done by the implementation, and we assume
053     * that the name of the library is "awt".  -br.
054     */
055    static void loadLibraries() {
056        java.security.AccessController.doPrivileged(
057                new sun.security.action.LoadLibraryAction("awt"));
058    }
059}
