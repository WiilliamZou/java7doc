038/**
039 * An event which indicates that a mouse action occurred in a component.
040 * A mouse action is considered to occur in a particular component if and only
041 * if the mouse cursor is over the unobscured part of the component's bounds
042 * when the action happens.
043 * For lightweight components, such as Swing's components, mouse events
044 * are only dispatched to the component if the mouse event type has been
045 * enabled on the component. A mouse event type is enabled by adding the
046 * appropriate mouse-based {@code EventListener} to the component
047 * ({@link MouseListener} or {@link MouseMotionListener}), or by invoking
048 * {@link Component#enableEvents(long)} with the appropriate mask parameter
049 * ({@code AWTEvent.MOUSE_EVENT_MASK} or {@code AWTEvent.MOUSE_MOTION_EVENT_MASK}).
050 * If the mouse event type has not been enabled on the component, the
051 * corresponding mouse events are dispatched to the first ancestor that
052 * has enabled the mouse event type.
053 *<p>
054 * For example, if a {@code MouseListener} has been added to a component, or
055 * {@code enableEvents(AWTEvent.MOUSE_EVENT_MASK)} has been invoked, then all
056 * the events defined by {@code MouseListener} are dispatched to the component.
057 * On the other hand, if a {@code MouseMotionListener} has not been added and
058 * {@code enableEvents} has not been invoked with
059 * {@code AWTEvent.MOUSE_MOTION_EVENT_MASK}, then mouse motion events are not
060 * dispatched to the component. Instead the mouse motion events are
061 * dispatched to the first ancestors that has enabled mouse motion
062 * events.
063 * <P>
064 * This low-level event is generated by a component object for:
065 * <ul>
066 * <li>Mouse Events
067 *     <ul>
068 *     <li>a mouse button is pressed
069 *     <li>a mouse button is released
070 *     <li>a mouse button is clicked (pressed and released)
071 *     <li>the mouse cursor enters the unobscured part of component's geometry
072 *     <li>the mouse cursor exits the unobscured part of component's geometry
073 *     </ul>
074 * <li> Mouse Motion Events
075 *     <ul>
076 *     <li>the mouse is moved
077 *     <li>the mouse is dragged
078 *     </ul>
079 * </ul>
080 * <P>
081 * A <code>MouseEvent</code> object is passed to every
082 * <code>MouseListener</code>
083 * or <code>MouseAdapter</code> object which is registered to receive
084 * the "interesting" mouse events using the component's
085 * <code>addMouseListener</code> method.
086 * (<code>MouseAdapter</code> objects implement the
087 * <code>MouseListener</code> interface.) Each such listener object
088 * gets a <code>MouseEvent</code> containing the mouse event.
089 * <P>
090 * A <code>MouseEvent</code> object is also passed to every
091 * <code>MouseMotionListener</code> or
092 * <code>MouseMotionAdapter</code> object which is registered to receive
093 * mouse motion events using the component's
094 * <code>addMouseMotionListener</code>
095 * method. (<code>MouseMotionAdapter</code> objects implement the
096 * <code>MouseMotionListener</code> interface.) Each such listener object
097 * gets a <code>MouseEvent</code> containing the mouse motion event.
098 * <P>
099 * When a mouse button is clicked, events are generated and sent to the
100 * registered <code>MouseListener</code>s.
101 * The state of modal keys can be retrieved using {@link InputEvent#getModifiers}
102 * and {@link InputEvent#getModifiersEx}.
103 * The button mask returned by {@link InputEvent#getModifiers} reflects
104 * only the button that changed state, not the current state of all buttons.
105 * (Note: Due to overlap in the values of ALT_MASK/BUTTON2_MASK and
106 * META_MASK/BUTTON3_MASK, this is not always true for mouse events involving
107 * modifier keys).
108 * To get the state of all buttons and modifier keys, use
109 * {@link InputEvent#getModifiersEx}.
110 * The button which has changed state is returned by {@link MouseEvent#getButton}
111 * <P>
112 * For example, if the first mouse button is pressed, events are sent in the
113 * following order:
114 * <PRE>
115 *    <b   >id           </b   >   <b   >modifiers   </b   > <b   >button </b   >
116 *    <code>MOUSE_PRESSED</code>:  <code>BUTTON1_MASK</code> <code>BUTTON1</code>
117 *    <code>MOUSE_RELEASED</code>: <code>BUTTON1_MASK</code> <code>BUTTON1</code>
118 *    <code>MOUSE_CLICKED</code>:  <code>BUTTON1_MASK</code> <code>BUTTON1</code>
119 * </PRE>
120 * When multiple mouse buttons are pressed, each press, release, and click
121 * results in a separate event.
122 * <P>
123 * For example, if the user presses <b>button 1</b> followed by
124 * <b>button 2</b>, and then releases them in the same order,
125 * the following sequence of events is generated:
126 * <PRE>
127 *    <b   >id           </b   >   <b   >modifiers   </b   > <b   >button </b   >
128 *    <code>MOUSE_PRESSED</code>:  <code>BUTTON1_MASK</code> <code>BUTTON1</code>
129 *    <code>MOUSE_PRESSED</code>:  <code>BUTTON2_MASK</code> <code>BUTTON2</code>
130 *    <code>MOUSE_RELEASED</code>: <code>BUTTON1_MASK</code> <code>BUTTON1</code>
131 *    <code>MOUSE_CLICKED</code>:  <code>BUTTON1_MASK</code> <code>BUTTON1</code>
132 *    <code>MOUSE_RELEASED</code>: <code>BUTTON2_MASK</code> <code>BUTTON2</code>
133 *    <code>MOUSE_CLICKED</code>:  <code>BUTTON2_MASK</code> <code>BUTTON2</code>
134 * </PRE>
135 * If <b>button 2</b> is released first, the
136 * <code>MOUSE_RELEASED</code>/<code>MOUSE_CLICKED</code> pair
137 * for <code>BUTTON2_MASK</code> arrives first,
138 * followed by the pair for <code>BUTTON1_MASK</code>.
139 * <p>
140 * Some extra mouse buttons are added to extend the standard set of buttons
141 * represented by the following constants:{@code BUTTON1}, {@code BUTTON2}, and {@code BUTTON3}.
142 * Extra buttons have no assigned {@code BUTTONx}
143 * constants as well as their button masks have no assigned {@code BUTTONx_DOWN_MASK}
144 * constants. Nevertheless, ordinal numbers starting from 4 may be
145 * used as button numbers (button ids). Values obtained by the
146 * {@link InputEvent#getMaskForButton(int) getMaskForButton(button)} method may be used
147 * as button masks.
148 * <p>
149 * <code>MOUSE_DRAGGED</code> events are delivered to the <code>Component</code>
150 * in which the mouse button was pressed until the mouse button is released
151 * (regardless of whether the mouse position is within the bounds of the
152 * <code>Component</code>).  Due to platform-dependent Drag&Drop implementations,
153 * <code>MOUSE_DRAGGED</code> events may not be delivered during a native
154 * Drag&Drop operation.
155 *
156 * In a multi-screen environment mouse drag events are delivered to the
157 * <code>Component</code> even if the mouse position is outside the bounds of the
158 * <code>GraphicsConfiguration</code> associated with that
159 * <code>Component</code>. However, the reported position for mouse drag events
160 * in this case may differ from the actual mouse position:
161 * <ul>
162 * <li>In a multi-screen environment without a virtual device:
163 * <br>
164 * The reported coordinates for mouse drag events are clipped to fit within the
165 * bounds of the <code>GraphicsConfiguration</code> associated with
166 * the <code>Component</code>.
167 * <li>In a multi-screen environment with a virtual device:
168 * <br>
169 * The reported coordinates for mouse drag events are clipped to fit within the
170 * bounds of the virtual device associated with the <code>Component</code>.
171 * </ul>
172 * <p>
173 * An unspecified behavior will be caused if the {@code id} parameter
174 * of any particular {@code MouseEvent} instance is not
175 * in the range from {@code MOUSE_FIRST} to {@code MOUSE_LAST}-1
176 * ({@code MOUSE_WHEEL} is not acceptable).
177 *
178 * @author Carl Quinn
179 *
180 * @see MouseAdapter
181 * @see MouseListener
182 * @see MouseMotionAdapter
183 * @see MouseMotionListener
184 * @see MouseWheelListener
185 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/mouselistener.html">Tutorial: Writing a Mouse Listener</a>
186 * @see <a href="http://java.sun.com/docs/books/tutorial/post1.0/ui/mousemotionlistener.html">Tutorial: Writing a Mouse Motion Listener</a>
187 *
188 * @since 1.1
189 */

462    /**
463     * Constructs a <code>MouseEvent</code> object with the
464     * specified source component,
465     * type, time, modifiers, coordinates, click count, popupTrigger flag,
466     * and button number.
467     * <p>
468     * Creating an invalid event (such
469     * as by using more than one of the old _MASKs, or modifier/button
470     * values which don't match) results in unspecified behavior.
471     * An invocation of the form
472     * <tt>MouseEvent(source, id, when, modifiers, x, y, clickCount, popupTrigger, button)</tt>
473     * behaves in exactly the same way as the invocation
474     * <tt> {@link #MouseEvent(Component, int, long, int, int, int,
475     * int, int, int, boolean, int) MouseEvent}(source, id, when, modifiers,
476     * x, y, xAbs, yAbs, clickCount, popupTrigger, button)</tt>
477     * where xAbs and yAbs defines as source's location on screen plus
478     * relative coordinates x and y.
479     * xAbs and yAbs are set to zero if the source is not showing.
480     * This method throws an
481     * <code>IllegalArgumentException</code> if <code>source</code>
482     * is <code>null</code>.
483     *
484     * @param source       The <code>Component</code> that originated the event
485     * @param id              An integer indicating the type of event.
486     *                     For information on allowable values, see
487     *                     the class description for {@link MouseEvent}
488     * @param when         A long integer that gives the time the event occurred.
489     *                     Passing negative or zero value
490     *                     is not recommended
491     * @param modifiers    The modifier keys down during event (e.g. shift, ctrl,
492     *                     alt, meta)
493     *                     Passing negative parameter
494     *                     is not recommended.
495     *                     Zero value means that no modifiers were passed.
496     *                     Use either an extended _DOWN_MASK or old _MASK modifiers,
497     *                     however do not mix models in the one event.
498     *                     The extended modifiers are preferred for using
499     * @param x            The horizontal x coordinate for the mouse location.
500     *                       It is allowed to pass negative values
501     * @param y            The vertical y coordinate for the mouse location.
502     *                       It is allowed to pass negative values
503     * @param clickCount   The number of mouse clicks associated with event.
504     *                       Passing negative value
505     *                       is not recommended
506     * @param popupTrigger A boolean that equals {@code true} if this event
507     *                     is a trigger for a popup menu
508     * @param button       An integer that indicates, which of the mouse buttons has
509     *                     changed its state.
510     * The following rules are applied to this parameter:
511     * <ul>
512     * <li>If support for the extended mouse buttons is
513     * {@link Toolkit#areExtraMouseButtonsEnabled() disabled} by Java
514     * then it is allowed to create {@code MouseEvent} objects only with the standard buttons:
515     * {@code NOBUTTON}, {@code BUTTON1}, {@code BUTTON2}, and
516     * {@code BUTTON3}.
517     * <li> If support for the extended mouse buttons is
518     * {@link Toolkit#areExtraMouseButtonsEnabled() enabled} by Java
519     * then it is allowed to create {@code MouseEvent} objects with
520     * the standard buttons.
521     * In case the support for extended mouse buttons is
522     * {@link Toolkit#areExtraMouseButtonsEnabled() enabled} by Java, then
523     * in addition to the standard buttons, {@code MouseEvent} objects can be created
524     * using buttons from the range starting from 4 to
525     * {@link java.awt.MouseInfo#getNumberOfButtons() MouseInfo.getNumberOfButtons()}
526     * if the mouse has more than three buttons.
527     * </ul>
528     * @throws IllegalArgumentException if {@code button} is less then zero
529     * @throws IllegalArgumentException if <code>source</code> is null
530     * @throws IllegalArgumentException if {@code button} is greater then BUTTON3 and the support for extended mouse buttons is
531     *                                  {@link Toolkit#areExtraMouseButtonsEnabled() disabled} by Java
532     * @throws IllegalArgumentException if {@code button} is greater then the
533     *                                  {@link java.awt.MouseInfo#getNumberOfButtons() current number of buttons} and the support
534     *                                  for extended mouse buttons is {@link Toolkit#areExtraMouseButtonsEnabled() enabled}
535     *                                  by Java
536     * @throws IllegalArgumentException if an invalid <code>button</code>
537     *            value is passed in
538     * @throws IllegalArgumentException if <code>source</code> is null
539     * @see #getSource()
540     * @see #getID()
541     * @see #getWhen()
542     * @see #getModifiers()
543     * @see #getX()
544     * @see #getY()
545     * @see #getClickCount()
546     * @see #isPopupTrigger()
547     * @see #getButton()
548     * @since 1.4
549     */
550    public MouseEvent(Component source, int id, long when, int modifiers,
551                      int x, int y, int clickCount, boolean popupTrigger,
552                      int button)
553    {
554        this(source, id, when, modifiers, x, y, 0, 0, clickCount, popupTrigger, button);
555        Point eventLocationOnScreen = new Point(0, 0);
556        try {
557          eventLocationOnScreen = source.getLocationOnScreen();
558          this.xAbs = eventLocationOnScreen.x + x;
559          this.yAbs = eventLocationOnScreen.y + y;
560        } catch (IllegalComponentStateException e){
561          this.xAbs = 0;
562          this.yAbs = 0;
563        }
564    }

566    /**
567     * Constructs a <code>MouseEvent</code> object with the
568     * specified source component,
569     * type, modifiers, coordinates, click count, and popupTrigger flag.
570     * An invocation of the form
571     * <tt>MouseEvent(source, id, when, modifiers, x, y, clickCount, popupTrigger)</tt>
572     * behaves in exactly the same way as the invocation
573     * <tt> {@link #MouseEvent(Component, int, long, int, int, int,
574     * int, int, int, boolean, int) MouseEvent}(source, id, when, modifiers,
575     * x, y, xAbs, yAbs, clickCount, popupTrigger, MouseEvent.NOBUTTON)</tt>
576     * where xAbs and yAbs defines as source's location on screen plus
577     * relative coordinates x and y.
578     * xAbs and yAbs are set to zero if the source is not showing.
579     * This method throws an <code>IllegalArgumentException</code>
580     * if <code>source</code> is <code>null</code>.
581     *
582     * @param source       The <code>Component</code> that originated the event
583     * @param id              An integer indicating the type of event.
584     *                     For information on allowable values, see
585     *                     the class description for {@link MouseEvent}
586     * @param when         A long integer that gives the time the event occurred.
587     *                     Passing negative or zero value
588     *                     is not recommended
589     * @param modifiers    The modifier keys down during event (e.g. shift, ctrl,
590     *                     alt, meta)
591     *                     Passing negative parameter
592     *                     is not recommended.
593     *                     Zero value means that no modifiers were passed.
594     *                     Use either an extended _DOWN_MASK or old _MASK modifiers,
595     *                     however do not mix models in the one event.
596     *                     The extended modifiers are preferred for using
597     * @param x            The horizontal x coordinate for the mouse location.
598     *                       It is allowed to pass negative values
599     * @param y            The vertical y coordinate for the mouse location.
600     *                       It is allowed to pass negative values
601     * @param clickCount   The number of mouse clicks associated with event.
602     *                       Passing negative value
603     *                       is not recommended
604     * @param popupTrigger A boolean that equals {@code true} if this event
605     *                     is a trigger for a popup menu
606     * @throws IllegalArgumentException if <code>source</code> is null
607     * @see #getSource()
608     * @see #getID()
609     * @see #getWhen()

637    /**
638     * Constructs a <code>MouseEvent</code> object with the
639     * specified source component,
640     * type, time, modifiers, coordinates, absolute coordinates, click count, popupTrigger flag,
641     * and button number.
642     * <p>
643     * Creating an invalid event (such
644     * as by using more than one of the old _MASKs, or modifier/button
645     * values which don't match) results in unspecified behavior.
646     * Even if inconsistent values for relative and absolute coordinates are
647     * passed to the constructor, the mouse event instance is still
648     * created and no exception is thrown.
649     * This method throws an
650     * <code>IllegalArgumentException</code> if <code>source</code>
651     * is <code>null</code>.
652     *
653     * @param source       The <code>Component</code> that originated the event
654     * @param id              An integer indicating the type of event.
655     *                     For information on allowable values, see
656     *                     the class description for {@link MouseEvent}
657     * @param when         A long integer that gives the time the event occurred.
658     *                     Passing negative or zero value
659     *                     is not recommended
660     * @param modifiers    The modifier keys down during event (e.g. shift, ctrl,
661     *                     alt, meta)
662     *                     Passing negative parameter
663     *                     is not recommended.
664     *                     Zero value means that no modifiers were passed.
665     *                     Use either an extended _DOWN_MASK or old _MASK modifiers,
666     *                     however do not mix models in the one event.
667     *                     The extended modifiers are preferred for using
668     * @param x            The horizontal x coordinate for the mouse location.
669     *                       It is allowed to pass negative values
670     * @param y            The vertical y coordinate for the mouse location.
671     *                       It is allowed to pass negative values
672     * @param xAbs           The absolute horizontal x coordinate for the mouse location
673     *                       It is allowed to pass negative values
674     * @param yAbs           The absolute vertical y coordinate for the mouse location
675     *                       It is allowed to pass negative values
676     * @param clickCount   The number of mouse clicks associated with event.
677     *                       Passing negative value
678     *                       is not recommended
679     * @param popupTrigger A boolean that equals {@code true} if this event
680     *                     is a trigger for a popup menu
681     * @param button       An integer that indicates, which of the mouse buttons has
682     *                     changed its state.
683     * The following rules are applied to this parameter:
684     * <ul>
685     * <li>If support for the extended mouse buttons is
686     * {@link Toolkit#areExtraMouseButtonsEnabled() disabled} by Java
687     * then it is allowed to create {@code MouseEvent} objects only with the standard buttons:
688     * {@code NOBUTTON}, {@code BUTTON1}, {@code BUTTON2}, and
689     * {@code BUTTON3}.
690     * <li> If support for the extended mouse buttons is
691     * {@link Toolkit#areExtraMouseButtonsEnabled() enabled} by Java
692     * then it is allowed to create {@code MouseEvent} objects with
693     * the standard buttons.
694     * In case the support for extended mouse buttons is
695     * {@link Toolkit#areExtraMouseButtonsEnabled() enabled} by Java, then
696     * in addition to the standard buttons, {@code MouseEvent} objects can be created
697     * using buttons from the range starting from 4 to
698     * {@link java.awt.MouseInfo#getNumberOfButtons() MouseInfo.getNumberOfButtons()}
699     * if the mouse has more than three buttons.
700     * </ul>
701     * @throws IllegalArgumentException if {@code button} is less then zero
702     * @throws IllegalArgumentException if <code>source</code> is null
703     * @throws IllegalArgumentException if {@code button} is greater then BUTTON3 and the support for extended mouse buttons is
704     *                                  {@link Toolkit#areExtraMouseButtonsEnabled() disabled} by Java
705     * @throws IllegalArgumentException if {@code button} is greater then the
706     *                                  {@link java.awt.MouseInfo#getNumberOfButtons() current number of buttons} and the support
707     *                                  for extended mouse buttons is {@link Toolkit#areExtraMouseButtonsEnabled() enabled}
708     *                                  by Java
709     * @throws IllegalArgumentException if an invalid <code>button</code>
710     *            value is passed in
711     * @throws IllegalArgumentException if <code>source</code> is null
712     * @see #getSource()
713     * @see #getID()
714     * @see #getWhen()
715     * @see #getModifiers()
716     * @see #getX()
717     * @see #getY()
718     * @see #getXOnScreen()
719     * @see #getYOnScreen()
720     * @see #getClickCount()
721     * @see #isPopupTrigger()
722     * @see #getButton()
723     * @see #button
724     * @see Toolkit#areExtraMouseButtonsEnabled()
725     * @see java.awt.MouseInfo#getNumberOfButtons()
726     * @see InputEvent#getMaskForButton(int)
727     * @since 1.6
728     */
729    public MouseEvent(Component source, int id, long when, int modifiers,
730                      int x, int y, int xAbs, int yAbs,
731                      int clickCount, boolean popupTrigger, int button)
732    {
733        super(source, id, when, modifiers);
734        this.x = x;
735        this.y = y;
736        this.xAbs = xAbs;
737        this.yAbs = yAbs;
738        this.clickCount = clickCount;
739        this.popupTrigger = popupTrigger;
740        if (button < NOBUTTON){
741            throw new IllegalArgumentException("Invalid button value :" + button);
742        }
743        if (button > BUTTON3) {
744            if (!Toolkit.getDefaultToolkit().areExtraMouseButtonsEnabled()){
745                throw new IllegalArgumentException("Extra mouse events are disabled " + button);
746            } else {
747                if (button > cachedNumberOfButtons) {
748                    throw new IllegalArgumentException("Nonexistent button " + button);
749                }
750            }
751            // XToolkit: extra buttons are not reporting about their state correctly.
752            // Being pressed they report the state=0 both on the press and on the release.
753            // For 1-3 buttons the state value equals zero on press and non-zero on release.
754            // Other modifiers like Shift, ALT etc seem report well with extra buttons.
755            // The problem reveals as follows: one button is pressed and then another button is pressed and released.
756            // So, the getModifiersEx() would not be zero due to a first button and we will skip this modifier.
757            // This may have to be moved into the peer code instead if possible.
758
759            if (getModifiersEx() != 0) { //There is at least one more button in a pressed state.
760                if (id == MouseEvent.MOUSE_RELEASED || id == MouseEvent.MOUSE_CLICKED){
761                    shouldExcludeButtonFromExtModifiers = true;
762                }
763            }
764        }
765
766        this.button = button;
767
768        if ((getModifiers() != 0) && (getModifiersEx() == 0)) {
769            setNewModifiers();
770        } else if ((getModifiers() == 0) &&
771                   (getModifiersEx() != 0 || button != NOBUTTON) &&
772                   (button <= BUTTON3))
773        {
774            setOldModifiers();
775        }
776    }

913    /**
914     * Returns a <code>String</code> instance describing the modifier keys and
915     * mouse buttons that were down during the event, such as "Shift",
916     * or "Ctrl+Shift". These strings can be localized by changing
917     * the <code>awt.properties</code> file.
918     * <p>
919     * Note that the <code>InputEvent.ALT_MASK</code> and
920     * <code>InputEvent.BUTTON2_MASK</code> have equal values,
921     * so the "Alt" string is returned for both modifiers.  Likewise,
922     * the <code>InputEvent.META_MASK</code> and
923     * <code>InputEvent.BUTTON3_MASK</code> have equal values,
924     * so the "Meta" string is returned for both modifiers.
925     * <p>
926     * Note that passing negative parameter is incorrect,
927     * and will cause the returning an unspecified string.
928     * Zero parameter means that no modifiers were passed and will
929     * cause the returning an empty string.
930     * <p>
931     * @param modifiers A modifier mask describing the modifier keys and
932     *                  mouse buttons that were down during the event
933     * @return string   string text description of the combination of modifier
934     *                  keys and mouse buttons that were down during the event
935     * @see InputEvent#getModifiersExText(int)
936     * @since 1.4
937     */
938    public static String getMouseModifiersText(int modifiers) {
939        StringBuilder buf = new StringBuilder();
940        if ((modifiers & InputEvent.ALT_MASK) != 0) {
941            buf.append(Toolkit.getProperty("AWT.alt", "Alt"));
942            buf.append("+");
943        }
944        if ((modifiers & InputEvent.META_MASK) != 0) {
945            buf.append(Toolkit.getProperty("AWT.meta", "Meta"));
946            buf.append("+");
947        }
948        if ((modifiers & InputEvent.CTRL_MASK) != 0) {
949            buf.append(Toolkit.getProperty("AWT.control", "Ctrl"));
950            buf.append("+");
951        }
952        if ((modifiers & InputEvent.SHIFT_MASK) != 0) {
953            buf.append(Toolkit.getProperty("AWT.shift", "Shift"));
954            buf.append("+");
955        }
956        if ((modifiers & InputEvent.ALT_GRAPH_MASK) != 0) {
957            buf.append(Toolkit.getProperty("AWT.altGraph", "Alt Graph"));
958            buf.append("+");
959        }
960        if ((modifiers & InputEvent.BUTTON1_MASK) != 0) {
961            buf.append(Toolkit.getProperty("AWT.button1", "Button1"));
962            buf.append("+");
963        }
964        if ((modifiers & InputEvent.BUTTON2_MASK) != 0) {
965            buf.append(Toolkit.getProperty("AWT.button2", "Button2"));
966            buf.append("+");
967        }
968        if ((modifiers & InputEvent.BUTTON3_MASK) != 0) {
969            buf.append(Toolkit.getProperty("AWT.button3", "Button3"));
970            buf.append("+");
971        }
972
973        int mask;
974
975        // TODO: add a toolkit field that holds a number of button on the mouse.
976        // As the method getMouseModifiersText() is static and obtain
977        // an integer as a parameter then we may not restrict this with the number
978        // of buttons installed on the mouse.
979        // It's a temporary solution. We need to somehow hold the number of buttons somewhere else.
980        for (int i = 1; i <= cachedNumberOfButtons; i++){
981            mask = InputEvent.getMaskForButton(i);
982            if ((modifiers & mask) != 0 &&
983                buf.indexOf(Toolkit.getProperty("AWT.button"+i, "Button"+i)) == -1) //1,2,3 buttons may already be there; so don't duplicate it.
984            {
985                buf.append(Toolkit.getProperty("AWT.button"+i, "Button"+i));
986                buf.append("+");
987            }
988        }
989
990        if (buf.length() > 0) {
991            buf.setLength(buf.length()-1); // remove trailing '+'
992        }
993        return buf.toString();
994    }

996    /**
997     * Returns a parameter string identifying this event.
998     * This method is useful for event-logging and for debugging.
999     *
1000     * @return a string identifying the event and its attributes
1001     */
1002    public String paramString() {
1003        StringBuilder str = new StringBuilder(80);
1004
1005        switch(id) {
1006          case MOUSE_PRESSED:
1007              str.append("MOUSE_PRESSED");
1008              break;
1009          case MOUSE_RELEASED:
1010              str.append("MOUSE_RELEASED");
1011              break;
1012          case MOUSE_CLICKED:
1013              str.append("MOUSE_CLICKED");
1014              break;
1015          case MOUSE_ENTERED:
1016              str.append("MOUSE_ENTERED");
1017              break;
1018          case MOUSE_EXITED:
1019              str.append("MOUSE_EXITED");
1020              break;
1021          case MOUSE_MOVED:
1022              str.append("MOUSE_MOVED");
1023              break;
1024          case MOUSE_DRAGGED:
1025              str.append("MOUSE_DRAGGED");
1026              break;
1027          case MOUSE_WHEEL:
1028              str.append("MOUSE_WHEEL");
1029              break;
1030           default:
1031              str.append("unknown type");
1032        }
1033
1034        // (x,y) coordinates
1035        str.append(",(").append(x).append(",").append(y).append(")");
1036        str.append(",absolute(").append(xAbs).append(",").append(yAbs).append(")");
1037
1038        if (id != MOUSE_DRAGGED && id != MOUSE_MOVED){
1039            str.append(",button=").append(getButton());
1040        }
1041
1042        if (getModifiers() != 0) {
1043            str.append(",modifiers=").append(getMouseModifiersText(modifiers));
1044        }
1045
1046        if (getModifiersEx() != 0) {
1047            //Using plain "modifiers" here does show an excluded extended buttons in the string event representation.
1048            //getModifiersEx() solves the problem.
1049            str.append(",extModifiers=").append(getModifiersExText(getModifiersEx()));
1050        }
1051
1052        str.append(",clickCount=").append(clickCount);
1053
1054        return str.toString();
1055    }

1057    /**
1058     * Sets new modifiers by the old ones.
1059     * Also sets button.
1060     */
1061    private void setNewModifiers() {
1062        if ((modifiers & BUTTON1_MASK) != 0) {
1063            modifiers |= BUTTON1_DOWN_MASK;
1064        }
1065        if ((modifiers & BUTTON2_MASK) != 0) {
1066            modifiers |= BUTTON2_DOWN_MASK;
1067        }
1068        if ((modifiers & BUTTON3_MASK) != 0) {
1069            modifiers |= BUTTON3_DOWN_MASK;
1070        }
1071        if (id == MOUSE_PRESSED
1072            || id == MOUSE_RELEASED
1073            || id == MOUSE_CLICKED)
1074        {
1075            if ((modifiers & BUTTON1_MASK) != 0) {
1076                button = BUTTON1;
1077                modifiers &= ~BUTTON2_MASK & ~BUTTON3_MASK;
1078                if (id != MOUSE_PRESSED) {
1079                    modifiers &= ~BUTTON1_DOWN_MASK;
1080                }
1081            } else if ((modifiers & BUTTON2_MASK) != 0) {
1082                button = BUTTON2;
1083                modifiers &= ~BUTTON1_MASK & ~BUTTON3_MASK;
1084                if (id != MOUSE_PRESSED) {
1085                    modifiers &= ~BUTTON2_DOWN_MASK;
1086                }
1087            } else if ((modifiers & BUTTON3_MASK) != 0) {
1088                button = BUTTON3;
1089                modifiers &= ~BUTTON1_MASK & ~BUTTON2_MASK;
1090                if (id != MOUSE_PRESSED) {
1091                    modifiers &= ~BUTTON3_DOWN_MASK;
1092                }
1093            }
1094        }
1095        if ((modifiers & InputEvent.ALT_MASK) != 0) {
1096            modifiers |= InputEvent.ALT_DOWN_MASK;
1097        }
1098        if ((modifiers & InputEvent.META_MASK) != 0) {
1099            modifiers |= InputEvent.META_DOWN_MASK;
1100        }
1101        if ((modifiers & InputEvent.SHIFT_MASK) != 0) {
1102            modifiers |= InputEvent.SHIFT_DOWN_MASK;
1103        }
1104        if ((modifiers & InputEvent.CTRL_MASK) != 0) {
1105            modifiers |= InputEvent.CTRL_DOWN_MASK;
1106        }
1107        if ((modifiers & InputEvent.ALT_GRAPH_MASK) != 0) {
1108            modifiers |= InputEvent.ALT_GRAPH_DOWN_MASK;
1109        }
1110    }

1112    /**
1113     * Sets old modifiers by the new ones.
1114     */
1115    private void setOldModifiers() {
1116        if (id == MOUSE_PRESSED
1117            || id == MOUSE_RELEASED
1118            || id == MOUSE_CLICKED)
1119        {
1120            switch(button) {
1121            case BUTTON1:
1122                modifiers |= BUTTON1_MASK;
1123                break;
1124            case BUTTON2:
1125                modifiers |= BUTTON2_MASK;
1126                break;
1127            case BUTTON3:
1128                modifiers |= BUTTON3_MASK;
1129                break;
1130            }
1131        } else {
1132            if ((modifiers & BUTTON1_DOWN_MASK) != 0) {
1133                modifiers |= BUTTON1_MASK;
1134            }
1135            if ((modifiers & BUTTON2_DOWN_MASK) != 0) {
1136                modifiers |= BUTTON2_MASK;
1137            }
1138            if ((modifiers & BUTTON3_DOWN_MASK) != 0) {
1139                modifiers |= BUTTON3_MASK;
1140            }
1141        }
1142        if ((modifiers & ALT_DOWN_MASK) != 0) {
1143            modifiers |= ALT_MASK;
1144        }
1145        if ((modifiers & META_DOWN_MASK) != 0) {
1146            modifiers |= META_MASK;
1147        }
1148        if ((modifiers & SHIFT_DOWN_MASK) != 0) {
1149            modifiers |= SHIFT_MASK;
1150        }
1151        if ((modifiers & CTRL_DOWN_MASK) != 0) {
1152            modifiers |= CTRL_MASK;
1153        }
1154        if ((modifiers & ALT_GRAPH_DOWN_MASK) != 0) {
1155            modifiers |= ALT_GRAPH_MASK;
1156        }
1157    }
