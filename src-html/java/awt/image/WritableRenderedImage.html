036package java.awt.image;
037import java.awt.Point;
038
039/**
040 * WriteableRenderedImage is a common interface for objects which
041 * contain or can produce image data in the form of Rasters and
042 * which can be modified and/or written over.  The image
043 * data may be stored/produced as a single tile or a regular array
044 * of tiles.
045 * <p>
046 * WritableRenderedImage provides notification to other interested
047 * objects when a tile is checked out for writing (via the
048 * getWritableTile method) and when the last writer of a particular
049 * tile relinquishes its access (via a call to releaseWritableTile).
050 * Additionally, it allows any caller to determine whether any tiles
051 * are currently checked out (via hasTileWriters), and to obtain a
052 * list of such tiles (via getWritableTileIndices, in the form of a Vector
053 * of Point objects).
054 * <p>
055 * Objects wishing to be notified of changes in tile writability must
056 * implement the TileObserver interface, and are added by a
057 * call to addTileObserver.  Multiple calls to
058 * addTileObserver for the same object will result in multiple
059 * notifications.  An existing observer may reduce its notifications
060 * by calling removeTileObserver; if the observer had no
061 * notifications the operation is a no-op.
062 * <p>
063 * It is necessary for a WritableRenderedImage to ensure that
064 * notifications occur only when the first writer acquires a tile and
065 * the last writer releases it.
066 *
067 */
068
069public interface WritableRenderedImage extends RenderedImage
070{
071
072  /**
073   * Adds an observer.  If the observer is already present,
074   * it will receive multiple notifications.
075   * @param to the specified <code>TileObserver</code>
076   */
077  public void addTileObserver(TileObserver to);
078
079  /**
080   * Removes an observer.  If the observer was not registered,
081   * nothing happens.  If the observer was registered for multiple
082   * notifications, it will now be registered for one fewer.
083   * @param to the specified <code>TileObserver</code>
084   */
085  public void removeTileObserver(TileObserver to);
086
087  /**
088   * Checks out a tile for writing.
089   *
090   * The WritableRenderedImage is responsible for notifying all
091   * of its TileObservers when a tile goes from having
092   * no writers to having one writer.
093   *
094   * @param tileX the X index of the tile.
095   * @param tileY the Y index of the tile.
096   * @return a writable tile.
097   */
098  public WritableRaster getWritableTile(int tileX, int tileY);
099
100  /**
101   * Relinquishes the right to write to a tile.  If the caller
102   * continues to write to the tile, the results are undefined.
103   * Calls to this method should only appear in matching pairs
104   * with calls to getWritableTile; any other use will lead
105   * to undefined results.
106   *
107   * The WritableRenderedImage is responsible for notifying all of
108   * its TileObservers when a tile goes from having one writer
109   * to having no writers.
110   *
111   * @param tileX the X index of the tile.
112   * @param tileY the Y index of the tile.
113   */
114  public void releaseWritableTile(int tileX, int tileY);
115
116  /**
117   * Returns whether a tile is currently checked out for writing.
118   *
119   * @param tileX the X index of the tile.
120   * @param tileY the Y index of the tile.
121   * @return <code>true</code> if specified tile is checked out
122   *         for writing; <code>false</code> otherwise.
123   */
124  public boolean isTileWritable(int tileX, int tileY);
125
126  /**
127   * Returns an array of Point objects indicating which tiles
128   * are checked out for writing.  Returns null if none are
129   * checked out.
130   * @return an array containing the locations of tiles that are
131   *         checked out for writing.
132   */
133  public Point[] getWritableTileIndices();
134
135  /**
136   * Returns whether any tile is checked out for writing.
137   * Semantically equivalent to (getWritableTileIndices() != null).
138   * @return <code>true</code> if any tiles are checked out for
139   *         writing; <code>false</code> otherwise.
140   */
141  public boolean hasTileWriters();
142
143  /**
144   * Sets a rect of the image to the contents of the Raster r, which is
145   * assumed to be in the same coordinate space as the WritableRenderedImage.
146   * The operation is clipped to the bounds of the WritableRenderedImage.
147   * @param r the specified <code>Raster</code>
148   */
149  public void setData(Raster r);
150
151}
