026package java.awt.image;
027
028/**
029 * The interface for objects which can produce the image data for Images.
030 * Each image contains an ImageProducer which is used to reconstruct
031 * the image whenever it is needed, for example, when a new size of the
032 * Image is scaled, or when the width or height of the Image is being
033 * requested.
034 *
035 * @see ImageConsumer
036 *
037 * @author      Jim Graham
038 */
039public interface ImageProducer {
040    /**
041     * Registers an <code>ImageConsumer</code> with the
042     * <code>ImageProducer</code> for access to the image data
043     * during a later reconstruction of the <code>Image</code>.
044     * The <code>ImageProducer</code> may, at its discretion,
045     * start delivering the image data to the consumer
046     * using the <code>ImageConsumer</code> interface immediately,
047     * or when the next available image reconstruction is triggered
048     * by a call to the <code>startProduction</code> method.
049     * @param ic the specified <code>ImageConsumer</code>
050     * @see #startProduction
051     */
052    public void addConsumer(ImageConsumer ic);
053
054    /**
055     * Determines if a specified <code>ImageConsumer</code>
056     * object is currently registered with this
057     * <code>ImageProducer</code> as one of its consumers.
058     * @param ic the specified <code>ImageConsumer</code>
059     * @return <code>true</code> if the specified
060     *         <code>ImageConsumer</code> is registered with
061     *         this <code>ImageProducer</code>;
062     *         <code>false</code> otherwise.
063     */
064    public boolean isConsumer(ImageConsumer ic);
065
066    /**
067     * Removes the specified <code>ImageConsumer</code> object
068     * from the list of consumers currently registered to
069     * receive image data.  It is not considered an error
070     * to remove a consumer that is not currently registered.
071     * The <code>ImageProducer</code> should stop sending data
072     * to this consumer as soon as is feasible.
073     * @param ic the specified <code>ImageConsumer</code>
074     */
075    public void removeConsumer(ImageConsumer ic);
076
077    /**
078     * Registers the specified <code>ImageConsumer</code> object
079     * as a consumer and starts an immediate reconstruction of
080     * the image data which will then be delivered to this
081     * consumer and any other consumer which might have already
082     * been registered with the producer.  This method differs
083     * from the addConsumer method in that a reproduction of
084     * the image data should be triggered as soon as possible.
085     * @param ic the specified <code>ImageConsumer</code>
086     * @see #addConsumer
087     */
088    public void startProduction(ImageConsumer ic);
089
090    /**
091     * Requests, on behalf of the <code>ImageConsumer</code>,
092     * that the <code>ImageProducer</code> attempt to resend
093     * the image data one more time in TOPDOWNLEFTRIGHT order
094     * so that higher quality conversion algorithms which
095     * depend on receiving pixels in order can be used to
096     * produce a better output version of the image.  The
097     * <code>ImageProducer</code> is free to
098     * ignore this call if it cannot resend the data in that
099     * order.  If the data can be resent, the
100     * <code>ImageProducer</code> should respond by executing
101     * the following minimum set of <code>ImageConsumer</code>
102     * method calls:
103     * <pre>
104     *  ic.setHints(TOPDOWNLEFTRIGHT | < otherhints >);
105     *  ic.setPixels(...);      // As many times as needed
106     *  ic.imageComplete();
107     * </pre>
108     * @param ic the specified <code>ImageConsumer</code>
109     * @see ImageConsumer#setHints
110     */
111    public void requestTopDownLeftRightResend(ImageConsumer ic);
112}
