025package java.awt;
026
027/**
028 * Defines an interface for classes that know how to layout Containers
029 * based on a layout constraints object.
030 *
031 * This interface extends the LayoutManager interface to deal with layouts
032 * explicitly in terms of constraint objects that specify how and where
033 * components should be added to the layout.
034 * <p>
035 * This minimal extension to LayoutManager is intended for tool
036 * providers who wish to the creation of constraint-based layouts.
037 * It does not yet provide full, general support for custom
038 * constraint-based layout managers.
039 *
040 * @see LayoutManager
041 * @see Container
042 *
043 * @author      Jonni Kanerva
044 */
045public interface LayoutManager2 extends LayoutManager {
046
047    /**
048     * Adds the specified component to the layout, using the specified
049     * constraint object.
050     * @param comp the component to be added
051     * @param constraints  where/how the component is added to the layout.
052     */
053    void addLayoutComponent(Component comp, Object constraints);
054
055    /**
056     * Calculates the maximum size dimensions for the specified container,
057     * given the components it contains.
058     * @see java.awt.Component#getMaximumSize
059     * @see LayoutManager
060     */
061    public Dimension maximumLayoutSize(Container target);
062
063    /**
064     * Returns the alignment along the x axis.  This specifies how
065     * the component would like to be aligned relative to other
066     * components.  The value should be a number between 0 and 1
067     * where 0 represents alignment along the origin, 1 is aligned
068     * the furthest away from the origin, 0.5 is centered, etc.
069     */
070    public float getLayoutAlignmentX(Container target);
071
072    /**
073     * Returns the alignment along the y axis.  This specifies how
074     * the component would like to be aligned relative to other
075     * components.  The value should be a number between 0 and 1
076     * where 0 represents alignment along the origin, 1 is aligned
077     * the furthest away from the origin, 0.5 is centered, etc.
078     */
079    public float getLayoutAlignmentY(Container target);
080
081    /**
082     * Invalidates the layout, indicating that if the layout manager
083     * has cached information it should be discarded.
084     */
085    public void invalidateLayout(Container target);
086
087}
