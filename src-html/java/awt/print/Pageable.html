026package java.awt.print;
027
028/**
029 * The <code>Pageable</code> implementation represents a set of
030 * pages to be printed. The <code>Pageable</code> object returns
031 * the total number of pages in the set as well as the
032 * {@link PageFormat} and {@link Printable} for a specified page.
033 * @see java.awt.print.PageFormat
034 * @see java.awt.print.Printable
035 */
036public interface Pageable {
037
038    /**
039     * This constant is returned from the
040     * {@link #getNumberOfPages() getNumberOfPages}
041     * method if a <code>Pageable</code> implementation does not know
042     * the number of pages in its set.
043     */
044    int UNKNOWN_NUMBER_OF_PAGES = -1;
045
046    /**
047     * Returns the number of pages in the set.
048     * To enable advanced printing features,
049     * it is recommended that <code>Pageable</code>
050     * implementations return the true number of pages
051     * rather than the
052     * UNKNOWN_NUMBER_OF_PAGES constant.
053     * @return the number of pages in this <code>Pageable</code>.
054     */
055    int getNumberOfPages();
056
057    /**
058     * Returns the <code>PageFormat</code> of the page specified by
059     * <code>pageIndex</code>.
060     * @param pageIndex the zero based index of the page whose
061     *            <code>PageFormat</code> is being requested
062     * @return the <code>PageFormat</code> describing the size and
063     *          orientation.
064     * @throws IndexOutOfBoundsException if
065     *          the <code>Pageable</code> does not contain the requested
066     *          page.
067     */
068    PageFormat getPageFormat(int pageIndex)
069        throws IndexOutOfBoundsException;
070
071    /**
072     * Returns the <code>Printable</code> instance responsible for
073     * rendering the page specified by <code>pageIndex</code>.
074     * @param pageIndex the zero based index of the page whose
075     *            <code>Printable</code> is being requested
076     * @return the <code>Printable</code> that renders the page.
077     * @throws IndexOutOfBoundsException if
078     *            the <code>Pageable</code> does not contain the requested
079     *            page.
080     */
081    Printable getPrintable(int pageIndex)
082        throws IndexOutOfBoundsException;
083}
