025package java.awt.font;
026
027import java.awt.Font;
028
029/**
030 * The <code>MultipleMaster</code> interface represents Type 1
031 * Multiple Master fonts.
032 * A particular {@link Font} object can implement this interface.
033 */
034public interface MultipleMaster {
035
036  /**
037   * Returns the number of multiple master design controls.
038   * Design axes include things like width, weight and optical scaling.
039   * @return the number of multiple master design controls
040   */
041  public  int getNumDesignAxes();
042
043  /**
044   * Returns an array of design limits interleaved in the form [from->to]
045   * for each axis.  For example,
046   * design limits for weight could be from 0.1 to 1.0. The values are
047   * returned in the same order returned by
048   * <code>getDesignAxisNames</code>.
049   * @return an array of design limits for each axis.
050   */
051  public  float[]  getDesignAxisRanges();
052
053  /**
054   * Returns an array of default design values for each axis.  For example,
055   * the default value for weight could be 1.6. The values are returned
056   * in the same order returned by <code>getDesignAxisNames</code>.
057   * @return an array of default design values for each axis.
058   */
059  public  float[]  getDesignAxisDefaults();
060
061  /**
062   * Returns the name for each design axis. This also determines the order in
063   * which the values for each axis are returned.
064   * @return an array containing the names of each design axis.
065   */
066  public  String[] getDesignAxisNames();
067
068  /**
069   * Creates a new instance of a multiple master font based on the design
070   * axis values contained in the specified array. The size of the array
071   * must correspond to the value returned from
072   * <code>getNumDesignAxes</code> and the values of the array elements
073   * must fall within limits specified by
074   * <code>getDesignAxesLimits</code>. In case of an error,
075   * <code>null</code> is returned.
076   * @param axes an array containing axis values
077   * @return a {@link Font} object that is an instance of
078   * <code>MultipleMaster</code> and is based on the design axis values
079   * provided by <code>axes</code>.
080   */
081  public Font deriveMMFont(float[] axes);
082
083  /**
084   * Creates a new instance of a multiple master font based on detailed metric
085   * information. In case of an error, <code>null</code> is returned.
086   * @param glyphWidths an array of floats representing the desired width
087   * of each glyph in font space
088   * @param avgStemWidth the average stem width for the overall font in
089   * font space
090   * @param typicalCapHeight the height of a typical upper case char
091   * @param typicalXHeight the height of a typical lower case char
092   * @param italicAngle the angle at which the italics lean, in degrees
093   * counterclockwise from vertical
094   * @return a <code>Font</code> object that is an instance of
095   * <code>MultipleMaster</code> and is based on the specified metric
096   * information.
097   */
098  public Font deriveMMFont(
099                                   float[] glyphWidths,
100                                   float avgStemWidth,
101                                   float typicalCapHeight,
102                                   float typicalXHeight,
103                                   float italicAngle);
104
105
106}
