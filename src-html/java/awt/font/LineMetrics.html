026package java.awt.font;
027
028/**
029* The <code>LineMetrics</code> class allows access to the
030* metrics needed to layout characters along a line
031* and to layout of a set of lines.  A <code>LineMetrics</code>
032* object encapsulates the measurement information associated
033* with a run of text.
034* <p>
035* Fonts can have different metrics for different ranges of
036* characters.  The <code>getLineMetrics</code> methods of
037* {@link java.awt.Font Font} take some text as an argument
038* and return a <code>LineMetrics</code> object describing the
039* metrics of the initial number of characters in that text, as
040* returned by {@link #getNumChars}.
041*/
042
043
044public abstract class LineMetrics {
045
046
047    /**
048     * Returns the number of characters (<code>char</code> values) in the text whose
049     * metrics are encapsulated by this <code>LineMetrics</code>
050     * object.
051     * @return the number of characters (<code>char</code> values) in the text with which
052     *         this <code>LineMetrics</code> was created.
053     */
054    public abstract int getNumChars();
055
056    /**
057     * Returns the ascent of the text.  The ascent
058     * is the distance from the baseline
059     * to the ascender line.  The ascent usually represents the
060     * the height of the capital letters of the text.  Some characters
061     * can extend above the ascender line.
062     * @return the ascent of the text.
063     */
064    public abstract float getAscent();
065
066    /**
067     * Returns the descent of the text.  The descent
068     * is the distance from the baseline
069     * to the descender line.  The descent usually represents
070     * the distance to the bottom of lower case letters like
071     * 'p'.  Some characters can extend below the descender
072     * line.
073     * @return the descent of the text.
074     */
075    public abstract float getDescent();
076
077    /**
078     * Returns the leading of the text. The
079     * leading is the recommended
080     * distance from the bottom of the descender line to the
081     * top of the next line.
082     * @return the leading of the text.
083     */
084    public abstract float getLeading();
085
086    /**
087     * Returns the height of the text.  The
088     * height is equal to the sum of the ascent, the
089     * descent and the leading.
090     * @return the height of the text.
091     */
092    public abstract float getHeight();
093
094    /**
095     * Returns the baseline index of the text.
096     * The index is one of
097     * {@link java.awt.Font#ROMAN_BASELINE ROMAN_BASELINE},
098     * {@link java.awt.Font#CENTER_BASELINE CENTER_BASELINE},
099     * {@link java.awt.Font#HANGING_BASELINE HANGING_BASELINE}.
100     * @return the baseline of the text.
101     */
102    public abstract int getBaselineIndex();
103
104    /**
105     * Returns the baseline offsets of the text,
106     * relative to the baseline of the text.  The
107     * offsets are indexed by baseline index.  For
108     * example, if the baseline index is
109     * <code>CENTER_BASELINE</code> then
110     * <code>offsets[HANGING_BASELINE]</code> is usually
111     * negative, <code>offsets[CENTER_BASELINE]</code>
112     * is zero, and <code>offsets[ROMAN_BASELINE]</code>
113     * is usually positive.
114     * @return the baseline offsets of the text.
115     */
116    public abstract float[] getBaselineOffsets();
117
118    /**
119     * Returns the position of the strike-through line
120     * relative to the baseline.
121     * @return the position of the strike-through line.
122     */
123    public abstract float getStrikethroughOffset();
124
125    /**
126     * Returns the thickness of the strike-through line.
127     * @return the thickness of the strike-through line.
128     */
129    public abstract float getStrikethroughThickness();
130
131    /**
132     * Returns the position of the underline relative to
133     * the baseline.
134     * @return the position of the underline.
135     */
136    public abstract float getUnderlineOffset();
137
138    /**
139     * Returns the thickness of the underline.
140     * @return the thickness of the underline.
141     */
142    public abstract float getUnderlineThickness();
143}
