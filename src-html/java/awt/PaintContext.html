026package java.awt;
027
028import java.awt.image.Raster;
029import java.awt.image.ColorModel;
030
031/**
032 * The <code>PaintContext</code> interface defines the encapsulated
033 * and optimized environment to generate color patterns in device
034 * space for fill or stroke operations on a
035 * {@link Graphics2D}.  The <code>PaintContext</code> provides
036 * the necessary colors for <code>Graphics2D</code> operations in the
037 * form of a {@link Raster} associated with a {@link ColorModel}.
038 * The <code>PaintContext</code> maintains state for a particular paint
039 * operation.  In a multi-threaded environment, several
040 * contexts can exist simultaneously for a single {@link Paint} object.
041 * @see Paint
042 */
043
044public interface PaintContext {
045    /**
046     * Releases the resources allocated for the operation.
047     */
048    public void dispose();
049
050    /**
051     * Returns the <code>ColorModel</code> of the output.  Note that
052     * this <code>ColorModel</code> might be different from the hint
053     * specified in the
054     * {@link Paint#createContext(ColorModel, Rectangle, Rectangle2D,
055AffineTransform, RenderingHints) createContext} method of
056     * <code>Paint</code>.  Not all <code>PaintContext</code> objects are
057     * capable of generating color patterns in an arbitrary
058     * <code>ColorModel</code>.
059     * @return the <code>ColorModel</code> of the output.
060     */
061    ColorModel getColorModel();
062
063    /**
064     * Returns a <code>Raster</code> containing the colors generated for
065     * the graphics operation.
066     * @param x the x coordinate of the area in device space
067     * for which colors are generated.
068     * @param y the y coordinate of the area in device space
069     * for which colors are generated.
070     * @param w the width of the area in device space
071     * @param h the height of the area in device space
072     * @return a <code>Raster</code> representing the specified
073     * rectangular area and containing the colors generated for
074     * the graphics operation.
075     */
076    Raster getRaster(int x,
077                     int y,
078                     int w,
079                     int h);
080
081}
