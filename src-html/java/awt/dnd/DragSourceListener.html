026package java.awt.dnd;
027
028import java.util.EventListener;
029
030/**
031 * The <code>DragSourceListener</code> defines the
032 * event interface for originators of
033 * Drag and Drop operations to track the state of the user's gesture, and to
034 * provide appropriate &quot;drag over&quot;
035 * feedback to the user throughout the
036 * Drag and Drop operation.
037 * <p>
038 * The drop site is <i>associated with the previous <code>dragEnter()</code>
039 * invocation</i> if the latest invocation of <code>dragEnter()</code> on this
040 * listener:
041 * <ul>
042 * <li>corresponds to that drop site and
043 * <li> is not followed by a <code>dragExit()</code> invocation on this listener.
044 * </ul>
045 *
046 * @since 1.2
047 */
048
049public interface DragSourceListener extends EventListener {
050
051    /**
052     * Called as the cursor's hotspot enters a platform-dependent drop site.
053     * This method is invoked when all the following conditions are true:
054     * <UL>
055     * <LI>The cursor's hotspot enters the operable part of a platform-
056     * dependent drop site.
057     * <LI>The drop site is active.
058     * <LI>The drop site accepts the drag.
059     * </UL>
060     *
061     * @param dsde the <code>DragSourceDragEvent</code>
062     */
063    void dragEnter(DragSourceDragEvent dsde);
064
065    /**
066     * Called as the cursor's hotspot moves over a platform-dependent drop site.
067     * This method is invoked when all the following conditions are true:
068     * <UL>
069     * <LI>The cursor's hotspot has moved, but still intersects the
070     * operable part of the drop site associated with the previous
071     * dragEnter() invocation.
072     * <LI>The drop site is still active.
073     * <LI>The drop site accepts the drag.
074     * </UL>
075     *
076     * @param dsde the <code>DragSourceDragEvent</code>
077     */
078    void dragOver(DragSourceDragEvent dsde);
079
080    /**
081     * Called when the user has modified the drop gesture.
082     * This method is invoked when the state of the input
083     * device(s) that the user is interacting with changes.
084     * Such devices are typically the mouse buttons or keyboard
085     * modifiers that the user is interacting with.
086     *
087     * @param dsde the <code>DragSourceDragEvent</code>
088     */
089    void dropActionChanged(DragSourceDragEvent dsde);
090
091    /**
092     * Called as the cursor's hotspot exits a platform-dependent drop site.
093     * This method is invoked when any of the following conditions are true:
094     * <UL>
095     * <LI>The cursor's hotspot no longer intersects the operable part
096     * of the drop site associated with the previous dragEnter() invocation.
097     * </UL>
098     * OR
099     * <UL>
100     * <LI>The drop site associated with the previous dragEnter() invocation
101     * is no longer active.
102     * </UL>
103     * OR
104     * <UL>
105     * <LI> The drop site associated with the previous dragEnter() invocation
106     * has rejected the drag.
107     * </UL>
108     *
109     * @param dse the <code>DragSourceEvent</code>
110     */
111    void dragExit(DragSourceEvent dse);
112
113    /**
114     * This method is invoked to signify that the Drag and Drop
115     * operation is complete. The getDropSuccess() method of
116     * the <code>DragSourceDropEvent</code> can be used to
117     * determine the termination state. The getDropAction() method
118     * returns the operation that the drop site selected
119     * to apply to the Drop operation. Once this method is complete, the
120     * current <code>DragSourceContext</code> and
121     * associated resources become invalid.
122     *
123     * @param dsde the <code>DragSourceDropEvent</code>
124     */
125    void dragDropEnd(DragSourceDropEvent dsde);
126}
