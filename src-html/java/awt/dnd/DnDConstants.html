026package java.awt.dnd;
027
028/**
029 * This class contains constant values representing
030 * the type of action(s) to be performed by a Drag and Drop operation.
031 * @since 1.2
032 */
033
034public final class DnDConstants {
035
036    private DnDConstants() {} // define null private constructor.
037
038    /**
039     * An <code>int</code> representing no action.
040     */
041    public static final int ACTION_NONE         = 0x0;
042
043    /**
044     * An <code>int</code> representing a &quot;copy&quot; action.
045     */
046    public static final int ACTION_COPY         = 0x1;
047
048    /**
049     * An <code>int</code> representing a &quot;move&quot; action.
050     */
051    public static final int ACTION_MOVE         = 0x2;
052
053    /**
054     * An <code>int</code> representing a &quot;copy&quot; or
055     * &quot;move&quot; action.
056     */
057    public static final int ACTION_COPY_OR_MOVE = ACTION_COPY | ACTION_MOVE;
058
059    /**
060     * An <code>int</code> representing a &quot;link&quot; action.
061     *
062     * The link verb is found in many, if not all native DnD platforms, and the
063     * actual interpretation of LINK semantics is both platform
064     * and application dependent. Broadly speaking, the
065     * semantic is "do not copy, or move the operand, but create a reference
066     * to it". Defining the meaning of "reference" is where ambiguity is
067     * introduced.
068     *
069     * The verb is provided for completeness, but its use is not recommended
070     * for DnD operations between logically distinct applications where
071     * misinterpretation of the operations semantics could lead to confusing
072     * results for the user.
073     */
074
075    public static final int ACTION_LINK         = 0x40000000;
076
077    /**
078     * An <code>int</code> representing a &quot;reference&quot;
079     * action (synonym for ACTION_LINK).
080     */
081    public static final int ACTION_REFERENCE    = ACTION_LINK;
082
083}
