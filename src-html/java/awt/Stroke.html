026package java.awt;
027
028/**
029 * The <code>Stroke</code> interface allows a
030 * {@link Graphics2D} object to obtain a {@link Shape} that is the
031 * decorated outline, or stylistic representation of the outline,
032 * of the specified <code>Shape</code>.
033 * Stroking a <code>Shape</code> is like tracing its outline with a
034 * marking pen of the appropriate size and shape.
035 * The area where the pen would place ink is the area enclosed by the
036 * outline <code>Shape</code>.
037 * <p>
038 * The methods of the <code>Graphics2D</code> interface that use the
039 * outline <code>Shape</code> returned by a <code>Stroke</code> object
040 * include <code>draw</code> and any other methods that are
041 * implemented in terms of that method, such as
042 * <code>drawLine</code>, <code>drawRect</code>,
043 * <code>drawRoundRect</code>, <code>drawOval</code>,
044 * <code>drawArc</code>, <code>drawPolyline</code>,
045 * and <code>drawPolygon</code>.
046 * <p>
047 * The objects of the classes implementing <code>Stroke</code>
048 * must be read-only because <code>Graphics2D</code> does not
049 * clone these objects either when they are set as an attribute
050 * with the <code>setStroke</code> method or when the
051 * <code>Graphics2D</code> object is itself cloned.
052 * If a <code>Stroke</code> object is modified after it is set in
053 * the <code>Graphics2D</code> context then the behavior
054 * of subsequent rendering would be undefined.
055 * @see BasicStroke
056 * @see Graphics2D#setStroke
057 */
058public interface Stroke {
059    /**
060     * Returns an outline <code>Shape</code> which encloses the area that
061     * should be painted when the <code>Shape</code> is stroked according
062     * to the rules defined by the
063     * object implementing the <code>Stroke</code> interface.
064     * @param p a <code>Shape</code> to be stroked
065     * @return the stroked outline <code>Shape</code>.
066     */
067    Shape createStrokedShape (Shape p);
068}
