026package javax.sound.sampled;
027
028/**
029 * {@link Line Lines} often have a set of controls, such as gain and pan, that affect
030 * the audio signal passing through the line.  Java Sound's <code>Line</code> objects
031 * let you obtain a particular control object by passing its class as the
032 * argument to a {@link Line#getControl(Control.Type) getControl} method.
033 * <p>
034 * Because the various types of controls have different purposes and features,
035 * all of their functionality is accessed from the subclasses that define
036 * each kind of control.
037 *
038 * @author Kara Kytle
039 *
040 * @see Line#getControls
041 * @see Line#isControlSupported
042 * @since 1.3
043 */
044public abstract class Control {
045
046
047    // INSTANCE VARIABLES
048
049    /**
050     * The control type.
051     */
052    private final Type type;
053
054
055
056    // CONSTRUCTORS
057
058    /**
059     * Constructs a Control with the specified type.
060     * @param type the kind of control desired
061     */
062    protected Control(Type type) {
063        this.type = type;
064    }
065
066
067    // METHODS
068
069    /**
070     * Obtains the control's type.
071     * @return the control's type.
072     */
073    public Type getType() {
074        return type;
075    }
076
077
078    // ABSTRACT METHODS
079
080    /**
081     * Obtains a String describing the control type and its current state.
082     * @return a String representation of the Control.
083     */
084    public String toString() {
085        return new String(getType() + " Control");
086    }
087
088
089    /**
090     * An instance of the <code>Type</code> class represents the type of
091     * the control.  Static instances are provided for the
092     * common types.
093     */
094    public static class Type {
095
096        // CONTROL TYPE DEFINES
097
098        // INSTANCE VARIABLES
099
100        /**
101         * Type name.
102         */
103        private String name;
104
105
106        // CONSTRUCTOR
107
108        /**
109         * Constructs a new control type with the name specified.
110         * The name should be a descriptive string appropriate for
111         * labelling the control in an application, such as "Gain" or "Balance."
112         * @param name  the name of the new control type.
113         */
114        protected Type(String name) {
115            this.name = name;
116        }
117
118
119        // METHODS
120
121        /**
122         * Finalizes the equals method
123         */
124        public final boolean equals(Object obj) {
125            return super.equals(obj);
126        }
127
128        /**
129         * Finalizes the hashCode method
130         */
131        public final int hashCode() {
132            return super.hashCode();
133        }
134
135        /**
136         * Provides the <code>String</code> representation of the control type.  This <code>String</code> is
137         * the same name that was passed to the constructor.
138         *
139         * @return the control type name
140         */
141        public final String toString() {
142            return name;
143        }
144    } // class Type
145
146} // class Control
