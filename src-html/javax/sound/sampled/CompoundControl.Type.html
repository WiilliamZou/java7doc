026package javax.sound.sampled;
027
028/**
029 * A <code>CompoundControl</code>, such as a graphic equalizer, provides control
030 * over two or more related properties, each of which is itself represented as
031 * a <code>Control</code>.
032 *
033 * @author Kara Kytle
034 * @since 1.3
035 */
036public abstract class CompoundControl extends Control {
037
038
039    // TYPE DEFINES
040
041
042    // INSTANCE VARIABLES
043
044
045    /**
046     * The set of member controls.
047     */
048    private Control[] controls;
049
050
051
052    // CONSTRUCTORS
053
054
055    /**
056     * Constructs a new compound control object with the given parameters.
057     *
058     * @param type the type of control represented this compound control object
059     * @param memberControls the set of member controls
060     */
061    protected CompoundControl(Type type, Control[] memberControls) {
062
063        super(type);
064        this.controls = memberControls;
065    }
066
067
068
069    // METHODS
070
071
072    /**
073     * Returns the set of member controls that comprise the compound control.
074     * @return the set of member controls.
075     */
076    public Control[] getMemberControls() {
077
078        Control[] localArray = new Control[controls.length];
079
080        for (int i = 0; i < controls.length; i++) {
081            localArray[i] = controls[i];
082        }
083
084        return localArray;
085    }
086
087
088    // ABSTRACT METHOD IMPLEMENTATIONS: CONTROL
089
090
091    /**
092     * Provides a string representation of the control
093     * @return a string description
094     */
095    public String toString() {
096
097        StringBuffer buf = new StringBuffer();
098        for (int i = 0; i < controls.length; i++) {
099            if (i != 0) {
100                buf.append(", ");
101                if ((i + 1) == controls.length) {
102                    buf.append("and ");
103                }
104            }
105            buf.append(controls[i].getType());
106        }
107
108        return new String(getType() + " Control containing " + buf + " Controls.");
109    }
110
111
112    // INNER CLASSES
113
114
115    /**
116     * An instance of the <code>CompoundControl.Type</code> inner class identifies one kind of
117     * compound control.  Static instances are provided for the
118     * common types.
119     *
120     * @author Kara Kytle
121     * @since 1.3
122     */
123    public static class Type extends Control.Type {
124
125
126        // TYPE DEFINES
127
128        // CONSTRUCTOR
129
130
131        /**
132         * Constructs a new compound control type.
133         * @param name  the name of the new compound control type
134         */
135        protected Type(String name) {
136            super(name);
137        }
138    } // class Type
139
140} // class CompoundControl
