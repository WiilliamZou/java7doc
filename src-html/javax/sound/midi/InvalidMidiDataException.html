026package javax.sound.midi;
027
028
029/**
030 * An <code>InvalidMidiDataException</code> indicates that inappropriate MIDI
031 * data was encountered. This often means that the data is invalid in and of
032 * itself, from the perspective of the MIDI specification.  An example would
033 * be an undefined status byte.  However, the exception might simply
034 * mean that the data was invalid in the context it was used, or that
035 * the object to which the data was given was unable to parse or use it.
036 * For example, a file reader might not be able to parse a Type 2 MIDI file, even
037 * though that format is defined in the MIDI specification.
038 *
039 * @author Kara Kytle
040 */
041public class InvalidMidiDataException extends Exception {
042
043    /**
044     * Constructs an <code>InvalidMidiDataException</code> with
045     * <code>null</code> for its error detail message.
046     */
047    public InvalidMidiDataException() {
048
049        super();
050    }
051
052
053    /**
054     *  Constructs an <code>InvalidMidiDataException</code> with the
055     * specified detail message.
056     *
057     * @param message the string to display as an error detail message
058     */
059    public InvalidMidiDataException(String message) {
060
061        super(message);
062    }
063}
