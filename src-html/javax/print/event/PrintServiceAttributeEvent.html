026package javax.print.event;
027
028import javax.print.PrintService;
029import javax.print.attribute.AttributeSetUtilities;
030import javax.print.attribute.PrintServiceAttributeSet;
031
032/**
033 *
034 * Class PrintServiceAttributeEvent encapsulates an event a
035 * Print Service instance reports to let the client know of
036 * changes in the print service state.
037 */
038
039public class PrintServiceAttributeEvent extends PrintEvent {
040
041    private static final long serialVersionUID = -7565987018140326600L;
042
043    private PrintServiceAttributeSet attributes;
044
045    /**
046     * Constructs a PrintServiceAttributeEvent object.
047     *
048     * @param source the print job generating  this event
049     * @param attributes the attribute changes being reported
050     * @throws IllegalArgumentException if <code>source</code> is
051     *         <code>null</code>.
052     */
053    public PrintServiceAttributeEvent(PrintService source,
054                                      PrintServiceAttributeSet attributes) {
055
056        super(source);
057        this.attributes = AttributeSetUtilities.unmodifiableView(attributes);
058    }
059
060
061    /**
062     * Returns the print service.
063
064     * @return  Print Service object.
065     */
066    public PrintService getPrintService() {
067
068        return (PrintService) getSource();
069    }
070
071
072    /**
073     * Determine the printing service attributes that changed and their new
074     * values.
075     *
076     * @return  Attributes containing the new values for the service
077     * attributes that changed. The returned set may be unmodifiable.
078     */
079    public PrintServiceAttributeSet getAttributes() {
080
081        return attributes;
082    }
083
084}
