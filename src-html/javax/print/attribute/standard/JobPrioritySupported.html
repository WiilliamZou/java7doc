025package javax.print.attribute.standard;
026
027import javax.print.attribute.Attribute;
028import javax.print.attribute.IntegerSyntax;
029import javax.print.attribute.SupportedValuesAttribute;
030
031/**
032 * Class JobPrioritySupported is an integer valued printing attribute class
033 * that specifies whether a Print Service instance supports the {@link
034 * JobPriority JobPriority} attribute and the number of different job priority
035 * levels supported.
036 * <P>
037 * The client can always specify any {@link JobPriority JobPriority} value
038 * from 1 to 100 for a job. However, the Print Service instance may support
039 * fewer than 100 different job priority levels. If this is the case, the
040 * Print Service instance automatically maps the client-specified job priority
041 * value to one of the supported job priority levels, dividing the 100 job
042 * priority values equally among the available job priority levels.
043 * <P>
044 * <B>IPP Compatibility:</B> The integer value gives the IPP integer value.
045 * The category name returned by <CODE>getName()</CODE> gives the IPP
046 * attribute name.
047 * <P>
048 *
049 * @author  Alan Kaminsky
050 */
051public final class JobPrioritySupported extends IntegerSyntax
052    implements SupportedValuesAttribute {
053
054    private static final long serialVersionUID = 2564840378013555894L;
055
056
057    /**
058     * Construct a new job priority supported attribute with the given integer
059     * value.
060     *
061     * @param  value  Number of different job priority levels supported.
062     *
063     * @exception  IllegalArgumentException
064     *     (Unchecked exception) Thrown if <CODE>value</CODE> is less than 1
065     *     or greater than 100.
066     */
067    public JobPrioritySupported(int value) {
068        super (value, 1, 100);
069    }
070
071    /**
072     * Returns whether this job priority supported attribute is equivalent to
073     * the passed in object. To be equivalent, all of the following conditions
074     * must be true:
075     * <OL TYPE=1>
076     * <LI>
077     * <CODE>object</CODE> is not null.
078     * <LI>
079     * <CODE>object</CODE> is an instance of class JobPrioritySupported.
080     * <LI>
081     * This job priority supported attribute's value and
082     * <CODE>object</CODE>'s value are equal.
083     * </OL>
084     *
085     * @param  object  Object to compare to.
086     *
087     * @return  True if <CODE>object</CODE> is equivalent to this job
088     *          priority supported attribute, false otherwise.
089     */
090    public boolean equals (Object object) {
091
092        return (super.equals(object) &&
093               object instanceof JobPrioritySupported);
094    }
095
096
097    /**
098     * Get the printing attribute class which is to be used as the "category"
099     * for this printing attribute value.
100     * <P>
101     * For class JobPrioritySupported, the
102     * category is class JobPrioritySupported itself.
103     *
104     * @return  Printing attribute class (category), an instance of class
105     *          {@link java.lang.Class java.lang.Class}.
106     */
107    public final Class<? extends Attribute> getCategory() {
108        return JobPrioritySupported.class;
109    }
110
111    /**
112     * Get the name of the category of which this attribute value is an
113     * instance.
114     * <P>
115     * For class JobPrioritySupported, the
116     * category name is <CODE>"job-priority-supported"</CODE>.
117     *
118     * @return  Attribute category name.
119     */
120    public final String getName() {
121        return "job-priority-supported";
122    }
123
124}
