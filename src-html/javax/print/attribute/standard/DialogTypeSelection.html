026package javax.print.attribute.standard;
027
028import javax.print.attribute.EnumSyntax;
029import javax.print.attribute.PrintRequestAttribute;
030
031/**
032 * Class DialogTypeSelection is a printing attribute class, an enumeration,
033 * that indicates the user dialog type to be used for specifying
034 * printing options.
035 * If {@code NATIVE} is specified, then where available, a native
036 * platform dialog is displayed.
037 * If {@code COMMON} is specified, a cross-platform print dialog is displayed.
038 *
039 * This option to specify a native dialog for use with an IPP attribute
040 * set provides a standard way to reflect back of the setting and option
041 * changes made by a user to the calling application, and integrates
042 * the native dialog into the Java printing APIs.
043 * But note that some options and settings in a native dialog may not
044 * necessarily map to IPP attributes as they may be non-standard platform,
045 * or even printer specific options.
046 * <P>
047 * <B>IPP Compatibility:</B> This is not an IPP attribute.
048 * <P>
049 * @since 1.7
050 *
051 */
052public final class DialogTypeSelection extends EnumSyntax
053        implements PrintRequestAttribute {
054
055    private static final long serialVersionUID = 7518682952133256029L;
056
057    /**
058     *
059     */
060    public static final DialogTypeSelection
061        NATIVE = new DialogTypeSelection(0);
062
063    /**
064     *
065     */
066    public static final DialogTypeSelection
067        COMMON = new DialogTypeSelection(1);
068
069    /**
070     * Construct a new dialog type selection enumeration value with the
071     * given integer value.
072     *
073     * @param  value  Integer value.
074     */
075    protected DialogTypeSelection(int value) {
076                super(value);
077    }
078
079    private static final String[] myStringTable = {
080        "native", "common"};
081
082
083    private static final DialogTypeSelection[] myEnumValueTable = {
084        NATIVE,
085        COMMON
086    };
087
088    /**
089     * Returns the string table for class DialogTypeSelection.
090     */
091    protected String[] getStringTable() {
092        return myStringTable;
093    }
094
095    /**
096     * Returns the enumeration value table for class DialogTypeSelection.
097     */
098    protected EnumSyntax[] getEnumValueTable() {
099        return myEnumValueTable;
100    }
101
102
103   /**
104     * Get the printing attribute class which is to be used as the "category"
105     * for this printing attribute value.
106     * <P>
107     * For class DialogTypeSelection the category is class
108     * DialogTypeSelection itself.
109     *
110     * @return  Printing attribute class (category), an instance of class
111     *          {@link java.lang.Class java.lang.Class}.
112     */
113    public final Class getCategory() {
114        return DialogTypeSelection.class;
115    }
116
117
118    /**
119     * Get the name of the category of which this attribute value is an
120     * instance.
121     * <P>
122     * For class DialogTypeSelection the category name is
123     * <CODE>"dialog-type-selection"</CODE>.

125     * @return  Attribute category name.

129    }
