025package javax.print.attribute.standard;
026
027import javax.print.attribute.Attribute;
028import javax.print.attribute.SetOfIntegerSyntax;
029import javax.print.attribute.SupportedValuesAttribute;
030
031/**
032 * Class JobImpressionsSupported is a printing attribute class, a set of
033 * integers, that gives the supported values for a {@link JobImpressions
034 * JobImpressions} attribute. It is restricted to a single contiguous range of
035 * integers; multiple non-overlapping ranges are not allowed. This gives the
036 * lower and upper bounds of the total sizes of print jobs in number of
037 * impressions that the printer will accept.
038 * <P>
039 * <B>IPP Compatibility:</B> The JobImpressionsSupported attribute's canonical
040 * array form gives the lower and upper bound for the range of values to be
041 * included in an IPP "job-impressions-supported" attribute. See class {@link
042 * javax.print.attribute.SetOfIntegerSyntax SetOfIntegerSyntax} for an
043 * explanation of canonical array form. The category name returned by
044 * <CODE>getName()</CODE> gives the IPP attribute name.
045 * <P>
046 *
047 * @author  Alan Kaminsky
048 */
049public final class JobImpressionsSupported extends SetOfIntegerSyntax
050        implements SupportedValuesAttribute {
051
052    private static final long serialVersionUID = -4887354803843173692L;
053
054
055    /**
056     * Construct a new job impressions supported attribute containing a single
057     * range of integers. That is, only those values of JobImpressions in the
058     * one range are supported.
059     *
060     * @param  lowerBound  Lower bound of the range.
061     * @param  upperBound  Upper bound of the range.
062     *
063     * @exception  IllegalArgumentException
064     *     (Unchecked exception) Thrown if a null range is specified or if a
065     *     non-null range is specified with <CODE>lowerBound</CODE> less than
066     *     0.
067     */
068    public JobImpressionsSupported(int lowerBound, int upperBound) {
069        super (lowerBound, upperBound);
070        if (lowerBound > upperBound) {
071            throw new IllegalArgumentException("Null range specified");
072        } else if (lowerBound < 0) {
073            throw new IllegalArgumentException(
074                                         "Job K octets value < 0 specified");
075        }
076    }
077
078
079    /**
080     * Returns whether this job impressions supported attribute is equivalent
081     * to the passed in object. To be equivalent, all of the following
082     * conditions must be true:
083     * <OL TYPE=1>
084     * <LI>
085     * <CODE>object</CODE> is not null.
086     * <LI>
087     * <CODE>object</CODE> is an instance of class JobImpressionsSupported.
088     * <LI>
089     * This job impressions supported attribute's members and
090     * <CODE>object</CODE>'s members are the same.
091     * </OL>
092     *
093     * @param  object  Object to compare to.
094     *
095     * @return  True if <CODE>object</CODE> is equivalent to this job
096     *          impressions supported attribute, false otherwise.
097     */
098    public boolean equals(Object object) {
099        return (super.equals (object) &&
100                object instanceof JobImpressionsSupported);
101    }
102
103    /**
104     * Get the printing attribute class which is to be used as the "category"
105     * for this printing attribute value.
106     * <P>
107     * For class JobImpressionsSupported, the category is class
108     * JobImpressionsSupported itself.
109     *
110     * @return  Printing attribute class (category), an instance of class
111     *          {@link java.lang.Class java.lang.Class}.
112     */
113    public final Class<? extends Attribute> getCategory() {
114        return JobImpressionsSupported.class;
115    }
116
117    /**
118     * Get the name of the category of which this attribute value is an
119     * instance.
120     * <P>
121     * For class JobImpressionsSupported, the category name is
122     * <CODE>"job-impressions-supported"</CODE>.
123     *
124     * @return  Attribute category name.
125     */
126    public final String getName() {
127        return "job-impressions-supported";
128    }
129
130}
