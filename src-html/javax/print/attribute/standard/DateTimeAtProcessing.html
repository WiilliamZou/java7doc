025package javax.print.attribute.standard;
026
027import java.util.Date;
028import javax.print.attribute.Attribute;
029import javax.print.attribute.DateTimeSyntax;
030import javax.print.attribute.PrintJobAttribute;
031
032/**
033 * Class DateTimeAtProcessing is a printing attribute class, a date-time
034 * attribute, that indicates the date and time at which the Print Job first
035 * began processing.
036 * <P>
037 * To construct a DateTimeAtProcessing attribute from separate values of the
038 * year, month, day, hour, minute, and so on, use a {@link java.util.Calendar
039 * Calendar} object to construct a {@link java.util.Date Date} object, then use
040 * the {@link java.util.Date Date} object to construct the DateTimeAtProcessing
041 * attribute. To convert a DateTimeAtProcessing attribute to separate values of
042 * the year, month, day, hour, minute, and so on, create a {@link
043 * java.util.Calendar Calendar} object and set it to the {@link java.util.Date
044 * Date} from the DateTimeAtProcessing attribute.
045 * <P>
046 * <B>IPP Compatibility:</B> The information needed to construct an IPP
047 * "date-time-at-processing" attribute can be obtained as described above. The
048 * category name returned by <CODE>getName()</CODE> gives the IPP attribute
049 * name.
050 * <P>
051 *
052 * @author  Alan Kaminsky
053 */
054public final class DateTimeAtProcessing extends DateTimeSyntax
055        implements PrintJobAttribute {
056
057    private static final long serialVersionUID = -3710068197278263244L;
058
059    /**
060     * Construct a new date-time at processing attribute with the given {@link
061     * java.util.Date Date} value.
062     *
063     * @param  dateTime  {@link java.util.Date Date} value.
064     *
065     * @exception  NullPointerException
066     *     (unchecked exception) Thrown if <CODE>dateTime</CODE> is null.
067     */
068    public DateTimeAtProcessing(Date dateTime) {
069        super (dateTime);
070    }
071
072    /**
073     * Returns whether this date-time at processing attribute is equivalent to
074     * the passed in object. To be equivalent, all of the following conditions
075     * must be true:
076     * <OL TYPE=1>
077     * <LI>
078     * <CODE>object</CODE> is not null.
079     * <LI>
080     * <CODE>object</CODE> is an instance of class DateTimeAtProcessing.
081     * <LI>
082     * This date-time at processing attribute's {@link java.util.Date Date}
083     * value and <CODE>object</CODE>'s {@link java.util.Date Date} value
084     * are equal.
085     * </OL>
086     *
087     * @param  object  Object to compare to.
088     *
089     * @return  True if <CODE>object</CODE> is equivalent to this date-time
090     *          at processing attribute, false otherwise.
091     */
092    public boolean equals(Object object) {
093        return(super.equals (object) &&
094               object instanceof DateTimeAtProcessing);
095    }
096
097    /**
098     * Get the printing attribute class which is to be used as the "category"
099     * for this printing attribute value.
100     * <P>
101     * For class DateTimeAtProcessing, the category is class
102     * DateTimeAtProcessing itself.
103     *
104     * @return  Printing attribute class (category), an instance of class
105     *          {@link java.lang.Class java.lang.Class}.
106     */
107    public final Class<? extends Attribute> getCategory() {
108        return DateTimeAtProcessing.class;
109    }
110
111    /**
112     * Get the name of the category of which this attribute value is an
113     * instance.
114     * <P>
115     * For class DateTimeAtProcessing, the category name is
116     * <CODE>"date-time-at-processing"</CODE>.
117     *
118     * @return  Attribute category name.
119     */
120    public final String getName() {
121        return "date-time-at-processing";
122    }
123
124}
