025package javax.print.attribute.standard;
026
027import java.util.Locale;
028
029import javax.print.attribute.Attribute;
030import javax.print.attribute.TextSyntax;
031import javax.print.attribute.DocAttribute;
032
033/**
034 * Class DocumentName is a printing attribute class, a text attribute, that
035 * specifies the name of a document. DocumentName is an attribute of the print
036 * data (the doc), not of the Print Job. A document's name is an arbitrary
037 * string defined by the client.
038 * However if a JobName is not specified, the DocumentName should be used
039 * instead, which implies that supporting specification of DocumentName
040 * requires reporting of JobName and vice versa.
041 * See {@link JobName JobName} for more information.
042 * <P>
043 * <B>IPP Compatibility:</B> The string value gives the IPP name value. The
044 * locale gives the IPP natural language. The category name returned by
045 * <CODE>getName()</CODE> gives the IPP attribute name.
046 * <P>
047 *
048 * @author  Alan Kaminsky
049 */
050public final class DocumentName extends TextSyntax implements DocAttribute {
051
052    private static final long serialVersionUID = 7883105848533280430L;
053
054    /**
055     * Constructs a new document name attribute with the given document name
056     * and locale.
057     *
058     * @param  documentName  Document name.
059     * @param  locale        Natural language of the text string. null
060     * is interpreted to mean the default locale as returned
061     * by <code>Locale.getDefault()</code>
062     *
063     * @exception  NullPointerException
064     *   (unchecked exception) Thrown if <CODE>documentName</CODE> is null.
065     */
066    public DocumentName(String documentName, Locale locale) {
067        super (documentName, locale);
068    }
069
070    /**
071     * Returns whether this document name attribute is equivalent to the
072     * passed in object.
073     * To be equivalent, all of the following conditions must be true:
074     * <OL TYPE=1>
075     * <LI>
076     * <CODE>object</CODE> is not null.
077     * <LI>
078     * <CODE>object</CODE> is an instance of class DocumentName.
079     * <LI>
080     * This document name attribute's underlying string and
081     * <CODE>object</CODE>'s underlying string are equal.
082     * <LI>
083     * This document name attribute's locale and <CODE>object</CODE>'s locale
084     * are equal.
085     * </OL>
086     *
087     * @param  object  Object to compare to.
088     *
089     * @return  True if <CODE>object</CODE> is equivalent to this document
090     *          name attribute, false otherwise.
091     */
092    public boolean equals(Object object) {
093        return (super.equals (object) && object instanceof DocumentName);
094    }
095
096    /**
097     * Get the printing attribute class which is to be used as the "category"
098     * for this printing attribute value.
099     * <P>
100     * For class DocumentName, the category is class DocumentName itself.
101     *
102     * @return  Printing attribute class (category), an instance of class
103     *          {@link java.lang.Class java.lang.Class}.
104     */
105    public final Class<? extends Attribute> getCategory() {
106        return DocumentName.class;
107    }
108
109    /**
110     * Get the name of the category of which this attribute value is an
111     * instance.
112     * <P>
113     * For class DocumentName, the category name is <CODE>"document-name"</CODE>.
114     *
115     * @return  Attribute category name.
116     */
117    public final String getName() {
118        return "document-name";
119    }
120
121}
