027package javax.print.attribute;
028
029/**
030 * Interface PrintRequestAttributeSet specifies the interface for a set of
031 * print request attributes, i.e. printing attributes that implement interface
032 * {@link PrintRequestAttribute PrintRequestAttribute}.
033 * The client uses a PrintRequestAttributeSet to specify the settings to be
034 * applied to a whole print job and to all the docs in the print job.
035 * <P>
036 * PrintRequestAttributeSet is just an {@link AttributeSet AttributeSet} whose
037 * constructors and mutating operations guarantee an additional invariant,
038 * namely that all attribute values in the PrintRequestAttributeSet must be
039 * instances of interface {@link PrintRequestAttribute PrintRequestAttribute}.
040 * The {@link #add(Attribute) <CODE>add(Attribute)</CODE>}, and
041 * {@link #addAll(AttributeSet) <CODE>addAll(AttributeSet)</CODE>} operations
042 * are respecified below to guarantee this additional invariant.
043 * <P>
044 *
045 * @author  Alan Kaminsky
046 */
047public interface PrintRequestAttributeSet extends AttributeSet {
048
049    /**
050     * Adds the specified attribute value to this attribute set if it is not
051     * already present, first removing any existing value in the same
052     * attribute category as the specified attribute value (optional
053     * operation).
054     *
055     * @param  attribute  Attribute value to be added to this attribute set.
056     *
057     * @return  <tt>true</tt> if this attribute set changed as a result of
058     *          the call, i.e., the given attribute value was not already a
059     *          member of this attribute set.
060     *
061     * @throws  UnmodifiableSetException
062     *     (unchecked exception) Thrown if this attribute set does not
063     *     support the <CODE>add()</CODE> operation.
064     * @throws  ClassCastException
065     *     (unchecked exception) Thrown if the <CODE>attribute</CODE> is
066     *     not an instance of interface
067     *     {@link PrintRequestAttribute PrintRequestAttribute}.
068     * @throws  NullPointerException
069     *     (unchecked exception) Thrown if the <CODE>attribute</CODE> is null.
070     */
071    public boolean add(Attribute attribute);
072
073    /**
074     * Adds all of the elements in the specified set to this attribute.
075     * The outcome is  the same as if the
076     * {@link #add(Attribute) <CODE>add(Attribute)</CODE>}
077     * operation had been applied to this attribute set successively with
078     * each element from the specified set. If none of the categories in the
079     * specified set  are the same as any categories in this attribute set,
080     * the <tt>addAll()</tt> operation effectively modifies this attribute
081     * set so that its value is the <i>union</i> of the two sets.
082     * <P>
083     * The behavior of the <CODE>addAll()</CODE> operation is unspecified if
084     * the specified set is modified while the operation is in progress.
085     * <P>
086     * If the <CODE>addAll()</CODE> operation throws an exception, the effect
087     * on this attribute set's state is implementation dependent; elements
088     * from the specified set before the point of the exception may or
089     * may not have been added to this attribute set.
090     *
091     * @param  attributes  whose elements are to be added to this attribute
092     *            set.
093     *
094     * @return  <tt>true</tt> if this attribute set changed as a result of
095     *          the call.
096     *
097     * @throws  UnmodifiableSetException
098     *     (Unchecked exception) Thrown if this attribute set does not
099     *     support the <tt>addAll()</tt> method.
100     * @throws  ClassCastException
101     *     (Unchecked exception) Thrown if some element in the specified
102     *     set is not an instance of interface {@link PrintRequestAttribute
103     *     PrintRequestAttribute}.
104     * @throws  NullPointerException
105     *     (Unchecked exception) Thrown if the specified  set is null.
106     *
107     * @see #add(Attribute)
108     */
109    public boolean addAll(AttributeSet attributes);
110
111}
