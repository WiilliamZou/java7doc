027package javax.print.attribute;
028
029import java.io.Serializable;
030
031/**
032 * Class HashDocAttributeSet provides an attribute set which
033 * inherits its implementation from class {@link HashAttributeSet
034 * HashAttributeSet} and enforces the semantic restrictions of interface {@link
035 * DocAttributeSet DocAttributeSet}.
036 * <P>
037 *
038 * @author  Alan Kaminsky
039 */
040public class HashDocAttributeSet extends HashAttributeSet
041    implements DocAttributeSet, Serializable {
042
043    private static final long serialVersionUID = -1128534486061432528L;
044
045    /**
046     * Construct a new, empty hash doc attribute set.
047     */
048    public HashDocAttributeSet() {
049        super (DocAttribute.class);
050    }
051
052    /**
053     * Construct a new hash doc attribute set,
054     * initially populated with the given value.
055     *
056     * @param  attribute  Attribute value to add to the set.
057     *
058     * @exception  NullPointerException
059     *     (unchecked exception) Thrown if <CODE>attribute</CODE> is null.
060     */
061    public HashDocAttributeSet(DocAttribute attribute) {
062        super (attribute, DocAttribute.class);
063    }
064
065    /**
066     * Construct a new hash doc attribute set,
067     * initially populated with the values from the given array.
068     * The new attribute set is populated by
069     * adding the elements of <CODE>attributes</CODE> array to the set in
070     * sequence, starting at index 0. Thus, later array elements may replace
071     * earlier array elements if the array contains duplicate attribute
072     * values or attribute categories.
073     *
074     * @param  attributes  Array of attribute values to add to the set.
075     *                     If null, an empty attribute set is constructed.
076     *
077     * @exception  NullPointerException
078     *  (unchecked exception)
079     * Thrown if any element of <CODE>attributes</CODE> is null.
080     */
081    public HashDocAttributeSet(DocAttribute[] attributes) {
082        super (attributes, DocAttribute.class);
083    }
084
085    /**
086     * Construct a new attribute set, initially populated with the
087     * values from the  given set where the members of the attribute set
088     * are restricted to the <code>DocAttribute</code> interface.
089     *
090     * @param  attributes set of attribute values to initialise the set. If
091     *                    null, an empty attribute set is constructed.
092     *
093     * @exception  ClassCastException
094     *     (unchecked exception) Thrown if any element of
095     * <CODE>attributes</CODE> is not an instance of
096     * <CODE>DocAttribute</CODE>.
097     */
098    public HashDocAttributeSet(DocAttributeSet attributes) {
099        super(attributes, DocAttribute.class);
100    }
101
102}
