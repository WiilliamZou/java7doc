026package javax.management;
027
028
029/**
030 * <p>Can be implemented by an MBean in order to
031 * carry out operations before and after being registered or unregistered from
032 * the MBean Server.  An MBean can also implement this interface in order
033 * to get a reference to the MBean Server and/or its name within that
034 * MBean Server.</p>
035 *
036 * @since 1.5
037 */
038public interface MBeanRegistration   {
039
040
041    /**
042     * Allows the MBean to perform any operations it needs before
043     * being registered in the MBean Server.  If the name of the MBean
044     * is not specified, the MBean can provide a name for its
045     * registration.  If any exception is raised, the MBean will not be
046     * registered in the MBean Server.
047     *
048     * @param server The MBean Server in which the MBean will be registered.
049     *
050     * @param name The object name of the MBean.  This name is null if
051     * the name parameter to one of the <code>createMBean</code> or
052     * <code>registerMBean</code> methods in the {@link MBeanServer}
053     * interface is null.  In that case, this method must return a
054     * non-null ObjectName for the new MBean.
055     *
056     * @return The name under which the MBean is to be registered.
057     * This value must not be null.  If the <code>name</code>
058     * parameter is not null, it will usually but not necessarily be
059     * the returned value.
060     *
061     * @exception java.lang.Exception This exception will be caught by
062     * the MBean Server and re-thrown as an {@link
063     * MBeanRegistrationException}.
064     */
065    public ObjectName preRegister(MBeanServer server,
066                                  ObjectName name) throws java.lang.Exception;
067
068    /**
069     * Allows the MBean to perform any operations needed after having been
070     * registered in the MBean server or after the registration has failed.
071     * <p>If the implementation of this method throws a {@link RuntimeException}
072     * or an {@link Error}, the MBean Server will rethrow those inside
073     * a {@link RuntimeMBeanException} or {@link RuntimeErrorException},
074     * respectively. However, throwing an exception in {@code postRegister}
075     * will not change the state of the MBean:
076     * if the MBean was already registered ({@code registrationDone} is
077     * {@code true}), the MBean will remain registered. </p>
078     * <p>This might be confusing for the code calling {@code createMBean()}
079     * or {@code registerMBean()}, as such code might assume that MBean
080     * registration has failed when such an exception is raised.
081     * Therefore it is recommended that implementations of
082     * {@code postRegister} do not throw Runtime Exceptions or Errors if it
083     * can be avoided.</p>
084     * @param registrationDone Indicates whether or not the MBean has
085     * been successfully registered in the MBean server. The value
086     * false means that the registration phase has failed.
087     */
088    public void postRegister(Boolean registrationDone);
089
090    /**
091     * Allows the MBean to perform any operations it needs before
092     * being unregistered by the MBean server.
093     *
094     * @exception java.lang.Exception This exception will be caught by
095     * the MBean server and re-thrown as an {@link
096     * MBeanRegistrationException}.
097     */
098    public void preDeregister() throws java.lang.Exception ;
099
100    /**
101     * Allows the MBean to perform any operations needed after having been
102     * unregistered in the MBean server.
103     * <p>If the implementation of this method throws a {@link RuntimeException}
104     * or an {@link Error}, the MBean Server will rethrow those inside
105     * a {@link RuntimeMBeanException} or {@link RuntimeErrorException},
106     * respectively. However, throwing an exception in {@code postDeregister}
107     * will not change the state of the MBean:
108     * the MBean was already successfully deregistered and will remain so. </p>
109     * <p>This might be confusing for the code calling
110     * {@code unregisterMBean()}, as it might assume that MBean deregistration
111     * has failed. Therefore it is recommended that implementations of
112     * {@code postDeregister} do not throw Runtime Exceptions or Errors if it
113     * can be avoided.</p>
114     */
115    public void postDeregister();
116
117 }
