026package javax.management;
027
028// java import
029import java.io.Serializable;
030
031/**
032 * Allows a query to be performed in the context of a specific MBean server.
033 *
034 * @since 1.5
035 */
036public abstract class QueryEval implements Serializable {
037
038    /* Serial version */
039    private static final long serialVersionUID = 2675899265640874796L;
040
041    private static ThreadLocal<MBeanServer> server =
042        new InheritableThreadLocal<MBeanServer>();
043
044    /**
045     * <p>Sets the MBean server on which the query is to be performed.
046     * The setting is valid for the thread performing the set.
047     * It is copied to any threads created by that thread at the moment
048     * of their creation.</p>
049     *
050     * <p>For historical reasons, this method is not static, but its
051     * behavior does not depend on the instance on which it is
052     * called.</p>
053     *
054     * @param s The MBean server on which the query is to be performed.
055     *
056     * @see #getMBeanServer
057     */
058    public void setMBeanServer(MBeanServer s) {
059        server.set(s);
060    }
061
062    /**
063     * <p>Return the MBean server that was most recently given to the
064     * {@link #setMBeanServer setMBeanServer} method by this thread.
065     * If this thread never called that method, the result is the
066     * value its parent thread would have obtained from
067     * <code>getMBeanServer</code> at the moment of the creation of
068     * this thread, or null if there is no parent thread.</p>
069     *
070     * @return the MBean server.
071     *
072     * @see #setMBeanServer
073     *
074     */
075    public static MBeanServer getMBeanServer() {
076        return server.get();
077    }
078}
