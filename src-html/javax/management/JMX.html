101    /**
102     * <p>Make a proxy for a Standard MBean in a local or remote
103     * MBean Server.</p>
104     *
105     * <p>If you have an MBean Server {@code mbs} containing an MBean
106     * with {@link ObjectName} {@code name}, and if the MBean's
107     * management interface is described by the Java interface
108     * {@code MyMBean}, you can construct a proxy for the MBean like
109     * this:</p>
110     *
111     * <pre>
112     * MyMBean proxy = JMX.newMBeanProxy(mbs, name, MyMBean.class);
113     * </pre>
114     *
115     * <p>Suppose, for example, {@code MyMBean} looks like this:</p>
116     *
117     * <pre>
118     * public interface MyMBean {
119     *     public String getSomeAttribute();
120     *     public void setSomeAttribute(String value);
121     *     public void someOperation(String param1, int param2);
122     * }
123     * </pre>
124     *
125     * <p>Then you can execute:</p>
126     *
127     * <ul>
128     *
129     * <li>{@code proxy.getSomeAttribute()} which will result in a
130     * call to {@code mbs.}{@link MBeanServerConnection#getAttribute
131     * getAttribute}{@code (name, "SomeAttribute")}.
132     *
133     * <li>{@code proxy.setSomeAttribute("whatever")} which will result
134     * in a call to {@code mbs.}{@link MBeanServerConnection#setAttribute
135     * setAttribute}{@code (name, new Attribute("SomeAttribute", "whatever"))}.
136     *
137     * <li>{@code proxy.someOperation("param1", 2)} which will be
138     * translated into a call to {@code mbs.}{@link
139     * MBeanServerConnection#invoke invoke}{@code (name, "someOperation", <etc>)}.
140     *
141     * </ul>
142     *
143     * <p>The object returned by this method is a
144     * {@link Proxy} whose {@code InvocationHandler} is an
145     * {@link MBeanServerInvocationHandler}.</p>
146     *
147     * <p>This method is equivalent to {@link
148     * #newMBeanProxy(MBeanServerConnection, ObjectName, Class,
149     * boolean) newMBeanProxy(connection, objectName, interfaceClass,
150     * false)}.</p>
151     *
152     * @param connection the MBean server to forward to.
153     * @param objectName the name of the MBean within
154     * {@code connection} to forward to.
155     * @param interfaceClass the management interface that the MBean
156     * exports, which will also be implemented by the returned proxy.
157     *
158     * @param <T> allows the compiler to know that if the {@code
159     * interfaceClass} parameter is {@code MyMBean.class}, for
160     * example, then the return type is {@code MyMBean}.
161     *
162     * @return the new proxy instance.
163     */
164    public static <T> T newMBeanProxy(MBeanServerConnection connection,
165                                      ObjectName objectName,
166                                      Class<T> interfaceClass) {
167        return newMBeanProxy(connection, objectName, interfaceClass, false);
168    }

170    /**
171     * <p>Make a proxy for a Standard MBean in a local or remote MBean
172     * Server that may also support the methods of {@link
173     * NotificationEmitter}.</p>
174     *
175     * <p>This method behaves the same as {@link
176     * #newMBeanProxy(MBeanServerConnection, ObjectName, Class)}, but
177     * additionally, if {@code notificationEmitter} is {@code
178     * true}, then the MBean is assumed to be a {@link
179     * NotificationBroadcaster} or {@link NotificationEmitter} and the
180     * returned proxy will implement {@link NotificationEmitter} as
181     * well as {@code interfaceClass}.  A call to {@link
182     * NotificationBroadcaster#addNotificationListener} on the proxy
183     * will result in a call to {@link
184     * MBeanServerConnection#addNotificationListener(ObjectName,
185     * NotificationListener, NotificationFilter, Object)}, and
186     * likewise for the other methods of {@link
187     * NotificationBroadcaster} and {@link NotificationEmitter}.</p>
188     *
189     * @param connection the MBean server to forward to.
190     * @param objectName the name of the MBean within
191     * {@code connection} to forward to.
192     * @param interfaceClass the management interface that the MBean
193     * exports, which will also be implemented by the returned proxy.
194     * @param notificationEmitter make the returned proxy
195     * implement {@link NotificationEmitter} by forwarding its methods
196     * via {@code connection}.
197     *
198     * @param <T> allows the compiler to know that if the {@code
199     * interfaceClass} parameter is {@code MyMBean.class}, for
200     * example, then the return type is {@code MyMBean}.
201     *
202     * @return the new proxy instance.
203     */
204    public static <T> T newMBeanProxy(MBeanServerConnection connection,
205                                      ObjectName objectName,
206                                      Class<T> interfaceClass,
207                                      boolean notificationEmitter) {
208        return createProxy(connection, objectName, interfaceClass, notificationEmitter, false);
209    }

211    /**
212     * <p>Make a proxy for an MXBean in a local or remote
213     * MBean Server.</p>
214     *
215     * <p>If you have an MBean Server {@code mbs} containing an
216     * MXBean with {@link ObjectName} {@code name}, and if the
217     * MXBean's management interface is described by the Java
218     * interface {@code MyMXBean}, you can construct a proxy for
219     * the MXBean like this:</p>
220     *
221     * <pre>
222     * MyMXBean proxy = JMX.newMXBeanProxy(mbs, name, MyMXBean.class);
223     * </pre>
224     *
225     * <p>Suppose, for example, {@code MyMXBean} looks like this:</p>
226     *
227     * <pre>
228     * public interface MyMXBean {
229     *     public String getSimpleAttribute();
230     *     public void setSimpleAttribute(String value);
231     *     public {@link java.lang.management.MemoryUsage} getMappedAttribute();
232     *     public void setMappedAttribute(MemoryUsage memoryUsage);
233     *     public MemoryUsage someOperation(String param1, MemoryUsage param2);
234     * }
235     * </pre>
236     *
237     * <p>Then:</p>
238     *
239     * <ul>
240     *
241     * <li><p>{@code proxy.getSimpleAttribute()} will result in a
242     * call to {@code mbs.}{@link MBeanServerConnection#getAttribute
243     * getAttribute}{@code (name, "SimpleAttribute")}.</p>
244     *
245     * <li><p>{@code proxy.setSimpleAttribute("whatever")} will result
246     * in a call to {@code mbs.}{@link
247     * MBeanServerConnection#setAttribute setAttribute}<code>(name,
248     * new Attribute("SimpleAttribute", "whatever"))</code>.<p>
249     *
250     *     <p>Because {@code String} is a <em>simple type</em>, in the
251     *     sense of {@link javax.management.openmbean.SimpleType}, it
252     *     is not changed in the context of an MXBean.  The MXBean
253     *     proxy behaves the same as a Standard MBean proxy (see
254     *     {@link #newMBeanProxy(MBeanServerConnection, ObjectName,
255     *     Class) newMBeanProxy}) for the attribute {@code
256     *     SimpleAttribute}.</p>
257     *
258     * <li><p>{@code proxy.getMappedAttribute()} will result in a call
259     * to {@code mbs.getAttribute("MappedAttribute")}.  The MXBean
260     * mapping rules mean that the actual type of the attribute {@code
261     * MappedAttribute} will be {@link
262     * javax.management.openmbean.CompositeData CompositeData} and
263     * that is what the {@code mbs.getAttribute} call will return.
264     * The proxy will then convert the {@code CompositeData} back into
265     * the expected type {@code MemoryUsage} using the MXBean mapping
266     * rules.</p>
267     *
268     * <li><p>Similarly, {@code proxy.setMappedAttribute(memoryUsage)}
269     * will convert the {@code MemoryUsage} argument into a {@code
270     * CompositeData} before calling {@code mbs.setAttribute}.</p>
271     *
272     * <li><p>{@code proxy.someOperation("whatever", memoryUsage)}
273     * will convert the {@code MemoryUsage} argument into a {@code
274     * CompositeData} and call {@code mbs.invoke}.  The value returned
275     * by {@code mbs.invoke} will be also be a {@code CompositeData},
276     * and the proxy will convert this into the expected type {@code
277     * MemoryUsage} using the MXBean mapping rules.</p>
278     *
279     * </ul>
280     *
281     * <p>The object returned by this method is a
282     * {@link Proxy} whose {@code InvocationHandler} is an
283     * {@link MBeanServerInvocationHandler}.</p>
284     *
285     * <p>This method is equivalent to {@link
286     * #newMXBeanProxy(MBeanServerConnection, ObjectName, Class,
287     * boolean) newMXBeanProxy(connection, objectName, interfaceClass,
288     * false)}.</p>
289     *
290     * @param connection the MBean server to forward to.
291     * @param objectName the name of the MBean within
292     * {@code connection} to forward to.
293     * @param interfaceClass the MXBean interface,
294     * which will also be implemented by the returned proxy.
295     *
296     * @param <T> allows the compiler to know that if the {@code
297     * interfaceClass} parameter is {@code MyMXBean.class}, for
298     * example, then the return type is {@code MyMXBean}.
299     *
300     * @return the new proxy instance.
301     */
302    public static <T> T newMXBeanProxy(MBeanServerConnection connection,
303                                       ObjectName objectName,
304                                       Class<T> interfaceClass) {
305        return newMXBeanProxy(connection, objectName, interfaceClass, false);
306    }

349    /**
350     * <p>Test whether an interface is an MXBean interface.
351     * An interface is an MXBean interface if it is annotated
352     * {@link MXBean &#64;MXBean} or {@code @MXBean(true)}
353     * or if it does not have an {@code @MXBean} annotation
354     * and its name ends with "{@code MXBean}".</p>
355     *
356     * @param interfaceClass The candidate interface.
357     *
358     * @return true if {@code interfaceClass} is an interface and
359     * meets the conditions described.
360     *
361     * @throws NullPointerException if {@code interfaceClass} is null.
362     */
363    public static boolean isMXBeanInterface(Class<?> interfaceClass) {
364        if (!interfaceClass.isInterface())
365            return false;
366        MXBean a = interfaceClass.getAnnotation(MXBean.class);
367        if (a != null)
368            return a.value();
369        return interfaceClass.getName().endsWith("MXBean");
370        // We don't bother excluding the case where the name is
371        // exactly the string "MXBean" since that would mean there
372        // was no package name, which is pretty unlikely in practice.
373    }

375    /**
376     * Centralised M(X)Bean proxy creation code
377     * @param connection {@linkplain MBeanServerConnection} to use
378     * @param objectName M(X)Bean object name
379     * @param interfaceClass M(X)Bean interface class
380     * @param notificationEmitter Is a notification emitter?
381     * @param isMXBean Is an MXBean?
382     * @return Returns an M(X)Bean proxy generated for the provided interface class
383     * @throws SecurityException
384     * @throws IllegalArgumentException
385     */
386    private static <T> T createProxy(MBeanServerConnection connection,
387                                     ObjectName objectName,
388                                     Class<T> interfaceClass,
389                                     boolean notificationEmitter,
390                                     boolean isMXBean) {
391
392        if (System.getSecurityManager() != null) {
393            checkProxyInterface(interfaceClass);
394        }
395        try {
396            if (isMXBean) {
397                // Check interface for MXBean compliance
398                Introspector.testComplianceMXBeanInterface(interfaceClass);
399            } else {
400                // Check interface for MBean compliance
401                Introspector.testComplianceMBeanInterface(interfaceClass);
402            }
403        } catch (NotCompliantMBeanException e) {
404            throw new IllegalArgumentException(e);
405        }
406
407        InvocationHandler handler = new MBeanServerInvocationHandler(
408                connection, objectName, isMXBean);
409        final Class<?>[] interfaces;
410        if (notificationEmitter) {
411            interfaces =
412                new Class<?>[] {interfaceClass, NotificationEmitter.class};
413        } else
414            interfaces = new Class<?>[] {interfaceClass};
415
416        Object proxy = Proxy.newProxyInstance(
417                interfaceClass.getClassLoader(),
418                interfaces,
419                handler);
420        return interfaceClass.cast(proxy);
421    }

423    /**
424     * Checks for the M(X)Bean proxy interface being public and not restricted
425     * @param interfaceClass MBean proxy interface
426     * @throws SecurityException when the proxy interface comes from a restricted
427     *                           package or is not public
428     */
429    private static void checkProxyInterface(Class<?> interfaceClass) {
430        if (!Modifier.isPublic(interfaceClass.getModifiers())) {
431            throw new SecurityException("mbean proxy interface non-public");
432        }
433        ReflectUtil.checkPackageAccess(interfaceClass);
434    }
