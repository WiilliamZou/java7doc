167    /**
168     * <p>Makes an <code>RMIConnectorServer</code> for the given MBean
169     * server.</p>
170     *
171     * @param url the URL defining how to create the connector server.
172     * Cannot be null.
173     *
174     * @param environment attributes governing the creation and
175     * storing of the RMI object.  Can be null, which is equivalent to
176     * an empty Map.
177     *
178     * @param rmiServerImpl An implementation of the RMIServer interface,
179     *  consistent with the protocol type specified in <var>url</var>.
180     *  If this parameter is non null, the protocol type specified by
181     *  <var>url</var> is not constrained, and is assumed to be valid.
182     *  Otherwise, only "rmi" and "iiop" will be recognized.
183     *
184     * @param mbeanServer the MBean server to which the new connector
185     * server is attached, or null if it will be attached by being
186     * registered as an MBean in the MBean server.
187     *
188     * @exception IllegalArgumentException if <code>url</code> is null.
189     *
190     * @exception MalformedURLException if <code>url</code> does not
191     * conform to the syntax for an RMI connector, or if its protocol
192     * is not recognized by this implementation. Only "rmi" and "iiop"
193     * are recognized when <var>rmiServerImpl</var> is null.
194     *
195     * @exception IOException if the connector server cannot be created
196     * for some reason or if it is inevitable that its {@link #start()
197     * start} method will fail.
198     *
199     * @see #start
200     */
201    public RMIConnectorServer(JMXServiceURL url, Map<String,?> environment,
202                              RMIServerImpl rmiServerImpl,
203                              MBeanServer mbeanServer)
204            throws IOException {
205        super(mbeanServer);
206
207        if (url == null) throw new
208            IllegalArgumentException("Null JMXServiceURL");
209        if (rmiServerImpl == null) {
210            final String prt = url.getProtocol();
211            if (prt == null || !(prt.equals("rmi") || prt.equals("iiop"))) {
212                final String msg = "Invalid protocol type: " + prt;
213                throw new MalformedURLException(msg);
214            }
215            final String urlPath = url.getURLPath();
216            if (!urlPath.equals("")
217                && !urlPath.equals("/")
218                && !urlPath.startsWith("/jndi/")) {
219                final String msg = "URL path must be empty or start with " +
220                    "/jndi/";
221                throw new MalformedURLException(msg);
222            }
223        }
224
225        if (environment == null)
226            this.attributes = Collections.emptyMap();
227        else {
228            EnvHelp.checkAttributes(environment);
229            this.attributes = Collections.unmodifiableMap(environment);
230        }
231
232        this.address = url;
233        this.rmiServerImpl = rmiServerImpl;
234    }

236    /**
237     * <p>Returns a client stub for this connector server.  A client
238     * stub is a serializable object whose {@link
239     * JMXConnector#connect(Map) connect} method can be used to make
240     * one new connection to this connector server.</p>
241     *
242     * @param env client connection parameters of the same sort that
243     * could be provided to {@link JMXConnector#connect(Map)
244     * JMXConnector.connect(Map)}.  Can be null, which is equivalent
245     * to an empty map.
246     *
247     * @return a client stub that can be used to make a new connection
248     * to this connector server.
249     *
250     * @exception UnsupportedOperationException if this connector
251     * server does not support the generation of client stubs.
252     *
253     * @exception IllegalStateException if the JMXConnectorServer is
254     * not started (see {@link #isActive()}).
255     *
256     * @exception IOException if a communications problem means that a
257     * stub cannot be created.
258     **/
259    public JMXConnector toJMXConnector(Map<String,?> env) throws IOException {
260        // The serialized for of rmiServerImpl is automatically
261        // a RMI server stub.
262        if (!isActive()) throw new
263            IllegalStateException("Connector is not active");
264
265        // Merge maps
266        Map<String, Object> usemap = new HashMap<String, Object>(
267                (this.attributes==null)?Collections.<String, Object>emptyMap():
268                    this.attributes);
269
270        if (env != null) {
271            EnvHelp.checkAttributes(env);
272            usemap.putAll(env);
273        }
274
275        usemap = EnvHelp.filterAttributes(usemap);
276
277        final RMIServer stub=(RMIServer)rmiServerImpl.toStub();
278
279        return new RMIConnector(stub, usemap);
280    }

282    /**
283     * <p>Activates the connector server, that is starts listening for
284     * client connections.  Calling this method when the connector
285     * server is already active has no effect.  Calling this method
286     * when the connector server has been stopped will generate an
287     * <code>IOException</code>.</p>
288     *
289     * <p>The behavior of this method when called for the first time
290     * depends on the parameters that were supplied at construction,
291     * as described below.</p>
292     *
293     * <p>First, an object of a subclass of {@link RMIServerImpl} is
294     * required, to export the connector server through RMI:</p>
295     *
296     * <ul>
297     *
298     * <li>If an <code>RMIServerImpl</code> was supplied to the
299     * constructor, it is used.
300     *
301     * <li>Otherwise, if the protocol part of the
302     * <code>JMXServiceURL</code> supplied to the constructor was
303     * <code>iiop</code>, an object of type {@link RMIIIOPServerImpl}
304     * is created.
305     *
306     * <li>Otherwise, if the <code>JMXServiceURL</code>
307     * was null, or its protocol part was <code>rmi</code>, an object
308     * of type {@link RMIJRMPServerImpl} is created.
309     *
310     * <li>Otherwise, the implementation can create an
311     * implementation-specific {@link RMIServerImpl} or it can throw
312     * {@link MalformedURLException}.
313     *
314     * </ul>
315     *
316     * <p>If the given address includes a JNDI directory URL as
317     * specified in the package documentation for {@link
318     * javax.management.remote.rmi}, then this
319     * <code>RMIConnectorServer</code> will bootstrap by binding the
320     * <code>RMIServerImpl</code> to the given address.</p>
321     *
322     * <p>If the URL path part of the <code>JMXServiceURL</code> was
323     * empty or a single slash (<code>/</code>), then the RMI object
324     * will not be bound to a directory.  Instead, a reference to it
325     * will be encoded in the URL path of the RMIConnectorServer
326     * address (returned by {@link #getAddress()}).  The encodings for
327     * <code>rmi</code> and <code>iiop</code> are described in the
328     * package documentation for {@link
329     * javax.management.remote.rmi}.</p>
330     *
331     * <p>The behavior when the URL path is neither empty nor a JNDI
332     * directory URL, or when the protocol is neither <code>rmi</code>
333     * nor <code>iiop</code>, is implementation defined, and may
334     * include throwing {@link MalformedURLException} when the
335     * connector server is created or when it is started.</p>
336     *
337     * @exception IllegalStateException if the connector server has
338     * not been attached to an MBean server.
339     * @exception IOException if the connector server cannot be
340     * started.
341     */
342    public synchronized void start() throws IOException {
343        final boolean tracing = logger.traceOn();
344
345        if (state == STARTED) {
346            if (tracing) logger.trace("start", "already started");
347            return;
348        } else if (state == STOPPED) {
349            if (tracing) logger.trace("start", "already stopped");
350            throw new IOException("The server has been stopped.");
351        }
352
353        if (getMBeanServer() == null)
354            throw new IllegalStateException("This connector server is not " +
355                                            "attached to an MBean server");
356
357        // Check the internal access file property to see
358        // if an MBeanServerForwarder is to be provided
359        //
360        if (attributes != null) {
361            // Check if access file property is specified
362            //
363            String accessFile =
364                (String) attributes.get("jmx.remote.x.access.file");
365            if (accessFile != null) {
366                // Access file property specified, create an instance
367                // of the MBeanServerFileAccessController class
368                //
369                MBeanServerForwarder mbsf;
370                try {
371                    mbsf = new MBeanServerFileAccessController(accessFile);
372                } catch (IOException e) {
373                    throw EnvHelp.initCause(
374                        new IllegalArgumentException(e.getMessage()), e);
375                }
376                // Set the MBeanServerForwarder
377                //
378                setMBeanServerForwarder(mbsf);
379            }
380        }
381
382        try {
383            if (tracing) logger.trace("start", "setting default class loader");
384            defaultClassLoader = EnvHelp.resolveServerClassLoader(
385                    attributes, getMBeanServer());
386        } catch (InstanceNotFoundException infc) {
387            IllegalArgumentException x = new
388                IllegalArgumentException("ClassLoader not found: "+infc);
389            throw EnvHelp.initCause(x,infc);
390        }
391
392        if (tracing) logger.trace("start", "setting RMIServer object");
393        final RMIServerImpl rmiServer;
394
395        if (rmiServerImpl != null)
396            rmiServer = rmiServerImpl;
397        else
398            rmiServer = newServer();
399
400        rmiServer.setMBeanServer(getMBeanServer());
401        rmiServer.setDefaultClassLoader(defaultClassLoader);
402        rmiServer.setRMIConnectorServer(this);
403        rmiServer.export();
404
405        try {
406            if (tracing) logger.trace("start", "getting RMIServer object to export");
407            final RMIServer objref = objectToBind(rmiServer, attributes);
408
409            if (address != null && address.getURLPath().startsWith("/jndi/")) {
410                final String jndiUrl = address.getURLPath().substring(6);
411
412                if (tracing)
413                    logger.trace("start", "Using external directory: " + jndiUrl);
414
415                String stringBoolean = (String) attributes.get(JNDI_REBIND_ATTRIBUTE);
416                final boolean rebind = EnvHelp.computeBooleanFromString( stringBoolean );
417
418                if (tracing)
419                    logger.trace("start", JNDI_REBIND_ATTRIBUTE + "=" + rebind);
420
421                try {
422                    if (tracing) logger.trace("start", "binding to " + jndiUrl);
423
424                    final Hashtable<?, ?> usemap = EnvHelp.mapToHashtable(attributes);
425
426                    bind(jndiUrl, usemap, objref, rebind);
427
428                    boundJndiUrl = jndiUrl;
429                } catch (NamingException e) {
430                    // fit e in the nested exception if we are on 1.4
431                    throw newIOException("Cannot bind to URL ["+jndiUrl+"]: "
432                                         + e, e);
433                }
434            } else {
435                // if jndiURL is null, we must encode the stub into the URL.
436                if (tracing) logger.trace("start", "Encoding URL");
437
438                encodeStubInAddress(objref, attributes);
439
440                if (tracing) logger.trace("start", "Encoded URL: " + this.address);
441            }
442        } catch (Exception e) {
443            try {
444                rmiServer.close();
445            } catch (Exception x) {
446                // OK: we are already throwing another exception
447            }
448            if (e instanceof RuntimeException)
449                throw (RuntimeException) e;
450            else if (e instanceof IOException)
451                throw (IOException) e;
452            else
453                throw newIOException("Got unexpected exception while " +
454                                     "starting the connector server: "
455                                     + e, e);
456        }
457
458        rmiServerImpl = rmiServer;
459
460        synchronized(openedServers) {
461            openedServers.add(this);
462        }
463
464        state = STARTED;
465
466        if (tracing) {
467            logger.trace("start", "Connector Server Address = " + address);
468            logger.trace("start", "started.");
469        }
470    }

472    /**
473     * <p>Deactivates the connector server, that is, stops listening for
474     * client connections.  Calling this method will also close all
475     * client connections that were made by this server.  After this
476     * method returns, whether normally or with an exception, the
477     * connector server will not create any new client
478     * connections.</p>
479     *
480     * <p>Once a connector server has been stopped, it cannot be started
481     * again.</p>
482     *
483     * <p>Calling this method when the connector server has already
484     * been stopped has no effect.  Calling this method when the
485     * connector server has not yet been started will disable the
486     * connector server object permanently.</p>
487     *
488     * <p>If closing a client connection produces an exception, that
489     * exception is not thrown from this method.  A {@link
490     * JMXConnectionNotification} is emitted from this MBean with the
491     * connection ID of the connection that could not be closed.</p>
492     *
493     * <p>Closing a connector server is a potentially slow operation.
494     * For example, if a client machine with an open connection has
495     * crashed, the close operation might have to wait for a network
496     * protocol timeout.  Callers that do not want to block in a close
497     * operation should do it in a separate thread.</p>
498     *
499     * <p>This method calls the method {@link RMIServerImpl#close()
500     * close} on the connector server's <code>RMIServerImpl</code>
501     * object.</p>
502     *
503     * <p>If the <code>RMIServerImpl</code> was bound to a JNDI
504     * directory by the {@link #start() start} method, it is unbound
505     * from the directory by this method.</p>
506     *
507     * @exception IOException if the server cannot be closed cleanly,
508     * or if the <code>RMIServerImpl</code> cannot be unbound from the
509     * directory.  When this exception is thrown, the server has
510     * already attempted to close all client connections, if
511     * appropriate; to call {@link RMIServerImpl#close()}; and to
512     * unbind the <code>RMIServerImpl</code> from its directory, if
513     * appropriate.  All client connections are closed except possibly
514     * those that generated exceptions when the server attempted to
515     * close them.
516     */
517    public void stop() throws IOException {
518        final boolean tracing = logger.traceOn();
519
520        synchronized (this) {
521            if (state == STOPPED) {
522                if (tracing) logger.trace("stop","already stopped.");
523                return;
524            } else if (state == CREATED) {
525                if (tracing) logger.trace("stop","not started yet.");
526            }
527
528            if (tracing) logger.trace("stop", "stopping.");
529            state = STOPPED;
530        }
531
532        synchronized(openedServers) {
533            openedServers.remove(this);
534        }
535
536        IOException exception = null;
537
538        // rmiServerImpl can be null if stop() called without start()
539        if (rmiServerImpl != null) {
540            try {
541                if (tracing) logger.trace("stop", "closing RMI server.");
542                rmiServerImpl.close();
543            } catch (IOException e) {
544                if (tracing) logger.trace("stop", "failed to close RMI server: " + e);
545                if (logger.debugOn()) logger.debug("stop",e);
546                exception = e;
547            }
548        }
549
550        if (boundJndiUrl != null) {
551            try {
552                if (tracing)
553                    logger.trace("stop",
554                          "unbind from external directory: " + boundJndiUrl);
555
556                final Hashtable<?, ?> usemap = EnvHelp.mapToHashtable(attributes);
557
558                InitialContext ctx =
559                    new InitialContext(usemap);
560
561                ctx.unbind(boundJndiUrl);
562
563                ctx.close();
564            } catch (NamingException e) {
565                if (tracing) logger.trace("stop", "failed to unbind RMI server: "+e);
566                if (logger.debugOn()) logger.debug("stop",e);
567                // fit e in as the nested exception if we are on 1.4
568                if (exception == null)
569                    exception = newIOException("Cannot bind to URL: " + e, e);
570            }
571        }
572
573        if (exception != null) throw exception;
574
575        if (tracing) logger.trace("stop", "stopped");
576    }

623    /**
624     * Bind a stub to a registry.
625     * @param jndiUrl URL of the stub in the registry, extracted
626     *        from the <code>JMXServiceURL</code>.
627     * @param attributes A Hashtable containing environment parameters,
628     *        built from the Map specified at this object creation.
629     * @param rmiServer The object to bind in the registry
630     * @param rebind true if the object must be rebound.
631     **/
632    void bind(String jndiUrl, Hashtable<?, ?> attributes,
633              RMIServer rmiServer, boolean rebind)
634        throws NamingException, MalformedURLException {
635        // if jndiURL is not null, we nust bind the stub to a
636        // directory.
637        InitialContext ctx =
638            new InitialContext(attributes);
639
640        if (rebind)
641            ctx.rebind(jndiUrl, rmiServer);
642        else
643            ctx.bind(jndiUrl, rmiServer);
644        ctx.close();
645    }

647    /**
648     * Creates a new RMIServerImpl.
649     **/
650    RMIServerImpl newServer() throws IOException {
651        final boolean iiop = isIiopURL(address,true);
652        final int port;
653        if (address == null)
654            port = 0;
655        else
656            port = address.getPort();
657        if (iiop)
658            return newIIOPServer(attributes);
659        else
660            return newJRMPServer(attributes, port);
661    }

663    /**
664     * Encode a stub into the JMXServiceURL.
665     * @param rmiServer The stub object to encode in the URL
666     * @param attributes A Map containing environment parameters,
667     *        built from the Map specified at this object creation.
668     **/
669    private void encodeStubInAddress(
670            RMIServer rmiServer, Map<String, ?> attributes)
671            throws IOException {
672
673        final String protocol, host;
674        final int port;
675
676        if (address == null) {
677            if (IIOPHelper.isStub(rmiServer))
678                protocol = "iiop";
679            else
680                protocol = "rmi";
681            host = null; // will default to local host name
682            port = 0;
683        } else {
684            protocol = address.getProtocol();
685            host = (address.getHost().equals("")) ? null : address.getHost();
686            port = address.getPort();
687        }
688
689        final String urlPath = encodeStub(rmiServer, attributes);
690
691        address = new JMXServiceURL(protocol, host, port, urlPath);
692    }

694    static boolean isIiopURL(JMXServiceURL directoryURL, boolean strict)
695        throws MalformedURLException {
696        String protocol = directoryURL.getProtocol();
697        if (protocol.equals("rmi"))
698            return false;
699        else if (protocol.equals("iiop"))
700            return true;
701        else if (strict) {
702
703            throw new MalformedURLException("URL must have protocol " +
704                                            "\"rmi\" or \"iiop\": \"" +
705                                            protocol + "\"");
706        }
707        return false;
708    }

768    private static String byteArrayToBase64(byte[] a) {
769        int aLen = a.length;
770        int numFullGroups = aLen/3;
771        int numBytesInPartialGroup = aLen - 3*numFullGroups;
772        int resultLen = 4*((aLen + 2)/3);
773        final StringBuilder result = new StringBuilder(resultLen);
774
775        // Translate all full groups from byte array elements to Base64
776        int inCursor = 0;
777        for (int i=0; i<numFullGroups; i++) {
778            int byte0 = a[inCursor++] & 0xff;
779            int byte1 = a[inCursor++] & 0xff;
780            int byte2 = a[inCursor++] & 0xff;
781            result.append(intToAlpha[byte0 >> 2]);
782            result.append(intToAlpha[(byte0 << 4)&0x3f | (byte1 >> 4)]);
783            result.append(intToAlpha[(byte1 << 2)&0x3f | (byte2 >> 6)]);
784            result.append(intToAlpha[byte2 & 0x3f]);
785        }
786
787        // Translate partial group if present
788        if (numBytesInPartialGroup != 0) {
789            int byte0 = a[inCursor++] & 0xff;
790            result.append(intToAlpha[byte0 >> 2]);
791            if (numBytesInPartialGroup == 1) {
792                result.append(intToAlpha[(byte0 << 4) & 0x3f]);
793                result.append("==");
794            } else {
795                // assert numBytesInPartialGroup == 2;
796                int byte1 = a[inCursor++] & 0xff;
797                result.append(intToAlpha[(byte0 << 4)&0x3f | (byte1 >> 4)]);
798                result.append(intToAlpha[(byte1 << 2)&0x3f]);
799                result.append('=');
800            }
801        }
802        // assert inCursor == a.length;
803        // assert result.length() == resultLen;
804        return result.toString();
805    }
