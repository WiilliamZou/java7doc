099    /**
100     * Constructs a new {@link RMIConnection}. This connection can be
101     * used with either the JRMP or IIOP transport. This object does
102     * not export itself: it is the responsibility of the caller to
103     * export it appropriately (see {@link
104     * RMIJRMPServerImpl#makeClient(String,Subject)} and {@link
105     * RMIIIOPServerImpl#makeClient(String,Subject)}.
106     *
107     * @param rmiServer The RMIServerImpl object for which this
108     * connection is created.  The behavior is unspecified if this
109     * parameter is null.
110     * @param connectionId The ID for this connection.  The behavior
111     * is unspecified if this parameter is null.
112     * @param defaultClassLoader The default ClassLoader to be used
113     * when deserializing marshalled objects.  Can be null, to signify
114     * the bootstrap class loader.
115     * @param subject the authenticated subject to be used for
116     * authorization.  Can be null, to signify that no subject has
117     * been authenticated.
118     * @param env the environment containing attributes for the new
119     * <code>RMIServerImpl</code>.  Can be null, equivalent to an
120     * empty map.
121     */
122    public RMIConnectionImpl(RMIServerImpl rmiServer,
123                             String connectionId,
124                             ClassLoader defaultClassLoader,
125                             Subject subject,
126                             Map<String,?> env) {
127        if (rmiServer == null || connectionId == null)
128            throw new NullPointerException("Illegal null argument");
129        if (env == null)
130            env = Collections.emptyMap();
131        this.rmiServer = rmiServer;
132        this.connectionId = connectionId;
133        this.defaultClassLoader = defaultClassLoader;
134
135        this.subjectDelegator = new SubjectDelegator();
136        this.subject = subject;
137        if (subject == null) {
138            this.acc = null;
139            this.removeCallerContext = false;
140        } else {
141            this.removeCallerContext =
142                SubjectDelegator.checkRemoveCallerContext(subject);
143            if (this.removeCallerContext) {
144                this.acc =
145                    JMXSubjectDomainCombiner.getDomainCombinerContext(subject);
146            } else {
147                this.acc =
148                    JMXSubjectDomainCombiner.getContext(subject);
149            }
150        }
151        this.mbeanServer = rmiServer.getMBeanServer();
152
153        final ClassLoader dcl = defaultClassLoader;
154
155        this.classLoaderWithRepository =
156            AccessController.doPrivileged(
157                new PrivilegedAction<ClassLoaderWithRepository>() {
158                    public ClassLoaderWithRepository run() {
159                        return new ClassLoaderWithRepository(
160                                      mbeanServer.getClassLoaderRepository(),
161                                      dcl);
162                    }
163                },
164
165                withPermissions( new MBeanPermission("*", "getClassLoaderRepository"),
166                                 new RuntimePermission("createClassLoader"))
167            );
168
169
170        this.defaultContextClassLoader =
171            AccessController.doPrivileged(
172                new PrivilegedAction<ClassLoader>() {
173            @Override
174                    public ClassLoader run() {
175                        return new CombinedClassLoader(Thread.currentThread().getContextClassLoader(),
176                                dcl);
177                    }
178                });
179
180        serverCommunicatorAdmin = new
181          RMIServerCommunicatorAdmin(EnvHelp.getServerConnectionTimeout(env));
182
183        this.env = env;
184    }

186    private static AccessControlContext withPermissions(Permission ... perms){
187        Permissions col = new Permissions();
188
189        for (Permission thePerm : perms ) {
190            col.add(thePerm);
191        }
192
193        final ProtectionDomain pd = new ProtectionDomain(null, col);
194        return new AccessControlContext( new ProtectionDomain[] { pd });
195    }

214    public void close() throws IOException {
215        final boolean debug = logger.debugOn();
216        final String  idstr = (debug?"["+this.toString()+"]":null);
217
218        synchronized (this) {
219            if (terminated) {
220                if (debug) logger.debug("close",idstr + " already terminated.");
221                return;
222            }
223
224            if (debug) logger.debug("close",idstr + " closing.");
225
226            terminated = true;
227
228            if (serverCommunicatorAdmin != null) {
229                serverCommunicatorAdmin.terminate();
230            }
231
232            if (serverNotifForwarder != null) {
233                serverNotifForwarder.terminate();
234            }
235        }
236
237        rmiServer.clientClosed(this);
238
239        if (debug) logger.debug("close",idstr + " closed.");
240    }

242    public void unreferenced() {
243        logger.debug("unreferenced", "called");
244        try {
245            close();
246            logger.debug("unreferenced", "done");
247        } catch (IOException e) {
248            logger.fine("unreferenced", e);
249        }
250    }

1534        case REMOVE_NOTIFICATION_LISTENER:
1535            getServerNotifFwd().removeNotificationListener(
1536                                                   (ObjectName)params[0],
1537                                                   (Integer[])params[1]);
1538            return null;
1539
1540        case REMOVE_NOTIFICATION_LISTENER_OBJECTNAME:
1541            mbeanServer.removeNotificationListener((ObjectName)params[0],
1542                                                   (ObjectName)params[1]);
1543            return null;
1544
1545        case REMOVE_NOTIFICATION_LISTENER_OBJECTNAME_FILTER_HANDBACK:
1546            mbeanServer.removeNotificationListener(
1547                                          (ObjectName)params[0],
1548                                          (ObjectName)params[1],
1549                                          (NotificationFilter)params[2],
1550                                          params[3]);
1551            return null;
1552
1553        default:
1554            throw new IllegalArgumentException("Invalid operation");
1555        }

1558    private static class SetCcl implements PrivilegedExceptionAction<ClassLoader> {
1559        private final ClassLoader classLoader;

1561        SetCcl(ClassLoader classLoader) {
1562            this.classLoader = classLoader;
1563        }
1564
1565        public ClassLoader run() {
1566            Thread currentThread = Thread.currentThread();
1567            ClassLoader old = currentThread.getContextClassLoader();
1568            currentThread.setContextClassLoader(classLoader);
1569            return old;
1570        }

1573    private static <T> T unwrap(final MarshalledObject<?> mo,
1574                                final ClassLoader cl,
1575                                final Class<T> wrappedClass)
1576            throws IOException {
1577        if (mo == null) {
1578            return null;
1579        }
1580        try {
1581            final ClassLoader old = AccessController.doPrivileged(new SetCcl(cl));
1582            try {
1583                return wrappedClass.cast(mo.get());
1584            } catch (ClassNotFoundException cnfe) {
1585                throw new UnmarshalException(cnfe.toString(), cnfe);
1586            } finally {
1587                AccessController.doPrivileged(new SetCcl(old));
1588            }
1589        } catch (PrivilegedActionException pe) {
1590            Exception e = extractException(pe);
1591            if (e instanceof IOException) {
1592                throw (IOException) e;
1593            }
1594            if (e instanceof ClassNotFoundException) {
1595                throw new UnmarshalException(e.toString(), e);
1596            }
1597            logger.warning("unwrap", "Failed to unmarshall object: " + e);
1598            logger.debug("unwrap", e);
1599        }
1600        return null;
1601    }
1602
1603    private static <T> T unwrap(final MarshalledObject<?> mo,
1604                                final ClassLoader cl1,
1605                                final ClassLoader cl2,
1606                                final Class<T> wrappedClass)
1607        throws IOException {
1608        if (mo == null) {
1609            return null;
1610        }
1611        try {
1612            ClassLoader orderCL = AccessController.doPrivileged(
1613                new PrivilegedExceptionAction<ClassLoader>() {
1614                    public ClassLoader run() throws Exception {
1615                        return new CombinedClassLoader(Thread.currentThread().getContextClassLoader(),
1616                                new OrderClassLoaders(cl1, cl2));
1617                    }
1618                }
1619            );
1620            return unwrap(mo, orderCL, wrappedClass);
1621        } catch (PrivilegedActionException pe) {
1622            Exception e = extractException(pe);
1623            if (e instanceof IOException) {
1624                throw (IOException) e;
1625            }
1626            if (e instanceof ClassNotFoundException) {
1627                throw new UnmarshalException(e.toString(), e);
1628            }
1629            logger.warning("unwrap", "Failed to unmarshall object: " + e);
1630            logger.debug("unwrap", e);
1631        }
1632        return null;
1633    }

1645    /**
1646     * Iterate until we extract the real exception
1647     * from a stack of PrivilegedActionExceptions.
1648     */
1649    private static Exception extractException(Exception e) {
1650        while (e instanceof PrivilegedActionException) {
1651            e = ((PrivilegedActionException)e).getException();

1793    private static final class CombinedClassLoader extends ClassLoader {
1794
1795        private final static class ClassLoaderWrapper extends ClassLoader {
1796            ClassLoaderWrapper(ClassLoader cl) {
1797                super(cl);
1798            }
1799
1800            @Override
1801            protected Class<?> loadClass(String name, boolean resolve)
1802                    throws ClassNotFoundException {
1803                return super.loadClass(name, resolve);
1804            }
1805        };
1806
1807        final ClassLoaderWrapper defaultCL;
1808
1809        private CombinedClassLoader(ClassLoader parent, ClassLoader defaultCL) {
1810            super(parent);
1811            this.defaultCL = new ClassLoaderWrapper(defaultCL);
1812        }
1813
1814        @Override
1815        protected Class<?> loadClass(String name, boolean resolve)
1816        throws ClassNotFoundException {
1817            ReflectUtil.checkPackageAccess(name);
1818            try {
1819                super.loadClass(name, resolve);
1820            } catch(Exception e) {
1821                for(Throwable t = e; t != null; t = t.getCause()) {
1822                    if(t instanceof SecurityException) {
1823                        throw t==e?(SecurityException)t:new SecurityException(t.getMessage(), e);
1824                    }
1825                }
1826            }
1827            final Class<?> cl = defaultCL.loadClass(name, resolve);
1828            return cl;
1829        }
