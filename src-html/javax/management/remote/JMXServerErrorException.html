027package javax.management.remote;
028
029import java.io.IOException;
030
031// imports for javadoc
032import javax.management.MBeanServer;
033
034/**
035 * Exception thrown as the result of a remote {@link MBeanServer}
036 * method invocation when an <code>Error</code> is thrown while
037 * processing the invocation in the remote MBean server.  A
038 * <code>JMXServerErrorException</code> instance contains the original
039 * <code>Error</code> that occurred as its cause.
040 *
041 * @see java.rmi.ServerError
042 * @since 1.5
043 */
044public class JMXServerErrorException extends IOException {
045
046    private static final long serialVersionUID = 3996732239558744666L;
047
048    /**
049     * Constructs a <code>JMXServerErrorException</code> with the specified
050     * detail message and nested error.
051     *
052     * @param s the detail message.
053     * @param err the nested error.  An instance of this class can be
054     * constructed where this parameter is null, but the standard
055     * connectors will never do so.
056     */
057    public JMXServerErrorException(String s, Error err) {
058        super(s);
059        cause = err;
060    }
061
062    public Throwable getCause() {
063        return cause;
064    }
065
066    /**
067     * @serial An {@link Error} that caused this exception to be thrown.
068     * @see #getCause()
069     **/
070    private final Error cause;
071}
