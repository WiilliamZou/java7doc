026package javax.management;
027
028
029/**
030 * This class represents a boolean value. A BooleanValueExp may be
031 * used anywhere a ValueExp is required.
032 * @serial include
033 *
034 * @since 1.5
035 */
036class BooleanValueExp extends QueryEval implements ValueExp {
037
038    /* Serial version */
039    private static final long serialVersionUID = 7754922052666594581L;
040
041    /**
042     * @serial The boolean value
043     */
044    private boolean val = false;
045
046
047    /** Creates a new BooleanValueExp representing the boolean literal <val>.*/
048    BooleanValueExp(boolean val) {
049        this.val = val;
050    }
051
052    /**Creates a new BooleanValueExp representing the Boolean object <val>.*/
053    BooleanValueExp(Boolean val) {
054        this.val = val.booleanValue();
055    }
056
057
058    /** Returns the  Boolean object representing the value of the BooleanValueExp object.*/
059    public Boolean getValue()  {
060        return Boolean.valueOf(val);
061    }
062
063    /**
064     * Returns the string representing the object.
065     */
066    public String toString()  {
067        return String.valueOf(val);
068    }
069
070    /**
071     * Applies the ValueExp on a MBean.
072     *
073     * @param name The name of the MBean on which the ValueExp will be applied.
074     *
075     * @return  The <CODE>ValueExp</CODE>.
076     *
077     * @exception BadStringOperationException
078     * @exception BadBinaryOpValueExpException
079     * @exception BadAttributeValueExpException
080     * @exception InvalidApplicationException
081     */
082    public ValueExp apply(ObjectName name) throws BadStringOperationException, BadBinaryOpValueExpException,
083        BadAttributeValueExpException, InvalidApplicationException  {
084        return this;
085    }
086
087    @Deprecated
088    public void setMBeanServer(MBeanServer s) {
089        super.setMBeanServer(s);
090    }
091
092
093 }
