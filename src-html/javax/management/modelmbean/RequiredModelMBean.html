089/**
090 * This class is the implementation of a ModelMBean. An appropriate
091 * implementation of a ModelMBean must be shipped with every JMX Agent
092 * and the class must be named RequiredModelMBean.
093 * <P>
094 * Java resources wishing to be manageable instantiate the
095 * RequiredModelMBean using the MBeanServer's createMBean method.
096 * The resource then sets the MBeanInfo and Descriptors for the
097 * RequiredModelMBean instance. The attributes and operations exposed
098 * via the ModelMBeanInfo for the ModelMBean are accessible
099 * from MBeans, connectors/adaptors like other MBeans. Through the
100 * Descriptors, values and methods in the managed application can be
101 * defined and mapped to attributes and operations of the ModelMBean.
102 * This mapping can be defined in an XML formatted file or dynamically and
103 * programmatically at runtime.
104 * <P>
105 * Every RequiredModelMBean which is instantiated in the MBeanServer
106 * becomes manageable:<br>
107 * its attributes and operations become remotely accessible through the
108 * connectors/adaptors connected to that MBeanServer.
109 * <P>
110 * A Java object cannot be registered in the MBeanServer unless it is a
111 * JMX compliant MBean. By instantiating a RequiredModelMBean, resources
112 * are guaranteed that the MBean is valid.
113 *
114 * MBeanException and RuntimeOperationsException must be thrown on every
115 * public method.  This allows for wrapping exceptions from distributed
116 * communications (RMI, EJB, etc.)
117 *
118 * @since 1.5
119 */
120

457    /**
458     * The resolveForCacheValue method checks the descriptor passed in to
459     * see if there is a valid cached value in the descriptor.
460     * The valid value will be in the 'value' field if there is one.
461     * If the 'currencyTimeLimit' field in the descriptor is:
462     * <ul>
463     *   <li><b>&lt;0</b> Then the value is not cached and is never valid.
464     *         Null is returned. The 'value' and 'lastUpdatedTimeStamp'
465     *         fields are cleared.</li>
466     *   <li><b>=0</b> Then the value is always cached and always valid.
467     *         The 'value' field is returned.
468     *         The 'lastUpdatedTimeStamp' field is not checked.</li>
469     *   <li><b>&gt;0</b> Represents the number of seconds that the
470     *         'value' field is valid.
471     *         The 'value' field is no longer valid when
472     *         'lastUpdatedTimeStamp' + 'currencyTimeLimit' &gt; Now.</li>
473     * </ul>
474     * <li>When 'value' is valid, 'valid' is returned.</li>
475     * <li>When 'value' is no longer valid then null is returned and
476     *     'value' and 'lastUpdatedTimeStamp' fields are cleared.</li>
477     *
478     **/
479    private Object resolveForCacheValue(Descriptor descr)
480        throws MBeanException, RuntimeOperationsException {
481
482        final boolean tracing = MODELMBEAN_LOGGER.isLoggable(Level.FINER);
483        final String mth = "resolveForCacheValue(Descriptor)";
484        if (tracing) {
485            MODELMBEAN_LOGGER.logp(Level.FINER,
486                    RequiredModelMBean.class.getName(),mth,"Entry");
487        }
488
489        Object response = null;
490        boolean resetValue = false, returnCachedValue = true;
491        long currencyPeriod = 0;
492
493        if (descr == null) {
494            if (tracing) {
495                MODELMBEAN_LOGGER.logp(Level.FINER,
496                        RequiredModelMBean.class.getName(),mth,
497                    "Input Descriptor is null");
498            }
499            return response;
500        }
501
502        if (tracing) {
503            MODELMBEAN_LOGGER.logp(Level.FINER,
504                    RequiredModelMBean.class.getName(),
505                    mth, "descriptor is " + descr);
506        }
507
508        final Descriptor mmbDescr = modelMBeanInfo.getMBeanDescriptor();
509        if (mmbDescr == null) {
510            if (tracing) {
511                MODELMBEAN_LOGGER.logp(Level.FINER,
512                    RequiredModelMBean.class.getName(),
513                        mth,"MBean Descriptor is null");
514            }
515            //return response;
516        }
517
518        Object objExpTime = descr.getFieldValue("currencyTimeLimit");
519
520        String expTime;
521        if (objExpTime != null) {
522            expTime = objExpTime.toString();
523        } else {
524            expTime = null;
525        }
526
527        if ((expTime == null) && (mmbDescr != null)) {
528            objExpTime = mmbDescr.getFieldValue("currencyTimeLimit");
529            if (objExpTime != null) {
530                expTime = objExpTime.toString();
531            } else {
532                expTime = null;
533            }
534        }
535
536        if (expTime != null) {
537            if (tracing) {
538                MODELMBEAN_LOGGER.logp(Level.FINER,
539                    RequiredModelMBean.class.getName(),
540                        mth,"currencyTimeLimit: " + expTime);
541            }
542
543            // convert seconds to milliseconds for time comparison
544            currencyPeriod = ((new Long(expTime)).longValue()) * 1000;
545            if (currencyPeriod < 0) {
546                /* if currencyTimeLimit is -1 then value is never cached */
547                returnCachedValue = false;
548                resetValue = true;
549                if (tracing) {
550                    MODELMBEAN_LOGGER.logp(Level.FINER,
551                            RequiredModelMBean.class.getName(),mth,
552                        currencyPeriod + ": never Cached");
553                }
554            } else if (currencyPeriod == 0) {
555                /* if currencyTimeLimit is 0 then value is always cached */
556                returnCachedValue = true;
557                resetValue = false;
558                if (tracing) {
559                    MODELMBEAN_LOGGER.logp(Level.FINER,
560                            RequiredModelMBean.class.getName(),mth,
561                        "always valid Cache");
562                }
563            } else {
564                Object objtStamp =
565                    descr.getFieldValue("lastUpdatedTimeStamp");
566
567                String tStamp;
568                if (objtStamp != null) tStamp = objtStamp.toString();
569                else tStamp = null;
570
571                if (tracing) {
572                    MODELMBEAN_LOGGER.logp(Level.FINER,
573                            RequiredModelMBean.class.getName(),mth,
574                        "lastUpdatedTimeStamp: " + tStamp);
575                }
576
577                if (tStamp == null)
578                    tStamp = "0";
579
580                long lastTime = (new Long(tStamp)).longValue();
581
582                if (tracing) {
583                    MODELMBEAN_LOGGER.logp(Level.FINER,
584                            RequiredModelMBean.class.getName(),mth,
585                        "currencyPeriod:" + currencyPeriod +
586                        " lastUpdatedTimeStamp:" + lastTime);
587                }
588
589                long now = (new Date()).getTime();
590
591                if (now < (lastTime + currencyPeriod)) {
592                    returnCachedValue = true;
593                    resetValue = false;
594                    if (tracing) {
595                        MODELMBEAN_LOGGER.logp(Level.FINER,
596                                RequiredModelMBean.class.getName(),mth,
597                            " timed valid Cache for " + now + " < " +
598                            (lastTime + currencyPeriod));
599                    }
600                } else { /* value is expired */
601                    returnCachedValue = false;
602                    resetValue = true;
603                    if (tracing) {
604                        MODELMBEAN_LOGGER.logp(Level.FINER,
605                                RequiredModelMBean.class.getName(),mth,
606                            "timed expired cache for " + now + " > " +
607                            (lastTime + currencyPeriod));
608                    }
609                }
610            }
611            if (tracing) {
612                MODELMBEAN_LOGGER.logp(Level.FINER,
613                        RequiredModelMBean.class.getName(),mth,
614                    "returnCachedValue:" + returnCachedValue +
615                    " resetValue: " + resetValue);
616            }
617
618            if (returnCachedValue == true) {
619                Object currValue = descr.getFieldValue("value");
620                if (currValue != null) {
621                    /* error/validity check return value here */
622                    response = currValue;
623                    /* need to cast string cached value to type */
624                    if (tracing) {
625                        MODELMBEAN_LOGGER.logp(Level.FINER,
626                                RequiredModelMBean.class.getName(),mth,
627                            "valid Cache value: " + currValue);
628                    }
629
630                } else {
631                    response = null;
632                    if (tracing) {
633                        MODELMBEAN_LOGGER.logp(Level.FINER,
634                            RequiredModelMBean.class.getName(),
635                                mth,"no Cached value");
636                    }
637                }
638            }
639
640            if (resetValue == true) {
641                /* value is not current, so remove it */
642                descr.removeField("lastUpdatedTimeStamp");
643                descr.removeField("value");
644                response = null;
645                modelMBeanInfo.setDescriptor(descr,null);
646                if (tracing) {
647                    MODELMBEAN_LOGGER.logp(Level.FINER,
648                        RequiredModelMBean.class.getName(),
649                            mth,"reset cached value to null");
650                }
651            }
652        }
653
654        if (tracing) {
655            MODELMBEAN_LOGGER.logp(Level.FINER,
656                    RequiredModelMBean.class.getName(),mth,"Exit");
657        }
658
659        return response;
660    }

804    /**
805     * Invokes a method on or through a RequiredModelMBean and returns
806     * the result of the method execution.
807     * <P>
808     * If the given method to be invoked, together with the provided
809     * signature, matches one of RequiredModelMbean
810     * accessible methods, this one will be call. Otherwise the call to
811     * the given method will be tried on the managed resource.
812     * <P>
813     * The last value returned by an operation may be cached in
814     * the operation's descriptor which
815     * is in the ModelMBeanOperationInfo's descriptor.
816     * The valid value will be in the 'value' field if there is one.
817     * If the 'currencyTimeLimit' field in the descriptor is:
818     * <UL>
819     * <LI><b>&lt;0</b> Then the value is not cached and is never valid.
820     *      The operation method is invoked.
821     *      The 'value' and 'lastUpdatedTimeStamp' fields are cleared.</LI>
822     * <LI><b>=0</b> Then the value is always cached and always valid.
823     *      The 'value' field is returned. If there is no 'value' field
824     *      then the operation method is invoked for the attribute.
825     *      The 'lastUpdatedTimeStamp' field and `value' fields are set to
826     *      the operation's return value and the current time stamp.</LI>
827     * <LI><b>&gt;0</b> Represents the number of seconds that the 'value'
828     *      field is valid.
829     *      The 'value' field is no longer valid when
830     *      'lastUpdatedTimeStamp' + 'currencyTimeLimit' &gt; Now.
831     *      <UL>
832     *         <LI>When 'value' is valid, 'value' is returned.</LI>
833     *         <LI>When 'value' is no longer valid then the operation
834     *             method is invoked. The 'lastUpdatedTimeStamp' field
835     *             and `value' fields are updated.</lI>
836     *      </UL>
837     * </LI>
838     * </UL>
839     *
840     * <p><b>Note:</b> because of inconsistencies in previous versions of
841     * this specification, it is recommended not to use negative or zero
842     * values for <code>currencyTimeLimit</code>.  To indicate that a
843     * cached value is never valid, omit the
844     * <code>currencyTimeLimit</code> field.  To indicate that it is
845     * always valid, use a very large number for this field.</p>
846     *
847     * @param opName The name of the method to be invoked. The
848     *     name can be the fully qualified method name including the
849     *     classname, or just the method name if the classname is
850     *     defined in the 'class' field of the operation descriptor.
851     * @param opArgs An array containing the parameters to be set
852     *     when the operation is invoked
853     * @param sig An array containing the signature of the
854     *     operation. The class objects will be loaded using the same
855     *     class loader as the one used for loading the MBean on which
856     *     the operation was invoked.
857     *
858     * @return  The object returned by the method, which represents the
859     *     result of invoking the method on the specified managed resource.
860     *
861     * @exception MBeanException  Wraps one of the following Exceptions:
862     * <UL>
863     * <LI> An Exception thrown by the managed object's invoked method.</LI>
864     * <LI> {@link ServiceNotFoundException}: No ModelMBeanOperationInfo or
865     *      no descriptor defined for the specified operation or the managed
866     *      resource is null.</LI>
867     * <LI> {@link InvalidTargetObjectTypeException}: The 'targetType'
868     *      field value is not 'objectReference'.</LI>
869     * </UL>
870     * @exception ReflectionException  Wraps an {@link java.lang.Exception}
871     *      thrown while trying to invoke the method.
872     * @exception RuntimeOperationsException Wraps an
873     *      {@link IllegalArgumentException} Method name is null.
874     *
875     **/
876    /*
877      The requirement to be able to invoke methods on the
878      RequiredModelMBean class itself makes this method considerably
879      more complicated than it might otherwise be.  Note that, unlike
880      earlier versions, we do not allow you to invoke such methods if
881      they are not explicitly mentioned in the ModelMBeanInfo.  Doing
882      so was potentially a security problem, and certainly very
883      surprising.
884
885      We do not look for the method in the RequiredModelMBean class
886      itself if:
887      (a) there is a "targetObject" field in the Descriptor for the
888      operation; or
889      (b) there is a "class" field in the Descriptor for the operation
890      and the named class is not RequiredModelMBean or one of its
891      superinterfaces; or
892      (c) the name of the operation is not the name of a method in
893      RequiredModelMBean (this is just an optimization).
894
895      In cases (a) and (b), if you have gone to the trouble of adding
896      those fields specifically for this operation then presumably you
897      do not want RequiredModelMBean's methods to be called.
898
899      We have to pay attention to class loading issues.  If the
900      "class" field is present, the named class has to be resolved
901      relative to RequiredModelMBean's class loader to test the
902      condition (b) above, and relative to the managed resource's
903      class loader to ensure that the managed resource is in fact of
904      the named class (or a subclass).  The class names in the sig
905      array likewise have to be resolved, first against
906      RequiredModelMBean's class loader, then against the managed
907      resource's class loader.  There is no point in using any other
908      loader because when we call Method.invoke we must call it on
909      a Method that is implemented by the target object.
910     */
911    public Object invoke(String opName, Object[] opArgs, String[] sig)
912            throws MBeanException, ReflectionException {
913
914        final boolean tracing = MODELMBEAN_LOGGER.isLoggable(Level.FINER);
915        final String mth = "invoke(String, Object[], String[])";
916
917        if (tracing) {
918            MODELMBEAN_LOGGER.logp(Level.FINER,
919                    RequiredModelMBean.class.getName(), mth, "Entry");
920        }
921
922        if (opName == null) {
923            final RuntimeException x =
924                new IllegalArgumentException("Method name must not be null");
925            throw new RuntimeOperationsException(x,
926                      "An exception occurred while trying to " +
927                      "invoke a method on a RequiredModelMBean");
928        }
929
930        String opClassName = null;
931        String opMethodName;
932
933        // Parse for class name and method
934        int opSplitter = opName.lastIndexOf(".");
935        if (opSplitter > 0) {
936            opClassName = opName.substring(0,opSplitter);
937            opMethodName = opName.substring(opSplitter+1);
938        } else
939            opMethodName = opName;
940
941        /* Ignore anything after a left paren.  We keep this for
942           compatibility but it isn't specified.  */
943        opSplitter = opMethodName.indexOf("(");
944        if (opSplitter > 0)
945            opMethodName = opMethodName.substring(0,opSplitter);
946
947        if (tracing) {
948            MODELMBEAN_LOGGER.logp(Level.FINER,
949                    RequiredModelMBean.class.getName(),
950                    mth, "Finding operation " + opName + " as " + opMethodName);
951        }
952
953        ModelMBeanOperationInfo opInfo =
954            modelMBeanInfo.getOperation(opMethodName);
955        if (opInfo == null) {
956            final String msg =
957                "Operation " + opName + " not in ModelMBeanInfo";
958            throw new MBeanException(new ServiceNotFoundException(msg), msg);
959        }
960
961        final Descriptor opDescr = opInfo.getDescriptor();
962        if (opDescr == null) {
963            final String msg = "Operation descriptor null";
964            throw new MBeanException(new ServiceNotFoundException(msg), msg);
965        }
966
967        final Object cached = resolveForCacheValue(opDescr);
968        if (cached != null) {
969            if (tracing) {
970                MODELMBEAN_LOGGER.logp(Level.FINER,
971                        RequiredModelMBean.class.getName(),
972                        mth,
973                        "Returning cached value");
974            }
975            return cached;
976        }
977
978        if (opClassName == null)
979            opClassName = (String) opDescr.getFieldValue("class");
980        // may still be null now
981
982        opMethodName = (String) opDescr.getFieldValue("name");
983        if (opMethodName == null) {
984            final String msg =
985                "Method descriptor must include `name' field";
986            throw new MBeanException(new ServiceNotFoundException(msg), msg);
987        }
988
989        final String targetTypeField = (String)
990            opDescr.getFieldValue("targetType");
991        if (targetTypeField != null
992            && !targetTypeField.equalsIgnoreCase("objectReference")) {
993            final String msg =
994                "Target type must be objectReference: " + targetTypeField;
995            throw new MBeanException(new InvalidTargetObjectTypeException(msg),
996                                     msg);
997        }
998
999        final Object targetObjectField = opDescr.getFieldValue("targetObject");
1000        if (tracing && targetObjectField != null)
1001                MODELMBEAN_LOGGER.logp(Level.FINER,
1002                    RequiredModelMBean.class.getName(),
1003                        mth, "Found target object in descriptor");
1004
1005        /* Now look for the method, either in RequiredModelMBean itself
1006           or in the target object.  Set "method" and "targetObject"
1007           appropriately.  */
1008        Method method;
1009        Object targetObject;
1010
1011        method = findRMMBMethod(opMethodName, targetObjectField,
1012                                opClassName, sig);
1013
1014        if (method != null)
1015            targetObject = this;
1016        else {
1017            if (tracing) {
1018                MODELMBEAN_LOGGER.logp(Level.FINER,
1019                    RequiredModelMBean.class.getName(),
1020                        mth, "looking for method in managedResource class");
1021            }
1022            if (targetObjectField != null)
1023                targetObject = targetObjectField;
1024            else {
1025                targetObject = managedResource;
1026                if (targetObject == null) {
1027                    final String msg =
1028                        "managedResource for invoke " + opName +
1029                        " is null";
1030                    Exception snfe = new ServiceNotFoundException(msg);
1031                    throw new MBeanException(snfe);
1032                }
1033            }
1034
1035            final Class<?> targetClass;
1036
1037            if (opClassName != null) {
1038                try {
1039                    AccessControlContext stack = AccessController.getContext();
1040                    final Object obj = targetObject;
1041                    final String className = opClassName;
1042                    final ClassNotFoundException[] caughtException = new ClassNotFoundException[1];
1043
1044                    targetClass = javaSecurityAccess.doIntersectionPrivilege(new PrivilegedAction<Class<?>>() {
1045
1046                        @Override
1047                        public Class<?> run() {
1048                            try {
1049                                ReflectUtil.checkPackageAccess(className);
1050                                final ClassLoader targetClassLoader =
1051                                    obj.getClass().getClassLoader();
1052                                return Class.forName(className, false,
1053                                                            targetClassLoader);
1054                            } catch (ClassNotFoundException e) {
1055                                caughtException[0] = e;
1056                            }
1057                            return null;
1058                        }
1059                    }, stack, acc);
1060
1061                    if (caughtException[0] != null) {
1062                        throw caughtException[0];
1063                    }
1064                } catch (ClassNotFoundException e) {
1065                    final String msg =
1066                        "class for invoke " + opName + " not found";
1067                    throw new ReflectionException(e, msg);
1068                }
1069            } else
1070                targetClass = targetObject.getClass();
1071
1072            method = resolveMethod(targetClass, opMethodName, sig);
1073        }
1074
1075        if (tracing) {
1076            MODELMBEAN_LOGGER.logp(Level.FINER,
1077                RequiredModelMBean.class.getName(),
1078                    mth, "found " + opMethodName + ", now invoking");
1079        }
1080
1081        final Object result =
1082            invokeMethod(opName, method, targetObject, opArgs);
1083
1084        if (tracing) {
1085            MODELMBEAN_LOGGER.logp(Level.FINER,
1086                RequiredModelMBean.class.getName(),
1087                    mth, "successfully invoked method");
1088        }
1089
1090        if (result != null)
1091            cacheResult(opInfo, opDescr, result);
1092
1093        return result;
1094    }

1096    private Method resolveMethod(Class<?> targetClass,
1097                                        String opMethodName,
1098                                        final String[] sig)
1099            throws ReflectionException {
1100        final boolean tracing = MODELMBEAN_LOGGER.isLoggable(Level.FINER);
1101
1102        if (tracing) {
1103            MODELMBEAN_LOGGER.logp(Level.FINER,
1104                RequiredModelMBean.class.getName(),"resolveMethod",
1105                  "resolving " + targetClass.getName() + "." + opMethodName);
1106        }
1107
1108        final Class<?>[] argClasses;
1109
1110        if (sig == null)
1111            argClasses = null;
1112        else {
1113            final AccessControlContext stack = AccessController.getContext();
1114            final ReflectionException[] caughtException = new ReflectionException[1];
1115            final ClassLoader targetClassLoader = targetClass.getClassLoader();
1116            argClasses = new Class<?>[sig.length];
1117
1118            javaSecurityAccess.doIntersectionPrivilege(new PrivilegedAction<Void>() {
1119
1120                @Override
1121                public Void run() {
1122                    for (int i = 0; i < sig.length; i++) {
1123                        if (tracing) {
1124                            MODELMBEAN_LOGGER.logp(Level.FINER,
1125                                RequiredModelMBean.class.getName(),"resolveMethod",
1126                                    "resolve type " + sig[i]);
1127                        }
1128                        argClasses[i] = (Class<?>) primitiveClassMap.get(sig[i]);
1129                        if (argClasses[i] == null) {
1130                            try {
1131                                ReflectUtil.checkPackageAccess(sig[i]);
1132                                argClasses[i] =
1133                                    Class.forName(sig[i], false, targetClassLoader);
1134                            } catch (ClassNotFoundException e) {
1135                                if (tracing) {
1136                                    MODELMBEAN_LOGGER.logp(Level.FINER,
1137                                            RequiredModelMBean.class.getName(),
1138                                            "resolveMethod",
1139                                            "class not found");
1140                                }
1141                                final String msg = "Parameter class not found";
1142                                caughtException[0] = new ReflectionException(e, msg);
1143                            }
1144                        }
1145                    }
1146                    return null;
1147                }
1148            }, stack, acc);
1149
1150            if (caughtException[0] != null) {
1151                throw caughtException[0];
1152            }
1153        }
1154
1155        try {
1156            return targetClass.getMethod(opMethodName, argClasses);
1157        } catch (NoSuchMethodException e) {
1158            final String msg =
1159                "Target method not found: " + targetClass.getName() + "." +
1160                opMethodName;
1161            throw new ReflectionException(e, msg);
1162        }
1163    }

1181       parameters.  Return null if there is none, or if the parameters
1182       exclude using it.  Called from invoke. */

1184                                         Object targetObjectField,
1185                                         String opClassName,

1188
1189        if (tracing) {
1190            MODELMBEAN_LOGGER.logp(Level.FINER,
1191                RequiredModelMBean.class.getName(),
1192                    "invoke(String, Object[], String[])",
1193                  "looking for method in RequiredModelMBean class");
1194        }
1195
1196        if (!isRMMBMethodName(opMethodName))
1197            return null;
1198        if (targetObjectField != null)
1199            return null;
1200        final Class<RequiredModelMBean> rmmbClass = RequiredModelMBean.class;
1201        final Class<?> targetClass;
1202        if (opClassName == null)
1203            targetClass = rmmbClass;
1204        else {
1205            AccessControlContext stack = AccessController.getContext();
1206            final String className = opClassName;
1207            targetClass = javaSecurityAccess.doIntersectionPrivilege(new PrivilegedAction<Class<?>>() {
1208
1209                @Override
1210                public Class<?> run() {
1211                    try {
1212                        ReflectUtil.checkPackageAccess(className);
1213                        final ClassLoader targetClassLoader =
1214                            rmmbClass.getClassLoader();

1216                                                    targetClassLoader);

1218                            return null;

1221                        return null;

1229            return null;
1230        }
1231    }

1233    /*
1234     * Invoke the given method, and throw the somewhat unpredictable
1235     * appropriate exception if the method itself gets an exception.
1236     */
1237    private Object invokeMethod(String opName, final Method method,
1238                                final Object targetObject, final Object[] opArgs)
1239            throws MBeanException, ReflectionException {
1240        try {
1241            final Throwable[] caughtException = new Throwable[1];
1242            AccessControlContext stack = AccessController.getContext();
1243            Object rslt = javaSecurityAccess.doIntersectionPrivilege(new PrivilegedAction<Object>() {
1244
1245                @Override
1246                public Object run() {
1247                    try {
1248                        ReflectUtil.checkPackageAccess(method.getDeclaringClass());
1249                        return MethodUtil.invoke(method, targetObject, opArgs);
1250                    } catch (InvocationTargetException e) {
1251                        caughtException[0] = e;
1252                    } catch (IllegalAccessException e) {
1253                        caughtException[0] = e;
1254                    }
1255                    return null;
1256                }
1257            }, stack, acc);
1258            if (caughtException[0] != null) {
1259                if (caughtException[0] instanceof Exception) {
1260                    throw (Exception)caughtException[0];
1261                } else if(caughtException[0] instanceof Error) {
1262                    throw (Error)caughtException[0];
1263                }
1264            }
1265            return rslt;
1266        } catch (RuntimeErrorException ree) {
1267            throw new RuntimeOperationsException(ree,
1268                      "RuntimeException occurred in RequiredModelMBean "+
1269                      "while trying to invoke operation " + opName);
1270        } catch (RuntimeException re) {
1271            throw new RuntimeOperationsException(re,
1272                      "RuntimeException occurred in RequiredModelMBean "+
1273                      "while trying to invoke operation " + opName);
1274        } catch (IllegalAccessException iae) {
1275            throw new ReflectionException(iae,
1276                      "IllegalAccessException occurred in " +
1277                      "RequiredModelMBean while trying to " +
1278                      "invoke operation " + opName);
1279        } catch (InvocationTargetException ite) {
1280            Throwable mmbTargEx = ite.getTargetException();
1281            if (mmbTargEx instanceof RuntimeException) {
1282                throw new MBeanException ((RuntimeException)mmbTargEx,
1283                      "RuntimeException thrown in RequiredModelMBean "+
1284                      "while trying to invoke operation " + opName);
1285            } else if (mmbTargEx instanceof Error) {
1286                throw new RuntimeErrorException((Error)mmbTargEx,
1287                      "Error occurred in RequiredModelMBean while trying "+
1288                      "to invoke operation " + opName);
1289            } else if (mmbTargEx instanceof ReflectionException) {
1290                throw (ReflectionException) mmbTargEx;
1291            } else {

2851        throws ClassNotFoundException {
2852        AccessControlContext stack = AccessController.getContext();
2853        final ClassNotFoundException[] caughtException = new ClassNotFoundException[1];
2854
2855        Class c = javaSecurityAccess.doIntersectionPrivilege(new PrivilegedAction<Class<?>>() {
2856
2857            @Override
2858            public Class<?> run() {
2859                try {
2860                    ReflectUtil.checkPackageAccess(className);
2861                    return Class.forName(className);
2862                } catch (ClassNotFoundException e) {
2863                    final ClassLoaderRepository clr =
2864                        getClassLoaderRepository();
2865                    try {
2866                        if (clr == null) throw new ClassNotFoundException(className);
2867                        return clr.loadClass(className);
2868                    } catch (ClassNotFoundException ex) {
2869                        caughtException[0] = ex;
2870                    }
2871                }
2872                return null;
2873            }
2874        }, stack, acc);
2875
2876        if (caughtException[0] != null) {
2877            throw caughtException[0];
2878        }
2879
2880        return c;
