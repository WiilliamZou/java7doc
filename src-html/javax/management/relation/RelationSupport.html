047/**
048 * A RelationSupport object is used internally by the Relation Service to
049 * represent simple relations (only roles, no properties or methods), with an
050 * unlimited number of roles, of any relation type. As internal representation,
051 * it is not exposed to the user.
052 * <P>RelationSupport class conforms to the design patterns of standard MBean. So
053 * the user can decide to instantiate a RelationSupport object himself as
054 * a MBean (as it follows the MBean design patterns), to register it in the
055 * MBean Server, and then to add it in the Relation Service.
056 * <P>The user can also, when creating his own MBean relation class, have it
057 * extending RelationSupport, to retrieve the implementations of required
058 * interfaces (see below).
059 * <P>It is also possible to have in a user relation MBean class a member
060 * being a RelationSupport object, and to implement the required interfaces by
061 * delegating all to this member.
062 * <P> RelationSupport implements the Relation interface (to be handled by the
063 * Relation Service).
064 * <P>It implements also the MBeanRegistration interface to be able to retrieve
065 * the MBean Server where it is registered (if registered as a MBean) to access
066 * to its Relation Service.
067 *
068 * @since 1.5
069 */
070public class RelationSupport
071    implements RelationSupportMBean, MBeanRegistration {

120    /**
121     * Creates a {@code RelationSupport} object.
122     * <P>This constructor has to be used when the RelationSupport object will
123     * be registered as a MBean by the user, or when creating a user relation
124     * MBean whose class extends RelationSupport.
125     * <P>Nothing is done at the Relation Service level, i.e.
126     * the {@code RelationSupport} object is not added to the
127     * {@code RelationService} and no checks are performed to
128     * see if the provided values are correct.
129     * The object is always created, EXCEPT if:
130     * <P>- any of the required parameters is {@code null}.
131     * <P>- the same name is used for two roles.
132     * <P>To be handled as a relation, the {@code RelationSupport} object has
133     * to be added to the Relation Service using the Relation Service method
134     * addRelation().
135     *
136     * @param relationId  relation identifier, to identify the relation in the
137     * Relation Service.
138     * <P>Expected to be unique in the given Relation Service.
139     * @param relationServiceName  ObjectName of the Relation Service where
140     * the relation will be registered.
141     * <P>This parameter is required as it is the Relation Service that is
142     * aware of the definition of the relation type of the given relation,
143     * so that will be able to check update operations (set).
144     * @param relationTypeName  Name of relation type.
145     * <P>Expected to have been created in the given Relation Service.
146     * @param list  list of roles (Role objects) to initialize the
147     * relation. Can be {@code null}.
148     * <P>Expected to conform to relation info in associated relation type.
149     *
150     * @exception InvalidRoleValueException  if the same name is used for two
151     * roles.
152     * @exception IllegalArgumentException  if any of the required parameters
153     * (relation id, relation service ObjectName, or relation type name) is
154     * {@code null}.
155     */
156    public RelationSupport(String relationId,
157                        ObjectName relationServiceName,
158                        String relationTypeName,
159                        RoleList list)
160        throws InvalidRoleValueException,
161               IllegalArgumentException {
162
163        super();
164
165        RELATION_LOGGER.entering(RelationSupport.class.getName(),
166                "RelationSupport");
167
168        // Can throw InvalidRoleValueException and IllegalArgumentException
169        initMembers(relationId,
170                    relationServiceName,
171                    null,
172                    relationTypeName,
173                    list);
174
175        RELATION_LOGGER.exiting(RelationSupport.class.getName(),
176                "RelationSupport");
177    }

197     * <P>- any of the required parameters is {@code null}.

200     * to be added to the Relation Service using the Relation Service method

203     * @param relationId  relation identifier, to identify the relation in the
204     * Relation Service.
205     * <P>Expected to be unique in the given Relation Service.
206     * @param relationServiceName  ObjectName of the Relation Service where
207     * the relation will be registered.

384    /**
385     * Returns the number of MBeans currently referenced in the given role.
386     *
387     * @param roleName  name of role
388     *
389     * @return the number of currently referenced MBeans in that role
390     *
391     * @exception IllegalArgumentException  if null role name
392     * @exception RoleNotFoundException  if there is no role with given name
393     */
394    public Integer getRoleCardinality(String roleName)
395        throws IllegalArgumentException,
396               RoleNotFoundException {
397
398        if (roleName == null) {
399            String excMsg = "Invalid parameter.";
400            throw new IllegalArgumentException(excMsg);
401        }
402
403        RELATION_LOGGER.entering(RelationSupport.class.getName(),
404                "getRoleCardinality", roleName);
405
406        // Try to retrieve the role
407        Role role;
408        synchronized(myRoleName2ValueMap) {
409            // No null Role is allowed, so direct use of get()
410            role = (myRoleName2ValueMap.get(roleName));
411        }
412        if (role == null) {
413            int pbType = RoleStatus.NO_ROLE_WITH_NAME;
414            // Will throw a RoleNotFoundException
415            //
416            // Will not throw InvalidRoleValueException, so catch it for the
417            // compiler
418            try {
419                RelationService.throwRoleProblemException(pbType,
420                                                          roleName);
421            } catch (InvalidRoleValueException exc) {
422                // OK : Do not throw InvalidRoleValueException as
423                //      a RoleNotFoundException will be thrown.
424            }
425        }
426
427        List<ObjectName> roleValue = role.getRoleValue();
428
429        RELATION_LOGGER.exiting(RelationSupport.class.getName(),
430                "getRoleCardinality");
431        return roleValue.size();
432    }

589    /**
590     * Retrieves MBeans referenced in the various roles of the relation.
591     *
592     * @return a HashMap mapping:
593     * <P> ObjectName -> ArrayList of String (role names)
594     */
595    public Map<ObjectName,List<String>> getReferencedMBeans() {
596
597        RELATION_LOGGER.entering(RelationSupport.class.getName(),
598                "getReferencedMBeans");
599
600        Map<ObjectName,List<String>> refMBeanMap =
601            new HashMap<ObjectName,List<String>>();
602
603        synchronized(myRoleName2ValueMap) {
604
605            for (Role currRole : myRoleName2ValueMap.values()) {
606
607                String currRoleName = currRole.getRoleName();
608                // Retrieves ObjectNames of MBeans referenced in current role
609                List<ObjectName> currRefMBeanList = currRole.getRoleValue();
610
611                for (ObjectName currRoleObjName : currRefMBeanList) {
612
613                    // Sees if current MBean has been already referenced in
614                    // roles already seen
615                    List<String> mbeanRoleNameList =
616                        refMBeanMap.get(currRoleObjName);
617
618                    boolean newRefFlg = false;
619                    if (mbeanRoleNameList == null) {
620                        newRefFlg = true;
621                        mbeanRoleNameList = new ArrayList<String>();
622                    }
623                    mbeanRoleNameList.add(currRoleName);
624                    if (newRefFlg) {
625                        refMBeanMap.put(currRoleObjName, mbeanRoleNameList);
626                    }
627                }
628            }
629        }
630
631        RELATION_LOGGER.exiting(RelationSupport.class.getName(),
632                "getReferencedMBeans");
633        return refMBeanMap;
634    }

803                // avoiding MBean Server

815                // So we have a MBean. We must access the Relation Service
816                // via the MBean Server.

818                params[0] = roleName;
819                params[1] = myRelTypeName;

821                signature[0] = "java.lang.String";
822                signature[1] = "java.lang.String";

824                // Service is not registered (to be catched in any case and

826                //

828                // or an InstanceNotFoundException
829                try {

856                //       cloning, null value not allowed, empty list if
857                //       nothing.

859
860            } else {

862                // role

864            }
865
866        } else {

871                // found or not readable, so raises a RoleNotFoundException.

874                                                              roleName);
875                    // To keep compiler happy :)
876                    return null;
877                } catch (InvalidRoleValueException exc) {
878                    throw new RuntimeException(exc.getMessage());
879                }
880
881            } else {
882                // Problem when retrieving a role in a multi-role retrieval:
883                // returns a RoleUnresolved object
884                result = new RoleUnresolved(roleName, null, pbType);
885            }
886        }

1597    //
1598    // -param list  role list. Can be null.
1599    //  As it is a RoleList object, it cannot include null (rejected).

1605        throws InvalidRoleValueException {
1606
1607        if (list == null) {
1608            return;

1611        RELATION_LOGGER.entering(RelationSupport.class.getName(),
1612                "initRoleMap", list);
1613
1614        synchronized(myRoleName2ValueMap) {
1615
1616            for (Role currRole : list.asList()) {
1617
1618                // No need to check if role is null, it is not allowed to store
1619                // a null role in a RoleList :)
1620                String currRoleName = currRole.getRoleName();

1639    // Callback used by the Relation Service when a MBean referenced in a role
1640    // is unregistered.
1641    // The Relation Service will call this method to let the relation
1642    // take action to reflect the impact of such unregistration.
1643    // Current implementation is to set the role with its current value
1644    // (list of ObjectNames of referenced MBeans) without the unregistered
1645    // one.
1646    //
1647    // -param objectName  ObjectName of unregistered MBean
1648    // -param roleName  name of role where the MBean is referenced
1649    // -param relationServCallFlg  true if call from the Relation Service; this
1650    //  will happen if the current RelationSupport object has been created by
1651    //  the Relation Service (via createRelation()) method, so direct access is
1652    //  possible.
1653    // -param relationServ  reference to Relation Service object, if internal
1654    //  relation
1655    //
1656    // -exception IllegalArgumentException if null parameter
1657    // -exception RoleNotFoundException  if:
1658    //  - the role does not exist
1659    //  or
1660    //  - role not writable.
1661    // -exception InvalidRoleValueException  if value provided for:
1662    //   - the number of referenced MBeans in given value is less than
1663    //     expected minimum degree
1664    //   or
1665    //   - the number of referenced MBeans in provided value exceeds expected
1666    //     maximum degree
1667    //   or
1668    //   - one referenced MBean in the value is not an Object of the MBean
1669    //     class expected for that role
1670    //   or
1671    //   - a MBean provided for that role does not exist
1672    // -exception RelationServiceNotRegisteredException  if the Relation
1673    //  Service is not registered in the MBean Server
1674    // -exception RelationTypeNotFoundException if unknown relation type
1675    // -exception RelationNotFoundException if current relation has not been
1676    //  added in the RS
1677    void handleMBeanUnregistrationInt(ObjectName objectName,
1678                                      String roleName,
1679                                      boolean relationServCallFlg,
1680                                      RelationService relationServ)
1681        throws IllegalArgumentException,
1682               RoleNotFoundException,
1683               InvalidRoleValueException,
1684               RelationServiceNotRegisteredException,
1685               RelationTypeNotFoundException,
1686               RelationNotFoundException {
1687
1688        if (objectName == null ||
1689            roleName == null ||
1690            (relationServCallFlg && relationServ == null)) {
1691            String excMsg = "Invalid parameter.";
1692            throw new IllegalArgumentException(excMsg);
1693        }
1694
1695        RELATION_LOGGER.entering(RelationSupport.class.getName(),
1696                "handleMBeanUnregistrationInt", new Object[] {objectName,
1697                roleName, relationServCallFlg, relationServ});
1698
1699        // Retrieves current role value
1700        Role role;
1701        synchronized(myRoleName2ValueMap) {
1702            role = (myRoleName2ValueMap.get(roleName));
1703        }
1704
1705        if (role == null) {
1706            StringBuilder excMsgStrB = new StringBuilder();

1709            excMsgStrB.append(roleName);

1717        newRoleValue.remove(objectName);
1718        Role newRole = new Role(roleName, newRoleValue);
1719
1720        // Can throw InvalidRoleValueException,
1721        // RelationTypeNotFoundException
1722        // (RoleNotFoundException already detected)
1723        Object result =
1724            setRoleInt(newRole, relationServCallFlg, relationServ, false);
1725
1726        RELATION_LOGGER.exiting(RelationSupport.class.getName(),
1727                "handleMBeanUnregistrationInt");
1728        return;
