026package javax.management.relation;
027
028/**
029 * This class describes the various problems which can be encountered when
030 * accessing a role.
031 *
032 * @since 1.5
033 */
034public class RoleStatus {
035
036    //
037    // Possible problems
038    //
039
040    /**
041     * Problem type when trying to access an unknown role.
042     */
043    public static final int NO_ROLE_WITH_NAME = 1;
044    /**
045     * Problem type when trying to read a non-readable attribute.
046     */
047    public static final int ROLE_NOT_READABLE = 2;
048    /**
049     * Problem type when trying to update a non-writable attribute.
050     */
051    public static final int ROLE_NOT_WRITABLE = 3;
052    /**
053     * Problem type when trying to set a role value with less ObjectNames than
054     * the minimum expected cardinality.
055     */
056    public static final int LESS_THAN_MIN_ROLE_DEGREE = 4;
057    /**
058     * Problem type when trying to set a role value with more ObjectNames than
059     * the maximum expected cardinality.
060     */
061    public static final int MORE_THAN_MAX_ROLE_DEGREE = 5;
062    /**
063     * Problem type when trying to set a role value including the ObjectName of
064     * a MBean not of the class expected for that role.
065     */
066    public static final int REF_MBEAN_OF_INCORRECT_CLASS = 6;
067    /**
068     * Problem type when trying to set a role value including the ObjectName of
069     * a MBean not registered in the MBean Server.
070     */
071    public static final int REF_MBEAN_NOT_REGISTERED = 7;
072
073    /**
074     * Returns true if given value corresponds to a known role status, false
075     * otherwise.
076     *
077     * @param status a status code.
078     *
079     * @return true if this value is a known role status.
080     */
081    public static boolean isRoleStatus(int status) {
082        if (status != NO_ROLE_WITH_NAME &&
083            status != ROLE_NOT_READABLE &&
084            status != ROLE_NOT_WRITABLE &&
085            status != LESS_THAN_MIN_ROLE_DEGREE &&
086            status != MORE_THAN_MAX_ROLE_DEGREE &&
087            status != REF_MBEAN_OF_INCORRECT_CLASS &&
088            status != REF_MBEAN_NOT_REGISTERED) {
089            return false;
090        }
091        return true;
092    }
093}
