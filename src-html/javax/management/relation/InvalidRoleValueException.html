026package javax.management.relation;
027
028/**
029 * Role value is invalid.
030 * This exception is raised when, in a role, the number of referenced MBeans
031 * in given value is less than expected minimum degree, or the number of
032 * referenced MBeans in provided value exceeds expected maximum degree, or
033 * one referenced MBean in the value is not an Object of the MBean
034 * class expected for that role, or an MBean provided for that role does not
035 * exist.
036 *
037 * @since 1.5
038 */
039public class InvalidRoleValueException extends RelationException {
040
041    /* Serial version */
042    private static final long serialVersionUID = -2066091747301983721L;
043
044    /**
045     * Default constructor, no message put in exception.
046     */
047    public InvalidRoleValueException() {
048        super();
049    }
050
051    /**
052     * Constructor with given message put in exception.
053     *
054     * @param message the detail message.
055     */
056    public InvalidRoleValueException(String message) {
057        super(message);
058    }
059}
