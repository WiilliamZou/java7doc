026package javax.accessibility;
027
028
029import java.util.*;
030import java.awt.*;
031import javax.swing.text.*;
032
033
034/**
035 * <P>The AccessibleExtendedText interface contains additional methods
036 * not provided by the AccessibleText interface
037 *
038 * Applications can determine if an object supports the AccessibleExtendedText
039 * interface by first obtaining its AccessibleContext (see {@link Accessible})
040 * and then calling the {@link AccessibleContext#getAccessibleText} method of
041 * AccessibleContext.  If the return value is an instance of
042 * AccessibleExtendedText, the object supports this interface.
043 *
044 * @see Accessible
045 * @see Accessible#getAccessibleContext
046 * @see AccessibleContext
047 * @see AccessibleContext#getAccessibleText
048 *
049 * @author       Peter Korn
050 * @author       Lynn Monsanto
051 * @since 1.5
052 */
053public interface AccessibleExtendedText {
054
055    /**
056     * Constant used to indicate that the part of the text that should be
057     * retrieved is a line of text.
058     *
059     * @see AccessibleText#getAtIndex
060     * @see AccessibleText#getAfterIndex
061     * @see AccessibleText#getBeforeIndex
062     */
063    public static final int LINE = 4; // BugID: 4849720
064
065    /**
066     * Constant used to indicate that the part of the text that should be
067     * retrieved is contiguous text with the same text attributes.
068     *
069     * @see AccessibleText#getAtIndex
070     * @see AccessibleText#getAfterIndex
071     * @see AccessibleText#getBeforeIndex
072     */
073    public static final int ATTRIBUTE_RUN = 5; // BugID: 4849720
074
075    /**
076     * Returns the text between two indices
077     *
078     * @param startIndex the start index in the text
079     * @param endIndex the end index in the text
080     * @return the text string if the indices are valid.
081     * Otherwise, null is returned.
082     */
083    public String getTextRange(int startIndex, int endIndex);
084
085    /**
086     * Returns the <code>AccessibleTextSequence</code> at a given index.
087     *
088     * @param part the <code>CHARACTER</code>, <code>WORD</code>,
089     * <code>SENTENCE</code>, <code>LINE</code> or <code>ATTRIBUTE_RUN</code>
090     * to retrieve
091     * @param index an index within the text
092     * @return an <code>AccessibleTextSequence</code> specifying the text
093     * if part and index are valid.  Otherwise, null is returned.
094     *
095     * @see AccessibleText#CHARACTER
096     * @see AccessibleText#WORD
097     * @see AccessibleText#SENTENCE
098     */
099    public AccessibleTextSequence getTextSequenceAt(int part, int index);
100
101    /**
102     * Returns the <code>AccessibleTextSequence</code> after a given index.
103     *
104     * @param part the <code>CHARACTER</code>, <code>WORD</code>,
105     * <code>SENTENCE</code>, <code>LINE</code> or <code>ATTRIBUTE_RUN</code>
106     * to retrieve
107     * @param index an index within the text
108     * @return an <code>AccessibleTextSequence</code> specifying the text
109     * if part and index are valid.  Otherwise, null is returned.
110     *
111     * @see AccessibleText#CHARACTER
112     * @see AccessibleText#WORD
113     * @see AccessibleText#SENTENCE
114     */
115    public AccessibleTextSequence getTextSequenceAfter(int part, int index);
116
117    /**
118     * Returns the <code>AccessibleTextSequence</code> before a given index.
119     *
120     * @param part the <code>CHARACTER</code>, <code>WORD</code>,
121     * <code>SENTENCE</code>, <code>LINE</code> or <code>ATTRIBUTE_RUN</code>
122     * to retrieve
123     * @param index an index within the text
124     * @return an <code>AccessibleTextSequence</code> specifying the text
125     * if part and index are valid.  Otherwise, null is returned.
126     *
127     * @see AccessibleText#CHARACTER
128     * @see AccessibleText#WORD
129     * @see AccessibleText#SENTENCE
130     */
131    public AccessibleTextSequence getTextSequenceBefore(int part, int index);
132
133    /**
134     * Returns the bounding rectangle of the text between two indices.
135     *
136     * @param startIndex the start index in the text
137     * @param endIndex the end index in the text
138     * @return the bounding rectangle of the text if the indices are valid.
139     * Otherwise, null is returned.
140     */
141    public Rectangle getTextBounds(int startIndex, int endIndex);
142}
