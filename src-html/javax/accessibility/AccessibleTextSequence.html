025package javax.accessibility;
026
027
028/**
029 * <P>The AccessibleTextSequence provides information about
030 * a contiguous sequence of text.
031 *
032 * @see Accessible
033 * @see Accessible#getAccessibleContext
034 * @see AccessibleContext
035 * @see AccessibleContext#getAccessibleText
036 * @see AccessibleAttributeSequence
037 *
038 * @author       Lynn Monsanto
039 */
040
041/**
042 * This class collects together key details of a span of text.  It
043 * is used by implementors of the class <code>AccessibleExtendedText</code> in
044 * order to return the requested triplet of a <code>String</code>, and the
045 * start and end indicies/offsets into a larger body of text that the
046 * <code>String</code> comes from.
047 *
048 * @see javax.accessibility.AccessibleExtendedText
049 */
050public class AccessibleTextSequence {
051
052    /* The start index of the text sequence */
053    public int startIndex;
054
055    /** The end index of the text sequence */
056    public int endIndex;
057
058    /** The text */
059    public String text;
060
061    /**
062     * Constructs an <code>AccessibleTextSequence</code> with the given
063     * parameters.
064     *
065     * @param start the beginning index of the span of text
066     * @param end the ending index of the span of text
067     * @param txt the <code>String</code> shared by this text span
068     *
069     * @since 1.6
070     */
071    public AccessibleTextSequence(int start, int end, String txt) {
072        startIndex = start;
073        endIndex = end;
074        text = txt;
075    }
076};
