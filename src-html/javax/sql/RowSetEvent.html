026package javax.sql;
027
028/**
029 * An <code>Event</code> object generated when an event occurs to a
030 * <code>RowSet</code> object.  A <code>RowSetEvent</code> object is
031 * generated when a single row in a rowset is changed, the whole rowset
032 * is changed, or the rowset cursor moves.
033 * <P>
034 * When an event occurs on a <code>RowSet</code> object, one of the
035 * <code>RowSetListener</code> methods will be sent to all registered
036 * listeners to notify them of the event.  An <code>Event</code> object
037 * is supplied to the <code>RowSetListener</code> method so that the
038 * listener can use it to find out which <code>RowSet</code> object is
039 * the source of the event.
040 *
041 * @since 1.4
042 */
043
044public class RowSetEvent extends java.util.EventObject {
045
046  /**
047   * Constructs a <code>RowSetEvent</code> object initialized with the
048   * given <code>RowSet</code> object.
049   *
050   * @param source the <code>RowSet</code> object whose data has changed or
051   *        whose cursor has moved
052   * @throws IllegalArgumentException if <code>source</code> is null.
053   */
054  public RowSetEvent(RowSet source)
055    { super(source); }
056
057  /**
058   * Private serial version unique ID to ensure serialization
059   * compatibility.
060   */
061  static final long serialVersionUID = -1875450876546332005L;
062}
