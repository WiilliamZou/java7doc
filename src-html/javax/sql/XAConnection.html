026package javax.sql;
027
028import java.sql.*;
029
030
031/**
032 * An object that provides support for distributed
033 * transactions.  An <code>XAConnection</code> object  may be enlisted
034 * in a distributed transaction by means of an <code>XAResource</code> object.
035 * A transaction manager, usually part of a middle tier server, manages an
036 * <code>XAConnection</code> object through the <code>XAResource</code> object.
037 * <P>
038 * An application programmer does not use this interface directly; rather,
039 * it is used by a transaction manager working in the middle tier server.
040 *
041 * @since 1.4
042 */
043
044public interface XAConnection extends PooledConnection {
045
046
047  /**
048   * Retrieves an <code>XAResource</code> object that
049   * the transaction manager will use
050   * to manage this <code>XAConnection</code> object's participation in a
051   * distributed transaction.
052   *
053   * @return the <code>XAResource</code> object
054   * @exception SQLException if a database access error occurs
055   * @exception SQLFeatureNotSupportedException if the JDBC driver does not support
056   * this method
057   * @since 1.4
058   */
059  javax.transaction.xa.XAResource getXAResource() throws SQLException;
060
061 }
