026package javax.sql.rowset;
027
028import java.sql.SQLException;
029
030/**
031 * An interface that defines the implementation of a factory that is used
032 * to obtain different types of {@code RowSet} implementations.
033 *
034 * @author Lance Andersen
035 * @since 1.7
036 */
037public interface RowSetFactory{
038
039    /**
040     * <p>Creates a new instance of a CachedRowSet.</p>
041     *
042     * @return A new instance of a CachedRowSet.
043     *
044     * @throws SQLException if a CachedRowSet cannot
045     *   be created.
046     *
047     * @since 1.7
048     */
049    public CachedRowSet createCachedRowSet() throws SQLException;
050
051    /**
052     * <p>Creates a new instance of a FilteredRowSet.</p>
053     *
054     * @return A new instance of a FilteredRowSet.
055     *
056     * @throws SQLException if a FilteredRowSet cannot
057     *   be created.
058     *
059     * @since 1.7
060     */
061    public FilteredRowSet createFilteredRowSet() throws SQLException;
062
063    /**
064     * <p>Creates a new instance of a JdbcRowSet.</p>
065     *
066     * @return A new instance of a JdbcRowSet.
067     *
068     * @throws SQLException if a JdbcRowSet cannot
069     *   be created.
070     *
071     * @since 1.7
072     */
073    public  JdbcRowSet createJdbcRowSet() throws SQLException;
074
075    /**
076     * <p>Creates a new instance of a JoinRowSet.</p>
077     *
078     * @return A new instance of a JoinRowSet.
079     *
080     * @throws SQLException if a JoinRowSet cannot
081     *   be created.
082     *
083     * @since 1.7
084     */
085    public  JoinRowSet createJoinRowSet() throws SQLException;
086
087    /**
088     * <p>Creates a new instance of a WebRowSet.</p>
089     *
090     * @return A new instance of a WebRowSet.
091     *
092     * @throws SQLException if a WebRowSet cannot
093     *   be created.
094     *
095     * @since 1.7
096     */
097    public  WebRowSet createWebRowSet() throws SQLException;
098
099}
