026package javax.sql;
027
028import java.sql.SQLException;
029
030/**
031 * <P>An <code>Event</code> object that provides information about the
032 * source of a connection-related event.  <code>ConnectionEvent</code>
033 * objects are generated when an application closes a pooled connection
034 * and when an error occurs.  The <code>ConnectionEvent</code> object
035 * contains two kinds of information:
036 * <UL>
037 *   <LI>The pooled connection closed by the application
038 *   <LI>In the case of an error event, the <code>SQLException</code>
039 *       about to be thrown to the application
040 * </UL>
041 *
042 * @since 1.4
043 */
044
045public class ConnectionEvent extends java.util.EventObject {
046
047  /**
048   * <P>Constructs a <code>ConnectionEvent</code> object initialized with
049   * the given <code>PooledConnection</code> object. <code>SQLException</code>
050   * defaults to <code>null</code>.
051   *
052   * @param con the pooled connection that is the source of the event
053   * @throws IllegalArgumentException if <code>con</code> is null.
054   */
055  public ConnectionEvent(PooledConnection con) {
056    super(con);
057  }
058
059  /**
060   * <P>Constructs a <code>ConnectionEvent</code> object initialized with
061   * the given <code>PooledConnection</code> object and
062   * <code>SQLException</code> object.
063   *
064   * @param con the pooled connection that is the source of the event
065   * @param ex the SQLException about to be thrown to the application
066   * @throws IllegalArgumentException if <code>con</code> is null.
067   */
068  public ConnectionEvent(PooledConnection con, SQLException ex) {
069    super(con);
070    this.ex = ex;
071  }
072
073  /**
074   * <P>Retrieves the <code>SQLException</code> for this
075   * <code>ConnectionEvent</code> object. May be <code>null</code>.
076   *
077   * @return the SQLException about to be thrown or <code>null</code>
078   */
079  public SQLException getSQLException() { return ex; }
080
081  /**
082   * The <code>SQLException</code> that the driver will throw to the
083   * application when an error occurs and the pooled connection is no
084   * longer usable.
085   * @serial
086   */
087  private SQLException ex = null;
088
089  /**
090   * Private serial version unique ID to ensure serialization
091   * compatibility.
092   */
093  static final long serialVersionUID = -4843217645290030002L;
094
095 }
