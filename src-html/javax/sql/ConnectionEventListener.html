026package javax.sql;
027
028/**
029 * <P>
030 * An object that registers to be notified of events generated by a
031 * <code>PooledConnection</code> object.
032 * <P>
033 * The <code>ConnectionEventListener</code> interface is implemented by a
034 * connection pooling component.  A connection pooling component will
035 * usually be provided by a JDBC driver vendor or another system software
036 * vendor.  A JDBC driver notifies a <code>ConnectionEventListener</code>
037 * object when an application is finished using a pooled connection with
038 * which the listener has registered.  The notification
039 * occurs after the application calls the method <code>close</code> on
040 * its representation of a <code>PooledConnection</code> object.  A
041 * <code>ConnectionEventListener</code> is also notified when a
042 * connection error occurs due to the fact that the <code>PooledConnection</code>
043 * is unfit for future use---the server has crashed, for example.
044 * The listener is notified by the JDBC driver just before the driver throws an
045 * <code>SQLException</code> to the application using the
046 * <code>PooledConnection</code> object.
047 *
048 * @since 1.4
049 */
050
051public interface ConnectionEventListener extends java.util.EventListener {
052
053  /**
054   * Notifies this <code>ConnectionEventListener</code> that
055   * the application has called the method <code>close</code> on its
056   * representation of a pooled connection.
057   *
058   * @param event an event object describing the source of
059   * the event
060   */
061  void connectionClosed(ConnectionEvent event);
062
063  /**
064   * Notifies this <code>ConnectionEventListener</code> that
065   * a fatal error has occurred and the pooled connection can
066   * no longer be used.  The driver makes this notification just
067   * before it throws the application the <code>SQLException</code>
068   * contained in the given <code>ConnectionEvent</code> object.
069   *
070   * @param event an event object describing the source of
071   * the event and containing the <code>SQLException</code> that the
072   * driver is about to throw
073   */
074  void connectionErrorOccurred(ConnectionEvent event);
075
076 }
