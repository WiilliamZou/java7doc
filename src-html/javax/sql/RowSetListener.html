026package javax.sql;
027
028/**
029 * An interface that must be implemented by a
030 * component that wants to be notified when a significant
031 * event happens in the life of a <code>RowSet</code> object.
032 * A component becomes a listener by being registered with a
033 * <code>RowSet</code> object via the method <code>RowSet.addRowSetListener</code>.
034 * How a registered component implements this interface determines what it does
035 * when it is notified of an event.
036 *
037 * @since 1.4
038 */
039
040public interface RowSetListener extends java.util.EventListener {
041
042  /**
043   * Notifies registered listeners that a <code>RowSet</code> object
044   * in the given <code>RowSetEvent</code> object has changed its entire contents.
045   * <P>
046   * The source of the event can be retrieved with the method
047   * <code>event.getSource</code>.
048   *
049   * @param event a <code>RowSetEvent</code> object that contains
050   *         the <code>RowSet</code> object that is the source of the event
051   */
052  void rowSetChanged(RowSetEvent event);
053
054  /**
055   * Notifies registered listeners that a <code>RowSet</code> object
056   * has had a change in one of its rows.
057   * <P>
058   * The source of the event can be retrieved with the method
059   * <code>event.getSource</code>.
060   *
061   * @param event a <code>RowSetEvent</code> object that contains
062   *         the <code>RowSet</code> object that is the source of the event
063   */
064  void rowChanged(RowSetEvent event);
065
066  /**
067   * Notifies registered listeners that a <code>RowSet</code> object's
068   * cursor has moved.
069   * <P>
070   * The source of the event can be retrieved with the method
071   * <code>event.getSource</code>.
072   *
073   * @param event a <code>RowSetEvent</code> object that contains
074   *         the <code>RowSet</code> object that is the source of the event
075   */
076  void cursorMoved(RowSetEvent event);
077}
