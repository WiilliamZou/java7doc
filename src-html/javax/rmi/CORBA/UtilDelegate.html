032package javax.rmi.CORBA;
033
034import java.io.Serializable;
035import java.rmi.Remote;
036import java.rmi.RemoteException;
037import javax.rmi.CORBA.Tie;
038import javax.rmi.CORBA.ValueHandler;
039import org.omg.CORBA.ORB;
040import org.omg.CORBA.portable.InputStream;
041import org.omg.CORBA.portable.OutputStream;
042import org.omg.CORBA.SystemException;
043
044/**
045 * Supports delegation for method implementations in {@link Util}.  The
046 * delegate is a singleton instance of a class that implements this
047 * interface and provides a replacement implementation for all the
048 * methods of <code>javax.rmi.CORBA.Util</code>.
049 *
050 * Delegation is enabled by providing the delegate's class name as the
051 * value of the
052 * <code>javax.rmi.CORBA.UtilClass</code>
053 * system property.
054 *
055 * @see Util
056 */
057public interface UtilDelegate {
058
059    /**
060     * Delegation call for {@link Util#mapSystemException}.
061     */
062    RemoteException mapSystemException(SystemException ex);
063
064    /**
065     * Delegation call for {@link Util#writeAny}.
066     */
067    void writeAny(OutputStream out, Object obj);
068
069    /**
070     * Delegation call for {@link Util#readAny}.
071     */
072    java.lang.Object readAny(InputStream in);
073
074    /**
075     * Delegation call for {@link Util#writeRemoteObject}.
076     */
077    void writeRemoteObject(OutputStream out, Object obj);
078
079    /**
080     * Delegation call for {@link Util#writeAbstractObject}.
081     */
082    void writeAbstractObject(OutputStream out, Object obj);
083
084    /**
085     * Delegation call for {@link Util#registerTarget}.
086     */
087    void registerTarget(Tie tie, Remote target);
088
089    /**
090     * Delegation call for {@link Util#unexportObject}.
091     */
092    void unexportObject(Remote target) throws java.rmi.NoSuchObjectException;
093
094    /**
095     * Delegation call for {@link Util#getTie}.
096     */
097    Tie getTie(Remote target);
098
099    /**
100     * Delegation call for {@link Util#createValueHandler}.
101     */
102    ValueHandler createValueHandler();
103
104    /**
105     * Delegation call for {@link Util#getCodebase}.
106     */
107    String getCodebase(Class clz);
108
109    /**
110     * Delegation call for {@link Util#loadClass}.
111     */
112    Class loadClass(String className, String remoteCodebase, ClassLoader loader)
113        throws ClassNotFoundException;
114
115    /**
116     * Delegation call for {@link Util#isLocal}.
117     */
118    boolean isLocal(Stub stub) throws RemoteException;
119
120    /**
121     * Delegation call for {@link Util#wrapException}.
122     */
123    RemoteException wrapException(Throwable obj);
124
125    /**
126     * Delegation call for {@link Util#copyObject}.
127     */
128    Object copyObject(Object obj, ORB orb) throws RemoteException;
129
130    /**
131     * Delegation call for {@link Util#copyObjects}.
132     */
133    Object[] copyObjects(Object[] obj, ORB orb) throws RemoteException;
134
135}
