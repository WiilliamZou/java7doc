025package javax.swing.event;
026
027import java.awt.event.*;
028import java.awt.*;
029import java.util.*;
030
031import javax.swing.*;
032
033/**
034 * AncestorListener
035 *
036 * Interface to support notification when changes occur to a JComponent or one
037 * of its ancestors.  These include movement and when the component becomes
038 * visible or invisible, either by the setVisible() method or by being added
039 * or removed from the component hierarchy.
040 *
041 * @author Dave Moore
042 */
043public interface AncestorListener extends EventListener {
044    /**
045     * Called when the source or one of its ancestors is made visible
046     * either by setVisible(true) being called or by its being
047     * added to the component hierarchy.  The method is only called
048     * if the source has actually become visible.  For this to be true
049     * all its parents must be visible and it must be in a hierarchy
050     * rooted at a Window
051     */
052    public void ancestorAdded(AncestorEvent event);
053
054    /**
055     * Called when the source or one of its ancestors is made invisible
056     * either by setVisible(false) being called or by its being
057     * remove from the component hierarchy.  The method is only called
058     * if the source has actually become invisible.  For this to be true
059     * at least one of its parents must by invisible or it is not in
060     * a hierarchy rooted at a Window
061     */
062    public void ancestorRemoved(AncestorEvent event);
063
064    /**
065     * Called when either the source or one of its ancestors is moved.
066     */
067    public void ancestorMoved(AncestorEvent event);
068
069}
