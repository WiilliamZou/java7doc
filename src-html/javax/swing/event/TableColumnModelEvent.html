026package javax.swing.event;
027
028import java.util.EventObject;
029import javax.swing.table.*;
030
031/**
032 * <B>TableColumnModelEvent</B> is used to notify listeners that a table
033 * column model has changed, such as a column was added, removed, or
034 * moved.
035 * <p>
036 * <strong>Warning:</strong>
037 * Serialized objects of this class will not be compatible with
038 * future Swing releases. The current serialization support is
039 * appropriate for short term storage or RMI between applications running
040 * the same version of Swing.  As of 1.4, support for long term storage
041 * of all JavaBeans<sup><font size="-2">TM</font></sup>
042 * has been added to the <code>java.beans</code> package.
043 * Please see {@link java.beans.XMLEncoder}.
044 *
045 * @author Alan Chung
046 * @see TableColumnModelListener
047 */
048public class TableColumnModelEvent extends java.util.EventObject
049{
050//
051//  Instance Variables
052//
053
054    /** The index of the column from where it was moved or removed */
055    protected int       fromIndex;
056
057    /** The index of the column to where it was moved or added */
058    protected int       toIndex;
059
060//
061// Constructors
062//
063
064    /**
065     * Constructs a {@code TableColumnModelEvent} object.
066     *
067     * @param source  the {@code TableColumnModel} that originated the event
068     * @param from    an int specifying the index from where the column was
069     *                moved or removed
070     * @param to      an int specifying the index to where the column was
071     *                moved or added
072     * @see #getFromIndex
073     * @see #getToIndex
074     */
075    public TableColumnModelEvent(TableColumnModel source, int from, int to) {
076        super(source);
077        fromIndex = from;
078        toIndex = to;
079    }
080
081//
082// Querying Methods
083//
084
085    /** Returns the fromIndex.  Valid for removed or moved events */
086    public int getFromIndex() { return fromIndex; };
087
088    /** Returns the toIndex.  Valid for add and moved events */
089    public int getToIndex() { return toIndex; };
090}
