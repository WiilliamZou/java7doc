026package javax.swing.event;
027
028import java.util.EventListener;
029
030/**
031 * Defines the interface for an object that listens
032 * to changes in a TreeModel.
033 * For further information and examples see
034 * <a
035 href="http://java.sun.com/docs/books/tutorial/uiswing/events/treemodellistener.html">How to Write a Tree Model Listener</a>,
036 * a section in <em>The Java Tutorial.</em>
037 *
038 * @author Rob Davis
039 * @author Ray Ryan
040 */
041public interface TreeModelListener extends EventListener {
042
043    /**
044     * <p>Invoked after a node (or a set of siblings) has changed in some
045     * way. The node(s) have not changed locations in the tree or
046     * altered their children arrays, but other attributes have
047     * changed and may affect presentation. Example: the name of a
048     * file has changed, but it is in the same location in the file
049     * system.</p>
050     * <p>To indicate the root has changed, childIndices and children
051     * will be null. </p>
052     *
053     * <p>Use <code>e.getPath()</code>
054     * to get the parent of the changed node(s).
055     * <code>e.getChildIndices()</code>
056     * returns the index(es) of the changed node(s).</p>
057     */
058    void treeNodesChanged(TreeModelEvent e);
059
060    /**
061     * <p>Invoked after nodes have been inserted into the tree.</p>
062     *
063     * <p>Use <code>e.getPath()</code>
064     * to get the parent of the new node(s).
065     * <code>e.getChildIndices()</code>
066     * returns the index(es) of the new node(s)
067     * in ascending order.</p>
068     */
069    void treeNodesInserted(TreeModelEvent e);
070
071    /**
072     * <p>Invoked after nodes have been removed from the tree.  Note that
073     * if a subtree is removed from the tree, this method may only be
074     * invoked once for the root of the removed subtree, not once for
075     * each individual set of siblings removed.</p>
076     *
077     * <p>Use <code>e.getPath()</code>
078     * to get the former parent of the deleted node(s).
079     * <code>e.getChildIndices()</code>
080     * returns, in ascending order, the index(es)
081     * the node(s) had before being deleted.</p>
082     */
083    void treeNodesRemoved(TreeModelEvent e);
084
085    /**
086     * <p>Invoked after the tree has drastically changed structure from a
087     * given node down.  If the path returned by e.getPath() is of length
088     * one and the first element does not identify the current root node
089     * the first element should become the new root of the tree.<p>
090     *
091     * <p>Use <code>e.getPath()</code>
092     * to get the path to the node.
093     * <code>e.getChildIndices()</code>
094     * returns null.</p>
095     */
096    void treeStructureChanged(TreeModelEvent e);
097
098}
