026package javax.swing.event;
027
028import java.util.EventListener;
029
030/**
031 * The listener interface for receiving internal frame events.
032 * This class is functionally equivalent to the WindowListener class
033 * in the AWT.
034 * <p>
035 * See <a href="http://java.sun.com/docs/books/tutorial/uiswing/events/internalframelistener.html">How to Write an Internal Frame Listener</a>
036 * in <em>The Java Tutorial</em> for further documentation.
037 *
038 * @see java.awt.event.WindowListener
039 *
040 * @author Thomas Ball
041 */
042public interface InternalFrameListener extends EventListener {
043    /**
044     * Invoked when a internal frame has been opened.
045     * @see javax.swing.JInternalFrame#show
046     */
047    public void internalFrameOpened(InternalFrameEvent e);
048
049    /**
050     * Invoked when an internal frame is in the process of being closed.
051     * The close operation can be overridden at this point.
052     * @see javax.swing.JInternalFrame#setDefaultCloseOperation
053     */
054    public void internalFrameClosing(InternalFrameEvent e);
055
056    /**
057     * Invoked when an internal frame has been closed.
058     * @see javax.swing.JInternalFrame#setClosed
059     */
060    public void internalFrameClosed(InternalFrameEvent e);
061
062    /**
063     * Invoked when an internal frame is iconified.
064     * @see javax.swing.JInternalFrame#setIcon
065     */
066    public void internalFrameIconified(InternalFrameEvent e);
067
068    /**
069     * Invoked when an internal frame is de-iconified.
070     * @see javax.swing.JInternalFrame#setIcon
071     */
072    public void internalFrameDeiconified(InternalFrameEvent e);
073
074    /**
075     * Invoked when an internal frame is activated.
076     * @see javax.swing.JInternalFrame#setSelected
077     */
078    public void internalFrameActivated(InternalFrameEvent e);
079
080    /**
081     * Invoked when an internal frame is de-activated.
082     * @see javax.swing.JInternalFrame#setSelected
083     */
084    public void internalFrameDeactivated(InternalFrameEvent e);
085}
