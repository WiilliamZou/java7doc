026package javax.swing.plaf;
027
028import java.awt.Font;
029import javax.swing.plaf.UIResource;
030
031
032/**
033 * A subclass of java.awt.Font that implements UIResource.
034 * UI classes which set default font properties should use
035 * this class.
036 * <p>
037 * <strong>Warning:</strong>
038 * Serialized objects of this class will not be compatible with
039 * future Swing releases. The current serialization support is
040 * appropriate for short term storage or RMI between applications running
041 * the same version of Swing.  As of 1.4, support for long term storage
042 * of all JavaBeans<sup><font size="-2">TM</font></sup>
043 * has been added to the <code>java.beans</code> package.
044 * Please see {@link java.beans.XMLEncoder}.
045 *
046 * @see javax.swing.plaf.UIResource
047 * @author Hans Muller
048 *
049 */
050public class FontUIResource extends Font implements UIResource
051{
052    public FontUIResource(String name, int style, int size) {
053        super(name, style, size);
054    }
055
056    public FontUIResource(Font font) {
057        super(font);
058    }
059}
