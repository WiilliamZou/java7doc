037    //of SliderTrackPainter to determine which region/state is being painted

068    //Array of current component colors, updated in each paint call
069    private Object[] componentColors;
070
071    public SliderTrackPainter(PaintContext ctx, int state) {
072        super();
073        this.state = state;
074        this.ctx = ctx;
075    }
076
077    @Override
078    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
079        //populate componentColors array with colors calculated in getExtendedCacheKeys call
080        componentColors = extendedCacheKeys;
081        //generate this entire method. Each state/bg/fg/border combo that has
082        //been painted gets its own KEY and paint method.
083        switch(state) {
084            case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
085            case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
086
087        }
088    }
089        
090
091
092    @Override
093    protected final PaintContext getPaintContext() {
094        return ctx;
095    }
096
097    private void paintBackgroundDisabled(Graphics2D g) {
098        roundRect = decodeRoundRect1();
099        g.setPaint(color1);
100        g.fill(roundRect);
101        roundRect = decodeRoundRect2();
102        g.setPaint(decodeGradient1(roundRect));
103        g.fill(roundRect);
104        roundRect = decodeRoundRect3();
105        g.setPaint(decodeGradient2(roundRect));
106        g.fill(roundRect);
107
108    }
109
110    private void paintBackgroundEnabled(Graphics2D g) {
111        roundRect = decodeRoundRect4();
112        g.setPaint(color6);
113        g.fill(roundRect);
114        roundRect = decodeRoundRect2();
115        g.setPaint(decodeGradient3(roundRect));
116        g.fill(roundRect);
117        roundRect = decodeRoundRect5();
118        g.setPaint(decodeGradient4(roundRect));
119        g.fill(roundRect);
120
121    }

129                               decodeY(2.8333333f) - decodeY(1.6f), //height
130                               8.705882f, 8.705882f); //rounding

172    private Paint decodeGradient1(Shape s) {
173        Rectangle2D bounds = s.getBounds2D();
174        float x = (float)bounds.getX();
175        float y = (float)bounds.getY();
176        float w = (float)bounds.getWidth();
177        float h = (float)bounds.getHeight();
178        return decodeGradient((0.25f * w) + x, (0.07647059f * h) + y, (0.25f * w) + x, (0.9117647f * h) + y,
179                new float[] { 0.0f,0.5f,1.0f },
180                new Color[] { color2,
181                            decodeColor(color2,color3,0.5f),
182                            color3});
183    }
