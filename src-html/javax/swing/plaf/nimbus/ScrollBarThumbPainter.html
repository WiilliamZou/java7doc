088    public ScrollBarThumbPainter(PaintContext ctx, int state) {
089        super();
090        this.state = state;
091        this.ctx = ctx;

094    @Override
095    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
096        //populate componentColors array with colors calculated in getExtendedCacheKeys call
097        componentColors = extendedCacheKeys;
098        //generate this entire method. Each state/bg/fg/border combo that has
099        //been painted gets its own KEY and paint method.
100        switch(state) {
101            case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
102            case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
103            case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
104
105        }
106    }
107
