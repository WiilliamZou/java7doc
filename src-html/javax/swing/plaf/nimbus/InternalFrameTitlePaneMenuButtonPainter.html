105    public InternalFrameTitlePaneMenuButtonPainter(PaintContext ctx, int state) {
106        super();
107        this.state = state;
108        this.ctx = ctx;
109    }

111    @Override
112    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
113        //populate componentColors array with colors calculated in getExtendedCacheKeys call
114        componentColors = extendedCacheKeys;
115        //generate this entire method. Each state/bg/fg/border combo that has
116        //been painted gets its own KEY and paint method.
117        switch(state) {
118            case ICON_ENABLED: painticonEnabled(g); break;
119            case ICON_DISABLED: painticonDisabled(g); break;
120            case ICON_MOUSEOVER: painticonMouseOver(g); break;
121            case ICON_PRESSED: painticonPressed(g); break;
122            case ICON_ENABLED_WINDOWNOTFOCUSED: painticonEnabledAndWindowNotFocused(g); break;
123            case ICON_MOUSEOVER_WINDOWNOTFOCUSED: painticonMouseOverAndWindowNotFocused(g); break;
124            case ICON_PRESSED_WINDOWNOTFOCUSED: painticonPressedAndWindowNotFocused(g); break;
125
126        }
127    }
128

133        return ctx;
