107    @Override
108    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
109        //populate componentColors array with colors calculated in getExtendedCacheKeys call
110        componentColors = extendedCacheKeys;
111        //generate this entire method. Each state/bg/fg/border combo that has
112        //been painted gets its own KEY and paint method.
113        switch(state) {
114            case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
115            case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
116            case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
117            case BACKGROUND_MOUSEOVER_FOCUSED: paintBackgroundMouseOverAndFocused(g); break;
118            case BACKGROUND_PRESSED_FOCUSED: paintBackgroundPressedAndFocused(g); break;
119            case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
120            case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
121            case FOREGROUND_DISABLED: paintForegroundDisabled(g); break;
122            case FOREGROUND_ENABLED: paintForegroundEnabled(g); break;
123            case FOREGROUND_FOCUSED: paintForegroundFocused(g); break;
124            case FOREGROUND_MOUSEOVER_FOCUSED: paintForegroundMouseOverAndFocused(g); break;
125            case FOREGROUND_PRESSED_FOCUSED: paintForegroundPressedAndFocused(g); break;
126            case FOREGROUND_MOUSEOVER: paintForegroundMouseOver(g); break;
127            case FOREGROUND_PRESSED: paintForegroundPressed(g); break;
128
129        }
130    }
