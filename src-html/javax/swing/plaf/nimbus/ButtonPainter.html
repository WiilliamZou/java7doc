128    @Override
129    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
130        //populate componentColors array with colors calculated in getExtendedCacheKeys call
131        componentColors = extendedCacheKeys;
132        //generate this entire method. Each state/bg/fg/border combo that has
133        //been painted gets its own KEY and paint method.
134        switch(state) {
135            case BACKGROUND_DEFAULT: paintBackgroundDefault(g); break;
136            case BACKGROUND_DEFAULT_FOCUSED: paintBackgroundDefaultAndFocused(g); break;
137            case BACKGROUND_MOUSEOVER_DEFAULT: paintBackgroundMouseOverAndDefault(g); break;
138            case BACKGROUND_MOUSEOVER_DEFAULT_FOCUSED: paintBackgroundMouseOverAndDefaultAndFocused(g); break;
139            case BACKGROUND_PRESSED_DEFAULT: paintBackgroundPressedAndDefault(g); break;
140            case BACKGROUND_PRESSED_DEFAULT_FOCUSED: paintBackgroundPressedAndDefaultAndFocused(g); break;
141            case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
142            case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
143            case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
144            case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
145            case BACKGROUND_MOUSEOVER_FOCUSED: paintBackgroundMouseOverAndFocused(g); break;
146            case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
147            case BACKGROUND_PRESSED_FOCUSED: paintBackgroundPressedAndFocused(g); break;
148
149        }
150    }
