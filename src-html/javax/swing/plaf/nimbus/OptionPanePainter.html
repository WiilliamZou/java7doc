106    @Override
107    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
108        //populate componentColors array with colors calculated in getExtendedCacheKeys call
109        componentColors = extendedCacheKeys;
110        //generate this entire method. Each state/bg/fg/border combo that has
111        //been painted gets its own KEY and paint method.
112        switch(state) {
113            case ERRORICON_ENABLED: painterrorIconEnabled(g); break;
114            case INFORMATIONICON_ENABLED: paintinformationIconEnabled(g); break;
115            case QUESTIONICON_ENABLED: paintquestionIconEnabled(g); break;
116            case WARNINGICON_ENABLED: paintwarningIconEnabled(g); break;
117
118        }
119    }
