119    public CheckBoxPainter(PaintContext ctx, int state) {
120        super();
121        this.state = state;
122        this.ctx = ctx;
123    }

125    @Override
126    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
127        //populate componentColors array with colors calculated in getExtendedCacheKeys call
128        componentColors = extendedCacheKeys;
129        //generate this entire method. Each state/bg/fg/border combo that has
130        //been painted gets its own KEY and paint method.
131        switch(state) {
132            case ICON_DISABLED: painticonDisabled(g); break;
133            case ICON_ENABLED: painticonEnabled(g); break;
134            case ICON_FOCUSED: painticonFocused(g); break;
135            case ICON_MOUSEOVER: painticonMouseOver(g); break;
136            case ICON_MOUSEOVER_FOCUSED: painticonMouseOverAndFocused(g); break;
137            case ICON_PRESSED: painticonPressed(g); break;
138            case ICON_PRESSED_FOCUSED: painticonPressedAndFocused(g); break;
139            case ICON_SELECTED: painticonSelected(g); break;
140            case ICON_SELECTED_FOCUSED: painticonSelectedAndFocused(g); break;
141            case ICON_PRESSED_SELECTED: painticonPressedAndSelected(g); break;
142            case ICON_PRESSED_SELECTED_FOCUSED: painticonPressedAndSelectedAndFocused(g); break;
143            case ICON_MOUSEOVER_SELECTED: painticonMouseOverAndSelected(g); break;
144            case ICON_MOUSEOVER_SELECTED_FOCUSED: painticonMouseOverAndSelectedAndFocused(g); break;
145            case ICON_DISABLED_SELECTED: painticonDisabledAndSelected(g); break;
146
147        }
148    }
149
