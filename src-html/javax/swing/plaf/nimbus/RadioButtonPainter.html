129    public RadioButtonPainter(PaintContext ctx, int state) {
130        super();
131        this.state = state;
132        this.ctx = ctx;

135    @Override
136    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
137        //populate componentColors array with colors calculated in getExtendedCacheKeys call
138        componentColors = extendedCacheKeys;
139        //generate this entire method. Each state/bg/fg/border combo that has
140        //been painted gets its own KEY and paint method.
141        switch(state) {
142            case ICON_DISABLED: painticonDisabled(g); break;
143            case ICON_ENABLED: painticonEnabled(g); break;
144            case ICON_FOCUSED: painticonFocused(g); break;
145            case ICON_MOUSEOVER: painticonMouseOver(g); break;
146            case ICON_MOUSEOVER_FOCUSED: painticonMouseOverAndFocused(g); break;
147            case ICON_PRESSED: painticonPressed(g); break;
148            case ICON_PRESSED_FOCUSED: painticonPressedAndFocused(g); break;
149            case ICON_SELECTED: painticonSelected(g); break;
150            case ICON_SELECTED_FOCUSED: painticonSelectedAndFocused(g); break;
151            case ICON_PRESSED_SELECTED: painticonPressedAndSelected(g); break;
152            case ICON_PRESSED_SELECTED_FOCUSED: painticonPressedAndSelectedAndFocused(g); break;
153            case ICON_MOUSEOVER_SELECTED: painticonMouseOverAndSelected(g); break;
154            case ICON_MOUSEOVER_SELECTED_FOCUSED: painticonMouseOverAndSelectedAndFocused(g); break;
155            case ICON_DISABLED_SELECTED: painticonDisabledAndSelected(g); break;
156
157        }
158    }
159
