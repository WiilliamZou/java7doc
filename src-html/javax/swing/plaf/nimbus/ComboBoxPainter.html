131    @Override
132    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
133        //populate componentColors array with colors calculated in getExtendedCacheKeys call
134        componentColors = extendedCacheKeys;
135        //generate this entire method. Each state/bg/fg/border combo that has
136        //been painted gets its own KEY and paint method.
137        switch(state) {
138            case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
139            case BACKGROUND_DISABLED_PRESSED: paintBackgroundDisabledAndPressed(g); break;
140            case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
141            case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
142            case BACKGROUND_MOUSEOVER_FOCUSED: paintBackgroundMouseOverAndFocused(g); break;
143            case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
144            case BACKGROUND_PRESSED_FOCUSED: paintBackgroundPressedAndFocused(g); break;
145            case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
146            case BACKGROUND_ENABLED_SELECTED: paintBackgroundEnabledAndSelected(g); break;
147            case BACKGROUND_DISABLED_EDITABLE: paintBackgroundDisabledAndEditable(g); break;
148            case BACKGROUND_ENABLED_EDITABLE: paintBackgroundEnabledAndEditable(g); break;
149            case BACKGROUND_FOCUSED_EDITABLE: paintBackgroundFocusedAndEditable(g); break;
150            case BACKGROUND_MOUSEOVER_EDITABLE: paintBackgroundMouseOverAndEditable(g); break;
151            case BACKGROUND_PRESSED_EDITABLE: paintBackgroundPressedAndEditable(g); break;
152
153        }
154    }
