197    @Override
198    protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
199        //populate componentColors array with colors calculated in getExtendedCacheKeys call
200        componentColors = extendedCacheKeys;
201        //generate this entire method. Each state/bg/fg/border combo that has
202        //been painted gets its own KEY and paint method.
203        switch(state) {
204            case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
205            case FILEICON_ENABLED: paintfileIconEnabled(g); break;
206            case DIRECTORYICON_ENABLED: paintdirectoryIconEnabled(g); break;
207            case UPFOLDERICON_ENABLED: paintupFolderIconEnabled(g); break;
208            case NEWFOLDERICON_ENABLED: paintnewFolderIconEnabled(g); break;
209            case HARDDRIVEICON_ENABLED: painthardDriveIconEnabled(g); break;
210            case FLOPPYDRIVEICON_ENABLED: paintfloppyDriveIconEnabled(g); break;
211            case HOMEFOLDERICON_ENABLED: painthomeFolderIconEnabled(g); break;
212            case DETAILSVIEWICON_ENABLED: paintdetailsViewIconEnabled(g); break;
213            case LISTVIEWICON_ENABLED: paintlistViewIconEnabled(g); break;
214
215        }
216    }
