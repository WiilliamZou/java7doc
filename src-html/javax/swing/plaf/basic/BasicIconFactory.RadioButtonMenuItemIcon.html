037/**
038 * Factory object that can vend Icons appropriate for the basic L & F.
039 * <p>
040 * <strong>Warning:</strong>
041 * Serialized objects of this class will not be compatible with

043 * appropriate for short term storage or RMI between applications running

046 * has been added to the <code>java.beans</code> package.
047 * Please see {@link java.beans.XMLEncoder}.
048 *
049 * @author David Kloba
050 * @author Georges Saab
051 */
052public class BasicIconFactory implements Serializable
053{
054    private static Icon frame_icon;
055    private static Icon checkBoxIcon;
056    private static Icon radioButtonIcon;
057    private static Icon checkBoxMenuItemIcon;
058    private static Icon radioButtonMenuItemIcon;

060    private static Icon menuItemArrowIcon;

118    private static class EmptyFrameIcon implements Icon, Serializable {
119        int height = 16;

121        public void paintIcon(Component c, Graphics g, int x, int y) {
122        }

124        public int getIconHeight() { return height; }

126

136

138            return csize;
139        }

142    private static class RadioButtonIcon implements Icon, UIResource, Serializable

145        }

148            return 13;

155
156
157    private static class CheckBoxMenuItemIcon implements Icon, UIResource, Serializable
158    {
159        public void paintIcon(Component c, Graphics g, int x, int y) {
160            AbstractButton b = (AbstractButton) c;
161            ButtonModel model = b.getModel();
162            boolean isSelected = model.isSelected();
163            if (isSelected) {
164                g.drawLine(x+7, y+1, x+7, y+3);
165                g.drawLine(x+6, y+2, x+6, y+4);
166                g.drawLine(x+5, y+3, x+5, y+5);
167                g.drawLine(x+4, y+4, x+4, y+6);
168                g.drawLine(x+3, y+5, x+3, y+7);
169                g.drawLine(x+2, y+4, x+2, y+6);
170                g.drawLine(x+1, y+3, x+1, y+5);
171            }
172        }
173        public int getIconWidth() { return 9; }
174        public int getIconHeight() { return 9; }
175
176    } // End class CheckBoxMenuItemIcon
177
178
179    private static class RadioButtonMenuItemIcon implements Icon, UIResource, Serializable
180    {
181        public void paintIcon(Component c, Graphics g, int x, int y) {
182            AbstractButton b = (AbstractButton) c;
183            ButtonModel model = b.getModel();
184            if (b.isSelected() == true) {
185                g.fillOval(x+1, y+1, getIconWidth(), getIconHeight());
186            }
187        }
188        public int getIconWidth() { return 6; }
189        public int getIconHeight() { return 6; }
190
191    } // End class RadioButtonMenuItemIcon
192

202    private static class MenuItemArrowIcon implements Icon, UIResource, Serializable {
203        public void paintIcon(Component c, Graphics g, int x, int y) {
204        }
205        public int getIconWidth() { return 4; }
206        public int getIconHeight() { return 8; }
207
208    } // End class MenuItemArrowIcon
209
210    private static class MenuArrowIcon implements Icon, UIResource, Serializable {
211        public void paintIcon(Component c, Graphics g, int x, int y) {
212            Polygon p = new Polygon();
213            p.addPoint(x, y);
214            p.addPoint(x+getIconWidth(), y+getIconHeight()/2);
215            p.addPoint(x, y+getIconHeight());
216            g.fillPolygon(p);
217
218        }
219        public int getIconWidth() { return 4; }
220        public int getIconHeight() { return 8; }
221    } // End class MenuArrowIcon
