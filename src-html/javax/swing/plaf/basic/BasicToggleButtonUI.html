026package javax.swing.plaf.basic;
027
028import sun.awt.AppContext;
029
030import java.awt.*;
031import java.awt.event.*;
032
033import javax.swing.*;
034import javax.swing.border.*;
035import javax.swing.plaf.*;
036import javax.swing.text.View;
037
038
039
040/**
041 * BasicToggleButton implementation
042 * <p>
043 *
044 * @author Jeff Dinkins
045 */
046public class BasicToggleButtonUI extends BasicButtonUI {
047
048    private static final Object BASIC_TOGGLE_BUTTON_UI_KEY = new Object();
049
050    private final static String propertyPrefix = "ToggleButton" + ".";
051
052    // ********************************
053    //          Create PLAF
054    // ********************************
055    public static ComponentUI createUI(JComponent b) {
056        AppContext appContext = AppContext.getAppContext();
057        BasicToggleButtonUI toggleButtonUI =
058                (BasicToggleButtonUI) appContext.get(BASIC_TOGGLE_BUTTON_UI_KEY);
059        if (toggleButtonUI == null) {
060            toggleButtonUI = new BasicToggleButtonUI();
061            appContext.put(BASIC_TOGGLE_BUTTON_UI_KEY, toggleButtonUI);
062        }
063        return toggleButtonUI;
064    }
065
066    protected String getPropertyPrefix() {
067        return propertyPrefix;
068    }
069
070
071    // ********************************
072    //          Paint Methods
073    // ********************************
074    public void paint(Graphics g, JComponent c) {
075        AbstractButton b = (AbstractButton) c;
076        ButtonModel model = b.getModel();
077
078        Dimension size = b.getSize();
079        FontMetrics fm = g.getFontMetrics();
080
081        Insets i = c.getInsets();
082
083        Rectangle viewRect = new Rectangle(size);
084
085        viewRect.x += i.left;
086        viewRect.y += i.top;
087        viewRect.width -= (i.right + viewRect.x);
088        viewRect.height -= (i.bottom + viewRect.y);
089
090        Rectangle iconRect = new Rectangle();
091        Rectangle textRect = new Rectangle();
092
093        Font f = c.getFont();
094        g.setFont(f);
095
096        // layout the text and icon
097        String text = SwingUtilities.layoutCompoundLabel(
098            c, fm, b.getText(), b.getIcon(),
099            b.getVerticalAlignment(), b.getHorizontalAlignment(),
100            b.getVerticalTextPosition(), b.getHorizontalTextPosition(),
101            viewRect, iconRect, textRect,
102            b.getText() == null ? 0 : b.getIconTextGap());
103
104        g.setColor(b.getBackground());
105
106        if (model.isArmed() && model.isPressed() || model.isSelected()) {
107            paintButtonPressed(g,b);
108        }
109
110        // Paint the Icon
111        if(b.getIcon() != null) {
112            paintIcon(g, b, iconRect);
113        }
114
115        // Draw the Text
116        if(text != null && !text.equals("")) {
117            View v = (View) c.getClientProperty(BasicHTML.propertyKey);
118            if (v != null) {
119               v.paint(g, textRect);
120            } else {
121               paintText(g, b, textRect, text);
122            }
123        }
124
125        // draw the dashed focus line.
126        if (b.isFocusPainted() && b.hasFocus()) {
127            paintFocus(g, b, viewRect, textRect, iconRect);
128        }
129    }
130
131    protected void paintIcon(Graphics g, AbstractButton b, Rectangle iconRect) {
132        ButtonModel model = b.getModel();
133        Icon icon = null;
134
135        if(!model.isEnabled()) {
136            if(model.isSelected()) {
137               icon = b.getDisabledSelectedIcon();
138            } else {
139               icon = b.getDisabledIcon();
140            }
141        } else if(model.isPressed() && model.isArmed()) {
142            icon = b.getPressedIcon();
143            if(icon == null) {
144                // Use selected icon
145                icon = b.getSelectedIcon();
146            }
147        } else if(model.isSelected()) {
148            if(b.isRolloverEnabled() && model.isRollover()) {
149                icon = b.getRolloverSelectedIcon();
150                if (icon == null) {
151                    icon = b.getSelectedIcon();
152                }
153            } else {
154                icon = b.getSelectedIcon();
155            }
156        } else if(b.isRolloverEnabled() && model.isRollover()) {
157            icon = b.getRolloverIcon();
158        }
159
160        if(icon == null) {
161            icon = b.getIcon();
162        }
163
164        icon.paintIcon(b, g, iconRect.x, iconRect.y);
165    }
166
167    /**
168     * Overriden so that the text will not be rendered as shifted for
169     * Toggle buttons and subclasses.
170     */
171    protected int getTextShiftOffset() {
172        return 0;
173    }
174
175}
