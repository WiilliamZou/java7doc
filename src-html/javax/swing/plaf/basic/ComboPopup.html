026package javax.swing.plaf.basic;
027
028import java.awt.event.MouseListener;
029import java.awt.event.MouseMotionListener;
030import java.awt.event.KeyListener;
031import javax.swing.JList;
032
033
034/**
035 * The interface which defines the methods required for the implementation of the popup
036 * portion of a combo box.
037 * <p>
038 * <strong>Warning:</strong>
039 * Serialized objects of this class will not be compatible with
040 * future Swing releases. The current serialization support is
041 * appropriate for short term storage or RMI between applications running
042 * the same version of Swing.  As of 1.4, support for long term storage
043 * of all JavaBeans<sup><font size="-2">TM</font></sup>
044 * has been added to the <code>java.beans</code> package.
045 * Please see {@link java.beans.XMLEncoder}.
046 *
047 * @author Tom Santos
048 */
049public interface ComboPopup {
050    /**
051     * Shows the popup
052     */
053    public void show();
054
055    /**
056     * Hides the popup
057     */
058    public void hide();
059
060    /**
061     * Returns true if the popup is visible (currently being displayed).
062     *
063     * @return <code>true</code> if the component is visible; <code>false</code> otherwise.
064     */
065    public boolean isVisible();
066
067    /**
068     * Returns the list that is being used to draw the items in the combo box.
069     * This method is highly implementation specific and should not be used
070     * for general list manipulation.
071     */
072    public JList getList();
073
074    /**
075     * Returns a mouse listener that will be added to the combo box or null.
076     * If this method returns null then it will not be added to the combo box.
077     *
078     * @return a <code>MouseListener</code> or null
079     */
080    public MouseListener getMouseListener();
081
082    /**
083     * Returns a mouse motion listener that will be added to the combo box or null.
084     * If this method returns null then it will not be added to the combo box.
085     *
086     * @return a <code>MouseMotionListener</code> or null
087     */
088    public MouseMotionListener getMouseMotionListener();
089
090    /**
091     * Returns a key listener that will be added to the combo box or null.
092     * If this method returns null then it will not be added to the combo box.
093     */
094    public KeyListener getKeyListener();
095
096    /**
097     * Called to inform the ComboPopup that the UI is uninstalling.
098     * If the ComboPopup added any listeners in the component, it should remove them here.
099     */
100    public void uninstallingUI();
101}
