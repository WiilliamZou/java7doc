026package javax.swing.plaf.synth;
027
028import javax.swing.JComponent;
029import java.awt.Graphics;
030import javax.swing.plaf.ComponentUI;
031
032
033/**
034 * Provides the Synth L&F UI delegate for
035 * {@link javax.swing.JCheckBox}.
036 *
037 * @author Jeff Dinkins
038 * @since 1.7
039 */
040public class SynthCheckBoxUI extends SynthRadioButtonUI {
041
042    // ********************************
043    //            Create PLAF
044    // ********************************
045    /**
046     * Creates a new UI object for the given component.
047     *
048     * @param b component to create UI object for
049     * @return the UI object
050     */
051    public static ComponentUI createUI(JComponent b) {
052        return new SynthCheckBoxUI();
053    }
054
055    /**
056     * @inheritDoc
057     */
058    @Override
059    protected String getPropertyPrefix() {
060        return "CheckBox.";
061    }
062
063    @Override
064    void paintBackground(SynthContext context, Graphics g, JComponent c) {
065        context.getPainter().paintCheckBoxBackground(context, g, 0, 0,
066                                                  c.getWidth(), c.getHeight());
067    }
068
069    /**
070     * @inheritDoc
071     */
072    @Override
073    public void paintBorder(SynthContext context, Graphics g, int x,
074                            int y, int w, int h) {
075        context.getPainter().paintCheckBoxBorder(context, g, x, y, w, h);
076    }
077}
