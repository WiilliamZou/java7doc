025package javax.swing.plaf.synth;
026
027import javax.swing.*;
028
029/**
030 * Constants used by Synth. Not all Components support all states. A
031 * Component will at least be in one of the primary states. That is, the
032 * return value from <code>SynthContext.getComponentState()</code> will at
033 * least be one of <code>ENABLED</code>, <code>MOUSE_OVER</code>,
034 * <code>PRESSED</code> or <code>DISABLED</code>, and may also contain
035 * <code>FOCUSED</code>, <code>SELECTED</code> or <code>DEFAULT</code>.
036 *
037 * @since 1.5
038 */
039public interface SynthConstants {
040    /**
041     * Primary state indicating the component is enabled.
042     */
043    public static final int ENABLED = 1 << 0;
044    /**
045     * Primary state indicating the mouse is over the region.
046     */
047    public static final int MOUSE_OVER = 1 << 1;
048    /**
049     * Primary state indicating the region is in a pressed state. Pressed
050     * does not necessarily mean the user has pressed the mouse button.
051     */
052    public static final int PRESSED = 1 << 2;
053    /**
054     * Primary state indicating the region is not enabled.
055     */
056    public static final int DISABLED = 1 << 3;
057
058    /**
059     * Indicates the region has focus.
060     */
061    public static final int FOCUSED = 1 << 8;
062    /**
063     * Indicates the region is selected.
064     */
065    public static final int SELECTED = 1 << 9;
066    /**
067     * Indicates the region is the default. This is typically used for buttons
068     * to indicate this button is somehow special.
069     */
070    public static final int DEFAULT = 1 << 10;
071}
