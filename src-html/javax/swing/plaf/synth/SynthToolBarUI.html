098    private void updateStyle(JToolBar c) {
099        SynthContext context = getContext(
100                c, Region.TOOL_BAR_CONTENT, null, ENABLED);
101        contentStyle = SynthLookAndFeel.updateStyle(context, this);
102        context.dispose();
103
104        context = getContext(c, Region.TOOL_BAR_DRAG_WINDOW, null, ENABLED);
105        dragWindowStyle = SynthLookAndFeel.updateStyle(context, this);
106        context.dispose();
107
108        context = getContext(c, ENABLED);
109        SynthStyle oldStyle = style;
110
111        style = SynthLookAndFeel.updateStyle(context, this);
112        if (oldStyle != style) {
113            handleIcon =
114                style.getIcon(context, "ToolBar.handleIcon");
115            if (oldStyle != null) {
116                uninstallKeyboardActions();
117                installKeyboardActions();
118            }
119        }
120        context.dispose();
121    }

126    @Override
127    protected void uninstallDefaults() {
128        SynthContext context = getContext(toolBar, ENABLED);
129
130        style.uninstallDefaults(context);
131        context.dispose();
132        style = null;
133
134        handleIcon = null;
135
136        context = getContext(toolBar, Region.TOOL_BAR_CONTENT,
137                             contentStyle, ENABLED);
138        contentStyle.uninstallDefaults(context);
139        context.dispose();
140        contentStyle = null;
141
142        context = getContext(toolBar, Region.TOOL_BAR_DRAG_WINDOW,
143                             dragWindowStyle, ENABLED);
144        dragWindowStyle.uninstallDefaults(context);
145        context.dispose();
146        dragWindowStyle = null;
147
148        toolBar.setLayout(null);
149    }

200    /**
201     * Notifies this UI delegate to repaint the specified component.
202     * This method paints the component background, then calls
203     * the {@link #paint(SynthContext,Graphics)} method.
204     *
205     * <p>In general, this method does not need to be overridden by subclasses.
206     * All Look and Feel rendering code should reside in the {@code paint} method.
207     *
208     * @param g the {@code Graphics} object used for painting
209     * @param c the component being painted
210     * @see #paint(SynthContext,Graphics)
211     */
212    @Override
213    public void update(Graphics g, JComponent c) {
214        SynthContext context = getContext(c);
215
216        SynthLookAndFeel.update(context, g);
217        context.getPainter().paintToolBarBackground(context,
218                          g, 0, 0, c.getWidth(), c.getHeight(),
219                          toolBar.getOrientation());
220        paint(context, g);
221        context.dispose();
222    }

224    /**
225     * Paints the specified component according to the Look and Feel.
226     * <p>This method is not used by Synth Look and Feel.
227     * Painting is handled by the {@link #paint(SynthContext,Graphics)} method.
228     *
229     * @param g the {@code Graphics} object used for painting
230     * @param c the component being painted
231     * @see #paint(SynthContext,Graphics)
232     */
233    @Override
234    public void paint(Graphics g, JComponent c) {
235        SynthContext context = getContext(c);
236
237        paint(context, g);

275    /**
276     * Paints the toolbar.
277     *
278     * @param context context for the component being painted
279     * @param g the {@code Graphics} object used for painting
280     * @see #update(Graphics,JComponent)
281     */
282    protected void paint(SynthContext context, Graphics g) {
283        if (handleIcon != null && toolBar.isFloatable()) {
284            int startX = toolBar.getComponentOrientation().isLeftToRight() ?
285                0 : toolBar.getWidth() -
286                    SynthIcon.getIconWidth(handleIcon, context);
287            SynthIcon.paintIcon(handleIcon, context, g, startX, 0,
288                    SynthIcon.getIconWidth(handleIcon, context),
289                    SynthIcon.getIconHeight(handleIcon, context));
290        }
291
292        SynthContext subcontext = getContext(
293                toolBar, Region.TOOL_BAR_CONTENT, contentStyle);
294        paintContent(subcontext, g, contentRect);
295        subcontext.dispose();
296    }
297
298    /**
299     * Paints the toolbar content.
300     *
301     * @param context context for the component being painted
302     * @param g {@code Graphics} object used for painting
303     * @param bounds bounding box for the toolbar
304     */
305    protected void paintContent(SynthContext context, Graphics g,
306            Rectangle bounds) {
307        SynthLookAndFeel.updateSubregion(context, g, bounds);
308        context.getPainter().paintToolBarContentBackground(context, g,
309                             bounds.x, bounds.y, bounds.width, bounds.height,
310                             toolBar.getOrientation());
311        context.getPainter().paintToolBarContentBorder(context, g,
312                             bounds.x, bounds.y, bounds.width, bounds.height,
313                             toolBar.getOrientation());
314    }

316    /**
317     * @inheritDoc
318     */
319    @Override
320    protected void paintDragWindow(Graphics g) {
321        int w = dragWindow.getWidth();
322        int h = dragWindow.getHeight();
323        SynthContext context = getContext(
324                toolBar, Region.TOOL_BAR_DRAG_WINDOW, dragWindowStyle);
325        SynthLookAndFeel.updateSubregion(
326                context, g, new Rectangle(0, 0, w, h));
327        context.getPainter().paintToolBarDragWindowBackground(context,
328                                                           g, 0, 0, w, h,
329                                                           dragWindow.getOrientation());
330        context.getPainter().paintToolBarDragWindowBorder(context, g, 0, 0, w, h,
331                                                          dragWindow.getOrientation());
332        context.dispose();
333    }

355        public Dimension minimumLayoutSize(Container parent) {
356            JToolBar tb = (JToolBar)parent;
357            Insets insets = tb.getInsets();
358            Dimension dim = new Dimension();
359            SynthContext context = getContext(tb);
360
361            if (tb.getOrientation() == JToolBar.HORIZONTAL) {
362                dim.width = tb.isFloatable() ?
363                    SynthIcon.getIconWidth(handleIcon, context) : 0;
364                Dimension compDim;
365                for (int i = 0; i < tb.getComponentCount(); i++) {
366                    Component component = tb.getComponent(i);
367                    if (component.isVisible()) {
368                        compDim = component.getMinimumSize();
369                        dim.width += compDim.width;
370                        dim.height = Math.max(dim.height, compDim.height);
371                    }
372                }
373            } else {
374                dim.height = tb.isFloatable() ?
375                    SynthIcon.getIconHeight(handleIcon, context) : 0;
376                Dimension compDim;
377                for (int i = 0; i < tb.getComponentCount(); i++) {
378                    Component component = tb.getComponent(i);
379                    if (component.isVisible()) {
380                        compDim = component.getMinimumSize();
381                        dim.width = Math.max(dim.width, compDim.width);
382                        dim.height += compDim.height;
383                    }
384                }
385            }
386            dim.width += insets.left + insets.right;
387            dim.height += insets.top + insets.bottom;
388
389            context.dispose();
390            return dim;
391        }

431        public void layoutContainer(Container parent) {
432            JToolBar tb = (JToolBar)parent;
433            Insets insets = tb.getInsets();
434            boolean ltr = tb.getComponentOrientation().isLeftToRight();
435            SynthContext context = getContext(tb);
436
437            Component c;
438            Dimension d;
439
440            // JToolBar by default uses a somewhat modified BoxLayout as
441            // its layout manager. For compatibility reasons, we want to
442            // support Box "glue" as a way to move things around on the
443            // toolbar. "glue" is represented in BoxLayout as a Box.Filler
444            // with a minimum and preferred size of (0,0).
445            // So what we do here is find the number of such glue fillers
446            // and figure out how much space should be allocated to them.
447            int glueCount = 0;
448            for (int i=0; i<tb.getComponentCount(); i++) {
449                if (isGlue(tb.getComponent(i))) glueCount++;
450            }
451
452            if (tb.getOrientation() == JToolBar.HORIZONTAL) {
453                int handleWidth = tb.isFloatable() ?
454                    SynthIcon.getIconWidth(handleIcon, context) : 0;
455
456                // Note: contentRect does not take insets into account
457                // since it is used for determining the bounds that are
458                // passed to paintToolBarContentBackground().
459                contentRect.x = ltr ? handleWidth : 0;
460                contentRect.y = 0;
461                contentRect.width = tb.getWidth() - handleWidth;
462                contentRect.height = tb.getHeight();
463
464                // However, we do take the insets into account here for
465                // the purposes of laying out the toolbar child components.
466                int x = ltr ?
467                    handleWidth + insets.left :
468                    tb.getWidth() - handleWidth - insets.right;
469                int baseY = insets.top;
470                int baseH = tb.getHeight() - insets.top - insets.bottom;
471
472                // we need to get the minimum width for laying things out
473                // so that we can calculate how much empty space needs to
474                // be distributed among the "glue", if any
475                int extraSpacePerGlue = 0;
476                if (glueCount > 0) {
477                    int minWidth = minimumLayoutSize(parent).width;
478                    extraSpacePerGlue = (tb.getWidth() - minWidth) / glueCount;
479                    if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
480                }
481
482                for (int i = 0; i < tb.getComponentCount(); i++) {
483                    c = tb.getComponent(i);
484                    if (c.isVisible()) {
485                        d = c.getPreferredSize();
486                        int y, h;
487                        if (d.height >= baseH || c instanceof JSeparator) {
488                            // Fill available height
489                            y = baseY;
490                            h = baseH;
491                        } else {
492                            // Center component vertically in the available space
493                            y = baseY + (baseH / 2) - (d.height / 2);
494                            h = d.height;
495                        }
496                        //if the component is a "glue" component then add to its
497                        //width the extraSpacePerGlue it is due
498                        if (isGlue(c)) d.width += extraSpacePerGlue;
499                        c.setBounds(ltr ? x : x - d.width, y, d.width, h);
500                        x = ltr ? x + d.width : x - d.width;
501                    }
502                }
503            } else {
504                int handleHeight = tb.isFloatable() ?
505                    SynthIcon.getIconHeight(handleIcon, context) : 0;
506
507                // See notes above regarding the use of insets
508                contentRect.x = 0;
509                contentRect.y = handleHeight;
510                contentRect.width = tb.getWidth();
511                contentRect.height = tb.getHeight() - handleHeight;
512
513                int baseX = insets.left;
514                int baseW = tb.getWidth() - insets.left - insets.right;
515                int y = handleHeight + insets.top;
516
517                // we need to get the minimum height for laying things out
518                // so that we can calculate how much empty space needs to
519                // be distributed among the "glue", if any
520                int extraSpacePerGlue = 0;
521                if (glueCount > 0) {
522                    int minHeight = minimumLayoutSize(parent).height;
523                    extraSpacePerGlue = (tb.getHeight() - minHeight) / glueCount;
524                    if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
525                }
526
527                for (int i = 0; i < tb.getComponentCount(); i++) {
528                    c = tb.getComponent(i);
529                    if (c.isVisible()) {
530                        d = c.getPreferredSize();
531                        int x, w;
532                        if (d.width >= baseW || c instanceof JSeparator) {
533                            // Fill available width
534                            x = baseX;
535                            w = baseW;
536                        } else {
537                            // Center component horizontally in the available space
538                            x = baseX + (baseW / 2) - (d.width / 2);
539                            w = d.width;
540                        }
541                        //if the component is a "glue" component then add to its
542                        //height the extraSpacePerGlue it is due
543                        if (isGlue(c)) d.height += extraSpacePerGlue;
544                        c.setBounds(x, y, w, d.height);
545                        y += d.height;
546                    }
547                }
548            }
549            context.dispose();
550        }
551
552        private boolean isGlue(Component c) {
553            if (c.isVisible() && c instanceof Box.Filler) {
554                Box.Filler f = (Box.Filler)c;
555                Dimension min = f.getMinimumSize();
556                Dimension pref = f.getPreferredSize();
557                return min.width == 0 &&  min.height == 0 &&
558                        pref.width == 0 && pref.height == 0;
559            }
560            return false;
561        }
