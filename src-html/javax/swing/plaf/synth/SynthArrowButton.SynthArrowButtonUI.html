025package javax.swing.plaf.synth;
026
027import java.awt.*;
028import javax.swing.*;
029import javax.swing.plaf.UIResource;
030
031/**
032 * JButton object that draws a scaled Arrow in one of the cardinal directions.
033 *
034 * @author Scott Violet
035 */
036class SynthArrowButton extends JButton implements SwingConstants, UIResource {
037    private int direction;
038
039    public SynthArrowButton(int direction) {
040        super();
041        super.setFocusable(false);
042        setDirection(direction);
043        setDefaultCapable(false);
044    }
045
046    public String getUIClassID() {
047        return "ArrowButtonUI";
048    }
049
050    public void updateUI() {
051        setUI(new SynthArrowButtonUI());
052    }
053
054    public void setDirection(int dir) {
055        direction = dir;
056        putClientProperty("__arrow_direction__", Integer.valueOf(dir));
057        repaint();
058    }
059
060    public int getDirection() {
061        return direction;
062    }
063
064    public void setFocusable(boolean focusable) {}
065
066    private static class SynthArrowButtonUI extends SynthButtonUI {
067        protected void installDefaults(AbstractButton b) {
068            super.installDefaults(b);
069            updateStyle(b);
070        }
071
072        protected void paint(SynthContext context, Graphics g) {
073            SynthArrowButton button = (SynthArrowButton)context.
074                                      getComponent();
075            context.getPainter().paintArrowButtonForeground(
076                context, g, 0, 0, button.getWidth(), button.getHeight(),
077                button.getDirection());
078        }
079
080        void paintBackground(SynthContext context, Graphics g, JComponent c) {
081            context.getPainter().paintArrowButtonBackground(context, g, 0, 0,
082                                                c.getWidth(), c.getHeight());
083        }
084
085        public void paintBorder(SynthContext context, Graphics g, int x,
086                                int y, int w, int h) {
087            context.getPainter().paintArrowButtonBorder(context, g, x, y, w,h);
088        }
089
090        public Dimension getMinimumSize() {
091            return new Dimension(5, 5);
092        }
093
094        public Dimension getMaximumSize() {
095            return new Dimension(Integer.MAX_VALUE, Integer.MAX_VALUE);
096        }
097
098        public Dimension getPreferredSize(JComponent c) {
099            SynthContext context = getContext(c);
100            Dimension dim = null;
101            if (context.getComponent().getName() == "ScrollBar.button") {
102                // ScrollBar arrow buttons can be non-square when
103                // the ScrollBar.squareButtons property is set to FALSE
104                // and the ScrollBar.buttonSize property is non-null
105                dim = (Dimension)
106                    context.getStyle().get(context, "ScrollBar.buttonSize");
107            }
108            if (dim == null) {
109                // For all other cases (including Spinner, ComboBox), we will
110                // fall back on the single ArrowButton.size value to create
111                // a square return value
112                int size =
113                    context.getStyle().getInt(context, "ArrowButton.size", 16);
114                dim = new Dimension(size, size);
115            }
116
117            // handle scaling for sizeVarients for special case components. The
118            // key "JComponent.sizeVariant" scales for large/small/mini
119            // components are based on Apples LAF
120            Container parent = context.getComponent().getParent();
121            if (parent instanceof JComponent && !(parent instanceof JComboBox)) {
122                Object scaleKey = ((JComponent)parent).
123                                    getClientProperty("JComponent.sizeVariant");
124                if (scaleKey != null){
125                    if ("large".equals(scaleKey)){
126                        dim = new Dimension(
127                                (int)(dim.width * 1.15),
128                                (int)(dim.height * 1.15));
129                    } else if ("small".equals(scaleKey)){
130                        dim = new Dimension(
131                                (int)(dim.width * 0.857),
132                                (int)(dim.height * 0.857));
133                    } else if ("mini".equals(scaleKey)){
134                        dim = new Dimension(
135                                (int)(dim.width * 0.714),
136                                (int)(dim.height * 0.714));
137                    }
138                }
139            }
140
141            context.dispose();
142            return dim;
143        }
144    }
145}
