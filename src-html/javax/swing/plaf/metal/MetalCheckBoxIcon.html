026package javax.swing.plaf.metal;
027
028import javax.swing.*;
029
030import java.awt.*;
031import java.awt.event.*;
032import java.io.*;
033import javax.swing.plaf.*;
034
035/**
036 * CheckboxIcon implementation for OrganicCheckBoxUI
037 * <p>
038 * <strong>Warning:</strong>
039 * Serialized objects of this class will not be compatible with
040 * future Swing releases. The current serialization support is
041 * appropriate for short term storage or RMI between applications running
042 * the same version of Swing.  As of 1.4, support for long term storage
043 * of all JavaBeans<sup><font size="-2">TM</font></sup>
044 * has been added to the <code>java.beans</code> package.
045 * Please see {@link java.beans.XMLEncoder}.
046 *
047 * @author Steve Wilson
048 */
049public class MetalCheckBoxIcon implements Icon, UIResource, Serializable {
050
051    protected int getControlSize() { return 13; }
052
053    public void paintIcon(Component c, Graphics g, int x, int y) {
054
055        JCheckBox cb = (JCheckBox)c;
056        ButtonModel model = cb.getModel();
057        int controlSize = getControlSize();
058
059        boolean drawCheck = model.isSelected();
060
061        if (model.isEnabled()) {
062            if(cb.isBorderPaintedFlat()) {
063                g.setColor(MetalLookAndFeel.getControlDarkShadow());
064                g.drawRect(x+1, y, controlSize-1, controlSize-1);
065            }
066            if (model.isPressed() && model.isArmed()) {
067                if(cb.isBorderPaintedFlat()) {
068                    g.setColor(MetalLookAndFeel.getControlShadow());
069                    g.fillRect(x+2, y+1, controlSize-2, controlSize-2);
070                } else {
071                    g.setColor(MetalLookAndFeel.getControlShadow());
072                    g.fillRect(x, y, controlSize-1, controlSize-1);
073                    MetalUtils.drawPressed3DBorder(g, x, y, controlSize, controlSize);
074                }
075            } else if(!cb.isBorderPaintedFlat()) {
076                MetalUtils.drawFlush3DBorder(g, x, y, controlSize, controlSize);
077            }
078            g.setColor( MetalLookAndFeel.getControlInfo() );
079        } else {
080            g.setColor( MetalLookAndFeel.getControlShadow() );
081            g.drawRect( x, y, controlSize-1, controlSize-1);
082        }
083
084
085        if(drawCheck) {
086            if (cb.isBorderPaintedFlat()) {
087                x++;
088            }
089            drawCheck(c,g,x,y);
090        }
091    }
092
093    protected void drawCheck(Component c, Graphics g, int x, int y) {
094        int controlSize = getControlSize();
095        g.fillRect( x+3, y+5, 2, controlSize-8 );
096        g.drawLine( x+(controlSize-4), y+3, x+5, y+(controlSize-6) );
097        g.drawLine( x+(controlSize-4), y+4, x+5, y+(controlSize-5) );
098    }
099
100    public int getIconWidth() {
101        return getControlSize();
102    }
103
104    public int getIconHeight() {
105        return getControlSize();
106    }
107 }
