026package javax.swing.plaf.metal;
027
028import javax.swing.plaf.*;
029import javax.swing.plaf.basic.*;
030import javax.swing.plaf.metal.*;
031import javax.swing.*;
032import javax.swing.border.*;
033import java.awt.*;
034
035/**
036 * A high contrast theme. This is used on Windows if the system property
037 * awt.highContrast.on is true.
038 *
039 * @author Michael C. Albers
040 */
041class MetalHighContrastTheme extends DefaultMetalTheme {
042    private static final ColorUIResource primary1 = new
043                              ColorUIResource(0, 0, 0);
044    private static final ColorUIResource primary2 = new ColorUIResource(
045                              204, 204, 204);
046    private static final ColorUIResource primary3 = new ColorUIResource(255,
047                              255, 255);
048    private static final ColorUIResource primaryHighlight = new
049                              ColorUIResource(102, 102, 102);
050    private static final ColorUIResource secondary2 = new ColorUIResource(
051                              204, 204, 204);
052    private static final ColorUIResource secondary3 = new ColorUIResource(
053                              255, 255, 255);
054    private static final ColorUIResource controlHighlight = new
055                              ColorUIResource(102, 102, 102);
056
057
058    // This does not override getSecondary1 (102,102,102)
059
060    public String getName() {
061        return "Contrast";
062    }
063
064    protected ColorUIResource getPrimary1() {
065        return primary1;
066    }
067
068    protected ColorUIResource getPrimary2() {
069        return primary2;
070    }
071
072    protected ColorUIResource getPrimary3() {
073        return primary3;
074    }
075
076    public ColorUIResource getPrimaryControlHighlight() {
077        return primaryHighlight;
078    }
079
080    protected ColorUIResource getSecondary2() {
081        return secondary2;
082    }
083
084    protected ColorUIResource getSecondary3() {
085        return secondary3;
086    }
087
088    public ColorUIResource getControlHighlight() {
089        // This was super.getSecondary3();
090        return secondary2;
091    }
092
093    public ColorUIResource getFocusColor() {
094        return getBlack();
095    }
096
097    public ColorUIResource getTextHighlightColor() {
098        return getBlack();
099    }
100
101    public ColorUIResource getHighlightedTextColor() {
102        return getWhite();
103    }
104
105    public ColorUIResource getMenuSelectedBackground() {
106        return getBlack();
107    }
108
109    public ColorUIResource getMenuSelectedForeground() {
110        return getWhite();
111    }
112
113    public ColorUIResource getAcceleratorForeground() {
114        return getBlack();
115    }
116
117    public ColorUIResource getAcceleratorSelectedForeground() {
118        return getWhite();
119    }
120
121    public void addCustomEntriesToTable(UIDefaults table) {
122        Border blackLineBorder = new BorderUIResource(new LineBorder(
123                    getBlack()));
124        Border whiteLineBorder = new BorderUIResource(new LineBorder(
125                    getWhite()));
126        Object textBorder = new BorderUIResource(new CompoundBorder(
127                   blackLineBorder, new BasicBorders.MarginBorder()));
128
129        Object[] defaults = new Object[] {
130            "ToolTip.border", blackLineBorder,
131
132            "TitledBorder.border", blackLineBorder,
133
134            "TextField.border", textBorder,
135
136            "PasswordField.border", textBorder,
137
138            "TextArea.border", textBorder,
139
140            "TextPane.border", textBorder,
141
142            "EditorPane.border", textBorder,
143
144            "ComboBox.background", getWindowBackground(),
145            "ComboBox.foreground", getUserTextColor(),
146            "ComboBox.selectionBackground", getTextHighlightColor(),
147            "ComboBox.selectionForeground", getHighlightedTextColor(),
148
149            "ProgressBar.foreground",  getUserTextColor(),
150            "ProgressBar.background", getWindowBackground(),
151            "ProgressBar.selectionForeground", getWindowBackground(),
152            "ProgressBar.selectionBackground", getUserTextColor(),
153
154            "OptionPane.errorDialog.border.background",
155                        getPrimary1(),
156            "OptionPane.errorDialog.titlePane.foreground",
157                        getPrimary3(),
158            "OptionPane.errorDialog.titlePane.background",
159                        getPrimary1(),
160            "OptionPane.errorDialog.titlePane.shadow",
161                        getPrimary2(),
162            "OptionPane.questionDialog.border.background",
163                        getPrimary1(),
164            "OptionPane.questionDialog.titlePane.foreground",
165                        getPrimary3(),
166            "OptionPane.questionDialog.titlePane.background",
167                        getPrimary1(),
168            "OptionPane.questionDialog.titlePane.shadow",
169                        getPrimary2(),
170            "OptionPane.warningDialog.border.background",
171                        getPrimary1(),
172            "OptionPane.warningDialog.titlePane.foreground",
173                        getPrimary3(),
174            "OptionPane.warningDialog.titlePane.background",
175                        getPrimary1(),
176            "OptionPane.warningDialog.titlePane.shadow",
177                        getPrimary2(),
178        };
179
180        table.putDefaults(defaults);
181    }
182
183    /**
184     * Returns true if this is a theme provided by the core platform.
185     */
186    boolean isSystemTheme() {
187        return (getClass() == MetalHighContrastTheme.class);
188    }
189}
