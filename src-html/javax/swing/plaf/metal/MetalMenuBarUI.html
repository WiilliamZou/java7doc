026package javax.swing.plaf.metal;
027
028import java.awt.*;
029import javax.swing.*;
030import javax.swing.plaf.ComponentUI;
031import javax.swing.plaf.UIResource;
032import javax.swing.plaf.basic.*;
033
034/**
035 * Metal implementation of <code>MenuBarUI</code>. This class is responsible
036 * for providing the metal look and feel for <code>JMenuBar</code>s.
037 *
038 * @see javax.swing.plaf.MenuBarUI
039 * @since 1.5
040 */
041public class MetalMenuBarUI extends BasicMenuBarUI  {
042    /**
043     * Creates the <code>ComponentUI</code> implementation for the passed
044     * in component.
045     *
046     * @param x JComponent to create the ComponentUI implementation for
047     * @return ComponentUI implementation for <code>x</code>
048     * @throws NullPointerException if <code>x</code> is null
049     */
050    public static ComponentUI createUI(JComponent x) {
051        if (x == null) {
052            throw new NullPointerException("Must pass in a non-null component");
053        }
054        return new MetalMenuBarUI();
055    }
056
057    /**
058     * Configures the specified component appropriate for the metal look and
059     * feel.
060     *
061     * @param c the component where this UI delegate is being installed
062     * @throws NullPointerException if <code>c</code> is null.
063     */
064    public void installUI(JComponent c) {
065        super.installUI(c);
066        MetalToolBarUI.register(c);
067    }
068
069    /**
070     * Reverses configuration which was done on the specified component during
071     * <code>installUI</code>.
072     *
073     * @param c the component where this UI delegate is being installed
074     * @throws NullPointerException if <code>c</code> is null.
075     */
076    public void uninstallUI(JComponent c) {
077        super.uninstallUI(c);
078        MetalToolBarUI.unregister(c);
079    }
080
081    /**
082     * If necessary paints the background of the component, then
083     * invokes <code>paint</code>.
084     *
085     * @param g Graphics to paint to
086     * @param c JComponent painting on
087     * @throws NullPointerException if <code>g</code> or <code>c</code> is
088     *         null
089     * @see javax.swing.plaf.ComponentUI#update
090     * @see javax.swing.plaf.ComponentUI#paint
091     * @since 1.5
092     */
093    public void update(Graphics g, JComponent c) {
094        boolean isOpaque = c.isOpaque();
095        if (g == null) {
096            throw new NullPointerException("Graphics must be non-null");
097        }
098        if (isOpaque && (c.getBackground() instanceof UIResource) &&
099                        UIManager.get("MenuBar.gradient") != null) {
100            if (MetalToolBarUI.doesMenuBarBorderToolBar((JMenuBar)c)) {
101                JToolBar tb = (JToolBar)MetalToolBarUI.
102                     findRegisteredComponentOfType(c, JToolBar.class);
103                if (tb.isOpaque() &&tb.getBackground() instanceof UIResource) {
104                    MetalUtils.drawGradient(c, g, "MenuBar.gradient", 0, 0,
105                                            c.getWidth(), c.getHeight() +
106                                            tb.getHeight(), true);
107                    paint(g, c);
108                    return;
109                }
110            }
111            MetalUtils.drawGradient(c, g, "MenuBar.gradient", 0, 0,
112                                    c.getWidth(), c.getHeight(),true);
113            paint(g, c);
114        }
115        else {
116            super.update(g, c);
117        }
118    }
119}
