025package javax.swing.table;
026
027/**
028 * TableStringConverter is used to convert objects from the model into
029 * strings.  This is useful in filtering and searching when the model returns
030 * objects that do not have meaningful <code>toString</code> implementations.
031 *
032 * @since 1.6
033 */
034public abstract class TableStringConverter {
035    /**
036     * Returns the string representation of the value at the specified
037     * location.
038     *
039     * @param model the <code>TableModel</code> to fetch the value from
040     * @param row the row the string is being requested for
041     * @param column the column the string is being requested for
042     * @return the string representation.  This should never return null.
043     * @throws NullPointerException if <code>model</code> is null
044     * @throws IndexOutOfBoundsException if the arguments are outside the
045     *         bounds of the model
046     */
047    public abstract String toString(TableModel model, int row, int column);
048}
