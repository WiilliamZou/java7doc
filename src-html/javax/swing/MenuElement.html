025package javax.swing;
026
027import java.awt.*;
028import java.awt.event.*;
029
030/**
031 * Any component that can be placed into a menu should implement this interface.
032 * This interface is used by <code>MenuSelectionManager</code>
033 * to handle selection and navigation in menu hierarchies.
034 *
035 * @author Arnaud Weber
036 */
037
038public interface MenuElement {
039
040    /**
041     * Processes a mouse event. <code>event</code> is a <code>MouseEvent</code>
042     * with source being the receiving element's component.
043     * <code>path</code> is the path of the receiving element in the menu
044     * hierarchy including the receiving element itself.
045     * <code>manager</code> is the <code>MenuSelectionManager</code>
046     * for the menu hierarchy.
047     * This method should process the <code>MouseEvent</code> and change
048     * the menu selection if necessary
049     * by using <code>MenuSelectionManager</code>'s API
050     * Note: you do not have to forward the event to sub-components.
051     * This is done automatically by the <code>MenuSelectionManager</code>.
052     */
053    public void processMouseEvent(MouseEvent event,MenuElement path[],MenuSelectionManager manager);
054
055
056    /**
057     *  Process a key event.
058     */
059    public void processKeyEvent(KeyEvent event,MenuElement path[],MenuSelectionManager manager);
060
061    /**
062     * Call by the <code>MenuSelectionManager</code> when the
063     * <code>MenuElement</code> is added or remove from
064     * the menu selection.
065     */
066    public void menuSelectionChanged(boolean isIncluded);
067
068    /**
069     * This method should return an array containing the sub-elements for the receiving menu element
070     *
071     * @return an array of MenuElements
072     */
073    public MenuElement[] getSubElements();
074
075    /**
076     * This method should return the java.awt.Component used to paint the receiving element.
077     * The returned component will be used to convert events and detect if an event is inside
078     * a MenuElement's component.
079     *
080     * @return the Component value
081     */
082    public Component getComponent();
083}
