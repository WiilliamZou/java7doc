048    private static final StringBuilder MENU_SELECTION_MANAGER_KEY =
049                       new StringBuilder("javax.swing.MenuSelectionManager");

051    /**
052     * Returns the default menu selection manager.
053     *
054     * @return a MenuSelectionManager object
055     */
056    public static MenuSelectionManager defaultManager() {
057        synchronized (MENU_SELECTION_MANAGER_KEY) {
058            AppContext context = AppContext.getAppContext();
059            MenuSelectionManager msm = (MenuSelectionManager)context.get(
060                                                 MENU_SELECTION_MANAGER_KEY);
061            if (msm == null) {
062                msm = new MenuSelectionManager();
063                context.put(MENU_SELECTION_MANAGER_KEY, msm);
064
065                // installing additional listener if found in the AppContext
066                Object o = context.get(SwingUtilities2.MENU_SELECTION_MANAGER_LISTENER_KEY);
067                if (o != null && o instanceof ChangeListener) {
068                    msm.addChangeListener((ChangeListener) o);
069                }
070            }
071
072            return msm;
073        }
074    }

076    /**
077     * Only one ChangeEvent is needed per button model instance since the
078     * event's only state is the source property.  The source of events
079     * generated is always "this".
080     */
081    protected transient ChangeEvent changeEvent = null;
082    protected EventListenerList listenerList = new EventListenerList();

084    /**
085     * Changes the selection in the menu hierarchy.  The elements
086     * in the array are sorted in order from the root menu
087     * element to the currently selected menu element.
088     * <p>
089     * Note that this method is public but is used by the look and
090     * feel engine and should not be called by client applications.
091     *
092     * @param path  an array of <code>MenuElement</code> objects specifying
093     *        the selected path
094     */
095    public void setSelectedPath(MenuElement[] path) {
096        int i,c;
097        int currentSelectionCount = selection.size();
098        int firstDifference = 0;
099
100        if(path == null) {
101            path = new MenuElement[0];
102        }
103
104        if (DEBUG) {
105            System.out.print("Previous:  "); printMenuElementArray(getSelectedPath());
106            System.out.print("New:  "); printMenuElementArray(path);
107        }
108
109        for(i=0,c=path.length;i<c;i++) {
110            if (i < currentSelectionCount && selection.elementAt(i) == path[i])
111                firstDifference++;
112            else
113                break;
114        }
115
116        for(i=currentSelectionCount - 1 ; i >= firstDifference ; i--) {
117            MenuElement me = selection.elementAt(i);
118            selection.removeElementAt(i);
119            me.menuSelectionChanged(false);
120        }
121
122        for(i = firstDifference, c = path.length ; i < c ; i++) {
123            if (path[i] != null) {
124                selection.addElement(path[i]);
125                path[i].menuSelectionChanged(true);
126            }
127        }
128
129        fireStateChanged();
130    }

132    /**
133     * Returns the path to the currently selected menu item
134     *
135     * @return an array of MenuElement objects representing the selected path
136     */
137    public MenuElement[] getSelectedPath() {
138        MenuElement res[] = new MenuElement[selection.size()];
139        int i,c;
140        for(i=0,c=selection.size();i<c;i++)
141            res[i] = selection.elementAt(i);
142        return res;
143    }

145    /**
146     * Tell the menu selection to close and unselect all the menu components. Call this method
147     * when a choice has been made
148     */
149    public void clearSelectedPath() {
150        if (selection.size() > 0) {
151            setSelectedPath(null);
152        }
153    }

185    /**
186     * Notifies all listeners that have registered interest for
187     * notification on this event type.  The event instance
188     * is created lazily.
189     *
190     * @see EventListenerList
191     */
192    protected void fireStateChanged() {
193        // Guaranteed to return a non-null array
194        Object[] listeners = listenerList.getListenerList();
195        // Process the listeners last to first, notifying
196        // those that are interested in this event
197        for (int i = listeners.length-2; i>=0; i-=2) {
198            if (listeners[i]==ChangeListener.class) {
199                // Lazily create the event:
200                if (changeEvent == null)
201                    changeEvent = new ChangeEvent(this);
202                ((ChangeListener)listeners[i+1]).stateChanged(changeEvent);
203            }
204        }
205    }

207    /**
208     * When a MenuElement receives an event from a MouseListener, it should never process the event
209     * directly. Instead all MenuElements should call this method with the event.
210     *
211     * @param event  a MouseEvent object
212     */
213    public void processMouseEvent(MouseEvent event) {
214        int screenX,screenY;
215        Point p;
216        int i,c,j,d;
217        Component mc;
218        Rectangle r2;
219        int cWidth,cHeight;
220        MenuElement menuElement;
221        MenuElement subElements[];
222        MenuElement path[];
223        Vector tmp;
224        int selectionSize;
225        p = event.getPoint();
226
227        Component source = event.getComponent();
228
229        if ((source != null) && !source.isShowing()) {
230            // This can happen if a mouseReleased removes the
231            // containing component -- bug 4146684
232            return;
233        }
234
235        int type = event.getID();
236        int modifiers = event.getModifiers();
237        // 4188027: drag enter/exit added in JDK 1.1.7A, JDK1.2
238        if ((type==MouseEvent.MOUSE_ENTERED||
239             type==MouseEvent.MOUSE_EXITED)
240            && ((modifiers & (InputEvent.BUTTON1_MASK |
241                              InputEvent.BUTTON2_MASK | InputEvent.BUTTON3_MASK)) !=0 )) {
242            return;
243        }
244
245        if (source != null) {
246            SwingUtilities.convertPointToScreen(p, source);
247        }
248
249        screenX = p.x;
250        screenY = p.y;
251
252        tmp = (Vector)selection.clone();
253        selectionSize = tmp.size();
254        boolean success = false;
255        for (i=selectionSize - 1;i >= 0 && success == false; i--) {
256            menuElement = (MenuElement) tmp.elementAt(i);
257            subElements = menuElement.getSubElements();
258
259            path = null;
260            for (j = 0, d = subElements.length;j < d && success == false; j++) {
261                if (subElements[j] == null)
262                    continue;
263                mc = subElements[j].getComponent();
264                if(!mc.isShowing())
265                    continue;
266                if(mc instanceof JComponent) {
267                    cWidth  = mc.getWidth();
268                    cHeight = mc.getHeight();
269                } else {
270                    r2 = mc.getBounds();
271                    cWidth  = r2.width;
272                    cHeight = r2.height;
273                }
274                p.x = screenX;
275                p.y = screenY;
276                SwingUtilities.convertPointFromScreen(p,mc);
277
278                /** Send the event to visible menu element if menu element currently in
279                 *  the selected path or contains the event location
280                 */
281                if(
282                   (p.x >= 0 && p.x < cWidth && p.y >= 0 && p.y < cHeight)) {
283                    int k;
284                    if(path == null) {
285                        path = new MenuElement[i+2];
286                        for(k=0;k<=i;k++)
287                            path[k] = (MenuElement)tmp.elementAt(k);
288                    }
289                    path[i+1] = subElements[j];
290                    MenuElement currentSelection[] = getSelectedPath();
291
292                    // Enter/exit detection -- needs tuning...
293                    if (currentSelection[currentSelection.length-1] !=
294                        path[i+1] &&
295                        (currentSelection.length < 2 ||
296                         currentSelection[currentSelection.length-2] !=
297                         path[i+1])) {
298                        Component oldMC = currentSelection[currentSelection.length-1].getComponent();
299
300                        MouseEvent exitEvent = new MouseEvent(oldMC, MouseEvent.MOUSE_EXITED,
301                                                              event.getWhen(),
302                                                              event.getModifiers(), p.x, p.y,
303                                                              event.getXOnScreen(),
304                                                              event.getYOnScreen(),
305                                                              event.getClickCount(),
306                                                              event.isPopupTrigger(),
307                                                              MouseEvent.NOBUTTON);
308                        currentSelection[currentSelection.length-1].
309                            processMouseEvent(exitEvent, path, this);
310
311                        MouseEvent enterEvent = new MouseEvent(mc,
312                                                               MouseEvent.MOUSE_ENTERED,
313                                                               event.getWhen(),
314                                                               event.getModifiers(), p.x, p.y,
315                                                               event.getXOnScreen(),
316                                                               event.getYOnScreen(),
317                                                               event.getClickCount(),
318                                                               event.isPopupTrigger(),
319                                                               MouseEvent.NOBUTTON);
320                        subElements[j].processMouseEvent(enterEvent, path, this);
321                    }
322                    MouseEvent mouseEvent = new MouseEvent(mc, event.getID(),event. getWhen(),
323                                                           event.getModifiers(), p.x, p.y,
324                                                           event.getXOnScreen(),
325                                                           event.getYOnScreen(),
326                                                           event.getClickCount(),
327                                                           event.isPopupTrigger(),
328                                                           MouseEvent.NOBUTTON);
329                    subElements[j].processMouseEvent(mouseEvent, path, this);
330                    success = true;
331                    event.consume();
332                }
333            }
334        }
335    }

337    private void printMenuElementArray(MenuElement path[]) {
338        printMenuElementArray(path, false);
339    }

341    private void printMenuElementArray(MenuElement path[], boolean dumpStack) {
342        System.out.println("Path is(");
343        int i, j;
344        for(i=0,j=path.length; i<j ;i++){
345            for (int k=0; k<=i; k++)
346                System.out.print("  ");
347            MenuElement me = path[i];
348            if(me instanceof JMenuItem) {
349                System.out.println(((JMenuItem)me).getText() + ", ");
350            } else if (me instanceof JMenuBar) {
351                System.out.println("JMenuBar, ");
352            } else if(me instanceof JPopupMenu) {
353                System.out.println("JPopupMenu, ");
354            } else if (me == null) {
355                System.out.println("NULL , ");
356            } else {
357                System.out.println("" + me + ", ");
358            }
359        }
360        System.out.println(")");
361
362        if (dumpStack == true)
363            Thread.dumpStack();
364    }

366    /**
367     * Returns the component in the currently selected path
368     * which contains sourcePoint.
369     *
370     * @param source The component in whose coordinate space sourcePoint
371     *        is given
372     * @param sourcePoint The point which is being tested
373     * @return The component in the currently selected path which
374     *         contains sourcePoint (relative to the source component's
375     *         coordinate space.  If sourcePoint is not inside a component
376     *         on the currently selected path, null is returned.
377     */
378    public Component componentForPoint(Component source, Point sourcePoint) {
379        int screenX,screenY;
380        Point p = sourcePoint;
381        int i,c,j,d;
382        Component mc;
383        Rectangle r2;
384        int cWidth,cHeight;
385        MenuElement menuElement;
386        MenuElement subElements[];
387        Vector tmp;
388        int selectionSize;
389
390        SwingUtilities.convertPointToScreen(p,source);
391
392        screenX = p.x;
393        screenY = p.y;
394
395        tmp = (Vector)selection.clone();
396        selectionSize = tmp.size();
397        for(i=selectionSize - 1 ; i >= 0 ; i--) {
398            menuElement = (MenuElement) tmp.elementAt(i);
399            subElements = menuElement.getSubElements();
400
401            for(j = 0, d = subElements.length ; j < d ; j++) {
402                if (subElements[j] == null)
403                    continue;
404                mc = subElements[j].getComponent();
405                if(!mc.isShowing())
406                    continue;
407                if(mc instanceof JComponent) {
408                    cWidth  = mc.getWidth();
409                    cHeight = mc.getHeight();
410                } else {
411                    r2 = mc.getBounds();
412                    cWidth  = r2.width;
413                    cHeight = r2.height;
414                }
415                p.x = screenX;
416                p.y = screenY;
417                SwingUtilities.convertPointFromScreen(p,mc);
418
419                /** Return the deepest component on the selection
420                 *  path in whose bounds the event's point occurs
421                 */
422                if (p.x >= 0 && p.x < cWidth && p.y >= 0 && p.y < cHeight) {
423                    return mc;
424                }
425            }
426        }
427        return null;
428    }

430    /**
431     * When a MenuElement receives an event from a KeyListener, it should never process the event
432     * directly. Instead all MenuElements should call this method with the event.
433     *
434     * @param e  a KeyEvent object
435     */
436    public void processKeyEvent(KeyEvent e) {
437        MenuElement[] sel2 = new MenuElement[0];
438        sel2 = selection.toArray(sel2);
439        int selSize = sel2.length;
440        MenuElement[] path;
441
442        if (selSize < 1) {
443            return;
444        }
445
446        for (int i=selSize-1; i>=0; i--) {
447            MenuElement elem = sel2[i];
448            MenuElement[] subs = elem.getSubElements();
449            path = null;
450
451            for (int j=0; j<subs.length; j++) {
452                if (subs[j] == null || !subs[j].getComponent().isShowing()
453                    || !subs[j].getComponent().isEnabled()) {
454                    continue;
455                }
456
457                if(path == null) {
458                    path = new MenuElement[i+2];
459                    System.arraycopy(sel2, 0, path, 0, i+1);
460                    }
461                path[i+1] = subs[j];
462                subs[j].processKeyEvent(e, path, this);
463                if (e.isConsumed()) {
464                    return;
465            }
466        }
467    }
468
469        // finally dispatch event to the first component in path
470        path = new MenuElement[1];
471        path[0] = sel2[0];
472        path[0].processKeyEvent(e, path, this);
473        if (e.isConsumed()) {
474            return;
475        }
476    }

478    /**
479     * Return true if c is part of the currently used menu
480     */
481    public boolean isComponentPartOfCurrentMenu(Component c) {
482        if(selection.size() > 0) {
483            MenuElement me = selection.elementAt(0);
484            return isComponentPartOfCurrentMenu(me,c);
485        } else
486            return false;
487    }
488
489    private boolean isComponentPartOfCurrentMenu(MenuElement root,Component c) {
490        MenuElement children[];
491        int i,d;
492
493        if (root == null)
494            return false;
495
496        if(root.getComponent() == c)
497            return true;
498        else {
499            children = root.getSubElements();
500            for(i=0,d=children.length;i<d;i++) {
501                if(isComponentPartOfCurrentMenu(children[i],c))
502                    return true;
503            }
504        }
505        return false;
506    }
