026package javax.swing;
027
028
029/**
030 * Constants used to control the window-closing operation.
031 * The <code>setDefaultCloseOperation</code> and
032 * <code>getDefaultCloseOperation</code> methods
033 * provided by <code>JFrame</code>,
034 * <code>JInternalFrame</code>, and
035 * <code>JDialog</code>
036 * use these constants.
037 * For examples of setting the default window-closing operation, see
038 * <a
039 href="http://java.sun.com/docs/books/tutorial/uiswing/components/frame.html#windowevents">Responding to Window-Closing Events</a>,
040 * a section in <em>The Java Tutorial</em>.
041 * @see JFrame#setDefaultCloseOperation(int)
042 * @see JDialog#setDefaultCloseOperation(int)
043 * @see JInternalFrame#setDefaultCloseOperation(int)
044 *
045 *
046 * @author Amy Fowler
047 */
048public interface WindowConstants
049{
050    /**
051     * The do-nothing default window close operation.
052     */
053    public static final int DO_NOTHING_ON_CLOSE = 0;
054
055    /**
056     * The hide-window default window close operation
057     */
058    public static final int HIDE_ON_CLOSE = 1;
059
060    /**
061     * The dispose-window default window close operation.
062     * <p>
063     * <b>Note</b>: When the last displayable window
064     * within the Java virtual machine (VM) is disposed of, the VM may
065     * terminate.  See <a href="../../java/awt/doc-files/AWTThreadIssues.html">
066     * AWT Threading Issues</a> for more information.
067     * @see java.awt.Window#dispose()
068     * @see JInternalFrame#dispose()
069     */
070    public static final int DISPOSE_ON_CLOSE = 2;
071
072    /**
073     * The exit application default window close operation. Attempting
074     * to set this on Windows that support this, such as
075     * <code>JFrame</code>, may throw a <code>SecurityException</code> based
076     * on the <code>SecurityManager</code>.
077     * It is recommended you only use this in an application.
078     *
079     * @since 1.4
080     * @see JFrame#setDefaultCloseOperation
081     */
082    public static final int EXIT_ON_CLOSE = 3;
083
084}
