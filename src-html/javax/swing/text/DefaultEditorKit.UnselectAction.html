038/**
039 * This is the set of things needed by a text component
040 * to be a reasonably functioning editor for some <em>type</em>
041 * of text document.  This implementation provides a default
042 * implementation which treats text as plain text and
043 * provides a minimal set of actions for a simple editor.
044 * <p>
045 * <dl>
046 * <dt><b><font size=+1>Newlines</font></b>
047 * <dd>
048 * There are two properties which deal with newlines.  The
049 * system property, <code>line.separator</code>, is defined to be
050 * platform-dependent, either "\n", "\r", or "\r\n".  There is also
051 * a property defined in <code>DefaultEditorKit</code>, called
052 * <a href=#EndOfLineStringProperty><code>EndOfLineStringProperty</code></a>,
053 * which is defined automatically when a document is loaded, to be
054 * the first occurrence of any of the newline characters.
055 * When a document is loaded, <code>EndOfLineStringProperty</code>
056 * is set appropriately, and when the document is written back out, the
057 * <code>EndOfLineStringProperty</code> is used.  But while the document
058 * is in memory, the "\n" character is used to define a
059 * newline, regardless of how the newline is defined when
060 * the document is on disk.  Therefore, for searching purposes,
061 * "\n" should always be used.  When a new document is created,
062 * and the <code>EndOfLineStringProperty</code> has not been defined,
063 * it will use the System property when writing out the
064 * document.
065 * <p>Note that <code>EndOfLineStringProperty</code> is set
066 * on the <code>Document</code> using the <code>get/putProperty</code>
067 * methods.  Subclasses may override this behavior.
068 *
069 * </dl>
070 *
071 * @author  Timothy Prinzing
072 */

187    /**
188     * Inserts content from the given stream, which will be
189     * treated as plain text.
190     *
191     * @param in  The stream to read from
192     * @param doc The destination for the insertion.
193     * @param pos The location in the document to place the
194     *   content >= 0.
195     * @exception IOException on any I/O error
196     * @exception BadLocationException if pos represents an invalid
197     *   location within the document.
198     */
199    public void read(Reader in, Document doc, int pos)
200        throws IOException, BadLocationException {
201
202        char[] buff = new char[4096];
203        int nch;
204        boolean lastWasCR = false;
205        boolean isCRLF = false;
206        boolean isCR = false;
207        int last;
208        boolean wasEmpty = (doc.getLength() == 0);
209        AttributeSet attr = getInputAttributes();
210
211        // Read in a block at a time, mapping \r\n to \n, as well as single
212        // \r's to \n's. If a \r\n is encountered, \r\n will be set as the
213        // newline string for the document, if \r is encountered it will
214        // be set as the newline character, otherwise the newline property
215        // for the document will be removed.
216        while ((nch = in.read(buff, 0, buff.length)) != -1) {
217            last = 0;
218            for(int counter = 0; counter < nch; counter++) {
219                switch(buff[counter]) {
220                case '\r':
221                    if (lastWasCR) {
222                        isCR = true;
223                        if (counter == 0) {
224                            doc.insertString(pos, "\n", attr);
225                            pos++;
226                        }
227                        else {
228                            buff[counter - 1] = '\n';
229                        }
230                    }
231                    else {
232                        lastWasCR = true;
233                    }
234                    break;
235                case '\n':
236                    if (lastWasCR) {
237                        if (counter > (last + 1)) {
238                            doc.insertString(pos, new String(buff, last,
239                                            counter - last - 1), attr);
240                            pos += (counter - last - 1);
241                        }
242                        // else nothing to do, can skip \r, next write will
243                        // write \n
244                        lastWasCR = false;
245                        last = counter;
246                        isCRLF = true;
247                    }
248                    break;
249                default:
250                    if (lastWasCR) {
251                        isCR = true;
252                        if (counter == 0) {
253                            doc.insertString(pos, "\n", attr);
254                            pos++;
255                        }
256                        else {
257                            buff[counter - 1] = '\n';
258                        }
259                        lastWasCR = false;
260                    }
261                    break;
262                }
263            }
264            if (last < nch) {
265                if(lastWasCR) {
266                    if (last < (nch - 1)) {
267                        doc.insertString(pos, new String(buff, last,
268                                         nch - last - 1), attr);
269                        pos += (nch - last - 1);
270                    }
271                }
272                else {
273                    doc.insertString(pos, new String(buff, last,
274                                     nch - last), attr);
275                    pos += (nch - last);
276                }
277            }
278        }
279        if (lastWasCR) {
280            doc.insertString(pos, "\n", attr);
281            isCR = true;
282        }
283        if (wasEmpty) {
284            if (isCRLF) {
285                doc.putProperty(EndOfLineStringProperty, "\r\n");
286            }
287            else if (isCR) {
288                doc.putProperty(EndOfLineStringProperty, "\r");
289            }
290            else {
291                doc.putProperty(EndOfLineStringProperty, "\n");
292            }
293        }
294    }

296    /**
297     * Writes content from a document to the given stream
298     * as plain text.
299     *
300     * @param out  The stream to write to
301     * @param doc The source for the write.
302     * @param pos The location in the document to fetch the
303     *   content from >= 0.
304     * @param len The amount to write out >= 0.
305     * @exception IOException on any I/O error
306     * @exception BadLocationException if pos is not within 0 and
307     *   the length of the document.
308     */
309    public void write(Writer out, Document doc, int pos, int len)
310        throws IOException, BadLocationException {
311
312        if ((pos < 0) || ((pos + len) > doc.getLength())) {
313            throw new BadLocationException("DefaultEditorKit.write", pos);
314        }
315        Segment data = new Segment();
316        int nleft = len;
317        int offs = pos;
318        Object endOfLineProperty = doc.getProperty(EndOfLineStringProperty);
319        if (endOfLineProperty == null) {
320            try {
321                endOfLineProperty = System.getProperty("line.separator");
322            } catch (SecurityException se) { }
323        }
324        String endOfLine;
325        if (endOfLineProperty instanceof String) {
326            endOfLine = (String)endOfLineProperty;
327        }
328        else {
329            endOfLine = null;
330        }
331        if (endOfLineProperty != null && !endOfLine.equals("\n")) {
332            // There is an end of line string that isn't \n, have to iterate
333            // through and find all \n's and translate to end of line string.
334            while (nleft > 0) {
335                int n = Math.min(nleft, 4096);
336                doc.getText(offs, n, data);
337                int last = data.offset;
338                char[] array = data.array;
339                int maxCounter = last + data.count;
340                for (int counter = last; counter < maxCounter; counter++) {
341                    if (array[counter] == '\n') {
342                        if (counter > last) {
343                            out.write(array, last, counter - last);
344                        }
345                        out.write(endOfLine);
346                        last = counter + 1;
347                    }
348                }
349                if (maxCounter > last) {
350                    out.write(array, last, maxCounter - last);
351                }
352                offs += n;
353                nleft -= n;
354            }
355        }
356        else {
357            // Just write out text, will already have \n, no mapping to
358            // do.
359            while (nleft > 0) {
360                int n = Math.min(nleft, 4096);
361                doc.getText(offs, n, data);
362                out.write(data.array, data.offset, data.count);
363                offs += n;
364                nleft -= n;
365            }
366        }
367        out.flush();
368    }

1402     * Scrolls up/down vertically.  The select version of this action extends
1403     * the selection, instead of simply moving the caret.
1404     *
1405     * @see DefaultEditorKit#pageUpAction
1406     * @see DefaultEditorKit#pageDownAction
1407     * @see DefaultEditorKit#getActions
1408     */
1409    static class VerticalPageAction extends TextAction {
1410
1411        /** Create this object with the appropriate identifier. */
1412        public VerticalPageAction(String nm, int direction, boolean select) {
1413            super(nm);
1414            this.select = select;
1415            this.direction = direction;
1416        }
1417
1418        /** The operation to perform when this action is triggered. */
1419        public void actionPerformed(ActionEvent e) {
1420            JTextComponent target = getTextComponent(e);
1421            if (target != null) {
1422                Rectangle visible = target.getVisibleRect();
1423                Rectangle newVis = new Rectangle(visible);
1424                int selectedIndex = target.getCaretPosition();
1425                int scrollAmount = direction *
1426                        target.getScrollableBlockIncrement(
1427                                  visible, SwingConstants.VERTICAL, direction);
1428                int initialY = visible.y;
1429                Caret caret = target.getCaret();
1430                Point magicPosition = caret.getMagicCaretPosition();
1431
1432                if (selectedIndex != -1) {
1433                    try {
1434                        Rectangle dotBounds = target.modelToView(
1435                                                     selectedIndex);
1436                        int x = (magicPosition != null) ? magicPosition.x :
1437                                                          dotBounds.x;
1438                        int h = dotBounds.height;
1439                        if (h > 0) {
1440                            // We want to scroll by a multiple of caret height,
1441                            // rounding towards lower integer
1442                            scrollAmount = scrollAmount / h * h;
1443                        }
1444                        newVis.y = constrainY(target,
1445                                initialY + scrollAmount, visible.height);
1446
1447                        int newIndex;
1448
1449                        if (visible.contains(dotBounds.x, dotBounds.y)) {
1450                            // Dot is currently visible, base the new
1451                            // location off the old, or
1452                            newIndex = target.viewToModel(
1453                                new Point(x, constrainY(target,
1454                                          dotBounds.y + scrollAmount, 0)));
1455                        }
1456                        else {
1457                            // Dot isn't visible, choose the top or the bottom
1458                            // for the new location.
1459                            if (direction == -1) {
1460                                newIndex = target.viewToModel(new Point(
1461                                    x, newVis.y));
1462                            }
1463                            else {
1464                                newIndex = target.viewToModel(new Point(
1465                                    x, newVis.y + visible.height));
1466                            }
1467                        }
1468                        newIndex = constrainOffset(target, newIndex);
1469                        if (newIndex != selectedIndex) {
1470                            // Make sure the new visible location contains

1477                                newVis.y = newY;

1480                                    target.moveCaretPosition(newIndex);
1481                                } else {
1482                                    target.setCaretPosition(newIndex);
1483                                }
1484                            }
1485                        }
1486                    } catch (BadLocationException ble) { }
1487                } else {
1488                    newVis.y = constrainY(target,
1489                            initialY + scrollAmount, visible.height);
1490                }
1491                if (magicPosition != null) {
1492                    caret.setMagicCaretPosition(magicPosition);

1503            if (y < 0) {
1504                y = 0;
1505            }
1506            else if (y + vis > target.getHeight()) {
1507                y = Math.max(0, target.getHeight() - vis);

1518

1521            }

1530         * after selecting <code>index</code>.
1531         */

1533            int result = visible.y;

1536                Rectangle dotBounds = text.modelToView(index);
1537

1539                    result = dotBounds.y;
1540                } else {

1542                            (dotBounds.y + dotBounds.height > visible.y + visible.height)) {
1543                        result = dotBounds.y + dotBounds.height - visible.height;
1544                    }
1545                }
1546            } catch (BadLocationException ble) {
1547            }
1548
1549            return result;
1550        }
