046/**

083    /**
084     * Creates a new DTD with the specified name.
085     * @param name the name, as a <code>String</code> of the new DTD
086     */
087    protected DTD(String name) {
088        this.name = name;
089        defEntity("#RE", GENERAL, '\r');
090        defEntity("#RS", GENERAL, '\n');
091        defEntity("#SPACE", GENERAL, ' ');
092        defineElement("unknown", EMPTY, false, true, null, null, null, null);
093    }

121    /**
122     * Returns <code>true</code> if the element is part of the DTD,
123     * otherwise returns <code>false</code>.
124     *
125     * @param  name the requested <code>String</code>
126     * @return <code>true</code> if <code>name</code> exists as
127     *   part of the DTD, otherwise returns <code>false</code>
128     */
129    boolean elementExists(String name) {
130        return !"unknown".equals(name) && (elementHash.get(name) != null);
131    }

133    /**
134     * Gets an element by name. A new element is
135     * created if the element doesn't exist.
136     *
137     * @param name the requested <code>String</code>
138     * @return the <code>Element</code> corresponding to
139     *   <code>name</code>, which may be newly created
140     */
141    public Element getElement(String name) {
142        Element e = elementHash.get(name);
143        if (e == null) {
144            e = new Element(name, elements.size());
145            elements.addElement(e);
146            elementHash.put(name, e);
147        }
148        return e;
149    }

162    /**
163     * Defines an entity.  If the <code>Entity</code> specified
164     * by <code>name</code>, <code>type</code>, and <code>data</code>
165     * exists, it is returned; otherwise a new <code>Entity</code>
166     * is created and is returned.
167     *
168     * @param name the name of the <code>Entity</code> as a <code>String</code>
169     * @param type the type of the <code>Entity</code>
170     * @param data the <code>Entity</code>'s data
171     * @return the <code>Entity</code> requested or a new <code>Entity</code>
172     *   if not found
173     */
174    public Entity defineEntity(String name, int type, char data[]) {
175        Entity ent = entityHash.get(name);
176        if (ent == null) {
177            ent = new Entity(name, type, data);
178            entityHash.put(name, ent);
179            if (((type & GENERAL) != 0) && (data.length == 1)) {
180                switch (type & ~GENERAL) {
181                  case CDATA:
182                  case SDATA:
183                      entityHash.put(Integer.valueOf(data[0]), ent);
184                    break;
185                }
186            }
187        }
188        return ent;
189    }

191    /**
192     * Returns the <code>Element</code> which matches the
193     * specified parameters.  If one doesn't exist, a new
194     * one is created and returned.
195     *
196     * @param name the name of the <code>Element</code>
197     * @param type the type of the <code>Element</code>
198     * @param omitStart <code>true</code> if start should be omitted
199     * @param omitEnd  <code>true</code> if end should be omitted
200     * @param content  the <code>ContentModel</code>
201     * @param atts the <code>AttributeList</code> specifying the
202     *    <code>Element</code>
203     * @return the <code>Element</code> specified
204     */
205    public Element defineElement(String name, int type,
206                       boolean omitStart, boolean omitEnd, ContentModel content,
207                       BitSet exclusions, BitSet inclusions, AttributeList atts) {
208        Element e = getElement(name);
209        e.type = type;
210        e.oStart = omitStart;
211        e.oEnd = omitEnd;
212        e.content = content;
213        e.exclusions = exclusions;
214        e.inclusions = inclusions;
215        e.atts = atts;
216        return e;
217    }

231    /**

234     * @return the new character <code>Entity</code>

237        char data[] = {(char)ch};
238        return defineEntity(name, type, data);

241    /**
242     * Creates and returns an <code>Entity</code>.
243     * @param name the entity's name
244     * @return the new <code>Entity</code>
245     */
246    protected Entity defEntity(String name, int type, String str) {
247        int len = str.length();
248        char data[] = new char[len];
249        str.getChars(0, len, data, 0);
250        return defineEntity(name, type, data);
251    }
252
253    /**
254     * Creates and returns an <code>Element</code>.
255     * @param name the element's name
256     * @return the new <code>Element</code>
257     */
258    protected Element defElement(String name, int type,
259                       boolean omitStart, boolean omitEnd, ContentModel content,
260                       String[] exclusions, String[] inclusions, AttributeList atts) {
261        BitSet excl = null;
262        if (exclusions != null && exclusions.length > 0) {
263            excl = new BitSet();
264            for (String str : exclusions) {
265                if (str.length() > 0) {
266                    excl.set(getElement(str).getIndex());
267                }
268            }
269        }
270        BitSet incl = null;
271        if (inclusions != null && inclusions.length > 0) {
272            incl = new BitSet();
273            for (String str : inclusions) {
274                if (str.length() > 0) {
275                    incl.set(getElement(str).getIndex());
276                }
277            }
278        }
279        return defineElement(name, type, omitStart, omitEnd, content, excl, incl, atts);
280    }

282    /**
283     * Creates and returns an <code>AttributeList</code>.
284     * @param name the attribute list's name
285     * @return the new <code>AttributeList</code>
286     */
287    protected AttributeList defAttributeList(String name, int type, int modifier, String value, String values, AttributeList atts) {
288        Vector<String> vals = null;
289        if (values != null) {
290            vals = new Vector<String>();
291            for (StringTokenizer s = new StringTokenizer(values, "|") ; s.hasMoreTokens() ;) {
292                String str = s.nextToken();
293                if (str.length() > 0) {
294                    vals.addElement(str);
295                }
296            }
297        }
298        return new AttributeList(name, type, modifier, value, vals, atts);
299    }

327    /**
328     * Returns a DTD with the specified <code>name</code>.  If
329     * a DTD with that name doesn't exist, one is created
330     * and returned.  Any uppercase characters in the name
331     * are converted to lowercase.
332     *
333     * @param name the name of the DTD
334     * @return the DTD which corresponds to <code>name</code>
335     */
336    public static DTD getDTD(String name) throws IOException {
337        name = name.toLowerCase();
338        DTD dtd = getDtdHash().get(name);
339        if (dtd == null)
340          dtd = new DTD(name);
341
342        return dtd;
343    }

345    private static Hashtable<String, DTD> getDtdHash() {
346        AppContext appContext = AppContext.getAppContext();
347
348        Hashtable<String, DTD> result = (Hashtable<String, DTD>) appContext.get(DTD_HASH_KEY);
349
350        if (result == null) {
351            result = new Hashtable<String, DTD>();
352
353            appContext.put(DTD_HASH_KEY, result);
354        }
355
356        return result;
357    }

359    /**
360     * Recreates a DTD from an archived format.
361     * @param in  the <code>DataInputStream</code> to read from
362     */
363    public void read(DataInputStream in) throws IOException {
364        if (in.readInt() != FILE_VERSION) {
365        }
366
367        //
368        // Read the list of names
369        //
370        String[] names = new String[in.readShort()];
371        for (int i = 0; i < names.length; i++) {
372            names[i] = in.readUTF();
373        }
374
375
376        //
377        // Read the entities
378        //
379        int num = in.readShort();
380        for (int i = 0; i < num; i++) {
381            short nameId = in.readShort();
382            int type = in.readByte();
383            String name = in.readUTF();
384            defEntity(names[nameId], type | GENERAL, name);
385        }
386
387        // Read the elements
388        //
389        num = in.readShort();
390        for (int i = 0; i < num; i++) {
391            short nameId = in.readShort();
392            int type = in.readByte();
393            byte flags = in.readByte();
394            ContentModel m = readContentModel(in, names);
395            String[] exclusions = readNameArray(in, names);
396            String[] inclusions = readNameArray(in, names);
397            AttributeList atts = readAttributeList(in, names);
398            defElement(names[nameId], type,
399                       ((flags & 0x01) != 0), ((flags & 0x02) != 0),
400                       m, exclusions, inclusions, atts);
401        }
402    }
403
404    private ContentModel readContentModel(DataInputStream in, String[] names)
405                throws IOException {
406        byte flag = in.readByte();
407        switch(flag) {
408            case 0:             // null
409                return null;
410            case 1: {           // content_c
411                int type = in.readByte();
412                ContentModel m = readContentModel(in, names);
413                ContentModel next = readContentModel(in, names);
414                return defContentModel(type, m, next);
415            }
416            case 2: {           // content_e
417                int type = in.readByte();
418                Element el = getElement(names[in.readShort()]);
419                ContentModel next = readContentModel(in, names);
420                return defContentModel(type, el, next);
421            }
422        default:
423                throw new IOException("bad bdtd");
424        }
425    }

427    private String[] readNameArray(DataInputStream in, String[] names)
428                throws IOException {
429        int num = in.readShort();
430        if (num == 0) {
431            return null;
432        }
433        String[] result = new String[num];
434        for (int i = 0; i < num; i++) {
435            result[i] = names[in.readShort()];
436        }
437        return result;
438    }

441    private AttributeList readAttributeList(DataInputStream in, String[] names)
442                throws IOException  {
443        AttributeList result = null;
444        for (int num = in.readByte(); num > 0; --num) {
445            short nameId = in.readShort();
446            int type = in.readByte();
447            int modifier = in.readByte();
448            short valueId = in.readShort();
449            String value = (valueId == -1) ? null : names[valueId];
450            Vector<String> values = null;
451            short numValues = in.readShort();
452            if (numValues > 0) {
453                values = new Vector<String>(numValues);
454                for (int i = 0; i < numValues; i++) {
455                    values.addElement(names[in.readShort()]);
456                }
457            }
458result = new AttributeList(names[nameId], type, modifier, value,
459                                       values, result);
460            // We reverse the order of the linked list by doing this, but
461            // that order isn't important.
462        }
463        return result;
464    }
