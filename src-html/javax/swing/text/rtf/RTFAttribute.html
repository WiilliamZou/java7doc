025package javax.swing.text.rtf;
026
027import javax.swing.text.AttributeSet;
028import javax.swing.text.MutableAttributeSet;
029import java.io.IOException;
030
031/**
032 * This interface describes a class which defines a 1-1 mapping between
033 * an RTF keyword and a SwingText attribute.
034 */
035interface RTFAttribute
036{
037    static final int D_CHARACTER = 0;
038    static final int D_PARAGRAPH = 1;
039    static final int D_SECTION = 2;
040    static final int D_DOCUMENT = 3;
041    static final int D_META = 4;
042
043    /* These next three should really be public variables,
044       but you can't declare public variables in an interface... */
045    /* int domain; */
046    public int domain();
047    /* String swingName; */
048    public Object swingName();
049    /* String rtfName; */
050    public String rtfName();
051
052    public boolean set(MutableAttributeSet target);
053    public boolean set(MutableAttributeSet target, int parameter);
054
055    public boolean setDefault(MutableAttributeSet target);
056
057    /* TODO: This method is poorly thought out */
058    public boolean write(AttributeSet source,
059                         RTFGenerator target,
060                         boolean force)
061        throws IOException;
062
063    public boolean writeValue(Object value,
064                              RTFGenerator target,
065                              boolean force)
066        throws IOException;
067}
