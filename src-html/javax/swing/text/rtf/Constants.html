025package javax.swing.text.rtf;
026
027/**
028   Class to hold dictionary keys used by the RTF reader/writer.
029   These should be moved into StyleConstants.
030*/
031class Constants
032{
033    /** An array of TabStops */
034    static final String Tabs = "tabs";
035
036    /** The name of the character set the original RTF file was in */
037    static final String RTFCharacterSet = "rtfCharacterSet";
038
039    /** Indicates the domain of a Style */
040    static final String StyleType = "style:type";
041
042    /** Value for StyleType indicating a section style */
043    static final String STSection = "section";
044    /** Value for StyleType indicating a paragraph style */
045    static final String STParagraph = "paragraph";
046    /** Value for StyleType indicating a character style */
047    static final String STCharacter = "character";
048
049    /** The style of the text following this style */
050    static final String StyleNext = "style:nextStyle";
051
052    /** Whether the style is additive */
053    static final String StyleAdditive = "style:additive";
054
055    /** Whether the style is hidden from the user */
056    static final String StyleHidden = "style:hidden";
057
058    /* Miscellaneous character attributes */
059    static final String Caps          = "caps";
060    static final String Deleted       = "deleted";
061    static final String Outline       = "outl";
062    static final String SmallCaps     = "scaps";
063    static final String Shadow        = "shad";
064    static final String Strikethrough = "strike";
065    static final String Hidden        = "v";
066
067    /* Miscellaneous document attributes */
068    static final String PaperWidth    = "paperw";
069    static final String PaperHeight   = "paperh";
070    static final String MarginLeft    = "margl";
071    static final String MarginRight   = "margr";
072    static final String MarginTop     = "margt";
073    static final String MarginBottom  = "margb";
074    static final String GutterWidth   = "gutter";
075
076    /* This is both a document and a paragraph attribute */
077    static final String WidowControl  = "widowctrl";
078}
