025package javax.swing.text;
026
027import java.awt.Font;
028import java.awt.Color;
029
030/**
031 * Interface for a generic styled document.
032 *
033 * @author  Timothy Prinzing
034 */
035public interface StyledDocument extends Document {
036
037    /**
038     * Adds a new style into the logical style hierarchy.  Style attributes
039     * resolve from bottom up so an attribute specified in a child
040     * will override an attribute specified in the parent.
041     *
042     * @param nm   the name of the style (must be unique within the
043     *   collection of named styles).  The name may be null if the style
044     *   is unnamed, but the caller is responsible
045     *   for managing the reference returned as an unnamed style can't
046     *   be fetched by name.  An unnamed style may be useful for things
047     *   like character attribute overrides such as found in a style
048     *   run.
049     * @param parent the parent style.  This may be null if unspecified
050     *   attributes need not be resolved in some other style.
051     * @return the style
052     */
053    public Style addStyle(String nm, Style parent);
054
055    /**
056     * Removes a named style previously added to the document.
057     *
058     * @param nm  the name of the style to remove
059     */
060    public void removeStyle(String nm);
061
062    /**
063     * Fetches a named style previously added.
064     *
065     * @param nm  the name of the style
066     * @return the style
067     */
068    public Style getStyle(String nm);
069
070    /**
071     * Changes the content element attributes used for the given range of
072     * existing content in the document.  All of the attributes
073     * defined in the given Attributes argument are applied to the
074     * given range.  This method can be used to completely remove
075     * all content level attributes for the given range by
076     * giving an Attributes argument that has no attributes defined
077     * and setting replace to true.
078     *
079     * @param offset the start of the change >= 0
080     * @param length the length of the change >= 0
081     * @param s    the non-null attributes to change to.  Any attributes
082     *  defined will be applied to the text for the given range.
083     * @param replace indicates whether or not the previous
084     *  attributes should be cleared before the new attributes
085     *  as set.  If true, the operation will replace the
086     *  previous attributes entirely.  If false, the new
087     *  attributes will be merged with the previous attributes.
088     */
089    public void setCharacterAttributes(int offset, int length, AttributeSet s, boolean replace);
090
091    /**
092     * Sets paragraph attributes.
093     *
094     * @param offset the start of the change >= 0
095     * @param length the length of the change >= 0
096     * @param s    the non-null attributes to change to.  Any attributes
097     *  defined will be applied to the text for the given range.
098     * @param replace indicates whether or not the previous
099     *  attributes should be cleared before the new attributes
100     *  are set.  If true, the operation will replace the
101     *  previous attributes entirely.  If false, the new
102     *  attributes will be merged with the previous attributes.
103     */
104    public void setParagraphAttributes(int offset, int length, AttributeSet s, boolean replace);
105
106    /**
107     * Sets the logical style to use for the paragraph at the
108     * given position.  If attributes aren't explicitly set
109     * for character and paragraph attributes they will resolve
110     * through the logical style assigned to the paragraph, which
111     * in turn may resolve through some hierarchy completely
112     * independent of the element hierarchy in the document.
113     *
114     * @param pos the starting position >= 0
115     * @param s the style to set
116     */
117    public void setLogicalStyle(int pos, Style s);
118
119    /**
120     * Gets a logical style for a given position in a paragraph.
121     *
122     * @param p the position >= 0
123     * @return the style
124     */
125    public Style getLogicalStyle(int p);
126
127    /**
128     * Gets the element that represents the paragraph that
129     * encloses the given offset within the document.
130     *
131     * @param pos the offset >= 0
132     * @return the element
133     */
134    public Element getParagraphElement(int pos);
135
136    /**
137     * Gets the element that represents the character that
138     * is at the given offset within the document.
139     *
140     * @param pos the offset >= 0
141     * @return the element
142     */
143    public Element getCharacterElement(int pos);
144
145
146    /**
147     * Takes a set of attributes and turn it into a foreground color
148     * specification.  This might be used to specify things
149     * like brighter, more hue, etc.
150     *
151     * @param attr the set of attributes
152     * @return the color
153     */
154    public Color getForeground(AttributeSet attr);
155
156    /**
157     * Takes a set of attributes and turn it into a background color
158     * specification.  This might be used to specify things
159     * like brighter, more hue, etc.
160     *
161     * @param attr the set of attributes
162     * @return the color
163     */
164    public Color getBackground(AttributeSet attr);
165
166    /**
167     * Takes a set of attributes and turn it into a font
168     * specification.  This can be used to turn things like
169     * family, style, size, etc into a font that is available
170     * on the system the document is currently being used on.
171     *
172     * @param attr the set of attributes
173     * @return the font
174     */
175    public Font getFont(AttributeSet attr);
176
177}
