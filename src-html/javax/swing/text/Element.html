025package javax.swing.text;
026
027/**
028 * Interface to describe a structural piece of a document.  It
029 * is intended to capture the spirit of an SGML element.
030 *
031 * @author  Timothy Prinzing
032 */
033public interface Element {
034
035    /**
036     * Fetches the document associated with this element.
037     *
038     * @return the document
039     */
040    public Document getDocument();
041
042    /**
043     * Fetches the parent element.  If the element is a root level
044     * element returns <code>null</code>.
045     *
046     * @return the parent element
047     */
048    public Element getParentElement();
049
050    /**
051     * Fetches the name of the element.  If the element is used to
052     * represent some type of structure, this would be the type
053     * name.
054     *
055     * @return the element name
056     */
057    public String getName();
058
059    /**
060     * Fetches the collection of attributes this element contains.
061     *
062     * @return the attributes for the element
063     */
064    public AttributeSet getAttributes();
065
066    /**
067     * Fetches the offset from the beginning of the document
068     * that this element begins at.  If this element has
069     * children, this will be the offset of the first child.
070     * As a document position, there is an implied forward bias.
071     *
072     * @return the starting offset >= 0 and < getEndOffset();
073     * @see Document
074     * @see AbstractDocument
075     */
076    public int getStartOffset();
077
078    /**
079     * Fetches the offset from the beginning of the document
080     * that this element ends at.  If this element has
081     * children, this will be the end offset of the last child.
082     * As a document position, there is an implied backward bias.
083     * <p>
084     * All the default <code>Document</code> implementations
085     * descend from <code>AbstractDocument</code>.
086     * <code>AbstractDocument</code> models an implied break at the end of
087     * the document. As a result of this, it is possible for this to
088     * return a value greater than the length of the document.
089     *
090     * @return the ending offset > getStartOffset() and
091     *     <= getDocument().getLength() + 1
092     * @see Document
093     * @see AbstractDocument
094     */
095    public int getEndOffset();
096
097    /**
098     * Gets the child element index closest to the given offset.
099     * The offset is specified relative to the beginning of the
100     * document.  Returns <code>-1</code> if the
101     * <code>Element</code> is a leaf, otherwise returns
102     * the index of the <code>Element</code> that best represents
103     * the given location.  Returns <code>0</code> if the location
104     * is less than the start offset. Returns
105     * <code>getElementCount() - 1</code> if the location is
106     * greater than or equal to the end offset.
107     *
108     * @param offset the specified offset >= 0
109     * @return the element index >= 0
110     */
111    public int getElementIndex(int offset);
112
113    /**
114     * Gets the number of child elements contained by this element.
115     * If this element is a leaf, a count of zero is returned.
116     *
117     * @return the number of child elements >= 0
118     */
119    public int getElementCount();
120
121    /**
122     * Fetches the child element at the given index.
123     *
124     * @param index the specified index >= 0
125     * @return the child element
126     */
127    public Element getElement(int index);
128
129    /**
130     * Is this element a leaf element? An element that
131     * <i>may</i> have children, even if it currently
132     * has no children, would return <code>false</code>.
133     *
134     * @return true if a leaf element else false
135     */
136    public boolean isLeaf();
137
138
139}
