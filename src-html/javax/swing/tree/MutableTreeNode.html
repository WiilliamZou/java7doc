026package javax.swing.tree;
027
028/**
029 * Defines the requirements for a tree node object that can change --
030 * by adding or removing child nodes, or by changing the contents
031 * of a user object stored in the node.
032 *
033 * @see DefaultMutableTreeNode
034 * @see javax.swing.JTree
035 *
036 * @author Rob Davis
037 * @author Scott Violet
038 */
039
040public interface MutableTreeNode extends TreeNode
041{
042    /**
043     * Adds <code>child</code> to the receiver at <code>index</code>.
044     * <code>child</code> will be messaged with <code>setParent</code>.
045     */
046    void insert(MutableTreeNode child, int index);
047
048    /**
049     * Removes the child at <code>index</code> from the receiver.
050     */
051    void remove(int index);
052
053    /**
054     * Removes <code>node</code> from the receiver. <code>setParent</code>
055     * will be messaged on <code>node</code>.
056     */
057    void remove(MutableTreeNode node);
058
059    /**
060     * Resets the user object of the receiver to <code>object</code>.
061     */
062    void setUserObject(Object object);
063
064    /**
065     * Removes the receiver from its parent.
066     */
067    void removeFromParent();
068
069    /**
070     * Sets the parent of the receiver to <code>newParent</code>.
071     */
072    void setParent(MutableTreeNode newParent);
073}
