026package javax.swing.tree;
027
028import java.awt.Component;
029import javax.swing.CellEditor;
030import javax.swing.JTree;
031
032/**
033  * Adds to CellEditor the extensions necessary to configure an editor
034  * in a tree.
035  *
036  * @see javax.swing.JTree
037  *
038  * @author Scott Violet
039  */
040
041public interface TreeCellEditor extends CellEditor
042{
043    /**
044     * Sets an initial <I>value</I> for the editor.  This will cause
045     * the editor to stopEditing and lose any partially edited value
046     * if the editor is editing when this method is called. <p>
047     *
048     * Returns the component that should be added to the client's
049     * Component hierarchy.  Once installed in the client's hierarchy
050     * this component will then be able to draw and receive user input.
051     *
052     * @param   tree            the JTree that is asking the editor to edit;
053     *                          this parameter can be null
054     * @param   value           the value of the cell to be edited
055     * @param   isSelected      true if the cell is to be rendered with
056     *                          selection highlighting
057     * @param   expanded        true if the node is expanded
058     * @param   leaf            true if the node is a leaf node
059     * @param   row             the row index of the node being edited
060     * @return  the component for editing
061     */
062    Component getTreeCellEditorComponent(JTree tree, Object value,
063                                         boolean isSelected, boolean expanded,
064                                         boolean leaf, int row);
065}
