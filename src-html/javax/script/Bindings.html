026package javax.script;
027import java.util.Map;
028
029/**
030 * A mapping of key/value pairs, all of whose keys are
031 * <code>Strings</code>.
032 *
033 * @author Mike Grogan
034 * @since 1.6
035 */
036public interface Bindings extends Map<String, Object> {
037    /**
038     * Set a named value.
039     *
040     * @param name The name associated with the value.
041     * @param value The value associated with the name.
042     *
043     * @return The value previously associated with the given name.
044     * Returns null if no value was previously associated with the name.
045     *
046     * @throws NullPointerException if the name is null.
047     * @throws IllegalArgumentException if the name is empty String.
048     */
049    public Object put(String name, Object value);
050
051    /**
052     * Adds all the mappings in a given <code>Map</code> to this <code>Bindings</code>.
053     * @param toMerge The <code>Map</code> to merge with this one.
054     *
055     * @throws NullPointerException
056     *         if toMerge map is null or if some key in the map is null.
057     * @throws IllegalArgumentException
058     *         if some key in the map is an empty String.
059     */
060    public void putAll(Map<? extends String, ? extends Object> toMerge);
061
062    /**
063     * Returns <tt>true</tt> if this map contains a mapping for the specified
064     * key.  More formally, returns <tt>true</tt> if and only if
065     * this map contains a mapping for a key <tt>k</tt> such that
066     * <tt>(key==null ? k==null : key.equals(k))</tt>.  (There can be
067     * at most one such mapping.)
068     *
069     * @param key key whose presence in this map is to be tested.
070     * @return <tt>true</tt> if this map contains a mapping for the specified
071     *         key.
072     *
073     * @throws NullPointerException if key is null
074     * @throws ClassCastException if key is not String
075     * @throws IllegalArgumentException if key is empty String
076     */
077    public boolean containsKey(Object key);
078
079    /**
080     * Returns the value to which this map maps the specified key.  Returns
081     * <tt>null</tt> if the map contains no mapping for this key.  A return
082     * value of <tt>null</tt> does not <i>necessarily</i> indicate that the
083     * map contains no mapping for the key; it's also possible that the map
084     * explicitly maps the key to <tt>null</tt>.  The <tt>containsKey</tt>
085     * operation may be used to distinguish these two cases.
086     *
087     * <p>More formally, if this map contains a mapping from a key
088     * <tt>k</tt> to a value <tt>v</tt> such that <tt>(key==null ? k==null :
089     * key.equals(k))</tt>, then this method returns <tt>v</tt>; otherwise
090     * it returns <tt>null</tt>.  (There can be at most one such mapping.)
091     *
092     * @param key key whose associated value is to be returned.
093     * @return the value to which this map maps the specified key, or
094     *         <tt>null</tt> if the map contains no mapping for this key.
095     *
096     * @throws NullPointerException if key is null
097     * @throws ClassCastException if key is not String
098     * @throws IllegalArgumentException if key is empty String
099     */
100    public Object get(Object key);
101
102    /**
103     * Removes the mapping for this key from this map if it is present
104     * (optional operation).   More formally, if this map contains a mapping
105     * from key <tt>k</tt> to value <tt>v</tt> such that
106     * <code>(key==null ?  k==null : key.equals(k))</code>, that mapping
107     * is removed.  (The map can contain at most one such mapping.)
108     *
109     * <p>Returns the value to which the map previously associated the key, or
110     * <tt>null</tt> if the map contained no mapping for this key.  (A
111     * <tt>null</tt> return can also indicate that the map previously
112     * associated <tt>null</tt> with the specified key if the implementation
113     * supports <tt>null</tt> values.)  The map will not contain a mapping for
114     * the specified  key once the call returns.
115     *
116     * @param key key whose mapping is to be removed from the map.
117     * @return previous value associated with specified key, or <tt>null</tt>
118     *         if there was no mapping for key.
119     *
120     * @throws NullPointerException if key is null
121     * @throws ClassCastException if key is not String
122     * @throws IllegalArgumentException if key is empty String
123     */
124    public Object remove(Object key);
125}
