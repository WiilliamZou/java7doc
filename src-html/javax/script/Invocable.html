026package javax.script;
027
028/**
029 * The optional interface implemented by ScriptEngines whose methods allow the invocation of
030 * procedures in scripts that have previously been executed.
031 *
032 * @author  Mike Grogan
033 * @author  A. Sundararajan
034 * @since 1.6
035 */
036public interface Invocable  {
037    /**
038     * Calls a method on a script object compiled during a previous script execution,
039     * which is retained in the state of the <code>ScriptEngine</code>.
040     *
041     * @param name The name of the procedure to be called.
042     *
043     * @param thiz If the procedure is a member  of a class
044     * defined in the script and thiz is an instance of that class
045     * returned by a previous execution or invocation, the named method is
046     * called through that instance.
047     *
048     * @param args Arguments to pass to the procedure.  The rules for converting
049     * the arguments to scripting variables are implementation-specific.
050     *
051     * @return The value returned by the procedure.  The rules for converting the scripting
052     * variable returned by the script method to a Java Object are implementation-specific.
053     *
054     * @throws ScriptException if an error occurrs during invocation of the method.
055     * @throws NoSuchMethodException if method with given name or matching argument types cannot be found.
056     * @throws NullPointerException if the method name is null.
057     * @throws IllegalArgumentException if the specified thiz is null or the specified Object is
058     * does not represent a scripting object.
059     */
060    public Object invokeMethod(Object thiz, String name, Object... args)
061        throws ScriptException, NoSuchMethodException;
062
063    /**
064     * Used to call top-level procedures and functions defined in scripts.
065     *
066     * @param args Arguments to pass to the procedure or function
067     * @return The value returned by the procedure or function
068     *
069     * @throws ScriptException if an error occurrs during invocation of the method.
070     * @throws NoSuchMethodException if method with given name or matching argument types cannot be found.
071     * @throws NullPointerException if method name is null.
072     */
073     public Object invokeFunction(String name, Object... args)
074        throws ScriptException, NoSuchMethodException;
075
076
077     /**
078     * Returns an implementation of an interface using functions compiled in
079     * the interpreter. The methods of the interface
080     * may be implemented using the <code>invokeFunction</code> method.
081     *
082     * @param clasz The <code>Class</code> object of the interface to return.
083     *
084     * @return An instance of requested interface - null if the requested interface is unavailable,
085     * i. e. if compiled functions in the <code>ScriptEngine</code> cannot be found matching
086     * the ones in the requested interface.
087     *
088     * @throws IllegalArgumentException if the specified <code>Class</code> object
089     * is null or is not an interface.
090     */
091    public <T> T getInterface(Class<T> clasz);
092
093    /**
094     * Returns an implementation of an interface using member functions of
095     * a scripting object compiled in the interpreter. The methods of the
096     * interface may be implemented using the <code>invokeMethod</code> method.
097     *
098     * @param thiz The scripting object whose member functions are used to implement the methods of the interface.
099     * @param clasz The <code>Class</code> object of the interface to return.
100     *
101     * @return An instance of requested interface - null if the requested interface is unavailable,
102     * i. e. if compiled methods in the <code>ScriptEngine</code> cannot be found matching
103     * the ones in the requested interface.
104     *
105     * @throws IllegalArgumentException if the specified <code>Class</code> object
106     * is null or is not an interface, or if the specified Object is
107     * null or does not represent a scripting object.
108     */
109     public <T> T getInterface(Object thiz, Class<T> clasz);
110
111}
