026package javax.script;
027
028/**
029 * The generic <code>Exception</code> class for the Scripting APIs.  Checked
030 * exception types thrown by underlying scripting implementations must be wrapped in instances of
031 * <code>ScriptException</code>.  The class has members to store line and column numbers and
032 * filenames if this information is available.
033 *
034 * @author Mike Grogan
035 * @since 1.6
036 */
037public class ScriptException extends Exception {
038
039    private String fileName;
040    private int lineNumber;
041    private int columnNumber;
042
043    /**
044     * Creates a <code>ScriptException</code> with a String to be used in its message.
045     * Filename, and line and column numbers are unspecified.
046     *
047     * @param s The String to use in the message.
048     */
049    public ScriptException(String s) {
050        super(s);
051        fileName = null;
052        lineNumber = -1;
053        columnNumber = -1;
054    }
055
056    /**
057     * Creates a <code>ScriptException</code> wrapping an <code>Exception</code> thrown by an underlying
058     * interpreter.  Line and column numbers and filename are unspecified.
059     *
060     * @param e The wrapped <code>Exception</code>.
061     */
062    public ScriptException(Exception e) {
063        super(e);
064        fileName = null;
065        lineNumber = -1;
066        columnNumber = -1;
067    }
068
069    /**
070     * Creates a <code>ScriptException</code> with message, filename and linenumber to
071     * be used in error messages.
072     *
073     * @param message The string to use in the message
074     *
075     * @param fileName The file or resource name describing the location of a script error
076     * causing the <code>ScriptException</code> to be thrown.
077     *
078     * @param lineNumber A line number describing the location of a script error causing
079     * the <code>ScriptException</code> to be thrown.
080     */
081    public ScriptException(String message, String fileName, int lineNumber) {
082        super(message);
083        this.fileName = fileName;
084        this.lineNumber = lineNumber;
085        this.columnNumber = -1;
086    }
087
088    /**
089     * <code>ScriptException</code> constructor specifying message, filename, line number
090     * and column number.
091     * @param message The message.
092     * @param fileName The filename
093     * @param lineNumber the line number.
094     * @param columnNumber the column number.
095     */
096    public ScriptException(String message,
097            String fileName,
098            int lineNumber,
099            int columnNumber) {
100        super(message);
101        this.fileName = fileName;
102        this.lineNumber = lineNumber;
103        this.columnNumber = columnNumber;
104    }
105
106    /**
107     * Returns a message containing the String passed to a constructor as well as
108     * line and column numbers and filename if any of these are known.
109     * @return The error message.
110     */
111    public String getMessage() {
112        String ret = super.getMessage();
113        if (fileName != null) {
114            ret += (" in " + fileName);
115            if (lineNumber != -1) {
116                ret += " at line number " + lineNumber;
117            }
118
119            if (columnNumber != -1) {
120                ret += " at column number " + columnNumber;
121            }
122        }
123
124        return ret;
125    }
126
127    /**
128     * Get the line number on which an error occurred.
129     * @return The line number.  Returns -1 if a line number is unavailable.
130     */
131    public int getLineNumber() {
132        return lineNumber;
133    }
134
135    /**
136     * Get the column number on which an error occurred.
137     * @return The column number.  Returns -1 if a column number is unavailable.
138     */
139    public int getColumnNumber() {
140        return columnNumber;
141    }
142
143    /**
144     * Get the source of the script causing the error.
145     * @return The file name of the script or some other string describing the script
146     * source.  May return some implementation-defined string such as <i>&lt;unknown&gt;</i>
147     * if a description of the source is unavailable.
148     */
149    public String getFileName() {
150        return fileName;
151    }
152}
