026package javax.security.sasl;
027
028/**
029 * This exception is thrown by a SASL mechanism implementation
030 * to indicate that the SASL
031 * exchange has failed due to reasons related to authentication, such as
032 * an invalid identity, passphrase, or key.
033 * <p>
034 * Note that the lack of an AuthenticationException does not mean that
035 * the failure was not due to an authentication error.  A SASL mechanism
036 * implementation might throw the more general SaslException instead of
037 * AuthenticationException if it is unable to determine the nature
038 * of the failure, or if does not want to disclose the nature of
039 * the failure, for example, due to security reasons.
040 *
041 * @since 1.5
042 *
043 * @author Rosanna Lee
044 * @author Rob Weltman
045 */
046public class AuthenticationException extends SaslException {
047    /**
048     * Constructs a new instance of <tt>AuthenticationException</tt>.
049     * The root exception and the detailed message are null.
050     */
051    public AuthenticationException () {
052        super();
053    }
054
055    /**
056     * Constructs a new instance of <tt>AuthenticationException</tt>
057     * with a detailed message.
058     * The root exception is null.
059     * @param detail A possibly null string containing details of the exception.
060     *
061     * @see java.lang.Throwable#getMessage
062     */
063    public AuthenticationException (String detail) {
064        super(detail);
065    }
066
067    /**
068     * Constructs a new instance of <tt>AuthenticationException</tt> with a detailed message
069     * and a root exception.
070     *
071     * @param detail A possibly null string containing details of the exception.
072     * @param ex A possibly null root exception that caused this exception.
073     *
074     * @see java.lang.Throwable#getMessage
075     * @see #getCause
076     */
077    public AuthenticationException (String detail, Throwable ex) {
078        super(detail, ex);
079    }
080
081    /** Use serialVersionUID from JSR 28 RI for interoperability */
082    private static final long serialVersionUID = -3579708765071815007L;
083}
