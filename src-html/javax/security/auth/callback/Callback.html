027package javax.security.auth.callback;

029/**
030 * <p> Implementations of this interface are passed to a
031 * <code>CallbackHandler</code>, allowing underlying security services
032 * the ability to interact with a calling application to retrieve specific
033 * authentication data such as usernames and passwords, or to display
034 * certain information, such as error and warning messages.
035 *
036 * <p> <code>Callback</code> implementations do not retrieve or
037 * display the information requested by underlying security services.
038 * <code>Callback</code> implementations simply provide the means
039 * to pass such requests to applications, and for applications,
040 * if appropriate, to return requested information back to the
041 * underlying security services.
042 *
043 * @see javax.security.auth.callback.CallbackHandler
044 * @see javax.security.auth.callback.ChoiceCallback
045 * @see javax.security.auth.callback.ConfirmationCallback
046 * @see javax.security.auth.callback.LanguageCallback
047 * @see javax.security.auth.callback.NameCallback
048 * @see javax.security.auth.callback.PasswordCallback
049 * @see javax.security.auth.callback.TextInputCallback
050 * @see javax.security.auth.callback.TextOutputCallback
051 */
052public interface Callback { }
