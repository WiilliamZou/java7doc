027package javax.security.auth.login;
028
029/**
030 * This class defines the <i>Service Provider Interface</i> (<b>SPI</b>)
031 * for the <code>Configuration</code> class.
032 * All the abstract methods in this class must be implemented by each
033 * service provider who wishes to supply a Configuration implementation.
034 *
035 * <p> Subclass implementations of this abstract class must provide
036 * a public constructor that takes a <code>Configuration.Parameters</code>
037 * object as an input parameter.  This constructor also must throw
038 * an IllegalArgumentException if it does not understand the
039 * <code>Configuration.Parameters</code> input.
040 *
041 *
042 * @since 1.6
043 */
044
045public abstract class ConfigurationSpi {
046    /**
047     * Retrieve the AppConfigurationEntries for the specified <i>name</i>.
048     *
049     * <p>
050     *
051     * @param name the name used to index the Configuration.
052     *
053     * @return an array of AppConfigurationEntries for the specified
054     *          <i>name</i>, or null if there are no entries.
055     */
056    protected abstract AppConfigurationEntry[] engineGetAppConfigurationEntry
057                                                        (String name);
058
059    /**
060     * Refresh and reload the Configuration.
061     *
062     * <p> This method causes this Configuration object to refresh/reload its
063     * contents in an implementation-dependent manner.
064     * For example, if this Configuration object stores its entries in a file,
065     * calling <code>refresh</code> may cause the file to be re-read.
066     *
067     * <p> The default implementation of this method does nothing.
068     * This method should be overridden if a refresh operation is supported
069     * by the implementation.
070     *
071     * @exception SecurityException if the caller does not have permission
072     *          to refresh its Configuration.
073     */
074    protected void engineRefresh() { }
075}
