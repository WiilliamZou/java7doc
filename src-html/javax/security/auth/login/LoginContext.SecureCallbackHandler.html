043/**
044 * <p> The <code>LoginContext</code> class describes the basic methods used
045 * to authenticate Subjects and provides a way to develop an
046 * application independent of the underlying authentication technology.
047 * A <code>Configuration</code> specifies the authentication technology, or
048 * <code>LoginModule</code>, to be used with a particular application.
049 * Different LoginModules can be plugged in under an application
050 * without requiring any modifications to the application itself.
051 *
052 * <p> In addition to supporting <i>pluggable</i> authentication, this class
053 * also supports the notion of <i>stacked</i> authentication.
054 * Applications may be configured to use more than one
055 * LoginModule.  For example, one could
056 * configure both a Kerberos LoginModule and a smart card
057 * LoginModule under an application.
058 *
059 * <p> A typical caller instantiates a LoginContext with
060 * a <i>name</i> and a <code>CallbackHandler</code>.
061 * LoginContext uses the <i>name</i> as the index into a
062 * Configuration to determine which LoginModules should be used,
063 * and which ones must succeed in order for the overall authentication to
064 * succeed.  The <code>CallbackHandler</code> is passed to the underlying
065 * LoginModules so they may communicate and interact with users
066 * (prompting for a username and password via a graphical user interface,
067 * for example).
068 *
069 * <p> Once the caller has instantiated a LoginContext,
070 * it invokes the <code>login</code> method to authenticate
071 * a <code>Subject</code>.  The <code>login</code> method invokes
072 * the configured modules to perform their respective types of authentication
073 * (username/password, smart card pin verification, etc.).
074 * Note that the LoginModules will not attempt authentication retries nor
075 * introduce delays if the authentication fails.
076 * Such tasks belong to the LoginContext caller.
077 *
078 * <p> If the <code>login</code> method returns without
079 * throwing an exception, then the overall authentication succeeded.
080 * The caller can then retrieve
081 * the newly authenticated Subject by invoking the
082 * <code>getSubject</code> method.  Principals and Credentials associated
083 * with the Subject may be retrieved by invoking the Subject's
084 * respective <code>getPrincipals</code>, <code>getPublicCredentials</code>,
085 * and <code>getPrivateCredentials</code> methods.
086 *
087 * <p> To logout the Subject, the caller calls
088 * the <code>logout</code> method.  As with the <code>login</code>
089 * method, this <code>logout</code> method invokes the <code>logout</code>
090 * method for the configured modules.
091 *
092 * <p> A LoginContext should not be used to authenticate
093 * more than one Subject.  A separate LoginContext
094 * should be used to authenticate each different Subject.
095 *
096 * <p> The following documentation applies to all LoginContext constructors:
097 * <ol>
098 *
099 * <li> <code>Subject</code>
100 * <ul>
101 * <li> If the constructor has a Subject
102 * input parameter, the LoginContext uses the caller-specified
103 * Subject object.
104 * <p>
105 * <li> If the caller specifies a <code>null</code> Subject
106 * and a <code>null</code> value is permitted,
107 * the LoginContext instantiates a new Subject.
108 * <p>
109 * <li> If the constructor does <b>not</b> have a Subject
110 * input parameter, the LoginContext instantiates a new Subject.
111 * <p>
112 * </ul>
113 *
114 * <li> <code>Configuration</code>
115 * <ul>
116 * <li> If the constructor has a Configuration
117 * input parameter and the caller specifies a non-null Configuration,
118 * the LoginContext uses the caller-specified Configuration.
119 * <p>
120 * If the constructor does <b>not</b> have a Configuration
121 * input parameter, or if the caller specifies a <code>null</code>
122 * Configuration object, the constructor uses the following call to
123 * get the installed Configuration:
124 * <pre>
125 *      config = Configuration.getConfiguration();
126 * </pre>
127 * For both cases,
128 * the <i>name</i> argument given to the constructor is passed to the
129 * <code>Configuration.getAppConfigurationEntry</code> method.
130 * If the Configuration has no entries for the specified <i>name</i>,
131 * then the <code>LoginContext</code> calls
132 * <code>getAppConfigurationEntry</code> with the name, "<i>other</i>"
133 * (the default entry name).  If there is no entry for "<i>other</i>",
134 * then a <code>LoginException</code> is thrown.
135 * <p>
136 * <li> When LoginContext uses the installed Configuration, the caller
137 * requires the createLoginContext.<em>name</em> and possibly
138 * createLoginContext.other AuthPermissions. Furthermore, the
139 * LoginContext will invoke configured modules from within an
140 * <code>AccessController.doPrivileged</code> call so that modules that
141 * perform security-sensitive tasks (such as connecting to remote hosts,
142 * and updating the Subject) will require the respective permissions, but
143 * the callers of the LoginContext will not require those permissions.
144 * <p>
145 * <li> When LoginContext uses a caller-specified Configuration, the caller
146 * does not require any createLoginContext AuthPermission.  The LoginContext
147 * saves the <code>AccessControlContext</code> for the caller,
148 * and invokes the configured modules from within an
149 * <tt>AccessController.doPrivileged</tt> call constrained by that context.
150 * This means the caller context (stored when the LoginContext was created)
151 * must have sufficient permissions to perform any security-sensitive tasks
152 * that the modules may perform.
153 * <p>
154 * </ul>
155 *
156 * <li> <code>CallbackHandler</code>
157 * <ul>
158 * <li> If the constructor has a CallbackHandler
159 * input parameter, the LoginContext uses the caller-specified
160 * CallbackHandler object.
161 * <p>
162 * <li> If the constructor does <b>not</b> have a CallbackHandler
163 * input parameter, or if the caller specifies a <code>null</code>
164 * CallbackHandler object (and a <code>null</code> value is permitted),
165 * the LoginContext queries the
166 * <i>auth.login.defaultCallbackHandler</i> security property
167 * for the fully qualified class name of a default handler implementation.
168 * If the security property is not set,
169 * then the underlying modules will not have a
170 * CallbackHandler for use in communicating
171 * with users.  The caller thus assumes that the configured
172 * modules have alternative means for authenticating the user.
173 *
174 * <p>
175 * <li> When the LoginContext uses the installed Configuration (instead of
176 * a caller-specified Configuration, see above),
177 * then this LoginContext must wrap any
178 * caller-specified or default CallbackHandler implementation
179 * in a new CallbackHandler implementation
180 * whose <code>handle</code> method implementation invokes the
181 * specified CallbackHandler's <code>handle</code> method in a
182 * <code>java.security.AccessController.doPrivileged</code> call
183 * constrained by the caller's current <code>AccessControlContext</code>.
184 * </ul>
185 * </ol>
186 *
187 * <p> Note that Security Properties
188 * (such as <code>auth.login.defaultCallbackHandler</code>)
189 * can be set programmatically via the
190 * <code>java.security.Security</code> class,
191 * or statically in the Java security properties file located in the
192 * file named &lt;JAVA_HOME&gt;/lib/security/java.security.
193 * &lt;JAVA_HOME&gt; refers to the value of the java.home system property,
194 * and specifies the directory where the JRE is installed.
195 *
196 * @see java.security.Security
197 * @see javax.security.auth.AuthPermission
198 * @see javax.security.auth.Subject
199 * @see javax.security.auth.callback.CallbackHandler
200 * @see javax.security.auth.login.Configuration
201 * @see javax.security.auth.spi.LoginModule
202 */

236    private void init(String name) throws LoginException {
237
238        SecurityManager sm = System.getSecurityManager();
239        if (sm != null && creatorAcc == null) {
240            sm.checkPermission(new AuthPermission
241                                ("createLoginContext." + name));
242        }
243
244        if (name == null)
245            throw new LoginException
246                (ResourcesMgr.getString("Invalid.null.input.name"));
247
248        // get the Configuration
249        if (config == null) {
250            config = java.security.AccessController.doPrivileged
251                (new java.security.PrivilegedAction<Configuration>() {
252                public Configuration run() {
253                    return Configuration.getConfiguration();
254                }
255            });
256        }
257
258        // get the LoginModules configured for this application
259        AppConfigurationEntry[] entries = config.getAppConfigurationEntry(name);
260        if (entries == null) {
261
262            if (sm != null && creatorAcc == null) {
263                sm.checkPermission(new AuthPermission
264                                ("createLoginContext." + OTHER));
265            }
266
267            entries = config.getAppConfigurationEntry(OTHER);
268            if (entries == null) {
269                MessageFormat form = new MessageFormat(ResourcesMgr.getString
270                        ("No.LoginModules.configured.for.name"));
271                Object[] source = {name};
272                throw new LoginException(form.format(source));
273            }
274        }
275        moduleStack = new ModuleInfo[entries.length];
276        for (int i = 0; i < entries.length; i++) {
277            // clone returned array
278            moduleStack[i] = new ModuleInfo
279                                (new AppConfigurationEntry
280                                        (entries[i].getLoginModuleName(),
281                                        entries[i].getControlFlag(),
282                                        entries[i].getOptions()),
283                                null);
284        }
285
286        contextClassLoader = java.security.AccessController.doPrivileged
287                (new java.security.PrivilegedAction<ClassLoader>() {
288                public ClassLoader run() {
289                    ClassLoader loader =
290                            Thread.currentThread().getContextClassLoader();
291                    if (loader == null) {
292                        // Don't use bootstrap class loader directly to ensure
293                        // proper package access control!
294                        loader = ClassLoader.getSystemClassLoader();
295                    }
296
297                    return loader;
298                }
299        });
300    }

302    private void loadDefaultCallbackHandler() throws LoginException {
303
304        // get the default handler class
305        try {
306
307            final ClassLoader finalLoader = contextClassLoader;
308
309            this.callbackHandler = java.security.AccessController.doPrivileged(
310                new java.security.PrivilegedExceptionAction<CallbackHandler>() {
311                public CallbackHandler run() throws Exception {
312                    String defaultHandler = java.security.Security.getProperty
313                        (DEFAULT_HANDLER);
314                    if (defaultHandler == null || defaultHandler.length() == 0)
315                        return null;
316                    Class<? extends CallbackHandler> c = Class.forName(
317                            defaultHandler, true,
318                            finalLoader).asSubclass(CallbackHandler.class);
319                    return c.newInstance();
320                }
321            });
322        } catch (java.security.PrivilegedActionException pae) {
323            throw new LoginException(pae.getException().toString());
324        }
325
326        // secure it with the caller's ACC
327        if (this.callbackHandler != null && creatorAcc == null) {
328            this.callbackHandler = new SecureCallbackHandler
329                                (java.security.AccessController.getContext(),
330                                this.callbackHandler);
331        }
332    }

512    public LoginContext(String name, Subject subject,
513                        CallbackHandler callbackHandler,
514                        Configuration config) throws LoginException {
515        this.config = config;
516        if (config != null) {
517            creatorAcc = java.security.AccessController.getContext();
518        }
519
520        init(name);
521        if (subject != null) {
522            this.subject = subject;
523            subjectProvided = true;
524        }
525        if (callbackHandler == null) {
526            loadDefaultCallbackHandler();
527        } else if (creatorAcc == null) {
528            this.callbackHandler = new SecureCallbackHandler
529                                (java.security.AccessController.getContext(),
530                                callbackHandler);
531        } else {
532            this.callbackHandler = callbackHandler;
533        }
534    }

536    /**
537     * Perform the authentication.
538     *
539     * <p> This method invokes the <code>login</code> method for each
540     * LoginModule configured for the <i>name</i> specified to the
541     * <code>LoginContext</code> constructor, as determined by the login
542     * <code>Configuration</code>.  Each <code>LoginModule</code>
543     * then performs its respective type of authentication
544     * (username/password, smart card pin verification, etc.).
545     *
546     * <p> This method completes a 2-phase authentication process by
547     * calling each configured LoginModule's <code>commit</code> method
548     * if the overall authentication succeeded (the relevant REQUIRED,
549     * REQUISITE, SUFFICIENT, and OPTIONAL LoginModules succeeded),
550     * or by calling each configured LoginModule's <code>abort</code> method
551     * if the overall authentication failed.  If authentication succeeded,
552     * each successful LoginModule's <code>commit</code> method associates
553     * the relevant Principals and Credentials with the <code>Subject</code>.
554     * If authentication failed, each LoginModule's <code>abort</code> method
555     * removes/destroys any previously stored state.
556     *
557     * <p> If the <code>commit</code> phase of the authentication process
558     * fails, then the overall authentication fails and this method
559     * invokes the <code>abort</code> method for each configured
560     * <code>LoginModule</code>.
561     *
562     * <p> If the <code>abort</code> phase
563     * fails for any reason, then this method propagates the
564     * original exception thrown either during the <code>login</code> phase
565     * or the <code>commit</code> phase.  In either case, the overall
566     * authentication fails.
567     *
568     * <p> In the case where multiple LoginModules fail,
569     * this method propagates the exception raised by the first
570     * <code>LoginModule</code> which failed.
571     *
572     * <p> Note that if this method enters the <code>abort</code> phase
573     * (either the <code>login</code> or <code>commit</code> phase failed),
574     * this method invokes all LoginModules configured for the
575     * application regardless of their respective <code>Configuration</code>
576     * flag parameters.  Essentially this means that <code>Requisite</code>
577     * and <code>Sufficient</code> semantics are ignored during the
578     * <code>abort</code> phase.  This guarantees that proper cleanup
579     * and state restoration can take place.
580     *
581     * <p>
582     *
583     * @exception LoginException if the authentication fails.
584     */
585    public void login() throws LoginException {
586
587        loginSucceeded = false;
588
589        if (subject == null) {
590            subject = new Subject();
591        }
592
593        try {
594            // module invoked in doPrivileged
595            invokePriv(LOGIN_METHOD);
596            invokePriv(COMMIT_METHOD);
597            loginSucceeded = true;
598        } catch (LoginException le) {
599            try {
600                invokePriv(ABORT_METHOD);
601            } catch (LoginException le2) {
602                throw le;
603            }
604            throw le;
605        }
606    }

608    /**
609     * Logout the <code>Subject</code>.
610     *
611     * <p> This method invokes the <code>logout</code> method for each
612     * <code>LoginModule</code> configured for this <code>LoginContext</code>.
613     * Each <code>LoginModule</code> performs its respective logout procedure
614     * which may include removing/destroying
615     * <code>Principal</code> and <code>Credential</code> information
616     * from the <code>Subject</code> and state cleanup.
617     *
618     * <p> Note that this method invokes all LoginModules configured for the
619     * application regardless of their respective
620     * <code>Configuration</code> flag parameters.  Essentially this means
621     * that <code>Requisite</code> and <code>Sufficient</code> semantics are
622     * ignored for this method.  This guarantees that proper cleanup
623     * and state restoration can take place.
624     *
625     * <p>
626     *
627     * @exception LoginException if the logout fails.
628     */
629    public void logout() throws LoginException {
630        if (subject == null) {
631            throw new LoginException(ResourcesMgr.getString
632                ("null.subject.logout.called.before.login"));
633        }
634
635        // module invoked in doPrivileged
636        invokePriv(LOGOUT_METHOD);
637    }

639    /**
640     * Return the authenticated Subject.
641     *
642     * <p>
643     *
644     * @return the authenticated Subject.  If the caller specified a
645     *          Subject to this LoginContext's constructor,
646     *          this method returns the caller-specified Subject.
647     *          If a Subject was not specified and authentication succeeds,
648     *          this method returns the Subject instantiated and used for
649     *          authentication by this LoginContext.
650     *          If a Subject was not specified, and authentication fails or
651     *          has not been attempted, this method returns null.
652     */
653    public Subject getSubject() {
654        if (!loginSucceeded && !subjectProvided)
655            return null;
656        return subject;
657    }

666    private void throwException(LoginException originalError, LoginException le)
667    throws LoginException {
668
669        // first clear state
670        clearState();
671
672        // throw the exception
673        LoginException error = (originalError != null) ? originalError : le;
674        throw error;

677    /**
678     * Invokes the login, commit, and logout methods
679     * from a LoginModule inside a doPrivileged block restricted
680     * by creatorAcc (may be null).
681     *
682     * This version is called if the caller did not instantiate
683     * the LoginContext with a Configuration object.
684     */
685    private void invokePriv(final String methodName) throws LoginException {
686        try {
687            java.security.AccessController.doPrivileged
688                (new java.security.PrivilegedExceptionAction<Void>() {
689                public Void run() throws LoginException {
690                    invoke(methodName);
691                    return null;
692                }
693            }, creatorAcc);
694        } catch (java.security.PrivilegedActionException pae) {
695            throw (LoginException)pae.getException();
696        }
697    }
698
699    private void invoke(String methodName) throws LoginException {
700
701        // start at moduleIndex
702        // - this can only be non-zero if methodName is LOGIN_METHOD
703
704        for (int i = moduleIndex; i < moduleStack.length; i++, moduleIndex++) {
705            try {
706
707                int mIndex = 0;
708                Method[] methods = null;
709
710                if (moduleStack[i].module != null) {
711                    methods = moduleStack[i].module.getClass().getMethods();
712                } else {
713
714                    // instantiate the LoginModule
715                    //
716                    // Allow any object to be a LoginModule as long as it
717                    // conforms to the interface.
718                    Class<?> c = Class.forName(
719                                moduleStack[i].entry.getLoginModuleName(),
720                                true,
721                                contextClassLoader);
722
723                    Constructor constructor = c.getConstructor(PARAMS);
724                    Object[] args = { };
725                    moduleStack[i].module = constructor.newInstance(args);
726
727                    // call the LoginModule's initialize method
728                    methods = moduleStack[i].module.getClass().getMethods();
729                    for (mIndex = 0; mIndex < methods.length; mIndex++) {
730                        if (methods[mIndex].getName().equals(INIT_METHOD)) {
731                            break;
732                        }
733                    }
734
735                    Object[] initArgs = {subject,
736                                        callbackHandler,
737                                        state,
738                                        moduleStack[i].entry.getOptions() };
739                    // invoke the LoginModule initialize method
740                    //
741                    // Throws ArrayIndexOutOfBoundsException if no such
742                    // method defined.  May improve to use LoginException in
743                    // the future.
744                    methods[mIndex].invoke(moduleStack[i].module, initArgs);
745                }
746
747                // find the requested method in the LoginModule
748                for (mIndex = 0; mIndex < methods.length; mIndex++) {
749                    if (methods[mIndex].getName().equals(methodName)) {
750                        break;
751                    }
752                }
753
754                // set up the arguments to be passed to the LoginModule method
755                Object[] args = { };
756
757                // invoke the LoginModule method
758                //
759                // Throws ArrayIndexOutOfBoundsException if no such
760                // method defined.  May improve to use LoginException in
761                // the future.
762                boolean status = ((Boolean)methods[mIndex].invoke
763                                (moduleStack[i].module, args)).booleanValue();
764
765                if (status == true) {
766
767                    // if SUFFICIENT, return if no prior REQUIRED errors
768                    if (!methodName.equals(ABORT_METHOD) &&
769                        !methodName.equals(LOGOUT_METHOD) &&
770                        moduleStack[i].entry.getControlFlag() ==
771                    AppConfigurationEntry.LoginModuleControlFlag.SUFFICIENT &&
772                        firstRequiredError == null) {
773
774                        // clear state
775                        clearState();
776
777                        if (debug != null)
778                            debug.println(methodName + " SUFFICIENT success");
779                        return;
780                    }
781
782                    if (debug != null)
783                        debug.println(methodName + " success");
784                    success = true;
785                } else {
786                    if (debug != null)
787                        debug.println(methodName + " ignored");
788                }
789
790            } catch (NoSuchMethodException nsme) {
791                MessageFormat form = new MessageFormat(ResourcesMgr.getString
792                        ("unable.to.instantiate.LoginModule.module.because.it.does.not.provide.a.no.argument.constructor"));
793                Object[] source = {moduleStack[i].entry.getLoginModuleName()};
794                throwException(null, new LoginException(form.format(source)));
795            } catch (InstantiationException ie) {
796                throwException(null, new LoginException(ResourcesMgr.getString
797                        ("unable.to.instantiate.LoginModule.") +
798                        ie.getMessage()));
799            } catch (ClassNotFoundException cnfe) {
800                throwException(null, new LoginException(ResourcesMgr.getString
801                        ("unable.to.find.LoginModule.class.") +
802                        cnfe.getMessage()));
803            } catch (IllegalAccessException iae) {
804                throwException(null, new LoginException(ResourcesMgr.getString
805                        ("unable.to.access.LoginModule.") +
806                        iae.getMessage()));
807            } catch (InvocationTargetException ite) {
808
809                // failure cases
810
811                LoginException le;
812
813                if (ite.getCause() instanceof PendingException &&
814                    methodName.equals(LOGIN_METHOD)) {
815
816                    // XXX
817                    //
818                    // if a module's LOGIN_METHOD threw a PendingException
819                    // then immediately throw it.
820                    //
821                    // when LoginContext is called again,
822                    // the module that threw the exception is invoked first
823                    // (the module list is not invoked from the start).
824                    // previously thrown exception state is still present.
825                    //
826                    // it is assumed that the module which threw
827                    // the exception can have its
828                    // LOGIN_METHOD invoked twice in a row
829                    // without any commit/abort in between.
830                    //
831                    // in all cases when LoginContext returns
832                    // (either via natural return or by throwing an exception)
833                    // we need to call clearState before returning.
834                    // the only time that is not true is in this case -
835                    // do not call throwException here.
836
837                    throw (PendingException)ite.getCause();
838
839                } else if (ite.getCause() instanceof LoginException) {
840
841                    le = (LoginException)ite.getCause();
842
843                } else if (ite.getCause() instanceof SecurityException) {
844
845                    // do not want privacy leak
846                    // (e.g., sensitive file path in exception msg)
847
848                    le = new LoginException("Security Exception");
849                    le.initCause(new SecurityException());
850                    if (debug != null) {
851                        debug.println
852                            ("original security exception with detail msg " +
853                            "replaced by new exception with empty detail msg");
854                        debug.println("original security exception: " +
855                                ite.getCause().toString());
856                    }
857                } else {
858
859                    // capture an unexpected LoginModule exception
860                    java.io.StringWriter sw = new java.io.StringWriter();
861                    ite.getCause().printStackTrace
862                                                (new java.io.PrintWriter(sw));
863                    sw.flush();
864                    le = new LoginException(sw.toString());
865                }
866
867                if (moduleStack[i].entry.getControlFlag() ==
868                    AppConfigurationEntry.LoginModuleControlFlag.REQUISITE) {
869
870                    if (debug != null)
871                        debug.println(methodName + " REQUISITE failure");
872
873                    // if REQUISITE, then immediately throw an exception
874                    if (methodName.equals(ABORT_METHOD) ||
875                        methodName.equals(LOGOUT_METHOD)) {
876                        if (firstRequiredError == null)
877                            firstRequiredError = le;
878                    } else {
879                        throwException(firstRequiredError, le);
880                    }
881
882                } else if (moduleStack[i].entry.getControlFlag() ==
883                    AppConfigurationEntry.LoginModuleControlFlag.REQUIRED) {
884
885                    if (debug != null)
886                        debug.println(methodName + " REQUIRED failure");
887
888                    // mark down that a REQUIRED module failed
889                    if (firstRequiredError == null)
890                        firstRequiredError = le;
891
892                } else {
893
894                    if (debug != null)
895                        debug.println(methodName + " OPTIONAL failure");
896
897                    // mark down that an OPTIONAL module failed
898                    if (firstError == null)
899                        firstError = le;
900                }
901            }
902        }
903
904        // we went thru all the LoginModules.
905        if (firstRequiredError != null) {
906            // a REQUIRED module failed -- return the error
907            throwException(firstRequiredError, null);
908        } else if (success == false && firstError != null) {
909            // no module succeeded -- return the first error
910            throwException(firstError, null);
911        } else if (success == false) {
912            // no module succeeded -- all modules were IGNORED
913            throwException(new LoginException
914                (ResourcesMgr.getString("Login.Failure.all.modules.ignored")),
915                null);
916        } else {
917            // success
918
919            clearState();
920            return;
921        }
922    }

924    /**
925     * Wrap the caller-specified CallbackHandler in our own
926     * and invoke it within a privileged block, constrained by
927     * the caller's AccessControlContext.
928     */
929    private static class SecureCallbackHandler implements CallbackHandler {
930
931        private final java.security.AccessControlContext acc;
932        private final CallbackHandler ch;
933
934        SecureCallbackHandler(java.security.AccessControlContext acc,
935                        CallbackHandler ch) {
936            this.acc = acc;
937            this.ch = ch;
938        }
939
940        public void handle(final Callback[] callbacks)
941                throws java.io.IOException, UnsupportedCallbackException {
942            try {
943                java.security.AccessController.doPrivileged
944                    (new java.security.PrivilegedExceptionAction<Void>() {
945                    public Void run() throws java.io.IOException,
946                                        UnsupportedCallbackException {
947                        ch.handle(callbacks);
948                        return null;
949                    }
950                }, acc);
951            } catch (java.security.PrivilegedActionException pae) {
952                if (pae.getException() instanceof java.io.IOException) {
953                    throw (java.io.IOException)pae.getException();
954                } else {
955                    throw (UnsupportedCallbackException)pae.getException();
956                }
957            }
958        }
959    }
