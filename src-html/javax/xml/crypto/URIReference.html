028package javax.xml.crypto;
029
030/**
031 * Identifies a data object via a URI-Reference, as specified by
032 * <a href="http://www.ietf.org/rfc/rfc2396.txt">RFC 2396</a>.
033 *
034 * <p>Note that some subclasses may not have a <code>type</code> attribute
035 * and for objects of those types, the {@link #getType} method always returns
036 * <code>null</code>.
037 *
038 * @author Sean Mullan
039 * @author JSR 105 Expert Group
040 * @since 1.6
041 * @see URIDereferencer
042 */
043public interface URIReference {
044
045    /**
046     * Returns the URI of the referenced data object.
047     *
048     * @return the URI of the data object in RFC 2396 format (may be
049     *    <code>null</code> if not specified)
050     */
051    String getURI();
052
053    /**
054     * Returns the type of data referenced by this URI.
055     *
056     * @return the type (a URI) of the data object (may be <code>null</code>
057     *    if not specified)
058     */
059    String getType();
060}
