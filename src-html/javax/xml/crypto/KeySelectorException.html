028package javax.xml.crypto;
029
030import java.io.PrintStream;
031import java.io.PrintWriter;
032
033/**
034 * Indicates an exceptional condition thrown by a {@link KeySelector}.
035 *
036 * <p>A <code>KeySelectorException</code> can contain a cause: another
037 * throwable that caused this <code>KeySelectorException</code> to get thrown.
038 *
039 * @author Sean Mullan
040 * @author JSR 105 Expert Group
041 * @since 1.6
042 */
043public class KeySelectorException extends Exception {
044
045    private static final long serialVersionUID = -7480033639322531109L;
046
047    /**
048     * The throwable that caused this exception to get thrown, or
049     * <code>null</code> if this exception was not caused by another throwable
050     * or if the causative throwable is unknown.
051     *
052     * @serial
053     */
054    private Throwable cause;
055
056    /**
057     * Constructs a new <code>KeySelectorException</code> with
058     * <code>null</code> as its detail message.
059     */
060    public KeySelectorException() {
061        super();
062    }
063
064    /**
065     * Constructs a new <code>KeySelectorException</code> with the specified
066     * detail message.
067     *
068     * @param message the detail message
069     */
070    public KeySelectorException(String message) {
071        super(message);
072    }
073
074    /**
075     * Constructs a new <code>KeySelectorException</code> with the
076     * specified detail message and cause.
077     * <p>Note that the detail message associated with
078     * <code>cause</code> is <i>not</i> automatically incorporated in
079     * this exception's detail message.
080     *
081     * @param message the detail message
082     * @param cause the cause (A <tt>null</tt> value is permitted, and
083     *        indicates that the cause is nonexistent or unknown.)
084     */
085    public KeySelectorException(String message, Throwable cause) {
086        super(message);
087        this.cause = cause;
088    }
089
090    /**
091     * Constructs a new <code>KeySelectorException</code> with the specified
092     * cause and a detail message of
093     * <code>(cause==null ? null : cause.toString())</code>
094     * (which typically contains the class and detail message of
095     * <code>cause</code>).
096     *
097     * @param cause the cause (A <tt>null</tt> value is permitted, and
098     *        indicates that the cause is nonexistent or unknown.)
099     */
100    public KeySelectorException(Throwable cause) {
101        super(cause==null ? null : cause.toString());
102        this.cause = cause;
103    }
104
105    /**
106     * Returns the cause of this <code>KeySelectorException</code> or
107     * <code>null</code> if the cause is nonexistent or unknown.  (The
108     * cause is the throwable that caused this
109     * <code>KeySelectorException</code> to get thrown.)
110     *
111     * @return the cause of this <code>KeySelectorException</code> or
112     *         <code>null</code> if the cause is nonexistent or unknown.
113     */
114    public Throwable getCause() {
115        return cause;
116    }
117
118    /**
119     * Prints this <code>KeySelectorException</code>, its backtrace and
120     * the cause's backtrace to the standard error stream.
121     */
122    public void printStackTrace() {
123        super.printStackTrace();
124        //XXX print backtrace of cause
125    }
126
127    /**
128     * Prints this <code>KeySelectorException</code>, its backtrace and
129     * the cause's backtrace to the specified print stream.
130     *
131     * @param s <code>PrintStream</code> to use for output
132     */
133    public void printStackTrace(PrintStream s) {
134        super.printStackTrace(s);
135        //XXX print backtrace of cause
136    }
137
138    /**
139     * Prints this <code>KeySelectorException</code>, its backtrace and
140     * the cause's backtrace to the specified print writer.
141     *
142     * @param s <code>PrintWriter</code> to use for output
143     */
144    public void printStackTrace(PrintWriter s) {
145        super.printStackTrace(s);
146        //XXX print backtrace of cause
147    }
148}
