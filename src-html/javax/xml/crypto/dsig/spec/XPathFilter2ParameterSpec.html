028package javax.xml.crypto.dsig.spec;
029
030import java.util.ArrayList;
031import java.util.Collections;
032import java.util.List;
033import javax.xml.crypto.dsig.Transform;
034
035/**
036 * Parameters for the W3C Recommendation
037 * <a href="http://www.w3.org/TR/xmldsig-filter2/">
038 * XPath Filter 2.0 Transform Algorithm</a>.
039 * The parameters include a list of one or more {@link XPathType} objects.
040 *
041 * @author Sean Mullan
042 * @author JSR 105 Expert Group
043 * @since 1.6
044 * @see Transform
045 * @see XPathFilterParameterSpec
046 */
047public final class XPathFilter2ParameterSpec implements TransformParameterSpec {
048
049    private final List xPathList;
050
051    /**
052     * Creates an <code>XPathFilter2ParameterSpec</code>.
053     *
054     * @param xPathList a list of one or more {@link XPathType} objects. The
055     *    list is defensively copied to protect against subsequent modification.
056     * @throws ClassCastException if <code>xPathList</code> contains any
057     *    entries that are not of type {@link XPathType}
058     * @throws IllegalArgumentException if <code>xPathList</code> is empty
059     * @throws NullPointerException if <code>xPathList</code> is
060     *    <code>null</code>
061     */
062    public XPathFilter2ParameterSpec(List xPathList) {
063        if (xPathList == null) {
064            throw new NullPointerException("xPathList cannot be null");
065        }
066        List xPathListCopy = new ArrayList(xPathList);
067        if (xPathListCopy.isEmpty()) {
068            throw new IllegalArgumentException("xPathList cannot be empty");
069        }
070        int size = xPathListCopy.size();
071        for (int i = 0; i < size; i++) {
072            if (!(xPathListCopy.get(i) instanceof XPathType)) {
073                throw new ClassCastException
074                    ("xPathList["+i+"] is not a valid type");
075            }
076        }
077        this.xPathList = Collections.unmodifiableList(xPathListCopy);
078    }
079
080    /**
081     * Returns a list of one or more {@link XPathType} objects.
082     * <p>
083     * This implementation returns an {@link Collections#unmodifiableList
084     * unmodifiable list}.
085     *
086     * @return a <code>List</code> of <code>XPathType</code> objects
087     *    (never <code>null</code> or empty)
088     */
089    public List getXPathList() {
090        return xPathList;
091    }
092}
