028package javax.xml.crypto.dsig.spec;
029
030import javax.xml.crypto.dsig.Transform;
031import javax.xml.crypto.XMLStructure;
032
033/**
034 * Parameters for the <a href="http://www.w3.org/TR/1999/REC-xslt-19991116">
035 * XSLT Transform Algorithm</a>.
036 * The parameters include a namespace-qualified stylesheet element.
037 *
038 * <p>An <code>XSLTTransformParameterSpec</code> is instantiated with a
039 * mechanism-dependent (ex: DOM) stylesheet element. For example:
040 * <pre>
041 *   DOMStructure stylesheet = new DOMStructure(element)
042 *   XSLTTransformParameterSpec spec = new XSLTransformParameterSpec(stylesheet);
043 * </pre>
044 * where <code>element</code> is an {@link org.w3c.dom.Element} containing
045 * the namespace-qualified stylesheet element.
046 *
047 * @author Sean Mullan
048 * @author JSR 105 Expert Group
049 * @since 1.6
050 * @see Transform
051 */
052public final class XSLTTransformParameterSpec implements TransformParameterSpec{
053    private XMLStructure stylesheet;
054
055    /**
056     * Creates an <code>XSLTTransformParameterSpec</code> with the specified
057     * stylesheet.
058     *
059     * @param stylesheet the XSLT stylesheet to be used
060     * @throws NullPointerException if <code>stylesheet</code> is
061     *    <code>null</code>
062     */
063    public XSLTTransformParameterSpec(XMLStructure stylesheet) {
064        if (stylesheet == null) {
065            throw new NullPointerException();
066        }
067        this.stylesheet = stylesheet;
068    }
069
070    /**
071     * Returns the stylesheet.
072     *
073     * @return the stylesheet
074     */
075    public XMLStructure getStylesheet() {
076        return stylesheet;
077    }
078}
