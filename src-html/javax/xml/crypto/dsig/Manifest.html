028package javax.xml.crypto.dsig;
029
030import javax.xml.crypto.XMLStructure;
031import java.util.List;
032
033/**
034 * A representation of the XML <code>Manifest</code> element as defined in
035 * the <a href="http://www.w3.org/TR/xmldsig-core/">
036 * W3C Recommendation for XML-Signature Syntax and Processing</a>.
037 * The XML Schema Definition is defined as:
038 * <pre><code>
039 * &lt;element name="Manifest" type="ds:ManifestType"/&gt;
040 *   &lt;complexType name="ManifestType"&gt;
041 *     &lt;sequence>
042 *       &lt;element ref="ds:Reference" maxOccurs="unbounded"/&gt;
043 *     &lt;/sequence&gt;
044 *     &lt;attribute name="Id" type="ID" use="optional"/&gt;
045 *   &lt;/complexType&gt;
046 * </code></pre>
047 *
048 * A <code>Manifest</code> instance may be created by invoking
049 * one of the {@link XMLSignatureFactory#newManifest newManifest}
050 * methods of the {@link XMLSignatureFactory} class; for example:
051 *
052 * <pre>
053 *   XMLSignatureFactory factory = XMLSignatureFactory.getInstance("DOM");
054 *   List references = Collections.singletonList(factory.newReference
055 *       ("#reference-1", DigestMethod.SHA1));
056 *   Manifest manifest = factory.newManifest(references, "manifest-1");
057 * </pre>
058 *
059 * @author Sean Mullan
060 * @author JSR 105 Expert Group
061 * @since 1.6
062 * @see XMLSignatureFactory#newManifest(List)
063 * @see XMLSignatureFactory#newManifest(List, String)
064 */
065public interface Manifest extends XMLStructure {
066
067    /**
068     * URI that identifies the <code>Manifest</code> element (this can be
069     * specified as the value of the <code>type</code> parameter of the
070     * {@link Reference} class to identify the referent's type).
071     */
072    final static String TYPE = "http://www.w3.org/2000/09/xmldsig#Manifest";
073
074    /**
075     * Returns the Id of this <code>Manifest</code>.
076     *
077     * @return the Id  of this <code>Manifest</code> (or <code>null</code>
078     *    if not specified)
079     */
080    String getId();
081
082    /**
083     * Returns an {@link java.util.Collections#unmodifiableList unmodifiable
084     * list} of one or more {@link Reference}s that are contained in this
085     * <code>Manifest</code>.
086     *
087     * @return an unmodifiable list of one or more <code>Reference</code>s
088     */
089    List getReferences();
090}
