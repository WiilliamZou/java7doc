028package javax.xml.crypto.dsig;
029
030import java.io.PrintStream;
031import java.io.PrintWriter;
032
033/**
034 * Indicates an exceptional condition that occured during the XML
035 * signature generation or validation process.
036 *
037 * <p>An <code>XMLSignatureException</code> can contain a cause: another
038 * throwable that caused this <code>XMLSignatureException</code> to get thrown.
039 *
040 * @since 1.6
041 */
042public class XMLSignatureException extends Exception {
043
044    private static final long serialVersionUID = -3438102491013869995L;
045
046    /**
047     * The throwable that caused this exception to get thrown, or null if this
048     * exception was not caused by another throwable or if the causative
049     * throwable is unknown.
050     *
051     * @serial
052     */
053    private Throwable cause;
054
055    /**
056     * Constructs a new <code>XMLSignatureException</code> with
057     * <code>null</code> as its detail message.
058     */
059    public XMLSignatureException() {
060        super();
061    }
062
063    /**
064     * Constructs a new <code>XMLSignatureException</code> with the specified
065     * detail message.
066     *
067     * @param message the detail message
068     */
069    public XMLSignatureException(String message) {
070        super(message);
071    }
072
073    /**
074     * Constructs a new <code>XMLSignatureException</code> with the
075     * specified detail message and cause.
076     * <p>Note that the detail message associated with
077     * <code>cause</code> is <i>not</i> automatically incorporated in
078     * this exception's detail message.
079     *
080     * @param message the detail message
081     * @param cause the cause (A <tt>null</tt> value is permitted, and
082     *        indicates that the cause is nonexistent or unknown.)
083     */
084    public XMLSignatureException(String message, Throwable cause) {
085        super(message);
086        this.cause = cause;
087    }
088
089    /**
090     * Constructs a new <code>XMLSignatureException</code> with the specified
091     * cause and a detail message of
092     * <code>(cause==null ? null : cause.toString())</code>
093     * (which typically contains the class and detail message of
094     * <code>cause</code>).
095     *
096     * @param cause the cause (A <tt>null</tt> value is permitted, and
097     *        indicates that the cause is nonexistent or unknown.)
098     */
099    public XMLSignatureException(Throwable cause) {
100        super(cause==null ? null : cause.toString());
101        this.cause = cause;
102    }
103
104    /**
105     * Returns the cause of this <code>XMLSignatureException</code> or
106     * <code>null</code> if the cause is nonexistent or unknown.  (The
107     * cause is the throwable that caused this
108     * <code>XMLSignatureException</code> to get thrown.)
109     *
110     * @return the cause of this <code>XMLSignatureException</code> or
111     *         <code>null</code> if the cause is nonexistent or unknown.
112     */
113    public Throwable getCause() {
114        return cause;
115    }
116
117    /**
118     * Prints this <code>XMLSignatureException</code>, its backtrace and
119     * the cause's backtrace to the standard error stream.
120     */
121    public void printStackTrace() {
122        super.printStackTrace();
123        if (cause != null) {
124            cause.printStackTrace();
125        }
126    }
127
128    /**
129     * Prints this <code>XMLSignatureException</code>, its backtrace and
130     * the cause's backtrace to the specified print stream.
131     *
132     * @param s <code>PrintStream</code> to use for output
133     */
134    public void printStackTrace(PrintStream s) {
135        super.printStackTrace(s);
136        if (cause != null) {
137            cause.printStackTrace(s);
138        }
139    }
140
141    /**
142     * Prints this <code>XMLSignatureException</code>, its backtrace and
143     * the cause's backtrace to the specified print writer.
144     *
145     * @param s <code>PrintWriter</code> to use for output
146     */
147    public void printStackTrace(PrintWriter s) {
148        super.printStackTrace(s);
149        if (cause != null) {
150            cause.printStackTrace(s);
151        }
152    }
153}
