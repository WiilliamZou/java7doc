026package javax.xml.transform;
027
028/**
029 * This interface is primarily for the purposes of reporting where
030 * an error occurred in the XML source or transformation instructions.
031 */
032public interface SourceLocator {
033
034    /**
035     * Return the public identifier for the current document event.
036     *
037     * <p>The return value is the public identifier of the document
038     * entity or of the external parsed entity in which the markup that
039     * triggered the event appears.</p>
040     *
041     * @return A string containing the public identifier, or
042     *         null if none is available.
043     * @see #getSystemId
044     */
045    public String getPublicId();
046
047    /**
048     * Return the system identifier for the current document event.
049     *
050     * <p>The return value is the system identifier of the document
051     * entity or of the external parsed entity in which the markup that
052     * triggered the event appears.</p>
053     *
054     * <p>If the system identifier is a URL, the parser must resolve it
055     * fully before passing it to the application.</p>
056     *
057     * @return A string containing the system identifier, or null
058     *         if none is available.
059     * @see #getPublicId
060     */
061    public String getSystemId();
062
063    /**
064     * Return the line number where the current document event ends.
065     *
066     * <p><strong>Warning:</strong> The return value from the method
067     * is intended only as an approximation for the sake of error
068     * reporting; it is not intended to provide sufficient information
069     * to edit the character content of the original XML document.</p>
070     *
071     * <p>The return value is an approximation of the line number
072     * in the document entity or external parsed entity where the
073     * markup that triggered the event appears.</p>
074     *
075     * @return The line number, or -1 if none is available.
076     * @see #getColumnNumber
077     */
078    public int getLineNumber();
079
080    /**
081     * Return the character position where the current document event ends.
082     *
083     * <p><strong>Warning:</strong> The return value from the method
084     * is intended only as an approximation for the sake of error
085     * reporting; it is not intended to provide sufficient information
086     * to edit the character content of the original XML document.</p>
087     *
088     * <p>The return value is an approximation of the column number
089     * in the document entity or external parsed entity where the
090     * markup that triggered the event appears.</p>
091     *
092     * @return The column number, or -1 if none is available.
093     * @see #getLineNumber
094     */
095    public int getColumnNumber();
096}
