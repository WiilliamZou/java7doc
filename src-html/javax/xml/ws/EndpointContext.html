026package javax.xml.ws;
027
028import javax.xml.ws.Endpoint;
029import java.util.Set;
030
031/**
032 * <code>EndpointContext</code> allows multiple endpoints in an application
033 * to share any information. For example, servlet application's war may
034 * contain multiple endpoints and these endpoints can get addresses of each
035 * other by sharing this context. If multiple endpoints share different
036 * ports of a WSDL, then the multiple port addresses can be patched
037 * when the WSDL is accessed. It also allows all endpoints to share any
038 * other runtime information.
039 *
040 * <p>
041 * This needs to be set by using {@link Endpoint#setEndpointContext}
042 * before {@link Endpoint#publish} methods.
043 *
044 * @author Jitendra Kotamraju
045 * @since JAX-WS 2.2
046 */
047public abstract class EndpointContext {
048
049    /**
050     * This gives list of endpoints in an application. For e.g in
051     * servlet container, a war file may contain multiple endpoints.
052     * In case of http, these endpoints are hosted on the same http
053     * server.
054     *
055     * @return list of all endpoints in an application
056     */
057    public abstract Set<Endpoint> getEndpoints();
058
059}
