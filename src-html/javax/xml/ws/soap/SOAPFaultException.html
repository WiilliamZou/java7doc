026package javax.xml.ws.soap;
027
028import javax.xml.soap.SOAPFault;
029
030/** The <code>SOAPFaultException</code> exception represents a
031 *  SOAP 1.1 or 1.2 fault.
032 *
033 *  <p>A <code>SOAPFaultException</code> wraps a SAAJ <code>SOAPFault</code>
034 *  that manages the SOAP-specific representation of faults.
035 *  The <code>createFault</code> method of
036 *  <code>javax.xml.soap.SOAPFactory</code> may be used to create an instance
037 *  of <code>javax.xml.soap.SOAPFault</code> for use with the
038 *  constructor. <code>SOAPBinding</code> contains an accessor for the
039 *  <code>SOAPFactory</code> used by the binding instance.
040 *
041 *  <p>Note that the value of <code>getFault</code> is the only part of the
042 *  exception used when searializing a SOAP fault.
043 *
044 *  <p>Refer to the SOAP specification for a complete
045 *  description of SOAP faults.
046 *
047 *  @see javax.xml.soap.SOAPFault
048 *  @see javax.xml.ws.soap.SOAPBinding#getSOAPFactory
049 *  @see javax.xml.ws.ProtocolException
050 *
051 *  @since JAX-WS 2.0
052 **/
053public class SOAPFaultException extends javax.xml.ws.ProtocolException  {
054
055    private SOAPFault fault;
056
057    /** Constructor for SOAPFaultException
058     *  @param fault   <code>SOAPFault</code> representing the fault
059     *
060     *  @see javax.xml.soap.SOAPFactory#createFault
061     **/
062    public SOAPFaultException(SOAPFault fault) {
063        super(fault.getFaultString());
064        this.fault = fault;
065    }
066
067    /** Gets the embedded <code>SOAPFault</code> instance.
068     *
069     *  @return <code>javax.xml.soap.SOAPFault</code> SOAP
070     *          fault element
071     **/
072    public javax.xml.soap.SOAPFault getFault() {
073        return this.fault;
074    }
075}
