026package javax.xml.ws;
027
028import java.lang.annotation.Documented;
029import java.lang.annotation.Target;
030import java.lang.annotation.Retention;
031import java.lang.annotation.ElementType;
032import java.lang.annotation.RetentionPolicy;
033
034/**
035 *  Used to annotate the <code>get<em>PortName</em>()</code>
036 *  methods of a generated service interface.
037 *
038 *  <p>The information specified in this annotation is sufficient
039 *  to uniquely identify a <code>wsdl:port</code> element
040 *  inside a <code>wsdl:service</code>. The latter is
041 *  determined based on the value of the <code>WebServiceClient</code>
042 *  annotation on the generated service interface itself.
043 *
044 *  @since JAX-WS 2.0
045 *
046 *  @see javax.xml.ws.WebServiceClient
047**/
048@Target({ElementType.METHOD})
049@Retention(RetentionPolicy.RUNTIME)
050@Documented
051public @interface WebEndpoint {
052  /**
053   *  The local name of the endpoint.
054  **/
055  String name() default "";
056}
