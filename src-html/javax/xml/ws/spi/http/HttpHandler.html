026package javax.xml.ws.spi.http;
027
028import javax.xml.ws.Endpoint;
029import java.io.IOException;
030
031/**
032 * A handler which is invoked to process HTTP requests.
033 * <p>
034 * JAX-WS runtime provides the implementation for this and sets
035 * it using {@link HttpContext#setHandler(HttpHandler)} during
036 * {@link Endpoint#publish(HttpContext) }
037 *
038 * @author Jitendra Kotamraju
039 * @since JAX-WS 2.2
040 */
041public abstract class HttpHandler {
042    /**
043     * Handles a given request and generates an appropriate response.
044     * See {@link HttpExchange} for a description of the steps
045     * involved in handling an exchange. Container invokes this method
046     * when it receives an incoming request.
047     *
048     * @param exchange the exchange containing the request from the
049     *      client and used to send the response
050     * @throws IOException when an I/O error happens during request
051     *      handling
052     */
053    public abstract void handle(HttpExchange exchange) throws IOException;
054}
