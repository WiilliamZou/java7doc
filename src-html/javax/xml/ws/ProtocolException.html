026package javax.xml.ws;
027
028/** The <code>ProtocolException</code> class is a
029 *  base class for exceptions related to a specific protocol binding. Subclasses
030 *  are used to communicate protocol level fault information to clients and may
031 *  be used on the server to control the protocol specific fault representation.
032 *
033 *  @since JAX-WS 2.0
034**/
035public class ProtocolException extends WebServiceException {
036    /**
037     * Constructs a new protocol exception with <code>null</code> as its detail message. The
038     * cause is not initialized, and may subsequently be initialized by a call
039     * to <code>Throwable.initCause(java.lang.Throwable)</code>.
040     */
041    public ProtocolException() {
042        super();
043    }
044
045    /**
046     * Constructs a new protocol exception with the specified detail message.
047     * The cause is not initialized, and may subsequently be initialized by a
048     * call to <code>Throwable.initCause(java.lang.Throwable)</code>.
049     *
050     * @param message the detail message. The detail message is saved for later
051     *   retrieval by the Throwable.getMessage() method.
052     */
053    public ProtocolException(String message) {
054        super(message);
055    }
056
057    /**
058     * Constructs a new runtime exception with the specified detail message and
059     * cause.
060     *
061     * Note that the detail message associated with  cause is not automatically
062     * incorporated in  this runtime exception's detail message.
063     *
064     * @param message the detail message (which is saved for later retrieval  by
065     *   the Throwable.getMessage() method).
066     * @param cause the cause (which is saved for later retrieval by the
067     * <code>Throwable.getCause()</code> method). (A <code>null</code> value is  permitted, and indicates
068     * that the cause is nonexistent or  unknown.)
069     */
070    public ProtocolException(String message,  Throwable cause) {
071        super(message, cause);
072    }
073
074    /**
075     * Constructs a new runtime exception with the specified cause and a  detail
076     * message of <code>(cause==null ? null : cause.toString())</code>  (which typically
077     * contains the class and detail message of  cause). This constructor is
078     * useful for runtime exceptions  that are little more than wrappers for
079     * other throwables.
080     *
081     * @param cause the cause (which is saved for later retrieval by the
082     * <code>Throwable.getCause()</code> method). (A <code>null</code> value is  permitted, and indicates
083     * that the cause is nonexistent or  unknown.)
084     */
085    public ProtocolException(Throwable cause) {
086        super(cause);
087    }
088}
