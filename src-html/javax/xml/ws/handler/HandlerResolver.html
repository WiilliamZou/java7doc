026package javax.xml.ws.handler;
027
028/**
029 *  <code>HandlerResolver</code> is an interface implemented
030 *  by an application to get control over the handler chain
031 *  set on proxy/dispatch objects at the time of their creation.
032 *  <p>
033 *  A <code>HandlerResolver</code> may be set on a <code>Service</code>
034 *  using the <code>setHandlerResolver</code> method.
035 * <p>
036 *  When the runtime invokes a <code>HandlerResolver</code>, it will
037 *  pass it a <code>PortInfo</code> object containing information
038 *  about the port that the proxy/dispatch object will be accessing.
039 *
040 *  @see javax.xml.ws.Service#setHandlerResolver
041 *
042 *  @since JAX-WS 2.0
043**/
044public interface HandlerResolver {
045
046  /**
047   *  Gets the handler chain for the specified port.
048   *
049   *  @param portInfo Contains information about the port being accessed.
050   *  @return java.util.List&lt;Handler> chain
051  **/
052  public java.util.List<Handler> getHandlerChain(PortInfo portInfo);
053}
