026package javax.xml.bind.helpers;
027
028import javax.xml.bind.PrintConversionEvent;
029import javax.xml.bind.ValidationEventLocator;
030
031/**
032 * Default implementation of the PrintConversionEvent interface.
033 *
034 * <p>
035 * JAXB providers are allowed to use whatever class that implements
036 * the ValidationEvent interface. This class is just provided for a
037 * convenience.
038 *
039 * @author <ul><li>Ryan Shoemaker, Sun Microsystems, Inc.</li></ul>
040 * @see javax.xml.bind.PrintConversionEvent
041 * @see javax.xml.bind.Validator
042 * @see javax.xml.bind.ValidationEventHandler
043 * @see javax.xml.bind.ValidationEvent
044 * @see javax.xml.bind.ValidationEventLocator
045 * @since JAXB1.0
046 */
047public class PrintConversionEventImpl
048    extends ValidationEventImpl
049    implements PrintConversionEvent {
050
051    /**
052     * Create a new PrintConversionEventImpl.
053     *
054     * @param _severity The severity value for this event.  Must be one of
055     * ValidationEvent.WARNING, ValidationEvent.ERROR, or
056     * ValidationEvent.FATAL_ERROR
057     * @param _message The text message for this event - may be null.
058     * @param _locator The locator object for this event - may be null.
059     * @throws IllegalArgumentException if an illegal severity field is supplied
060     */
061    public PrintConversionEventImpl( int _severity, String _message,
062                                      ValidationEventLocator _locator) {
063
064        super(_severity, _message, _locator);
065    }
066
067    /**
068     * Create a new PrintConversionEventImpl.
069     *
070     * @param _severity The severity value for this event.  Must be one of
071     * ValidationEvent.WARNING, ValidationEvent.ERROR, or
072     * ValidationEvent.FATAL_ERROR
073     * @param _message The text message for this event - may be null.
074     * @param _locator The locator object for this event - may be null.
075     * @param _linkedException An optional linked exception that may provide
076     * additional information about the event - may be null.
077     * @throws IllegalArgumentException if an illegal severity field is supplied
078     */
079    public PrintConversionEventImpl( int _severity, String _message,
080                                      ValidationEventLocator _locator,
081                                      Throwable _linkedException) {
082
083        super(_severity, _message, _locator, _linkedException);
084    }
085
086}
