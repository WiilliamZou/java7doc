026package javax.xml.bind;
027import  javax.xml.namespace.QName;
028
029/**
030 * Provide access to JAXB xml binding data for a JAXB object.
031 *
032 * <p>
033 * Intially, the intent of this class is to just conceptualize how
034 * a JAXB application developer can access xml binding information,
035 * independent if binding model is java to schema or schema to java.
036 * Since accessing the XML element name related to a JAXB element is
037 * a highly requested feature, demonstrate access to this
038 * binding information.
039 *
040 * The factory method to get a <code>JAXBIntrospector</code> instance is
041 * {@link JAXBContext#createJAXBIntrospector()}.
042 *
043 * @see JAXBContext#createJAXBIntrospector()
044 * @since JAXB2.0
045 */
046public abstract class JAXBIntrospector {
047
048    /**
049     * <p>Return true iff <code>object</code> represents a JAXB element.</p>
050     * <p>Parameter <code>object</code> is a JAXB element for following cases:
051     * <ol>
052     *   <li>It is an instance of <code>javax.xml.bind.JAXBElement</code>.</li>
053     *   <li>The class of <code>object</code> is annotated with
054     *       <code>&#64XmlRootElement</code>.
055     *   </li>
056     * </ol>
057     *
058     * @see #getElementName(Object)
059     */
060    public abstract boolean isElement(Object object);
061
062    /**
063     * <p>Get xml element qname for <code>jaxbElement</code>.</p>
064     *
065     * @param jaxbElement is an object that {@link #isElement(Object)} returned true.
066     *
067     * @return xml element qname associated with jaxbElement;
068     *         null if <code>jaxbElement</code> is not a JAXB Element.
069     */
070    public abstract QName getElementName(Object jaxbElement);
071
072    /**
073     * <p>Get the element value of a JAXB element.</p>
074     *
075     * <p>Convenience method to abstract whether working with either
076     *    a javax.xml.bind.JAXBElement instance or an instance of
077     *    <tt>&#64XmlRootElement</tt> annotated Java class.</p>
078     *
079     * @param jaxbElement  object that #isElement(Object) returns true.
080     *
081     * @return The element value of the <code>jaxbElement</code>.
082     */
083    public static Object getValue(Object jaxbElement) {
084        if (jaxbElement instanceof JAXBElement) {
085            return ((JAXBElement)jaxbElement).getValue();
086        } else {
087            // assume that class of this instance is
088            // annotated with @XmlRootElement.
089            return jaxbElement;
090        }
091    }
092}
