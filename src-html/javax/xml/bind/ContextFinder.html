062        try {

064                // disconnect the logger from a bigger framework (if any)
065                // and take the matters into our own hands
066                logger.setUseParentHandlers(false);

068                ConsoleHandler handler = new ConsoleHandler();
069                handler.setLevel(Level.ALL);

074                // have done on configurations.
075            }
076        } catch(Throwable t) {
077            // just to be extra safe. in particular System.getProperty may throw
078            // SecurityException.
079        }

082    /**
083     * If the {@link InvocationTargetException} wraps an exception that shouldn't be wrapped,
084     * throw the wrapped exception.
085     */
086    private static void handleInvocationTargetException(InvocationTargetException x) throws JAXBException {
087        Throwable t = x.getTargetException();
088        if( t != null ) {
089            if( t instanceof JAXBException )
090                // one of our exceptions, just re-throw
091                throw (JAXBException)t;
092            if( t instanceof RuntimeException )
093                // avoid wrapping exceptions unnecessarily
094                throw (RuntimeException)t;
095            if( t instanceof Error )
096                throw (Error)t;
097        }
098    }

101    /**
102     * Determine if two types (JAXBContext in this case) will generate a ClassCastException.
103     *
104     * For example, (targetType)originalType
105     *
106     * @param originalType
107     *          The Class object of the type being cast
108     * @param targetType
109     *          The Class object of the type that is being cast to
110     * @return JAXBException to be thrown.
111     */
112    private static JAXBException handleClassCastException(Class originalType, Class targetType) {
113        final URL targetTypeURL = which(targetType);
114
115        return new JAXBException(Messages.format(Messages.ILLEGAL_CAST,
116                // we don't care where the impl class is, we want to know where JAXBContext lives in the impl
117                // class' ClassLoader
118                originalType.getClassLoader().getResource("javax/xml/bind/JAXBContext.class"),
119                targetTypeURL));
120    }

122    /**
123     * Create an instance of a class using the specified ClassLoader
124     */
125    static JAXBContext newInstance( String contextPath,
126                               String className,
127                               ClassLoader classLoader,
128                               Map properties )
129        throws JAXBException {
130        try {
131            Class spFactory = safeLoadClass(className,classLoader);
132            return newInstance(contextPath, spFactory, classLoader, properties);
133        } catch (ClassNotFoundException x) {
134            throw new JAXBException(
135                Messages.format( Messages.PROVIDER_NOT_FOUND, className ),
136                x);
137        } catch (RuntimeException x) {
138            // avoid wrapping RuntimeException to JAXBException,
139            // because it indicates a bug in this code.
140            throw x;
141        } catch (Exception x) {
142            // can't catch JAXBException because the method is hidden behind
143            // reflection.  Root element collisions detected in the call to
144            // createContext() are reported as JAXBExceptions - just re-throw it
145            // some other type of exception - just wrap it
146            throw new JAXBException(
147                Messages.format( Messages.COULD_NOT_INSTANTIATE, className, x ),
148                x);
149        }
150    }

152    static JAXBContext newInstance( String contextPath,
153                               Class spFactory,
154                               ClassLoader classLoader,
155                               Map properties )
156        throws JAXBException
157    {
158        try {
159            /*
160             * javax.xml.bind.context.factory points to a class which has a
161             * static method called 'createContext' that
162             * returns a javax.xml.JAXBContext.
163             */
164
165            Object context = null;
166
167            // first check the method that takes Map as the third parameter.
168            // this is added in 2.0.
169            try {
170                Method m = spFactory.getMethod("createContext",String.class,ClassLoader.class,Map.class);
171                // any failure in invoking this method would be considered fatal
172                context = m.invoke(null,contextPath,classLoader,properties);
173            } catch (NoSuchMethodException e) {
174                // it's not an error for the provider not to have this method.
175            }
176
177            if(context==null) {
178                // try the old method that doesn't take properties. compatible with 1.0.
179                // it is an error for an implementation not to have both forms of the createContext method.
180                Method m = spFactory.getMethod("createContext",String.class,ClassLoader.class);
181                // any failure in invoking this method would be considered fatal
182                context = m.invoke(null,contextPath,classLoader);
183            }
184
185            if(!(context instanceof JAXBContext)) {
186                // the cast would fail, so generate an exception with a nice message
187                throw handleClassCastException(context.getClass(), JAXBContext.class);
188            }
189            return (JAXBContext)context;
190        } catch (InvocationTargetException x) {
191            handleInvocationTargetException(x);
192            // for other exceptions, wrap the internal target exception
193            // with a JAXBException
194            Throwable e = x;
195            if(x.getTargetException()!=null)
196                e = x.getTargetException();
197
198            throw new JAXBException( Messages.format( Messages.COULD_NOT_INSTANTIATE, spFactory, e ), e );
199        } catch (RuntimeException x) {
200            // avoid wrapping RuntimeException to JAXBException,
201            // because it indicates a bug in this code.
202            throw x;
203        } catch (Exception x) {
204            // can't catch JAXBException because the method is hidden behind
205            // reflection.  Root element collisions detected in the call to
206            // createContext() are reported as JAXBExceptions - just re-throw it
207            // some other type of exception - just wrap it
208            throw new JAXBException(
209                Messages.format( Messages.COULD_NOT_INSTANTIATE, spFactory, x ),
210                x);
211        }
212    }

215    /**
216     * Create an instance of a class using the thread context ClassLoader
217     */
218    static JAXBContext newInstance(
219                              Class[] classes,
220                              Map properties,
221                              String className) throws JAXBException {
222        ClassLoader cl = Thread.currentThread().getContextClassLoader();
223        Class spi;
224        try {
225            spi = safeLoadClass(className,cl);
226        } catch (ClassNotFoundException e) {
227            throw new JAXBException(e);
228        }
229
230        if(logger.isLoggable(Level.FINE)) {
231            // extra check to avoid costly which operation if not logged
232            logger.log(Level.FINE, "loaded {0} from {1}", new Object[]{className, which(spi)});
233        }
234
235        return newInstance(classes, properties, spi);
236    }

238    static JAXBContext newInstance(Class[] classes,
239                                   Map properties,
240                                   Class spFactory) throws JAXBException {
241        Method m;
242        try {
243            m = spFactory.getMethod("createContext", Class[].class, Map.class);
244        } catch (NoSuchMethodException e) {
245            throw new JAXBException(e);
246        }
247        try {
248            Object context = m.invoke(null, classes, properties);
249            if(!(context instanceof JAXBContext)) {
250                // the cast would fail, so generate an exception with a nice message
251                throw handleClassCastException(context.getClass(), JAXBContext.class);
252            }
253            return (JAXBContext)context;
254        } catch (IllegalAccessException e) {
255            throw new JAXBException(e);
256        } catch (InvocationTargetException e) {
257            handleInvocationTargetException(e);
258
259            Throwable x = e;
260            if (e.getTargetException() != null)
261                x = e.getTargetException();
262
263            throw new JAXBException(x);
264        }
265    }

267    static JAXBContext find(String factoryId, String contextPath, ClassLoader classLoader, Map properties ) throws JAXBException {
268
269        // TODO: do we want/need another layer of searching in $java.home/lib/jaxb.properties like JAXP?
270
271        final String jaxbContextFQCN = JAXBContext.class.getName();
272
273        // search context path for jaxb.properties first
274        StringBuilder propFileName;
275        StringTokenizer packages = new StringTokenizer( contextPath, ":" );
276        String factoryClassName;
277
278        if(!packages.hasMoreTokens())
279            // no context is specified
280            throw new JAXBException(Messages.format(Messages.NO_PACKAGE_IN_CONTEXTPATH));
281
282
283        logger.fine("Searching jaxb.properties");
284
285        while( packages.hasMoreTokens() ) {
286            String packageName = packages.nextToken(":").replace('.','/');
287            // com.acme.foo - > com/acme/foo/jaxb.properties
288             propFileName = new StringBuilder().append(packageName).append("/jaxb.properties");
289
290            Properties props = loadJAXBProperties( classLoader, propFileName.toString() );
291            if (props != null) {
292                if (props.containsKey(factoryId)) {
293                    factoryClassName = props.getProperty(factoryId);
294                    return newInstance( contextPath, factoryClassName, classLoader, properties );
295                } else {
296                    throw new JAXBException(Messages.format(Messages.MISSING_PROPERTY, packageName, factoryId));
297                }
298            }
299        }
300
301        logger.fine("Searching the system property");
302
303        // search for a system property second (javax.xml.bind.JAXBContext)
304        factoryClassName = AccessController.doPrivileged(new GetPropertyAction(JAXBContext.JAXB_CONTEXT_FACTORY));
305        if(  factoryClassName != null ) {
306            return newInstance( contextPath, factoryClassName, classLoader, properties );
307        } else { // leave this here to assure compatibility
308            factoryClassName = AccessController.doPrivileged(new GetPropertyAction(jaxbContextFQCN));
309            if(  factoryClassName != null ) {
310                return newInstance( contextPath, factoryClassName, classLoader, properties );
311            }
312        }
313
314        if (Thread.currentThread().getContextClassLoader() == classLoader) {
315            Class factory = lookupUsingOSGiServiceLoader("javax.xml.bind.JAXBContext");
316            if (factory != null) {
317                logger.fine("OSGi environment detected");
318                return newInstance(contextPath, factory, classLoader, properties);
319            }
320        }
321
322        logger.fine("Searching META-INF/services");
323        // search META-INF services next
324        BufferedReader r;
325        try {
326            final StringBuilder resource = new StringBuilder().append("META-INF/services/").append(jaxbContextFQCN);
327            final InputStream resourceStream =
328                    classLoader.getResourceAsStream(resource.toString());
329
330            if (resourceStream != null) {
331                r = new BufferedReader(new InputStreamReader(resourceStream, "UTF-8"));
332                factoryClassName = r.readLine().trim();
333                r.close();
334                return newInstance(contextPath, factoryClassName, classLoader, properties);
335            } else {
336                logger.log(Level.FINE, "Unable to load:{0}", resource.toString());
337            }
338        } catch (UnsupportedEncodingException e) {
339            // should never happen
340            throw new JAXBException(e);
341        } catch (IOException e) {
342            throw new JAXBException(e);
343        }
344
345        // else no provider found
346        logger.fine("Trying to create the platform default provider");
347        return newInstance(contextPath, PLATFORM_DEFAULT_FACTORY_CLASS, classLoader, properties);
348    }

350    static JAXBContext find( Class[] classes, Map properties ) throws JAXBException {
351
352        final String jaxbContextFQCN = JAXBContext.class.getName();
353        String factoryClassName;
354
355        // search for jaxb.properties in the class loader of each class first
356        for (final Class c : classes) {
357            // this classloader is used only to load jaxb.properties, so doing this should be safe.
358            ClassLoader classLoader = AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
359                public ClassLoader run() {
360                    return c.getClassLoader();
361                }
362            });
363            Package pkg = c.getPackage();
364            if(pkg==null)
365                continue;       // this is possible for primitives, arrays, and classes that are loaded by poorly implemented ClassLoaders
366            String packageName = pkg.getName().replace('.', '/');
367
368            // TODO: do we want to optimize away searching the same package?  org.Foo, org.Bar, com.Baz
369            //       classes from the same package might come from different class loades, so it might be a bad idea
370
371            // TODO: it's easier to look things up from the class
372            // c.getResourceAsStream("jaxb.properties");
373
374            // build the resource name and use the property loader code
375            String resourceName = packageName+"/jaxb.properties";
376            logger.log(Level.FINE, "Trying to locate {0}", resourceName);
377            Properties props = loadJAXBProperties(classLoader, resourceName);
378            if (props == null) {
379                logger.fine("  not found");
380            } else {
381                logger.fine("  found");
382                if (props.containsKey(JAXB_CONTEXT_FACTORY)) {
383                    // trim() seems redundant, but adding to satisfy customer complaint
384                    factoryClassName = props.getProperty(JAXB_CONTEXT_FACTORY).trim();
385                    return newInstance(classes, properties, factoryClassName);
386                } else {
387                    throw new JAXBException(Messages.format(Messages.MISSING_PROPERTY, packageName, JAXB_CONTEXT_FACTORY));
388                }
389            }
390        }
391
392        // search for a system property second (javax.xml.bind.JAXBContext)
393        logger.log(Level.FINE, "Checking system property {0}", JAXBContext.JAXB_CONTEXT_FACTORY);
394        factoryClassName = AccessController.doPrivileged(new GetPropertyAction(JAXBContext.JAXB_CONTEXT_FACTORY));
395        if (factoryClassName != null) {
396            logger.log(Level.FINE, "  found {0}", factoryClassName);
397            return newInstance( classes, properties, factoryClassName );
398        } else { // leave it here for compatibility reasons
399            logger.fine("  not found");
400            logger.log(Level.FINE, "Checking system property {0}", jaxbContextFQCN);
401            factoryClassName = AccessController.doPrivileged(new GetPropertyAction(jaxbContextFQCN));
402            if (factoryClassName != null) {
403                logger.log(Level.FINE, "  found {0}", factoryClassName);
404                return newInstance( classes, properties, factoryClassName );
405            } else {
406                logger.fine("  not found");
407            }
408        }
409
410        Class factory = lookupUsingOSGiServiceLoader("javax.xml.bind.JAXBContext");
411        if (factory != null) {
412            logger.fine("OSGi environment detected");
413            return newInstance(classes, properties, factory);
414        }
415
416        // search META-INF services next
417        logger.fine("Checking META-INF/services");
418        BufferedReader r;
419        try {
420            final String resource = new StringBuilder("META-INF/services/").append(jaxbContextFQCN).toString();
421            ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
422            URL resourceURL;
423            if(classLoader==null)
424                resourceURL = ClassLoader.getSystemResource(resource);
425            else
426                resourceURL = classLoader.getResource(resource);
427
428            if (resourceURL != null) {
429                logger.log(Level.FINE, "Reading {0}", resourceURL);
430                r = new BufferedReader(new InputStreamReader(resourceURL.openStream(), "UTF-8"));
431                factoryClassName = r.readLine().trim();
432                return newInstance(classes, properties, factoryClassName);
433            } else {
434                logger.log(Level.FINE, "Unable to find: {0}", resource);
435            }
436        } catch (UnsupportedEncodingException e) {
437            // should never happen
438            throw new JAXBException(e);
439        } catch (IOException e) {
440            throw new JAXBException(e);
441        }
442
443        // else no provider found
444        logger.fine("Trying to create the platform default provider");
445        return newInstance(classes, properties, PLATFORM_DEFAULT_FACTORY_CLASS);
446    }

448    private static Class lookupUsingOSGiServiceLoader(String factoryId) {
449        try {
450            // Use reflection to avoid having any dependendcy on ServiceLoader class
451            Class serviceClass = Class.forName(factoryId);
452            Class target = Class.forName("com.sun.org.glassfish.hk2.osgiresourcelocator.ServiceLoader");
453            Method m = target.getMethod("lookupProviderClasses", Class.class);
454            Iterator iter = ((Iterable) m.invoke(null, serviceClass)).iterator();
455            return iter.hasNext() ? (Class)iter.next() : null;
456        } catch(Exception e) {
457            logger.log(Level.FINE, "Unable to find from OSGi: {0}", factoryId);
458            return null;
459        }
460    }

462    private static Properties loadJAXBProperties( ClassLoader classLoader,
463                                                  String propFileName )
464        throws JAXBException {
465
466        Properties props = null;
467
468        try {
469            URL url;
470            if(classLoader==null)
471                url = ClassLoader.getSystemResource(propFileName);
472            else
473                url = classLoader.getResource( propFileName );
474
475            if( url != null ) {
476                logger.log(Level.FINE, "loading props from {0}", url);
477                props = new Properties();
478                InputStream is = url.openStream();
479                props.load( is );
480                is.close();
481            }
482        } catch( IOException ioe ) {
483            logger.log(Level.FINE,"Unable to load "+propFileName,ioe);
484            throw new JAXBException( ioe.toString(), ioe );
485        }
486
487        return props;
488    }

491    /**
492     * Search the given ClassLoader for an instance of the specified class and
493     * return a string representation of the URL that points to the resource.
494     *
495     * @param clazz
496     *          The class to search for
497     * @param loader
498     *          The ClassLoader to search.  If this parameter is null, then the
499     *          system class loader will be searched
500     * @return
501     *          the URL for the class or null if it wasn't found
502     */
503    static URL which(Class clazz, ClassLoader loader) {
504
505        String classnameAsResource = clazz.getName().replace('.', '/') + ".class";
506
507        if(loader == null) {
508            loader = ClassLoader.getSystemClassLoader();
509        }
510
511        return loader.getResource(classnameAsResource);
512    }

514    /**
515     * Get the URL for the Class from it's ClassLoader.
516     *
517     * Convenience method for {@link #which(Class, ClassLoader)}.
518     *
519     * Equivalent to calling: which(clazz, clazz.getClassLoader())
520     *
521     * @param clazz
522     *          The class to search for
523     * @return
524     *          the URL for the class or null if it wasn't found
525     */
526    static URL which(Class clazz) {
527        return which(clazz, clazz.getClassLoader());
528    }

530    /**
531     * When JAXB is in J2SE, rt.jar has to have a JAXB implementation.
532     * However, rt.jar cannot have META-INF/services/javax.xml.bind.JAXBContext
533     * because if it has, it will take precedence over any file that applications have
534     * in their jar files.
535     *
536     * <p>
537     * When the user bundles his own JAXB implementation, we'd like to use it, and we
538     * want the platform default to be used only when there's no other JAXB provider.
539     *
540     * <p>
541     * For this reason, we have to hard-code the class name into the API.
542     */
543    private static final String PLATFORM_DEFAULT_FACTORY_CLASS = "com.sun.xml.internal.bind.v2.ContextFactory";
544
545    /**
546     * Loads the class, provided that the calling thread has an access to the class being loaded.
547     */
548    private static Class safeLoadClass(String className, ClassLoader classLoader) throws ClassNotFoundException {
549       logger.log(Level.FINE, "Trying to load {0}", className);
550       try {
551          // make sure that the current thread has an access to the package of the given name.
552          SecurityManager s = System.getSecurityManager();
553          if (s != null) {
554              int i = className.lastIndexOf('.');
555              if (i != -1) {
556                  s.checkPackageAccess(className.substring(0,i));
557              }
558          }
559
560          if (classLoader == null) {
561              return Class.forName(className);
562          } else {
563              return classLoader.loadClass(className);
564          }
565       } catch (SecurityException se) {
566           // anyone can access the platform default factory class without permission
567           if (PLATFORM_DEFAULT_FACTORY_CLASS.equals(className)) {
568              return Class.forName(className);
569           }
570           throw se;
571       }
572    }
