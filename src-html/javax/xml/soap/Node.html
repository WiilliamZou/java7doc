026package javax.xml.soap;
027
028/**
029 * A representation of a node (element) in an XML document.
030 * This interface extnends the standard DOM Node interface with methods for
031 * getting and setting the value of a node, for
032 * getting and setting the parent of a node, and for removing a node.
033 */
034public interface Node extends org.w3c.dom.Node {
035    /**
036     * Returns the value of this node if this is a <code>Text</code> node or the
037     * value of the immediate child of this node otherwise.
038     * If there is an immediate child of this <code>Node</code> that it is a
039     * <code>Text</code> node then it's value will be returned. If there is
040     * more than one <code>Text</code> node then the value of the first
041     * <code>Text</code> Node will be returned.
042     * Otherwise <code>null</code> is returned.
043     *
044     * @return a <code>String</code> with the text of this node if this is a
045     *          <code>Text</code> node or the text contained by the first
046     *          immediate child of this <code>Node</code> object that is a
047     *          <code>Text</code> object if such a child exists;
048     *          <code>null</code> otherwise.
049     */
050    public String getValue();
051
052    /**
053     * If this is a Text node then this method will set its value,
054     * otherwise it sets the value of  the immediate (Text) child of this node.
055     * The value of the immediate child of this node can be set only if, there is
056     * one child node and that node is a <code>Text</code> node, or if
057     * there are no children in which case a child <code>Text</code> node will be
058     * created.
059     *
060     * @exception IllegalStateException if the node is not a <code>Text</code>
061     *              node and either has more than one child node or has a child
062     *              node that is not a <code>Text</code> node.
063     *
064     * @since SAAJ 1.2
065     */
066    public void setValue(String value);
067
068    /**
069     * Sets the parent of this <code>Node</code> object to the given
070     * <code>SOAPElement</code> object.
071     *
072     * @param parent the <code>SOAPElement</code> object to be set as
073     *       the parent of this <code>Node</code> object
074     *
075     * @exception SOAPException if there is a problem in setting the
076     *                          parent to the given element
077     * @see #getParentElement
078     */
079    public void setParentElement(SOAPElement parent) throws SOAPException;
080
081    /**
082     * Returns the parent element of this <code>Node</code> object.
083     * This method can throw an <code>UnsupportedOperationException</code>
084     * if the tree is not kept in memory.
085     *
086     * @return the <code>SOAPElement</code> object that is the parent of
087     *         this <code>Node</code> object or <code>null</code> if this
088     *         <code>Node</code> object is root
089     *
090     * @exception UnsupportedOperationException if the whole tree is not
091     *            kept in memory
092     * @see #setParentElement
093     */
094    public SOAPElement getParentElement();
095
096    /**
097     * Removes this <code>Node</code> object from the tree.
098     */
099    public void detachNode();
100
101    /**
102     * Notifies the implementation that this <code>Node</code>
103     * object is no longer being used by the application and that the
104     * implementation is free to reuse this object for nodes that may
105     * be created later.
106     * <P>
107     * Calling the method <code>recycleNode</code> implies that the method
108     * <code>detachNode</code> has been called previously.
109     */
110    public void recycleNode();
111
112}
