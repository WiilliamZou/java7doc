026package javax.xml.soap;
027
028/**
029 * An exception that signals that a SOAP exception has occurred. A
030 * <code>SOAPException</code> object may contain a <code>String</code>
031 * that gives the reason for the exception, an embedded
032 * <code>Throwable</code> object, or both. This class provides methods
033 * for retrieving reason messages and for retrieving the embedded
034 * <code>Throwable</code> object.
035 *
036 * <P> Typical reasons for throwing a <code>SOAPException</code>
037 * object are problems such as difficulty setting a header, not being
038 * able to send a message, and not being able to get a connection with
039 * the provider.  Reasons for embedding a <code>Throwable</code>
040 * object include problems such as input/output errors or a parsing
041 * problem, such as an error in parsing a header.
042 */
043public class SOAPException extends Exception {
044    private Throwable cause;
045
046    /**
047     * Constructs a <code>SOAPException</code> object with no
048     * reason or embedded <code>Throwable</code> object.
049     */
050    public SOAPException() {
051        super();
052        this.cause = null;
053    }
054
055    /**
056     * Constructs a <code>SOAPException</code> object with the given
057     * <code>String</code> as the reason for the exception being thrown.
058     *
059     * @param reason a description of what caused the exception
060     */
061    public SOAPException(String reason) {
062        super(reason);
063        this.cause = null;
064    }
065
066    /**
067     * Constructs a <code>SOAPException</code> object with the given
068     * <code>String</code> as the reason for the exception being thrown
069     * and the given <code>Throwable</code> object as an embedded
070     * exception.
071     *
072     * @param reason a description of what caused the exception
073     * @param cause a <code>Throwable</code> object that is to
074     *        be embedded in this <code>SOAPException</code> object
075     */
076    public SOAPException(String reason, Throwable cause) {
077        super(reason);
078        initCause(cause);
079    }
080
081    /**
082     * Constructs a <code>SOAPException</code> object initialized
083     * with the given <code>Throwable</code> object.
084     */
085    public SOAPException(Throwable cause) {
086        super(cause.toString());
087        initCause(cause);
088    }
089
090    /**
091     * Returns the detail message for this <code>SOAPException</code>
092     * object.
093     * <P>
094     * If there is an embedded <code>Throwable</code> object, and if the
095     * <code>SOAPException</code> object has no detail message of its
096     * own, this method will return the detail message from the embedded
097     * <code>Throwable</code> object.
098     *
099     * @return the error or warning message for this
100     *         <code>SOAPException</code> or, if it has none, the
101     *         message of the embedded <code>Throwable</code> object,
102     *         if there is one
103     */
104    public String getMessage() {
105        String message = super.getMessage();
106        if (message == null && cause != null) {
107            return cause.getMessage();
108        } else {
109            return message;
110        }
111    }
112
113    /**
114     * Returns the <code>Throwable</code> object embedded in this
115     * <code>SOAPException</code> if there is one. Otherwise, this method
116     * returns <code>null</code>.
117     *
118     * @return the embedded <code>Throwable</code> object or <code>null</code>
119     *         if there is none
120     */
121
122    public Throwable getCause() {
123        return cause;
124    }
125
126    /**
127     * Initializes the <code>cause</code> field of this <code>SOAPException</code>
128     * object with the given <code>Throwable</code> object.
129     * <P>
130     * This method can be called at most once.  It is generally called from
131     * within the constructor or immediately after the constructor has
132     * returned a new <code>SOAPException</code> object.
133     * If this <code>SOAPException</code> object was created with the
134     * constructor {@link #SOAPException(Throwable)} or
135     * {@link #SOAPException(String,Throwable)}, meaning that its
136     * <code>cause</code> field already has a value, this method cannot be
137     * called even once.
138     *
139     * @param  cause the <code>Throwable</code> object that caused this
140     *         <code>SOAPException</code> object to be thrown.  The value of this
141     *         parameter is saved for later retrieval by the
142     *         {@link #getCause()} method.  A <tt>null</tt> value is
143     *         permitted and indicates that the cause is nonexistent or
144     *         unknown.
145     * @return  a reference to this <code>SOAPException</code> instance
146     * @throws IllegalArgumentException if <code>cause</code> is this
147     *         <code>Throwable</code> object.  (A <code>Throwable</code> object
148     *         cannot be its own cause.)
149     * @throws IllegalStateException if the cause for this <code>SOAPException</code> object
150     *         has already been initialized
151     */
152    public synchronized Throwable initCause(Throwable cause) {
153        if (this.cause != null) {
154            throw new IllegalStateException("Can't override cause");
155        }
156        if (cause == this) {
157            throw new IllegalArgumentException("Self-causation not permitted");
158        }
159        this.cause = cause;
160
161        return this;
162    }
163}
