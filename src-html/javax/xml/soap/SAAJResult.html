026package javax.xml.soap;
027
028import javax.xml.transform.dom.DOMResult;
029
030/**
031 * Acts as a holder for the results of a JAXP transformation or a JAXB
032 * marshalling, in the form of a SAAJ tree. These results should be accessed
033 * by using the {@link #getResult()} method. The {@link DOMResult#getNode()}
034 * method should be avoided in almost all cases.
035 *
036 * @author XWS-Security Development Team
037 *
038 * @since SAAJ 1.3
039 */
040public class SAAJResult extends DOMResult {
041
042    /**
043     * Creates a <code>SAAJResult</code> that will present results in the form
044     * of a SAAJ tree that supports the default (SOAP 1.1) protocol.
045     * <p>
046     * This kind of <code>SAAJResult</code> is meant for use in situations where the
047     * results will be used as a parameter to a method that takes a parameter
048     * whose type, such as <code>SOAPElement</code>, is drawn from the SAAJ
049     * API. When used in a transformation, the results are populated into the
050     * <code>SOAPPart</code> of a <code>SOAPMessage</code> that is created internally.
051     * The <code>SOAPPart</code> returned by {@link DOMResult#getNode()}
052     * is not guaranteed to be well-formed.
053     *
054     * @throws SOAPException if there is a problem creating a <code>SOAPMessage</code>
055     *
056     * @since SAAJ 1.3
057     */
058    public SAAJResult() throws SOAPException {
059        this(MessageFactory.newInstance().createMessage());
060    }
061
062    /**
063     * Creates a <code>SAAJResult</code> that will present results in the form
064     * of a SAAJ tree that supports the specified protocol. The
065     * <code>DYNAMIC_SOAP_PROTOCOL</code> is ambiguous in this context and will
066     * cause this constructor to throw an <code>UnsupportedOperationException</code>.
067     * <p>
068     * This kind of <code>SAAJResult</code> is meant for use in situations where the
069     * results will be used as a parameter to a method that takes a parameter
070     * whose type, such as <code>SOAPElement</code>, is drawn from the SAAJ
071     * API. When used in a transformation the results are populated into the
072     * <code>SOAPPart</code> of a <code>SOAPMessage</code> that is created
073     * internally. The <code>SOAPPart</code> returned by {@link DOMResult#getNode()}
074     * is not guaranteed to be well-formed.
075     *
076     * @param protocol - the name of the SOAP protocol that the resulting SAAJ
077     *                      tree should support
078     *
079     * @throws SOAPException if a <code>SOAPMessage</code> supporting the
080     *             specified protocol cannot be created
081     *
082     * @since SAAJ 1.3
083     */
084    public SAAJResult(String protocol) throws SOAPException {
085        this(MessageFactory.newInstance(protocol).createMessage());
086    }
087
088    /**
089     * Creates a <code>SAAJResult</code> that will write the results into the
090     * <code>SOAPPart</code> of the supplied <code>SOAPMessage</code>.
091     * In the normal case these results will be written using DOM APIs and,
092     * as a result, the finished <code>SOAPPart</code> will not be guaranteed
093     * to be well-formed unless the data used to create it is also well formed.
094     * When used in a transformation the validity of the <code>SOAPMessage</code>
095     * after the transformation can be guaranteed only by means outside SAAJ
096     * specification.
097     *
098     * @param message - the message whose <code>SOAPPart</code> will be
099     *                  populated as a result of some transformation or
100     *                  marshalling operation
101     *
102     * @since SAAJ 1.3
103     */
104    public SAAJResult(SOAPMessage message) {
105        super(message.getSOAPPart());
106    }
107
108    /**
109     * Creates a <code>SAAJResult</code> that will write the results as a
110     * child node of the <code>SOAPElement</code> specified. In the normal
111     * case these results will be written using DOM APIs and as a result may
112     * invalidate the structure of the SAAJ tree. This kind of
113     * <code>SAAJResult</code> should only be used when the validity of the
114     * incoming data can be guaranteed by means outside of the SAAJ
115     * specification.
116     *
117     * @param rootNode - the root to which the results will be appended
118     *
119     * @since SAAJ 1.3
120     */
121    public SAAJResult(SOAPElement rootNode) {
122        super(rootNode);
123    }
124
125
126    /**
127     * @return the resulting Tree that was created under the specified root Node.
128     * @since SAAJ 1.3
129     */
130    public javax.xml.soap.Node getResult() {
131        return (javax.xml.soap.Node)super.getNode().getFirstChild();
132     }
133}
