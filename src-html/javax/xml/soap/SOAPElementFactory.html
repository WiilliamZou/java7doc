026package javax.xml.soap;
027
028/**
029 * <code>SOAPElementFactory</code> is a factory for XML fragments that
030 * will eventually end up in the SOAP part. These fragments
031 * can be inserted as children of the <code>SOAPHeader</code> or
032 * <code>SOAPBody</code> or <code>SOAPEnvelope</code>.
033 *
034 * <p>Elements created using this factory do not have the properties
035 * of an element that lives inside a SOAP header document. These
036 * elements are copied into the XML document tree when they are
037 * inserted.
038 * @deprecated - Use <code>javax.xml.soap.SOAPFactory</code> for creating SOAPElements.
039 * @see javax.xml.soap.SOAPFactory
040 */
041public class SOAPElementFactory {
042
043    private SOAPFactory soapFactory;
044
045    private SOAPElementFactory(SOAPFactory soapFactory) {
046        this.soapFactory = soapFactory;
047    }
048
049    /**
050     * Create a <code>SOAPElement</code> object initialized with the
051     * given <code>Name</code> object.
052     *
053     * @param name a <code>Name</code> object with the XML name for
054     *             the new element
055     *
056     * @return the new <code>SOAPElement</code> object that was
057     *         created
058     *
059     * @exception SOAPException if there is an error in creating the
060     *            <code>SOAPElement</code> object
061     *
062     * @deprecated Use
063     * javax.xml.soap.SOAPFactory.createElement(javax.xml.soap.Name)
064     * instead
065     *
066     * @see javax.xml.soap.SOAPFactory#createElement(javax.xml.soap.Name)
067     * @see javax.xml.soap.SOAPFactory#createElement(javax.xml.namespace.QName)
068     */
069    public SOAPElement create(Name name) throws SOAPException {
070        return soapFactory.createElement(name);
071    }
072
073    /**
074     * Create a <code>SOAPElement</code> object initialized with the
075     * given local name.
076     *
077     * @param localName a <code>String</code> giving the local name for
078     *             the new element
079     *
080     * @return the new <code>SOAPElement</code> object that was
081     *         created
082     *
083     * @exception SOAPException if there is an error in creating the
084     *            <code>SOAPElement</code> object
085     *
086     * @deprecated Use
087     * javax.xml.soap.SOAPFactory.createElement(String localName) instead
088     *
089     * @see javax.xml.soap.SOAPFactory#createElement(java.lang.String)
090     */
091    public SOAPElement create(String localName) throws SOAPException {
092        return soapFactory.createElement(localName);
093    }
094
095    /**
096     * Create a new <code>SOAPElement</code> object with the given
097     * local name, prefix and uri.
098     *
099     * @param localName a <code>String</code> giving the local name
100     *                  for the new element
101     * @param prefix the prefix for this <code>SOAPElement</code>
102     * @param uri a <code>String</code> giving the URI of the
103     *            namespace to which the new element belongs
104     *
105     * @exception SOAPException if there is an error in creating the
106     *            <code>SOAPElement</code> object
107     *
108     * @deprecated Use
109     * javax.xml.soap.SOAPFactory.createElement(String localName,
110     *                      String prefix,
111     *                      String uri)
112     * instead
113     *
114     * @see javax.xml.soap.SOAPFactory#createElement(java.lang.String, java.lang.String, java.lang.String)
115     */
116    public SOAPElement create(String localName, String prefix, String uri)
117        throws SOAPException {
118        return soapFactory.createElement(localName, prefix, uri);
119    }
120
121    /**
122     * Creates a new instance of <code>SOAPElementFactory</code>.
123     *
124     * @return a new instance of a <code>SOAPElementFactory</code>
125     *
126     * @exception SOAPException if there was an error creating the
127     *            default <code>SOAPElementFactory</code>
128     */
129    public static SOAPElementFactory newInstance() throws SOAPException {
130        try {
131            return new SOAPElementFactory(SOAPFactory.newInstance());
132        } catch (Exception ex) {
133            throw new SOAPException(
134                "Unable to create SOAP Element Factory: " + ex.getMessage());
135        }
136    }
137}
