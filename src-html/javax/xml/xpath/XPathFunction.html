026package javax.xml.xpath;
027
028import java.util.List;
029
030/**
031 * <p><code>XPathFunction</code> provides access to XPath functions.</p>
032 *
033 * <p>Functions are identified by QName and arity in XPath.</p>
034 *
035 * @author  <a href="mailto:Norman.Walsh@Sun.com">Norman Walsh</a>
036 * @author  <a href="mailto:Jeff.Suttor@Sun.com">Jeff Suttor</a>
037 * @since 1.5
038 */
039public interface XPathFunction {
040  /**
041   * <p>Evaluate the function with the specified arguments.</p>
042   *
043   * <p>To the greatest extent possible, side-effects should be avoided in the
044   * definition of extension functions. The implementation evaluating an
045   * XPath expression is under no obligation to call extension functions in
046   * any particular order or any particular number of times.</p>
047   *
048   * @param args The arguments, <code>null</code> is a valid value.
049   *
050   * @return The result of evaluating the <code>XPath</code> function as an <code>Object</code>.
051   *
052   * @throws XPathFunctionException If <code>args</code> cannot be evaluated with this <code>XPath</code> function.
053   */
054  public Object evaluate(List args)
055    throws XPathFunctionException;
056}
