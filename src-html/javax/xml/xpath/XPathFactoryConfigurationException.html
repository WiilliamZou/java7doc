026package javax.xml.xpath;
027
028/**
029 * <code>XPathFactoryConfigurationException</code> represents a configuration error in a <code>XPathFactory</code> environment.</p>
030 *
031 * @author  <a href="mailto:Norman.Walsh@Sun.com">Norman Walsh</a>
032 * @author  <a href="mailto:Jeff.Suttor@Sun.com">Jeff Suttor</a>
033 * @since 1.5
034 */
035public class XPathFactoryConfigurationException extends XPathException {
036
037    /**
038     * <p>Stream Unique Identifier.</p>
039     */
040    private static final long serialVersionUID = -1837080260374986980L;
041
042    /**
043     * <p>Constructs a new <code>XPathFactoryConfigurationException</code> with the specified detail <code>message</code>.</p>
044     *
045     * <p>The <code>cause</code> is not initialized.</p>
046     *
047     * <p>If <code>message</code> is <code>null</code>,
048     * then a <code>NullPointerException</code> is thrown.</p>
049     *
050     * @param message The detail message.
051     *
052     * @throws NullPointerException When <code>message</code> is
053     *   <code>null</code>.
054     */
055    public XPathFactoryConfigurationException(String message) {
056        super(message);
057    }
058
059    /**
060     * <p>Constructs a new <code>XPathFactoryConfigurationException</code>
061     * with the specified <code>cause</code>.</p>
062     *
063     * <p>If <code>cause</code> is <code>null</code>,
064     * then a <code>NullPointerException</code> is thrown.</p>
065     *
066     * @param cause The cause.
067     *
068     * @throws NullPointerException if <code>cause</code> is <code>null</code>.
069     */
070    public XPathFactoryConfigurationException(Throwable cause) {
071        super(cause);
072    }
073}
