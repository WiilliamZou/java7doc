029package javax.xml.stream;
030import javax.xml.stream.events.*;
031import javax.xml.namespace.NamespaceContext;
032import javax.xml.namespace.QName;
033import java.util.Iterator;
034/**
035 * This interface defines a utility class for creating instances of
036 * XMLEvents
037 * @version 1.2
038 * @author Copyright (c) 2009 by Oracle Corporation. All Rights Reserved.
039 * @see javax.xml.stream.events.StartElement
040 * @see javax.xml.stream.events.EndElement
041 * @see javax.xml.stream.events.ProcessingInstruction
042 * @see javax.xml.stream.events.Comment
043 * @see javax.xml.stream.events.Characters
044 * @see javax.xml.stream.events.StartDocument
045 * @see javax.xml.stream.events.EndDocument
046 * @see javax.xml.stream.events.DTD
047 * @since 1.6
048 */
049public abstract class XMLEventFactory {
050  protected XMLEventFactory(){}
051
052    static final String JAXPFACTORYID = "javax.xml.stream.XMLEventFactory";
053    static final String DEFAULIMPL = "com.sun.xml.internal.stream.events.XMLEventFactoryImpl";
054
055
056  /**
057   * Create a new instance of the factory
058   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
059   */
060  public static XMLEventFactory newInstance()
061    throws FactoryConfigurationError
062  {
063    return (XMLEventFactory) FactoryFinder.find(JAXPFACTORYID, DEFAULIMPL, true);
064  }
065
066  /**
067   * Create a new instance of the factory.
068   * This static method creates a new factory instance.
069   * This method uses the following ordered lookup procedure to determine
070   * the XMLEventFactory implementation class to load:
071   *   Use the javax.xml.stream.XMLEventFactory system property.
072   *   Use the properties file "lib/stax.properties" in the JRE directory.
073   *     This configuration file is in standard java.util.Properties format
074   *     and contains the fully qualified name of the implementation class
075   *     with the key being the system property defined above.
076   *   Use the Services API (as detailed in the JAR specification), if available,
077   *     to determine the classname. The Services API will look for a classname
078   *     in the file META-INF/services/javax.xml.stream.XMLEventFactory in jars
079   *     available to the runtime.
080   *   Platform default XMLEventFactory instance.
081   *
082   *   Once an application has obtained a reference to a XMLEventFactory it
083   *   can use the factory to configure and obtain stream instances.
084   *
085   *   Note that this is a new method that replaces the deprecated newInstance() method.
086   *     No changes in behavior are defined by this replacement method relative to
087   *     the deprecated method.
088   *
089   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
090   */
091  public static XMLEventFactory newFactory()
092    throws FactoryConfigurationError
093  {
094    return (XMLEventFactory) FactoryFinder.find(JAXPFACTORYID, DEFAULIMPL, true);
095  }
096
097  /**
098   * Create a new instance of the factory
099   *
100   * @param factoryId             Name of the factory to find, same as
101   *                              a property name
102   * @param classLoader           classLoader to use
103   * @return the factory implementation
104   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
105   *
106   * @deprecated  This method has been deprecated to maintain API consistency.
107   *              All newInstance methods have been replaced with corresponding
108   *              newFactory methods. The replacement {@link
109   *              #newFactory(java.lang.String, java.lang.ClassLoader)}
110   *              method defines no changes in behavior.
111   */
112  public static XMLEventFactory newInstance(String factoryId,
113          ClassLoader classLoader)
114          throws FactoryConfigurationError {
115      try {
116          //do not fallback if given classloader can't find the class, throw exception
117            return (XMLEventFactory) FactoryFinder.find(factoryId, classLoader,
118                    null, factoryId.equals(JAXPFACTORYID) ? true : false);
119      } catch (FactoryFinder.ConfigurationError e) {
120          throw new FactoryConfigurationError(e.getException(),
121                  e.getMessage());
122      }
123  }
124
125  /**
126   * Create a new instance of the factory.
127   * If the classLoader argument is null, then the ContextClassLoader is used.
128   *
129   * Note that this is a new method that replaces the deprecated
130   *   newInstance(String factoryId, ClassLoader classLoader) method.
131   * No changes in behavior are defined by this replacement method relative
132   * to the deprecated method.
133   *
134   * @param factoryId             Name of the factory to find, same as
135   *                              a property name
136   * @param classLoader           classLoader to use
137   * @return the factory implementation
138   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
139   */
140  public static XMLEventFactory newFactory(String factoryId,
141          ClassLoader classLoader)
142          throws FactoryConfigurationError {
143      try {
144          //do not fallback if given classloader can't find the class, throw exception
145            return (XMLEventFactory) FactoryFinder.find(factoryId, classLoader,
146                    null, factoryId.equals(JAXPFACTORYID) ? true : false);
147      } catch (FactoryFinder.ConfigurationError e) {
148          throw new FactoryConfigurationError(e.getException(),
149                  e.getMessage());
150      }
151  }

153 /**
154   * This method allows setting of the Location on each event that
155   * is created by this factory.  The values are copied by value into
156   * the events created by this factory.  To reset the location

158   * @param location the location to set on each event created

160  public abstract void setLocation(Location location);

163   * Create a new Attribute

165   * @param namespaceURI the attribute value is set to this value, may not be null

168   * @return the Attribute with specified values
169   */
170  public abstract Attribute createAttribute(String prefix, String namespaceURI, String localName, String value);
171
172  /**
173   * Create a new Attribute
174   * @param localName the local name of the XML name of the attribute, localName cannot be null
175   * @param value the attribute value to set, may not be null
176   * @return the Attribute with specified values
177   */
178  public abstract Attribute createAttribute(String localName, String value);
179
180  /**

203  /**
204   * Create a new StartElement.  Namespaces can be added to this StartElement
205   * by passing in an Iterator that walks over a set of Namespace interfaces.
206   * Attributes can be added to this StartElement by passing an iterator
207   * that walks over a set of Attribute interfaces.
208   *
209   * @param name the qualified name of the attribute, may not be null
210   * @param attributes an optional unordered set of objects that
211   * implement Attribute to add to the new StartElement, may be null
212   * @param namespaces an optional unordered set of objects that
213   * implement Namespace to add to the new StartElement, may be null
214   * @return an instance of the requested StartElement
215   */
216  public abstract StartElement createStartElement(QName name,
217                                                  Iterator attributes,
218                                                  Iterator namespaces);
219
220  /**
221   * Create a new StartElement.  This defaults the NamespaceContext to
222   * an empty NamespaceContext.  Querying this event for its namespaces or
223   * attributes will result in an empty iterator being returned.
224   *
225   * @param namespaceUri the uri of the QName of the new StartElement
226   * @param localName the local name of the QName of the new StartElement
227   * @param prefix the prefix of the QName of the new StartElement
228   * @return an instance of the requested StartElement
229   */
230  public abstract StartElement createStartElement(String prefix,
231                                                  String namespaceUri,
232                                                  String localName);
233  /**
234   * Create a new StartElement.  Namespaces can be added to this StartElement
235   * by passing in an Iterator that walks over a set of Namespace interfaces.
236   * Attributes can be added to this StartElement by passing an iterator
237   * that walks over a set of Attribute interfaces.
238   *
239   * @param namespaceUri the uri of the QName of the new StartElement
240   * @param localName the local name of the QName of the new StartElement
241   * @param prefix the prefix of the QName of the new StartElement
242   * @param attributes an unordered set of objects that implement
243   * Attribute to add to the new StartElement
244   * @param namespaces an unordered set of objects that implement
245   * Namespace to add to the new StartElement
246   * @return an instance of the requested StartElement
247   */
248  public abstract StartElement createStartElement(String prefix,
249                                                  String namespaceUri,
250                                                  String localName,
251                                                  Iterator attributes,
252                                                  Iterator namespaces

254  /**
255   * Create a new StartElement.  Namespaces can be added to this StartElement
256   * by passing in an Iterator that walks over a set of Namespace interfaces.
257   * Attributes can be added to this StartElement by passing an iterator
258   * that walks over a set of Attribute interfaces.
259   *
260   * @param namespaceUri the uri of the QName of the new StartElement
261   * @param localName the local name of the QName of the new StartElement
262   * @param prefix the prefix of the QName of the new StartElement
263   * @param attributes an unordered set of objects that implement
264   * Attribute to add to the new StartElement, may be null
265   * @param namespaces an unordered set of objects that implement
266   * Namespace to add to the new StartElement, may be null
267   * @param context the namespace context of this element
268   * @return an instance of the requested StartElement
269   */
270  public abstract StartElement createStartElement(String prefix,
271                                                  String namespaceUri,
272                                                  String localName,
273                                                  Iterator attributes,
274                                                  Iterator namespaces,
275                                                  NamespaceContext context
276                                                  );
