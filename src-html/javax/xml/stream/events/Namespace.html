029package javax.xml.stream.events;
030
031import javax.xml.namespace.QName;
032
033/**
034 * An interface that contains information about a namespace.
035 * Namespaces are accessed from a StartElement.
036 *
037 * @version 1.0
038 * @author Copyright (c) 2009 by Oracle Corporation. All Rights Reserved.
039 * @see StartElement
040 * @since 1.6
041 */
042public interface Namespace extends Attribute {
043
044  /**
045   * Gets the prefix, returns "" if this is a default
046   * namespace declaration.
047   */
048  public String getPrefix();
049
050  /**
051   * Gets the uri bound to the prefix of this namespace
052   */
053  public String getNamespaceURI();
054
055  /**
056   * returns true if this attribute declares the default namespace
057   */
058  public boolean isDefaultNamespaceDeclaration();
059}
