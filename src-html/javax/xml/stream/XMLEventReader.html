029package javax.xml.stream;
030
031import javax.xml.stream.events.XMLEvent;
032
033import java.util.Iterator;
034
035/**
036 *
037 * This is the top level interface for parsing XML Events.  It provides
038 * the ability to peek at the next event and returns configuration
039 * information through the property interface.
040 *
041 * @version 1.0
042 * @author Copyright (c) 2009 by Oracle Corporation. All Rights Reserved.
043 * @see XMLInputFactory
044 * @see XMLEventWriter
045 * @since 1.6
046 */
047public interface XMLEventReader extends Iterator {
048  /**
049   * Get the next XMLEvent
050   * @see XMLEvent
051   * @throws XMLStreamException if there is an error with the underlying XML.
052   * @throws NoSuchElementException iteration has no more elements.
053   */
054  public XMLEvent nextEvent() throws XMLStreamException;
055
056  /**
057   * Check if there are more events.
058   * Returns true if there are more events and false otherwise.
059   * @return true if the event reader has more events, false otherwise
060   */
061  public boolean hasNext();
062
063  /**
064   * Check the next XMLEvent without reading it from the stream.
065   * Returns null if the stream is at EOF or has no more XMLEvents.
066   * A call to peek() will be equal to the next return of next().
067   * @see XMLEvent
068   * @throws XMLStreamException
069   */
070  public XMLEvent peek() throws XMLStreamException;
071
072  /**
073   * Reads the content of a text-only element. Precondition:
074   * the current event is START_ELEMENT. Postcondition:
075   * The current event is the corresponding END_ELEMENT.
076   * @throws XMLStreamException if the current event is not a START_ELEMENT
077   * or if a non text element is encountered
078   */
079  public String getElementText() throws XMLStreamException;
080
081  /**
082   * Skips any insignificant space events until a START_ELEMENT or
083   * END_ELEMENT is reached. If anything other than space characters are
084   * encountered, an exception is thrown. This method should
085   * be used when processing element-only content because
086   * the parser is not able to recognize ignorable whitespace if
087   * the DTD is missing or not interpreted.
088   * @throws XMLStreamException if anything other than space characters are encountered
089   */
090  public XMLEvent nextTag() throws XMLStreamException;
091
092  /**
093   * Get the value of a feature/property from the underlying implementation
094   * @param name The name of the property
095   * @return The value of the property
096   * @throws IllegalArgumentException if the property is not supported
097   */
098  public Object getProperty(java.lang.String name) throws java.lang.IllegalArgumentException;
099
100  /**
101   * Frees any resources associated with this Reader.  This method does not close the
102   * underlying input source.
103   * @throws XMLStreamException if there are errors freeing associated resources
104   */
105  public void close() throws XMLStreamException;
106}
