029package javax.xml.stream;
030
031/**
032 * Provides information on the location of an event.
033 *
034 * All the information provided by a Location is optional.  For example
035 * an application may only report line numbers.
036 *
037 * @version 1.0
038 * @author Copyright (c) 2009 by Oracle Corporation. All Rights Reserved.
039 * @since 1.6
040 */
041public interface Location {
042  /**
043   * Return the line number where the current event ends,
044   * returns -1 if none is available.
045   * @return the current line number
046   */
047  int getLineNumber();
048
049  /**
050   * Return the column number where the current event ends,
051   * returns -1 if none is available.
052   * @return the current column number
053   */
054  int getColumnNumber();
055
056  /**
057   * Return the byte or character offset into the input source this location
058   * is pointing to. If the input source is a file or a byte stream then
059   * this is the byte offset into that stream, but if the input source is
060   * a character media then the offset is the character offset.
061   * Returns -1 if there is no offset available.
062   * @return the current offset
063   */
064  int getCharacterOffset();
065
066  /**
067   * Returns the public ID of the XML
068   * @return the public ID, or null if not available
069   */
070  public String getPublicId();
071
072  /**
073   * Returns the system ID of the XML
074   * @return the system ID, or null if not available
075   */
076  public String getSystemId();
077}
