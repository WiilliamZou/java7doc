029package javax.xml.stream;
030
031/**
032 * This interface is used to report non-fatal errors.
033 * Only warnings should be echoed through this interface.
034 * @version 1.0
035 * @author Copyright (c) 2009 by Oracle Corporation. All Rights Reserved.
036 * @since 1.6
037 */
038public interface XMLReporter {
039
040    /**
041
042     * Report the desired message in an application specific format.
043
044     * Only warnings and non-fatal errors should be reported through
045
046     * this interface.
047
048     * Fatal errors should be thrown as XMLStreamException.
049
050     *
051
052     * @param message the error message
053
054     * @param errorType an implementation defined error type
055
056     * @param relatedInformation information related to the error, if available
057
058     * @param location the location of the error, if available
059
060     * @throws XMLStreamException
061
062     */
063    public void report(String message, String errorType, Object relatedInformation, Location location)
064            throws XMLStreamException;
065}
