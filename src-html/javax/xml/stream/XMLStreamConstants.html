025package javax.xml.stream;
026
027/**
028 * This interface declares the constants used in this API.
029 * Numbers in the range 0 to 256 are reserved for the specification,
030 * user defined events must use event codes outside that range.
031 *
032 * @since 1.6
033 */
034
035public interface XMLStreamConstants {
036  /**
037   * Indicates an event is a start element
038   * @see javax.xml.stream.events.StartElement
039   */
040  public static final int START_ELEMENT=1;
041  /**
042   * Indicates an event is an end element
043   * @see javax.xml.stream.events.EndElement
044   */
045  public static final int END_ELEMENT=2;
046  /**
047   * Indicates an event is a processing instruction
048   * @see javax.xml.stream.events.ProcessingInstruction
049   */
050  public static final int PROCESSING_INSTRUCTION=3;
051
052  /**
053   * Indicates an event is characters
054   * @see javax.xml.stream.events.Characters
055   */
056  public static final int CHARACTERS=4;
057
058  /**
059   * Indicates an event is a comment
060   * @see javax.xml.stream.events.Comment
061   */
062  public static final int COMMENT=5;
063
064  /**
065   * The characters are white space
066   * (see [XML], 2.10 "White Space Handling").
067   * Events are only reported as SPACE if they are ignorable white
068   * space.  Otherwise they are reported as CHARACTERS.
069   * @see javax.xml.stream.events.Characters
070   */
071  public static final int SPACE=6;
072
073  /**
074   * Indicates an event is a start document
075   * @see javax.xml.stream.events.StartDocument
076   */
077  public static final int START_DOCUMENT=7;
078
079  /**
080   * Indicates an event is an end document
081   * @see javax.xml.stream.events.EndDocument
082   */
083  public static final int END_DOCUMENT=8;
084
085  /**
086   * Indicates an event is an entity reference
087   * @see javax.xml.stream.events.EntityReference
088   */
089  public static final int ENTITY_REFERENCE=9;
090
091  /**
092   * Indicates an event is an attribute
093   * @see javax.xml.stream.events.Attribute
094   */
095  public static final int ATTRIBUTE=10;
096
097  /**
098   * Indicates an event is a DTD
099   * @see javax.xml.stream.events.DTD
100   */
101  public static final int DTD=11;
102
103  /**
104   * Indicates an event is a CDATA section
105   * @see javax.xml.stream.events.Characters
106   */
107  public static final int CDATA=12;
108
109  /**
110   * Indicates the event is a namespace declaration
111   *
112   * @see javax.xml.stream.events.Namespace
113   */
114  public static final int NAMESPACE=13;
115
116  /**
117   * Indicates a Notation
118   * @see javax.xml.stream.events.NotationDeclaration
119   */
120  public static final int NOTATION_DECLARATION=14;
121
122  /**
123   * Indicates a Entity Declaration
124   * @see javax.xml.stream.events.NotationDeclaration
125   */
126  public static final int ENTITY_DECLARATION=15;
127}
