034/**
035 * Defines an abstract implementation of a factory for getting streams.
036 *
037 * The following table defines the standard properties of this specification.
038 * Each property varies in the level of support required by each implementation.
039 * The level of support required is described in the 'Required' column.
040 *
041 *   <table border="2" rules="all" cellpadding="4">
042 *    <thead>
043 *      <tr>
044 *        <th align="center" colspan="5">
045 *          Configuration parameters
046 *        </th>
047 *      </tr>
048 *    </thead>
049 *    <tbody>
050 *      <tr>
051 *        <th>Property Name</th>
052 *        <th>Behavior</th>
053 *        <th>Return type</th>
054 *        <th>Default Value</th>
055 *        <th>Required</th>
056 *      </tr>
057 * <tr><td>javax.xml.stream.isValidating</td><td>Turns on/off implementation specific DTD validation</td><td>Boolean</td><td>False</td><td>No</td></tr>
058 * <tr><td>javax.xml.stream.isNamespaceAware</td><td>Turns on/off namespace processing for XML 1.0 support</td><td>Boolean</td><td>True</td><td>True (required) / False (optional)</td></tr>
059 * <tr><td>javax.xml.stream.isCoalescing</td><td>Requires the processor to coalesce adjacent character data</td><td>Boolean</td><td>False</td><td>Yes</td></tr>
060 * <tr><td>javax.xml.stream.isReplacingEntityReferences</td><td>replace internal entity references with their replacement text and report them as characters</td><td>Boolean</td><td>True</td><td>Yes</td></tr>
061 *<tr><td>javax.xml.stream.isSupportingExternalEntities</td><td>Resolve external parsed entities</td><td>Boolean</td><td>Unspecified</td><td>Yes</td></tr>
062 *<tr><td>javax.xml.stream.supportDTD</td><td>Use this property to request processors that do not support DTDs</td><td>Boolean</td><td>True</td><td>Yes</td></tr>
063 *<tr><td>javax.xml.stream.reporter</td><td>sets/gets the impl of the XMLReporter </td><td>javax.xml.stream.XMLReporter</td><td>Null</td><td>Yes</td></tr>
064 *<tr><td>javax.xml.stream.resolver</td><td>sets/gets the impl of the XMLResolver interface</td><td>javax.xml.stream.XMLResolver</td><td>Null</td><td>Yes</td></tr>
065 *<tr><td>javax.xml.stream.allocator</td><td>sets/gets the impl of the XMLEventAllocator interface</td><td>javax.xml.stream.util.XMLEventAllocator</td><td>Null</td><td>Yes</td></tr>
066 *    </tbody>
067 *  </table>

069 *
070 * @version 1.2

073 * @see XMLEventReader

142  static final String JAXPFACTORYID = "javax.xml.stream.XMLInputFactory";
143  static final String DEFAULIMPL = "com.sun.xml.internal.stream.XMLInputFactoryImpl";
144
145  protected XMLInputFactory(){}

147  /**
148   * Create a new instance of the factory.
149   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
150   */
151  public static XMLInputFactory newInstance()
152    throws FactoryConfigurationError
153  {
154    return (XMLInputFactory) FactoryFinder.find(JAXPFACTORYID, DEFAULIMPL, true);
155  }
156
157  /**
158   * Create a new instance of the factory.
159   * This static method creates a new factory instance.
160   * This method uses the following ordered lookup procedure to determine
161   * the XMLInputFactory implementation class to load:
162   *   Use the javax.xml.stream.XMLInputFactory system property.
163   *   Use the properties file "lib/stax.properties" in the JRE directory.
164   *     This configuration file is in standard java.util.Properties format
165   *     and contains the fully qualified name of the implementation class
166   *     with the key being the system property defined above.
167   *   Use the Services API (as detailed in the JAR specification), if available,
168   *     to determine the classname. The Services API will look for a classname
169   *     in the file META-INF/services/javax.xml.stream.XMLInputFactory in jars
170   *     available to the runtime.
171   *   Platform default XMLInputFactory instance.
172   *
173   *   Once an application has obtained a reference to a XMLInputFactory it
174   *   can use the factory to configure and obtain stream instances.
175   *
176   *   Note that this is a new method that replaces the deprecated newInstance() method.
177   *     No changes in behavior are defined by this replacement method relative to
178   *     the deprecated method.
179   *
180   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
181   */
182  public static XMLInputFactory newFactory()
183    throws FactoryConfigurationError
184  {
185    return (XMLInputFactory) FactoryFinder.find(JAXPFACTORYID, DEFAULIMPL, true);
186  }
187
188  /**
189   * Create a new instance of the factory
190   *
191   * @param factoryId             Name of the factory to find, same as
192   *                              a property name
193   * @param classLoader           classLoader to use
194   * @return the factory implementation
195   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
196   *
197   * @deprecated  This method has been deprecated to maintain API consistency.
198   *              All newInstance methods have been replaced with corresponding
199   *              newFactory methods. The replacement {@link
200   *              #newFactory(java.lang.String, java.lang.ClassLoader)} method
201   *              defines no changes in behavior.
202   */
203  public static XMLInputFactory newInstance(String factoryId,
204          ClassLoader classLoader)
205          throws FactoryConfigurationError {
206      try {
207          //do not fallback if given classloader can't find the class, throw exception
208          return (XMLInputFactory) FactoryFinder.find(factoryId, classLoader,
209                  null, factoryId.equals(JAXPFACTORYID) ? true : false);
210      } catch (FactoryFinder.ConfigurationError e) {
211          throw new FactoryConfigurationError(e.getException(),
212                  e.getMessage());
213      }
214  }
215

218   * If the classLoader argument is null, then the ContextClassLoader is used.
219   *
220   * Note that this is a new method that replaces the deprecated

222   * No changes in behavior are defined by this replacement method relative
223   * to the deprecated method.
224   *
225   * @param factoryId             Name of the factory to find, same as
226   *                              a property name
227   * @param classLoader           classLoader to use
228   * @return the factory implementation
229   * @throws FactoryConfigurationError if an instance of this factory cannot be loaded
230   */
