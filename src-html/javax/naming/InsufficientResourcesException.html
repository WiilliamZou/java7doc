026package javax.naming;
027
028/**
029  * This exception is thrown when resources are not available to complete
030  * the requested operation. This might due to a lack of resources on
031  * the server or on the client. There are no restrictions to resource types,
032  * as different services might make use of different resources. Such
033  * restrictions might be due to physical limits and/or adminstrative quotas.
034  * Examples of limited resources are internal buffers, memory, network bandwidth.
035  *<p>
036  * InsufficientResourcesException is different from LimitExceededException in that
037  * the latter is due to user/system specified limits. See LimitExceededException
038  * for details.
039  * <p>
040  * Synchronization and serialization issues that apply to NamingException
041  * apply directly here.
042  *
043  * @author Rosanna Lee
044  * @author Scott Seligman
045  * @since 1.3
046  */
047
048public class InsufficientResourcesException extends NamingException {
049    /**
050     * Constructs a new instance of InsufficientResourcesException using an
051     * explanation. All other fields default to null.
052     *
053     * @param   explanation     Possibly null additional detail about this exception.
054     * @see java.lang.Throwable#getMessage
055     */
056    public InsufficientResourcesException(String explanation) {
057        super(explanation);
058    }
059
060    /**
061      * Constructs a new instance of InsufficientResourcesException with
062      * all name resolution fields and explanation initialized to null.
063      */
064    public InsufficientResourcesException() {
065        super();
066    }
067
068    /**
069     * Use serialVersionUID from JNDI 1.1.1 for interoperability
070     */
071    private static final long serialVersionUID = 6227672693037844532L;
072}
