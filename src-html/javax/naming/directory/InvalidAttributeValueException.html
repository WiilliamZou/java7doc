026package javax.naming.directory;
027
028import javax.naming.NamingException;
029
030/**
031  * This class is thrown when an attempt is
032  * made to add to an attribute a value that conflicts with the attribute's
033  * schema definition.  This could happen, for example, if attempting
034  * to add an attribute with no value when the attribute is required
035  * to have at least one value, or if attempting to add more than
036  * one value to a single valued-attribute, or if attempting to
037  * add a value that conflicts with the syntax of the attribute.
038  * <p>
039  * Synchronization and serialization issues that apply to NamingException
040  * apply directly here.
041  *
042  * @author Rosanna Lee
043  * @author Scott Seligman
044  * @since 1.3
045  */
046
047public class InvalidAttributeValueException extends NamingException {
048    /**
049     * Constructs a new instance of InvalidAttributeValueException using
050     * an explanation. All other fields are set to null.
051     * @param   explanation     Additional detail about this exception. Can be null.
052     * @see java.lang.Throwable#getMessage
053     */
054    public InvalidAttributeValueException(String explanation) {
055        super(explanation);
056    }
057
058    /**
059      * Constructs a new instance of InvalidAttributeValueException.
060      * All fields are set to null.
061      */
062    public InvalidAttributeValueException() {
063        super();
064    }
065
066    /**
067     * Use serialVersionUID from JNDI 1.1.1 for interoperability
068     */
069    private static final long serialVersionUID = 8720050295499275011L;
070}
