026package javax.naming.ldap;
027
028/**
029 * This class represents an event fired in response to an unsolicited
030 * notification sent by the LDAP server.
031 *
032 * @author Rosanna Lee
033 * @author Scott Seligman
034 * @author Vincent Ryan
035 *
036 * @see UnsolicitedNotification
037 * @see UnsolicitedNotificationListener
038 * @see javax.naming.event.EventContext#addNamingListener
039 * @see javax.naming.event.EventDirContext#addNamingListener
040 * @see javax.naming.event.EventContext#removeNamingListener
041 * @since 1.3
042 */
043
044public class UnsolicitedNotificationEvent extends java.util.EventObject {
045    /**
046     * The notification that caused this event to be fired.
047     * @serial
048     */
049    private UnsolicitedNotification notice;
050
051    /**
052     * Constructs a new instance of <tt>UnsolicitedNotificationEvent</tt>.
053     *
054     * @param src The non-null source that fired the event.
055     * @param notice The non-null unsolicited notification.
056     */
057    public UnsolicitedNotificationEvent(Object src,
058        UnsolicitedNotification notice) {
059        super(src);
060        this.notice = notice;
061    }
062
063
064    /**
065     * Returns the unsolicited notification.
066     * @return The non-null unsolicited notification that caused this
067     * event to be fired.
068     */
069    public UnsolicitedNotification getNotification() {
070        return notice;
071    }
072
073    /**
074     * Invokes the <tt>notificationReceived()</tt> method on
075     * a listener using this event.
076     * @param listener The non-null listener on which to invoke
077     * <tt>notificationReceived</tt>.
078     */
079    public void dispatch(UnsolicitedNotificationListener listener) {
080        listener.notificationReceived(this);
081    }
082
083    private static final long serialVersionUID = -2382603380799883705L;
084}
