026package javax.naming.ldap;
027
028/**
029  * This interface represents an LDAPv3 control as defined in
030  * <A HREF="http://www.ietf.org/rfc/rfc2251.txt">RFC 2251</A>.
031  *<p>
032  * The LDAPv3 protocol uses controls to send and receive additional data
033  * to affect the behavior of predefined operations.
034  * Controls can be sent along with any LDAP operation to the server.
035  * These are referred to as <em>request controls</em>. For example, a
036  * "sort" control can be sent with an LDAP search operation to
037  * request that the results be returned in a particular order.
038  * Solicited and unsolicited controls can also be returned with
039  * responses from the server. Such controls are referred to as
040  * <em>response controls</em>. For example, an LDAP server might
041  * define a special control to return change notifications.
042  *<p>
043  * This interface is used to represent both request and response controls.
044  *
045  * @author Rosanna Lee
046  * @author Scott Seligman
047  * @author Vincent Ryan
048  *
049  * @see ControlFactory
050  * @since 1.3
051  */
052public interface Control extends java.io.Serializable {
053    /**
054      * Indicates a critical control.
055      * The value of this constant is <tt>true</tt>.
056      */
057    public static final boolean CRITICAL = true;
058
059    /**
060      * Indicates a non-critical control.
061      * The value of this constant is <tt>false</tt>.
062      */
063    public static final boolean NONCRITICAL = false;
064
065    /**
066      * Retrieves the object identifier assigned for the LDAP control.
067      *
068      * @return The non-null object identifier string.
069      */
070    public String getID();
071
072    /**
073      * Determines the criticality of the LDAP control.
074      * A critical control must not be ignored by the server.
075      * In other words, if the server receives a critical control
076      * that it does not support, regardless of whether the control
077      * makes sense for the operation, the operation will not be performed
078      * and an <tt>OperationNotSupportedException</tt> will be thrown.
079      * @return true if this control is critical; false otherwise.
080      */
081    public boolean isCritical();
082
083    /**
084      * Retrieves the ASN.1 BER encoded value of the LDAP control.
085      * The result is the raw BER bytes including the tag and length of
086      * the control's value. It does not include the controls OID or criticality.
087      *
088      * Null is returned if the value is absent.
089      *
090      * @return A possibly null byte array representing the ASN.1 BER encoded
091      *         value of the LDAP control.
092      */
093    public byte[] getEncodedValue();
094
095    // static final long serialVersionUID = -591027748900004825L;
096}
