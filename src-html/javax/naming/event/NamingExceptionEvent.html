026package javax.naming.event;
027
028import javax.naming.NamingException;
029
030/**
031  * This class represents an event fired when the procedures/processes
032  * used to collect information for notifying listeners of
033  * <tt>NamingEvent</tt>s threw a <tt>NamingException</tt>.
034  * This can happen, for example, if the server which the listener is using
035  * aborts subsequent to the <tt>addNamingListener()</tt> call.
036  *
037  * @author Rosanna Lee
038  * @author Scott Seligman
039  *
040  * @see NamingListener#namingExceptionThrown
041  * @see EventContext
042  * @since 1.3
043  */
044
045public class NamingExceptionEvent extends java.util.EventObject {
046    /**
047     * Contains the exception that was thrown
048     * @serial
049     */
050    private NamingException exception;
051
052    /**
053     * Constructs an instance of <tt>NamingExceptionEvent</tt> using
054     * the context in which the <tt>NamingException</tt> was thrown and the exception
055     * that was thrown.
056     *
057     * @param source The non-null context in which the exception was thrown.
058     * @param exc    The non-null <tt>NamingException</tt> that was thrown.
059     *
060     */
061    public NamingExceptionEvent(EventContext source, NamingException exc) {
062        super(source);
063        exception = exc;
064    }
065
066    /**
067     * Retrieves the exception that was thrown.
068     * @return The exception that was thrown.
069     */
070    public NamingException getException() {
071        return exception;
072    }
073
074    /**
075     * Retrieves the <tt>EventContext</tt> that fired this event.
076     * This returns the same object as <tt>EventObject.getSource()</tt>.
077     * @return The non-null <tt>EventContext</tt> that fired this event.
078     */
079    public EventContext getEventContext() {
080        return (EventContext)getSource();
081    }
082
083    /**
084     * Invokes the <tt>namingExceptionThrown()</tt> method on
085     * a listener using this event.
086     * @param listener The non-null naming listener on which to invoke
087     * the method.
088     */
089    public void dispatch(NamingListener listener) {
090        listener.namingExceptionThrown(this);
091    }
092
093    private static final long serialVersionUID = -4877678086134736336L;
094}
