026package javax.annotation;
027
028import java.lang.annotation.*;
029import static java.lang.annotation.ElementType.*;
030import static java.lang.annotation.RetentionPolicy.*;
031
032/**
033 * The PostConstruct annotation is used on a method that needs to be executed
034 * after dependency injection is done to perform any initialization. This
035 * method MUST be invoked before the class is put into service. This
036 * annotation MUST be supported on all classes that support dependency
037 * injection. The method annotated with PostConstruct MUST be invoked even
038 * if the class does not request any resources to be injected. Only one
039 * method can be annotated with this annotation. The method on which the
040 * PostConstruct annotation is applied MUST fulfill all of the following
041 * criteria -
042- The method MUST NOT have any parameters except in the case of EJB
043 * interceptors   in which case it takes an InvocationC ontext object as
044 * defined by the EJB   specification.
045 * - The return type of the method MUST be void.
046 * - The method MUST NOT throw a checked exception.
047 * - The method on which PostConstruct is applied MAY be public, protected,
048 * package private or private.
049 * - The method MUST NOT be static except for the application client.
050 * - The method MAY be final.
051 * - If the method throws an unchecked exception the class MUST NOT be put into
052 * service except in the case of EJBs where the EJB can handle exceptions and
053 * even   recover from them.
054 * @since Common Annotations 1.0
055 * @see javax.annotation.PreDestroy
056 * @see javax.annotation.Resource
057 */
058@Documented
059@Retention (RUNTIME)
060@Target(METHOD)
061public @interface PostConstruct {
062}
