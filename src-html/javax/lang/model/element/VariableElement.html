026package javax.lang.model.element;
027
028import javax.lang.model.type.TypeMirror;
029import javax.lang.model.util.Elements;
030
031/**
032 * Represents a field, {@code enum} constant, method or constructor
033 * parameter, local variable, resource variable, or exception
034 * parameter.
035 *
036 * @author Joseph D. Darcy
037 * @author Scott Seligman
038 * @author Peter von der Ah&eacute;
039 * @since 1.6
040 */
041public interface VariableElement extends Element {
042
043    /**
044     * Returns the value of this variable if this is a {@code final}
045     * field initialized to a compile-time constant.  Returns {@code
046     * null} otherwise.  The value will be of a primitive type or a
047     * {@code String}.  If the value is of a primitive type, it is
048     * wrapped in the appropriate wrapper class (such as {@link
049     * Integer}).
050     *
051     * <p>Note that not all {@code final} fields will have
052     * constant values.  In particular, {@code enum} constants are
053     * <em>not</em> considered to be compile-time constants.  To have a
054     * constant value, a field's type must be either a primitive type
055     * or {@code String}.
056     *
057     * @return the value of this variable if this is a {@code final}
058     * field initialized to a compile-time constant, or {@code null}
059     * otherwise
060     *
061     * @see Elements#getConstantExpression(Object)
062     * @jls 15.28 Constant Expression
063     * @jls 4.12.4 final Variables
064     */
065    Object getConstantValue();
066}
