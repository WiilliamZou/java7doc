026package javax.lang.model.type;
027
028import javax.lang.model.UnknownEntityException;
029
030/**
031 * Indicates that an unknown kind of type was encountered.  This can
032 * occur if the language evolves and new kinds of types are added to
033 * the {@code TypeMirror} hierarchy.  May be thrown by a {@linkplain
034 * TypeVisitor type visitor} to indicate that the visitor was created
035 * for a prior version of the language.
036 *
037 * @author Joseph D. Darcy
038 * @author Scott Seligman
039 * @author Peter von der Ah&eacute;
040 * @see TypeVisitor#visitUnknown
041 * @since 1.6
042 */
043public class UnknownTypeException extends UnknownEntityException {
044
045    private static final long serialVersionUID = 269L;
046
047    private transient TypeMirror type;
048    private transient Object parameter;
049
050    /**
051     * Creates a new {@code UnknownTypeException}.The {@code p}
052     * parameter may be used to pass in an additional argument with
053     * information about the context in which the unknown type was
054     * encountered; for example, the visit methods of {@link
055     * TypeVisitor} may pass in their additional parameter.
056     *
057     * @param t the unknown type, may be {@code null}
058     * @param p an additional parameter, may be {@code null}
059     */
060    public UnknownTypeException(TypeMirror t, Object p) {
061        super("Unknown type: " + t);
062        type = t;
063        this.parameter = p;
064    }
065
066    /**
067     * Returns the unknown type.
068     * The value may be unavailable if this exception has been
069     * serialized and then read back in.
070     *
071     * @return the unknown type, or {@code null} if unavailable
072     */
073    public TypeMirror getUnknownType() {
074        return type;
075    }
076
077    /**
078     * Returns the additional argument.
079     *
080     * @return the additional argument
081     */
082    public Object getArgument() {
083        return parameter;
084    }
085}
