026package javax.lang.model.type;
027
028
029import java.util.List;
030
031import javax.lang.model.element.ExecutableElement;
032
033
034/**
035 * Represents the type of an executable.  An <i>executable</i>
036 * is a method, constructor, or initializer.
037 *
038 * <p> The executable is
039 * represented as when viewed as a method (or constructor or
040 * initializer) of some reference type.
041 * If that reference type is parameterized, then its actual
042 * type arguments are substituted into any types returned by the methods of
043 * this interface.
044 *
045 * @author Joseph D. Darcy
046 * @author Scott Seligman
047 * @author Peter von der Ah&eacute;
048 * @see ExecutableElement
049 * @since 1.6
050 */
051public interface ExecutableType extends TypeMirror {
052
053    /**
054     * Returns the type variables declared by the formal type parameters
055     * of this executable.
056     *
057     * @return the type variables declared by the formal type parameters,
058     *          or an empty list if there are none
059     */
060    List<? extends TypeVariable> getTypeVariables();
061
062    /**
063     * Returns the return type of this executable.
064     * Returns a {@link NoType} with kind {@link TypeKind#VOID VOID}
065     * if this executable is not a method, or is a method that does not
066     * return a value.
067     *
068     * @return the return type of this executable
069     */
070    TypeMirror getReturnType();
071
072    /**
073     * Returns the types of this executable's formal parameters.
074     *
075     * @return the types of this executable's formal parameters,
076     *          or an empty list if there are none
077     */
078    List<? extends TypeMirror> getParameterTypes();
079
080    /**
081     * Returns the exceptions and other throwables listed in this
082     * executable's {@code throws} clause.
083     *
084     * @return the exceptions and other throwables listed in this
085     *          executable's {@code throws} clause,
086     *          or an empty list if there are none.
087     */
088    List<? extends TypeMirror> getThrownTypes();
089}
